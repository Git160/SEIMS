142 void SEDR_SBAGNOLD::PointSourceLoading() {
144  for (auto it = m_ptSrcFactory.begin(); it != m_ptSrcFactory.end(); it++) {
145  //cout<<"unique Point Source Factory ID: "<<it->first<<endl;
146  vector<int> m_ptSrcMgtSeqs = it->second->GetPointSrcMgtSeqs();
147  map<int, PointSourceMgtParams *> m_pointSrcMgtMap = it->second->GetPointSrcMgtMap();
148  vector<int> m_ptSrcIDs = it->second->GetPointSrcIDs();
149  map<int, PointSourceLocations *> m_pointSrcLocsMap = it->second->GetPointSrcLocsMap();
150  // 1. looking for management operations from m_pointSrcMgtMap
151  for (auto seqIter = m_ptSrcMgtSeqs.begin(); seqIter != m_ptSrcMgtSeqs.end(); seqIter++) {
152  PointSourceMgtParams *curPtMgt = m_pointSrcMgtMap.at(*seqIter);
153  // 1.1 If current day is beyond the date range, then continue to next management
154  if (curPtMgt->GetStartDate() != 0 && curPtMgt->GetEndDate() != 0) {
155  if (m_date < curPtMgt->GetStartDate() || m_date > curPtMgt->GetEndDate()) {
156  continue;
157  }
158  }
159  // 1.2 Otherwise, get the water volume
160  float per_sed = curPtMgt->GetSedment();
161  // 1.3 Sum up all point sources
162  for (auto locIter = m_ptSrcIDs.begin(); locIter != m_ptSrcIDs.end(); locIter++) {
163  if (m_pointSrcLocsMap.find(*locIter) != m_pointSrcLocsMap.end()) {
164  PointSourceLocations *curPtLoc = m_pointSrcLocsMap.at(*locIter);
165  int curSubID = curPtLoc->GetSubbasinID();
167  m_ptSub[curSubID] += per_sed * curPtLoc->GetSize() * 1000.f * m_dt / 86400.f;
168  }
169  }
170  }
171  }

174 int SEDR_SBAGNOLD::Execute() {
175  //check the data
176  CheckInputData();
177  initialOutputs();
179  PointSourceLoading();
180  for (auto it = m_reachLayers.begin(); it != m_reachLayers.end(); it++) {
181  // There are not any flow relationship within each routing layer.
182  // So parallelization can be done here.
183  int nReaches = it->second.size();
184 #pragma omp parallel for
185  for (int i = 0; i < nReaches; ++i) {
186  int reachIndex = it->second[i]; // index in the array, which is equal to reach ID
187  SedChannelRouting(reachIndex);
188  // compute changes in channel dimensions caused by downcutting and widening
189  if (m_VCD) {
190  doChannelDowncuttingAndWidening(reachIndex);
191  }
192  }
193  }
194  return 0;

345 void SEDR_SBAGNOLD::SedChannelRouting(int i) {
346  float qOutV = 0.f; // water volume (m^3) of flow out
347  float allWater = 0.f; // // water in reach during time step, qdin in SWAT
348  // initialize sediment in reach during time step
350  float allSediment = 0.f; // all sediment in reach, kg
351  float sedUp = 0.f; // sediment from upstream channels, kg
352  for (size_t j = 0; j < m_reachUpStream[i].size(); ++j) {
353  int upReachId = m_reachUpStream[i][j];
354  sedUp += m_sedOut[upReachId];
355  }
356  allSediment = sedUp + m_sedtoCh[i] + m_sedStorage[i];
358  if (nullptr != m_ptSub && m_ptSub[i] > 0.f) {
359  allSediment += m_ptSub[i];
360  }
361  //if (i == 4) cout<<"\tall sed: "<<allSediment<<", sed up: "<<sedUp<<", sed to ch: "<<m_sedtoCh[i]<<", strg: "<<m_sedStorage[i]<<endl;
362 
363  // initialize water in reach during time step
364  qOutV = m_qchOut[i] * m_dt; // m^3
365  //allWater = m_chStorage[i] + qOutV;
366  allWater = m_preChStorage[i];
367  if ((m_qchOut[i] < UTIL_ZERO && m_chWTdepth[i] < UTIL_ZERO) || allWater < 0.01f) {
371  m_sedDeg[i] = 0.f;
372  m_sedDep[i] = 0.f;
373  m_sedOut[i] = 0.f;
374  m_sedConc[i] = 0.f;
375  m_rchSand[i] = 0.f;
376  m_rchSilt[i] = 0.f;
377  m_rchClay[i] = 0.f;
378  m_rchSag[i] = 0.f;
379  m_rchLag[i] = 0.f;
380  m_rchGra[i] = 0.f;
381  m_rchBankEro[i] = 0.f;
382  m_rchDeg[i] = 0.f;
383  m_rchDep[i] = 0.f;
384  m_flplainDep[i] = 0.f;
385  m_sedStorage[i] = allSediment;
386  return;
387  }
388  // initialize reach peak runoff rate and calculate flow velocity
389  float peakFlowRate = m_qchOut[i] * m_prf;
390  float crossarea = allWater / m_chLen[i]; // SWAT, eq. 7:1.1.7
391  float peakVelocity = 0.f;
392  if (m_preChWTDepth[i] < 0.01f) {
393  peakVelocity = 0.01f;
394  } else {
395  peakVelocity = peakFlowRate / crossarea;
396  }
397  if (peakVelocity > 5.f) {
398  // cout << "subbasin id: " << i <<", peakVelocity: " << peakVelocity << endl;
399  peakVelocity = 5.f;
400  }
401  float tbase = m_chLen[i] / (m_dt * peakVelocity);
402  if (tbase > 1.f) tbase = 1.f;
403 
404  //if (i == 4) cout << "qchOut: " << m_qchOut[i] << ", allwater: " << allWater <<
405  // ", chLen: " << m_chLen[i] << ", peakVelocity: " << peakVelocity <<
406  // ", m_preChWTDepth: " << m_preChWTDepth[i] << ", tbase: "<< tbase << endl;
407 
409  float initCon = 0.f; // cyin
410  float maxCon = 0.f; // cych
411  float sedDeposition = 0.f; // depnet, and dep
412  float sedDegradation = 0.f; // deg
413  float sedDegradation1 = 0.f; // deg1
414  float sedDegradation2 = 0.f; // deg2
415 
416  //deposition and degradation
417  initCon = allSediment / allWater; // kg/m^3
418  //max concentration
419  maxCon = m_spcon * pow(peakVelocity, m_spexp) * 1000.f; // kg/m^3
420  //if (i == 4) cout<<"iniCon: "<<initCon<<", maxCon: "<<maxCon<<endl;
421  //initial concentration,mix sufficiently
422  sedDeposition = allWater * (initCon - maxCon); // kg
423  if (peakVelocity < m_vcrit) {
424  sedDeposition = 0.f;
425  }
426 
427  if (sedDeposition < 0.f) //degradation
428  {
429  //cout << "initCon < maxCon, subbasin id: " << i << endl;
430  sedDegradation = -sedDeposition * tbase;
431  // first the deposited material will be degraded before channel bed
432  if (sedDegradation >= m_sedDep[i]) {
433  sedDegradation1 = m_sedDep[i];
434  sedDegradation2 = (sedDegradation - sedDegradation1) * m_chErod[i] * m_chCover[i];
435  } else {
436  sedDegradation1 = sedDegradation;
437  sedDegradation2 = 0.f;
438  }
439  sedDeposition = 0.f;
440  } else {
441  sedDegradation = 0.f;
442  sedDegradation1 = 0.f;
443  sedDegradation2 = 0.f;
444  }
445  //update sed deposition
446  m_sedDep[i] += (sedDeposition - sedDegradation1);
447  if (m_sedDep[i] < UTIL_ZERO) m_sedDep[i] = 0.f;
448  m_sedDeg[i] += (sedDegradation1 + sedDegradation2);
449 
450  //get sediment after deposition and degradation
451  allSediment += (sedDegradation1 + sedDegradation2 - sedDeposition);
452  if (allSediment < UTIL_ZERO) allSediment = 0.f;
453  //get out flow water fraction
454  float outFraction = qOutV / allWater;
455  if (outFraction > 1.f) outFraction = 1.f;
456  m_sedOut[i] = allSediment * outFraction;
457  if (m_sedOut[i] < UTIL_ZERO) m_sedOut[i] = 0.f;
458  // update sediment storage
459  m_sedStorage[i] = allSediment - m_sedOut[i];
460  if (m_sedStorage[i] < UTIL_ZERO) m_sedStorage[i] = 0.f;
461 
462  // get final sediment in water, cannot large than 0.848 ton/m3
463  float maxSedinWt = 0.848f * qOutV * 1000.f;
464  if (m_sedOut[i] > maxSedinWt) {
465  m_sedDep[i] += (m_sedOut[i] - maxSedinWt);
466  m_sedOut[i] = maxSedinWt;
467  }
469  m_sedConc[i] = m_sedOut[i] / qOutV;
470  m_rchSand[i] = 0.f;
472  m_rchSilt[i] = m_sedOut[i]; // the sediments is assumed to be silt for mass conservation
473  m_rchClay[i] = 0.f;
474  m_rchSag[i] = 0.f;
475  m_rchLag[i] = 0.f;
476  m_rchGra[i] = 0.f;
