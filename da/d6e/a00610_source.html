54 int SNO_SP::Execute() {
55  CheckInputData();
56  InitialOutputs();
59  if (m_snowCoverCoef1 == NODATA_VALUE || m_snowCoverCoef2 == NODATA_VALUE) {
60  GetScurveShapeParameter(0.5f, 0.95f, m_snowCover50, 0.95f, &m_snowCoverCoef1, &m_snowCoverCoef2);
61  }
63  // which only need to computed once.
64  int d = utils_time::JulianDay(m_date);
65  float sinv = float(sin(2.f * PI / 365.f * (d - 81.f)));
66  float cmelt = (m_csnow6 + m_csnow12) * 0.5f + (m_csnow6 - m_csnow12) * 0.5f * sinv;
67 #pragma omp parallel for
68  for (int rw = 0; rw < m_nCells; rw++) {
70  m_packT[rw] = m_packT[rw] * (1 - m_lagSnow) + m_meanTemp[rw] * m_lagSnow;
72  m_SA[rw] += m_snowAccum[rw] - m_SE[rw];
73  if (m_meanTemp[rw] < m_snowTemp) {
75  m_SA[rw] += m_kblow * m_netPcp[rw];
76  m_netPcp[rw] *= (1.f - m_kblow);
77  }
78 
79  if (m_SA[rw] < 0.01) {
80  m_snowMelt[rw] = 0.f;
81  } else {
82  float dt = m_maxTemp[rw] - m_t0;
83  if (dt < 0) {
84  m_snowMelt[rw] = 0.f; //if temperature is lower than t0, the snowmelt is 0.
85  } else {
86  //calculate using eq. 1:2.5.2 SWAT p58
87  m_snowMelt[rw] = cmelt * ((m_packT[rw] + m_maxTemp[rw]) * 0.5f - m_t0);
88  // adjust for areal extent of snow cover
89  float snowCoverFrac = 0.f; //fraction of HRU area covered with snow
90  if (m_SA[rw] < m_snowCoverMax) {
91  float xx = m_SA[rw] / m_snowCoverMax;
92  snowCoverFrac = xx / (xx + exp(m_snowCoverCoef1 = m_snowCoverCoef2 * xx));
93  } else {
94  snowCoverFrac = 1.f;
95  }
96  m_snowMelt[rw] *= snowCoverFrac;
97  if (m_snowMelt[rw] < 0.f) m_snowMelt[rw] = 0.f;
98  if (m_snowMelt[rw] > m_SA[rw]) m_snowMelt[rw] = m_SA[rw];
99  m_SA[rw] -= m_snowMelt[rw];
100  m_netPcp[rw] += m_snowMelt[rw];
101  if (m_netPcp[rw] < 0.f) m_netPcp[rw] = 0.f;
102  }
103  }
104  }
105  return 0;
106 }
