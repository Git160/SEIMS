183 bool DeleteDirectory(const string& dirpath, bool del_subdirs/* = true */) {
184  string abspath = GetAbsolutePath(dirpath);
185  if (!DirectoryExists(abspath)) return true;
186 #ifdef windows
187  bool b_subdirectory = false; // Flag, indicating whether subdirectories have been found
188  string str_file_path; // Filepath
189  WIN32_FIND_DATA file_information; // File information
190 
191  string str_pattern = abspath + SEP + "*.*";
192  HANDLE h_file = ::FindFirstFile(str_pattern.c_str(), &file_information);
193  if (h_file != INVALID_HANDLE_VALUE) {
194  do {
195  if (file_information.cFileName[0] != '.') {
196  str_file_path.erase();
197  str_file_path = abspath + SEP + file_information.cFileName;
198  if (file_information.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY) {
199  if (del_subdirs) {
200  // Delete subdirectory
201  bool i_rc = DeleteDirectory(str_file_path, del_subdirs);
202  if (!i_rc) return false;
203  } else b_subdirectory = true;
204  } else {
205  // Set file attributes
206  if (::SetFileAttributes(str_file_path.c_str(), FILE_ATTRIBUTE_NORMAL) == FALSE) {
207  return false;
208  }
209  // Delete file
210  if (::DeleteFile(str_file_path.c_str()) == FALSE) {
211  return false;
212  }
213  }
214  }
215  } while (::FindNextFile(h_file, &file_information) == TRUE);
216  // Close handle
217  ::FindClose(h_file);
218 
219  DWORD dw_error = ::GetLastError();
220  if (dw_error != ERROR_NO_MORE_FILES) {
221  return false;
222  }
223  if (!b_subdirectory) {
224  // Set directory attributes
225  if (::SetFileAttributes(abspath.c_str(), FILE_ATTRIBUTE_NORMAL) == FALSE) {
226  return false;
227  }
228  // Delete directory
229  if (::RemoveDirectory(abspath.c_str()) == FALSE) {
230  return false;
231  }
232  }
233  }
234  return true;
235 #else
236  DIR *dir;
237  struct dirent *entry;
238  char path[PATH_MAX];
239 
240  dir = opendir(abspath.c_str());
241  if (dir == nullptr) {
242  perror("Error opendir()");
243  return true;
244  }
245 
246  while ((entry = readdir(dir)) != nullptr) {
247  if (strcmp(entry->d_name, ".") && strcmp(entry->d_name, "..")) {
248  snprintf(path, (size_t) PATH_MAX, "%s/%s", abspath.c_str(), entry->d_name);
249  if (entry->d_type == DT_DIR && del_subdirs) {
250  DeleteDirectory(path, del_subdirs);
251  }
252  printf("Deleting: %s\n", path);
253  remove(path);
254  }
255  }
256  closedir(dir);
257  printf("Deleting: %s\n", abspath.c_str());
258  remove(abspath.c_str());
259  return true;
260 #endif /* windows */
261 }
