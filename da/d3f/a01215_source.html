191  } else {
192  // check to see if there is an aggregate array to add data to
193  if (m_2DData == NULL) {
194  // create the aggregate array
195  m_nRows = nRows;

198  m_Counter = 0;

200 
201  switch (m_AggregationType) {
202  case AT_Average:

215  case AT_Sum:
216 #pragma omp parallel for
217  for (int i = 0; i < m_nRows; i++) {

220  if (FloatEqual(m_2DData[i][j], NODATA_VALUE)) {
221  m_2DData[i][j] = 0.f;
222  }
223  m_2DData[i][j] += data[i][j];
224  }
225  }
226  }
227  break;
228  case AT_Minimum:
229 #pragma omp parallel for
230  for (int i = 0; i < m_nRows; i++) {
231  for (int j = 0; j < m_nLayers; j++) {

234  m_2DData[i][j] = MAXIMUMFLOAT;

237  m_2DData[i][j] = data[i][j];
238  }
239  }

245  for (int i = 0; i < m_nRows; i++) {

247  if (!FloatEqual(data[i][j], NODATA_VALUE)) {

326 void PrintInfoItem::AggregateData(int numrows, float **data, AggregationType type, float NoDataValue) {
327  // check to see if there is an aggregate array to add data to
328  if (m_1DDataWithRowCol == NULL) {
329  // create the aggregate array
330  m_nRows = numrows;
331  m_1DDataWithRowCol = new float *[m_nRows];
332  for (int i = 0; i < m_nRows; i++) {
333  m_1DDataWithRowCol[i] = new float[3];
334  m_1DDataWithRowCol[i][0] = NoDataValue;
335  m_1DDataWithRowCol[i][1] = NoDataValue;
336  m_1DDataWithRowCol[i][2] = NoDataValue;
337  }
338  m_Counter = 0;
339  }
340 
341  // depending on the type of aggregation
342  switch (type) {
343  case AT_Average:
344 #pragma omp parallel for
345  for (int rw = 0; rw < m_nRows; rw++) {
346  if (!FloatEqual(data[rw][2], NoDataValue)) {
347  // initialize value to 0.0 if this is the first time
348  if (FloatEqual(m_1DDataWithRowCol[rw][2], NoDataValue)) m_1DDataWithRowCol[rw][2] = 0.0f;
349  m_1DDataWithRowCol[rw][0] = data[rw][0]; // store the row number
350  m_1DDataWithRowCol[rw][1] = data[rw][1]; // store the column number
351  if (m_Counter == 0) {
352  // first value so average = value
353  m_1DDataWithRowCol[rw][2] = data[rw][2]; // store the value
354  } else {
355  // calculate the incremental average
356  m_1DDataWithRowCol[rw][2] =
357  ((m_1DDataWithRowCol[rw][2] * m_Counter) + data[rw][2]) / (m_Counter + 1);
358  }
359  }
360  }
361  m_Counter++;
362  break;
363  case AT_Sum:
364 #pragma omp parallel for
365  for (int rw = 0; rw < m_nRows; rw++) {
366  if (!FloatEqual(data[rw][2], NoDataValue)) {
367  // initialize value to 0.0 if this is the first time
368  if (FloatEqual(m_1DDataWithRowCol[rw][2], NoDataValue)) m_1DDataWithRowCol[rw][2] = 0.0f;
369  m_1DDataWithRowCol[rw][0] = data[rw][0];
370  m_1DDataWithRowCol[rw][1] = data[rw][1];
371  // add the next value to the current sum
372  m_1DDataWithRowCol[rw][2] += data[rw][2];
373  }
374  }
375  break;
376  case AT_Minimum:
377 #pragma omp parallel for
378  for (int rw = 0; rw < m_nRows; rw++) {
379  if (!FloatEqual(data[rw][2], NoDataValue)) {
380  // initialize value to 0.0 if this is the first time
381  if (FloatEqual(m_1DDataWithRowCol[rw][2], NoDataValue)) m_1DDataWithRowCol[rw][2] = 0.0f;
382  m_1DDataWithRowCol[rw][0] = data[rw][0];
383  m_1DDataWithRowCol[rw][1] = data[rw][1];
384  // if the next value is smaller than the current value
385  if (data[rw][2] <= m_1DDataWithRowCol[rw][2]) {
386  // set the current value to the next (smaller) value
387  m_1DDataWithRowCol[rw][2] = data[rw][2];
388  }
389  }
390  }
391  break;
392  case AT_Maximum:
393 #pragma omp parallel for
394  for (int rw = 0; rw < m_nRows; rw++) {
395  if (!FloatEqual(data[rw][2], NoDataValue)) {
396  // initialize value to 0.0 if this is the first time
397  if (FloatEqual(m_1DDataWithRowCol[rw][2], NoDataValue)) m_1DDataWithRowCol[rw][2] = 0.0f;
398  m_1DDataWithRowCol[rw][0] = data[rw][0];
399  m_1DDataWithRowCol[rw][1] = data[rw][1];
400  // if the next value is larger than the current value
401  if (data[rw][2] >= m_1DDataWithRowCol[rw][2]) {
402  // set the current value to the next (larger) value
403  m_1DDataWithRowCol[rw][2] = data[rw][2];
404  }
405  }
406  }
407  break;
408  default:break;
409  }
410 }
