116 void MUSK_CH::PointSourceLoading() {
118  for (auto it = m_ptSrcFactory.begin(); it != m_ptSrcFactory.end(); ++it) {
120  for (int i = 0; i <= m_nreach; i++) {
121  m_ptSub[i] = 0.f;
122  }
123  //cout<<"unique Point Source Factory ID: "<<it->first<<endl;
124  vector<int>& m_ptSrcMgtSeqs = it->second->GetPointSrcMgtSeqs();
125  map<int, PointSourceMgtParams *>& m_pointSrcMgtMap = it->second->GetPointSrcMgtMap();
126  vector<int>& m_ptSrcIDs = it->second->GetPointSrcIDs();
127  map<int, PointSourceLocations *>& m_pointSrcLocsMap = it->second->GetPointSrcLocsMap();
128  // 1. looking for management operations from m_pointSrcMgtMap
129  for (auto seqIter = m_ptSrcMgtSeqs.begin(); seqIter != m_ptSrcMgtSeqs.end(); ++seqIter) {
130  PointSourceMgtParams* curPtMgt = m_pointSrcMgtMap.at(*seqIter);
131  // 1.1 If current day is beyond the date range, then continue to next management
132  if (curPtMgt->GetStartDate() != 0 && curPtMgt->GetEndDate() != 0) {
133  if (m_date < curPtMgt->GetStartDate() || m_date > curPtMgt->GetEndDate()) {
134  continue;
135  }
136  }
137  // 1.2 Otherwise, get the water volume
138  float per_wtrVol = curPtMgt->GetWaterVolume();
139  // 1.3 Sum up all point sources
140  for (auto locIter = m_ptSrcIDs.begin(); locIter != m_ptSrcIDs.end(); ++locIter) {

142  PointSourceLocations* curPtLoc = m_pointSrcLocsMap.at(*locIter);
143  int curSubID = curPtLoc->GetSubbasinID();

151 int MUSK_CH::Execute() {
152  InitialOutputs();
154  PointSourceLoading();
155  for (auto it = m_rteLyrs.begin(); it != m_rteLyrs.end(); ++it) {
156  // There are not any flow relationship within each routing layer.
157  // So parallelization can be done here.
158  int reachNum = CVT_INT(it->second.size());
159  // the size of m_reachLayers (map) is equal to the maximum stream order
160 #pragma omp parallel for
161  for (int i = 0; i < reachNum; i++) {
162  int reachIndex = it->second[i]; // index in the array, i.e., subbasinID
163  if (m_inputSubbsnID == 0 || m_inputSubbsnID == reachIndex) {
164  // for OpenMP version, all reaches will be executed,
165  // for MPI version, only the current reach will be executed.
166  ChannelFlow(reachIndex);
167  }
168  }
169  }
170  return 0;

367 void MUSK_CH::GetCoefficients(const float reachLength, const float v0, MuskWeights& weights) {
368  float K = (4.64f - 3.64f * m_co1) * reachLength / (5.f * v0 / 3.f);
369 
370  float min = 2.f * K * m_x;
371  float max = 2.f * K * (1.f - m_x);
372  float dt;
373  int n;
374  GetDt(float(m_dt), min, max, dt, n);
375  weights.dt = dt;
376 
377  //get coefficient
378  float temp = max + dt;
379  weights.c1 = (dt - min) / temp;
380  weights.c2 = (dt + min) / temp;
381  weights.c3 = (max - dt) / temp;
382  weights.c4 = 2 * dt / temp;
383  weights.n = n;
384 
385  //make sure any coefficient is positive
386  if (weights.c1 < 0) {
387  weights.c2 += weights.c1;
388  weights.c1 = 0.f;
389  }
390  if (weights.c3 < 0) {
391  weights.c2 += weights.c1;
392  weights.c3 = 0.f;
393  }

409 void MUSK_CH::ChannelFlow(const int i) {
410  float st0 = m_chStorage[i];
411  float qiSub = 0.f;
412  if (nullptr != m_qiSub && m_qiSub[i] >= 0.f) {
413  qiSub = m_qiSub[i];
414  }
415  float qgSub = 0.f;
416  if (nullptr != m_qgSub && m_qgSub[i] >= 0.f) {
417  qgSub = m_qgSub[i];
418  }
419  float ptSub = 0.f;
420  if (nullptr != m_ptSub && m_ptSub[i] >= 0.f) {
421  ptSub = m_ptSub[i];
422  }
424  // first add all the inflow water
425  // 1. water from this subbasin
426  float qIn = m_qsSub[i] + qiSub + qgSub + ptSub + m_deepGroundwater;
427  // 2. water from upstream reaches
428  float qsUp = 0.f;
429  float qiUp = 0.f;
430  float qgUp = 0.f;
431  for (size_t j = 0; j < m_reachUpStream[i].size(); ++j) {
432  int upReachId = m_reachUpStream[i][j];
433  qsUp += m_qsCh[upReachId];
434  qiUp += m_qiCh[upReachId];
435  qgUp += m_qgCh[upReachId];
436  }
437  qIn += (qsUp + qiUp + qgUp); //qIn is equivalent to the wtrin variable in rtmusk.f of SWAT
438  //if(i == 12)
439  // cout <<"surfaceQ: "<< m_qsSub[i] << ", subsurfaceQ: " << qiSub << ", groundQ: " << qgSub << ", pointQ: " << ptSub <<
440  // ", UPsurfaceQ: "<<qsUp<<", UPsubsurface: "<<qiUp<<", UPground: "<<qgUp<<", \n";
441  // 3. water from bank storage
442  float bankOut = m_bankStorage[i] * (1.f - exp(-m_aBank));
443 
444  m_bankStorage[i] -= bankOut;
445  qIn += bankOut / m_dt;
446 
447  // add inflow water to storage
448  //if (i==12) cout<<"initial chStorage: "<<m_chStorage[i]<<", ";
449  m_chStorage[i] += qIn * m_dt;
451  updateWaterWidthDepth(i);
452  //if (i==12) cout<<"added chStorage: "<<m_chStorage[i]<<endl;
453  //if(i == 2) cout <<"qIn:"<< qIn<<", chStorage: "<<m_chStorage[i]<<endl;
455  // then subtract all the outflow water
456  // 1. transmission losses to deep aquifer, which is lost from the system
457  // the unit of kchb is mm/hr, 1. / 1000. / 3600. = 2.7777777777777776e-07
458  float seepage = m_Kchb[i] * 2.7777777777777776e-07f * m_chBtmWidth[i] * m_chLen[i] * m_dt;

462  m_seepage[i] = seepage;
463  m_chStorage[i] -= seepage;
464  } else {
465  m_seepage[i] = m_chStorage[i];
466  m_chStorage[i] = 0.f;
467  }
468  } else {
469  m_seepage[i] = 0.f;
470  }
471 
472  // 2. calculate transmission losses to bank storage
473  //float dch = m_chStorage[i] / (m_chWTWidth[i] * m_chLen[i]);
474  float dch = m_chWTdepth[i];
475  float bankLen = dch * sqrt(1.f + m_chSideSlope[i] * m_chSideSlope[i]);
476  float bankInLoss = 2.f * m_Kbank[i] * 2.7777777777777776e-07f * bankLen * m_chLen[i] * m_dt; // m^3
477  //bankInLoss = 0.f; //TODO, why not consider bankage loss?
478  if (m_chStorage[i] > bankInLoss) {
479  m_chStorage[i] -= bankInLoss;
480  } else {
481  bankInLoss = m_chStorage[i];
482  m_chStorage[i] = 0.f;
483  }
484  // water balance of the bank storage
485  // loss the water from bank storage to the adjacent unsaturated zone and groundwater storage
486  float bankOutGw = m_bankStorage[i] * (1.f - exp(-m_bBank));
487  //bankOutGw = 0.f; //TODO, why not consider?
488  m_bankStorage[i] = m_bankStorage[i] + bankInLoss - bankOutGw;
489  if (nullptr != m_gwStorage) {
490  m_gwStorage[i] += bankOutGw / m_area[i] * 1000.f;
491  } // updated groundwater storage
492 
493  // 3. evaporation losses
494  float et = 0.f;
495  if (nullptr != m_petCh) {
496  et = m_Epch * m_petCh[i] * 0.001f * m_chWTWidth[i] * m_chLen[i]; //m3
497  if (m_chStorage[i] > et) {
498  m_chStorage[i] -= et;
499  } else {
500  et = m_chStorage[i];
501  m_chStorage[i] = 0.f;
502  }
503  }
504  if (FloatEqual(m_chStorage[i], 0.f)) {
505  m_qRchOut[i] = 0.f;
506  m_qsCh[i] = 0.f;
507  m_qiCh[i] = 0.f;
508  m_qgCh[i] = 0.f;
509  m_chWTdepth[i] = 0.f;
510  m_chWTWidth[i] = 0.f;
511  return;
512  }
513 
514  //if(i == 2) cout << "chStorage before routing " << m_chStorage[i] << endl;
516  // routing, there are water in the channel after inflow and transmission loss
517  float totalLoss = m_seepage[i] + bankInLoss + et;
518 
519  m_preChStorage[i] = m_chStorage[i];
520  m_preChWTDepth[i] = m_chWTdepth[i];
521  m_chStorage[i] = st0;
522 
523  // calculate coefficients
524  MuskWeights wt;
525  GetCoefficients(m_chLen[i], m_chVel[i], wt);
526  int n = wt.n;
527  float q = 0.f;
528  for (int j = 0; j < n; j++) {
529  m_qRchOut[i] = wt.c1 * qIn + wt.c2 * m_qIn[i] + wt.c3 * m_qRchOut[i];
530  m_qIn[i] = qIn;
531  float tmp = m_chStorage[i] + (qIn - totalLoss / m_dt - m_qRchOut[i]) * wt.dt;
532  if (tmp < 0.f) {
533  m_qRchOut[i] = m_chStorage[i] / wt.dt + qIn;
534  m_chStorage[i] = 0.f;
535  } else {
536  m_chStorage[i] = tmp;
537  }
538  q += m_qRchOut[i];
539  }
540  m_qRchOut[i] = q / n;
541 
542  float qInSum = m_qsSub[i] + qiSub + qgSub + qsUp + qiUp + qgUp;
543  m_qsCh[i] = m_qRchOut[i] * (m_qsSub[i] + qsUp) / qInSum;
544  m_qiCh[i] = m_qRchOut[i] * (qiSub + qiUp) / qInSum;
545  m_qgCh[i] = m_qRchOut[i] * (qgSub + qgUp) / qInSum;
546 
547  // set variables for next time step
548  m_qIn[i] = qIn;
549  updateWaterWidthDepth(i);
