41  def run(function_name, in_files, in_params = None, out_files = None, mpi_params = None, log_params = None):
42  """
43  Run TauDEM function.
44  The command will not execute if:
45  1) any input file does not exist;
46  2) all output files exist and were last modified no earlier than any input file.
47  An error will be detected after running the TauDEM command if:
48  1) any output file does not exist ;
49  2) any output file was last modified earilier than any input file.
50 
51  :param function_name: TauDEM function name (with or without path) as string
52  :param in_files: Required. Dict of pairs of parameter id (string) and file path (string) for input files.
53  :param in_params: Optional. Dict of pairs of parameter id (string) and value (or None for a flag parameter
54  without a value) for input parameters.
55  :param out_files: Optional. Dict of pairs of parameter id (string) and file path (string) for output files.
56  :param mpi_params: Optional. Dict of pairs of parameter id (string) and value or path for MPI setting.
57  e.g. {'mpipath':'/soft/bin','hostfile':'/soft/bin/cluster.node','n':4} or
58  {'mpipath':'/soft/bin', 'n':4} or {'n':4}
59  :param log_params: Optional. Dict of pairs of parameter id (string) and value or path for runtime and log
60  output parameters. e.g. {'logfile': '/home/user/log.txt'}
61  :return True if TauDEM run successfully, otherwise False.
62  """
63  # check the log parameter
64  log_file = None
65  if log_params is not None:
66  if type(log_params) != dict:
67  TauDEM.error("The log parameter must be a dict!\n")
68  if 'logfile' in log_params and log_params['logfile'] is not None:
69  log_file = log_params['logfile']
70 
71  # check input files
72  if in_files is None:
73  TauDEM.error("Input files parameter is required!\n", log_file)
74  if type(in_files) != dict:
75  TauDEM.error("The input files parameter must be a dict!\n")
76  for (pid, path) in in_files.items():
77  if path is not None and not os.path.exists(path):
78  TauDEM.error("Input files parameter %s: %s is not existed!\n" % (pid, path), log_file)

95  if out_files is not None:

98  for (pid, out_file) in out_files.items():
99  if out_file is not None:

102  # concatenate command line
103  commands = []
104  # MPI header
105  if mpi_params is not None:
106  if type(mpi_params) != dict:
107  TauDEM.error("The MPI settings parameter must be a dict!\n")
108  if 'mpipath' in mpi_params and mpi_params['mpipath'] is not None:
109  commands.append(mpi_params['mpipath'] + os.sep + 'mpiexec')
110  else:

118  else: # If number of processor is less equal than 1, then do not call mpiexec.
119  commands = []

121  commands.append(function_name)

124  if infile is None:

126  if pid[0] != '-':
127  pid = '-' + pid
128  commands.append(pid)
129  commands.append(infile)
130  # append input parameters
131  if in_params is not None:
132  if type(in_params) != dict:
133  TauDEM.error("The input parameters must be a dict!\n")
134  for (pid, v) in in_params.items():
135  if pid[0] != '-':
136  pid = '-' + pid
137  commands.append(pid)
138  # allow for parameter which is an flag without value
139  if v != '' and v is not None:
140  if MathClass.isnumerical(v):
141  commands.append(str(v))
142  else:
143  commands.append(v)
144  # append output parameters
145  if out_files is not None:
146  for (pid, outfile) in out_files.items():
147  if outfile is None:

150  pid = '-' + pid

152  commands.append(outfile)
153  # run command

156  TauDEM.log(runmsg, log_file)
157  return True
