107  @staticmethod
108  def run(function_name, in_files, in_params=None, out_files=None, mpi_params=None,
109  log_params=None):
110  """
111  Run TauDEM function.
112  The command will not execute if:
113  1) any input file does not exist;
114  2) all output files exist and were last modified no earlier than any input file.
115  An error will be detected after running the TauDEM command if:
116  1) any output file does not exist ;
117  2) any output file was last modified earlier than any input file.
118 
119  Args:
120  function_name: TauDEM function name (with or without path) as string
121  in_files: Required. Dict of pairs of parameter id (string) and file path
122  (string or list) for input files.
123  in_params: Optional. Dict of pairs of parameter id (string) and value (or None for a
124  flag parameter without a value) for input parameters.
125  out_files: Optional. Dict of pairs of parameter id (string) and file path (string) for
126  output files.
127  mpi_params: Optional. Dict of pairs of parameter id (string) and value or path for
128  MPI setting.
129  e.g. {'mpipath':'/soft/bin','hostfile':'/soft/bin/cluster.node','n':4} or
130  {'mpipath':'/soft/bin', 'n':4} or {'n':4}
131  log_params: Optional. Dict of pairs of parameter id (string) and value or path for
132  runtime and log output parameters. e.g. {'logfile': '/home/user/log.txt'}
133 
134  Returns:
135  True if TauDEM run successfully, otherwise False.
136  """
137  # check the log parameter
138  log_file = None
139  if log_params is not None:
140  if not isinstance(log_params, dict):
141  TauDEM.error('The log parameter must be a dict!\n')
142  if 'logfile' in log_params and log_params['logfile'] is not None:
143  log_file = log_params['logfile']
144 
145  # check input files
146  if in_files is None:
147  TauDEM.error('Input files parameter is required!\n', log_file)
148  if not isinstance(in_files, dict):
149  TauDEM.error('The input files parameter must be a dict!\n')
150  for (pid, infile) in in_files.items():
151  if infile is None:
152  continue
153  if isinstance(infile, list) or isinstance(infile, tuple):
154  for inf in infile:
155  if inf is not None and not os.path.exists(inf):
156  TauDEM.error('Input files parameter %s: %s is not '
157  'existed!\n' % (pid, inf), log_file)
158  elif len(StringClass.split_string(infile, ' ')) > 0: # not mean to a file
159  continue
160  elif not os.path.exists(infile):
161  TauDEM.error('Input files parameter %s: %s is not existed!\n' % (pid, infile),
162  log_file)
163 
164  # check output files to figure out if current run is necessary.
165  # need_run = True
166  # basedatetime = 0.
167  # for (pid, path) in in_files.items():
168  # if path is not None and os.path.getmtime(path) > basedatetime:
169  # basedatetime = os.path.getmtime(path)
170  # if out_files is not None:
171  # for (pid, out_file) in out_files.items():
172  # if out_file is not None and FileClass.is_up_to_date(out_file, basedatetime):
173  # need_run = False
174  # break
175  # if not need_run:
176  # return True
177 
178  # remove out_files to avoid any file IO related error
179  if out_files is not None:
180  if not isinstance(out_files, dict):
181  TauDEM.error('The output files parameter must be a dict!\n')
182  for (pid, out_file) in out_files.items():
183  if isinstance(out_file, list) or isinstance(out_file, tuple):
184  for outf in out_file:
185  if outf is not None:
186  FileClass.remove_files(outf)
187  elif out_file is not None:
188  FileClass.remove_files(out_file)
189 
190  # concatenate command line
191  commands = []
192  # MPI header
193  if mpi_params is not None:
194  if not isinstance(mpi_params, dict):
195  TauDEM.error('The MPI settings parameter must be a dict!\n')
196  if 'mpipath' in mpi_params and mpi_params['mpipath'] is not None:
197  commands.append(mpi_params['mpipath'] + os.sep + 'mpiexec')
198  else:
199  commands.append('mpiexec')
200  if 'hostfile' in mpi_params and mpi_params['hostfile'] is not None \
201  and not StringClass.string_match(mpi_params['hostfile'], 'none') \
202  and os.path.isfile(mpi_params['hostfile']):
203  commands.append('-f')
204  commands.append(mpi_params['hostfile'])
205  if 'n' in mpi_params and mpi_params['n'] > 1:
206  commands.append('-n')
207  commands.append(str(mpi_params['n']))
208  else: # If number of processor is less equal than 1, then do not call mpiexec.
209  commands = []
210  # append TauDEM function name, which can be full path or just one name
211  commands.append(function_name)
212  # append input files
213  for (pid, infile) in in_files.items():
214  if infile is None:
215  continue
216  if pid[0] != '-':
217  pid = '-' + pid
218  commands.append(pid)
219  if isinstance(infile, list) or isinstance(infile, tuple):
220  commands.append(' '.join(tmpf for tmpf in infile))
221  else:
222  commands.append(infile)
223  # append input parameters
224  if in_params is not None:
225  if not isinstance(in_params, dict):
226  TauDEM.error('The input parameters must be a dict!\n')
227  for (pid, v) in in_params.items():
228  if pid[0] != '-':
229  pid = '-' + pid
230  commands.append(pid)
231  # allow for parameter which is an flag without value
232  if v != '' and v is not None:
233  if MathClass.isnumerical(v):
234  commands.append(str(v))
235  else:
236  commands.append(v)
237  # append output parameters
238  if out_files is not None:
239  for (pid, outfile) in out_files.items():
240  if outfile is None:
241  continue
242  if pid[0] != '-':
243  pid = '-' + pid
244  commands.append(pid)
245  if isinstance(outfile, list) or isinstance(outfile, tuple):
246  commands.append(' '.join(tmpf for tmpf in outfile))
247  else:
248  commands.append(outfile)
249  # run command
250  # print (commands)
251  runmsg = UtilClass.run_command(commands)
252  TauDEM.log(runmsg, log_file)
253  return True
