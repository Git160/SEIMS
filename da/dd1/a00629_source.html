69 void MUSK_CH::InitialOutputs() {
70  CHECK_POSITIVE(MID_MUSK_CH, m_nreach);
71  if (nullptr != m_chStorage) return; // DO NOT Initial Outputs repeatedly.
72  if (m_mskX < 0.f) m_mskX = 0.2f;
73  if (m_mskCoef1 <= 0.f || m_mskCoef2 <= 0.f) {
74  m_mskCoef1 = 0.75f;
75  m_mskCoef2 = 0.25f;
76  } else {
77  // Make sure m_mskCoef1 + m_mskCoef2 = 1.
78  float msk1 = m_mskCoef1 / (m_mskCoef1 + m_mskCoef2);
79  float msk2 = m_mskCoef2 / (m_mskCoef1 + m_mskCoef2);
80  m_mskCoef1 = msk1;
81  m_mskCoef2 = msk2;
82  }
83  //initial channel storage
84  m_chStorage = new(nothrow) float[m_nreach + 1];
85  m_preChStorage = new(nothrow) float[m_nreach + 1];
86  m_qIn = new(nothrow) float[m_nreach + 1];
87  m_qRchOut = new(nothrow) float[m_nreach + 1];
88  m_bankStorage = new(nothrow) float[m_nreach + 1];
89  m_seepage = new(nothrow) float[m_nreach + 1];
90  m_qsCh = new(nothrow) float[m_nreach + 1];
91  m_qiCh = new(nothrow) float[m_nreach + 1];
92  m_qgCh = new(nothrow) float[m_nreach + 1];
93  m_chWTdepth = new(nothrow) float[m_nreach + 1];
94  m_preChWTDepth = new(nothrow) float[m_nreach + 1];
95  m_chWTWidth = new(nothrow) float[m_nreach + 1];
96  m_chBtmWidth = new(nothrow) float[m_nreach + 1];
97 
98  for (int i = 1; i <= m_nreach; i++) {
99  float qiSub = 0.f;
100  float qgSub = 0.f;
101  if (nullptr != m_qiSub) {
102  // interflow (subsurface flow)
103  qiSub = m_qiSub[i];
104  }
105  if (nullptr != m_qgSub) {
106  // groundwater
107  qgSub = m_qgSub[i];
108  }
109  m_seepage[i] = 0.f;
110  m_bankStorage[i] = m_Bnk0 * m_chLen[i];
111  updateChannleBottomWidth(i);
112  m_chWTdepth[i] = m_chDepth[i] * m_Chs0_perc;
113  m_chWTWidth[i] = m_chBtmWidth[i] + 2.f * m_chSideSlope[i] * m_chWTdepth[i];
114  m_preChWTDepth[i] = m_chWTWidth[i];
115  m_chStorage[i] = m_chLen[i] * m_chWTdepth[i] * (m_chBtmWidth[i] + m_chSideSlope[i] * m_chWTdepth[i]);
116  m_preChStorage[i] = m_chStorage[i];
117  m_qIn[i] = 0.f;
118  m_qRchOut[i] = m_qsSub[i] + qiSub + qgSub;
119  m_qsCh[i] = m_qsSub[i];
120  m_qiCh[i] = qiSub;
121  m_qgCh[i] = qgSub;
122  }
124  if (nullptr == m_ptSub) {
125  Initialize1DArray(m_nreach + 1, m_ptSub, 0.f);
126  }

129 void MUSK_CH::PointSourceLoading() {
131  for (auto it = m_ptSrcFactory.begin(); it != m_ptSrcFactory.end(); ++it) {
133  for (int i = 0; i <= m_nreach; i++) {
134  m_ptSub[i] = 0.f;
135  }
136  //cout<<"unique Point Source Factory ID: "<<it->first<<endl;
137  vector<int>& ptSrcMgtSeqs = it->second->GetPointSrcMgtSeqs();
138  map<int, PointSourceMgtParams *>& pointSrcMgtMap = it->second->GetPointSrcMgtMap();
139  vector<int>& ptSrcIDs = it->second->GetPointSrcIDs();
140  map<int, PointSourceLocations *>& pointSrcLocsMap = it->second->GetPointSrcLocsMap();
141  // 1. looking for management operations from m_pointSrcMgtMap
142  for (auto seqIter = ptSrcMgtSeqs.begin(); seqIter != ptSrcMgtSeqs.end(); ++seqIter) {
143  PointSourceMgtParams* curPtMgt = pointSrcMgtMap.at(*seqIter);
144  // 1.1 If current day is beyond the date range, then continue to next management
145  if (curPtMgt->GetStartDate() != 0 && curPtMgt->GetEndDate() != 0) {
146  if (m_date < curPtMgt->GetStartDate() || m_date > curPtMgt->GetEndDate()) {
147  continue;
148  }
149  }
150  // 1.2 Otherwise, get the water volume
151  float per_wtrVol = curPtMgt->GetWaterVolume();
152  // 1.3 Sum up all point sources
153  for (auto locIter = ptSrcIDs.begin(); locIter != ptSrcIDs.end(); ++locIter) {
154  if (pointSrcLocsMap.find(*locIter) != pointSrcLocsMap.end()) {
155  PointSourceLocations* curPtLoc = pointSrcLocsMap.at(*locIter);
156  int curSubID = curPtLoc->GetSubbasinID();
157  m_ptSub[curSubID] += per_wtrVol * curPtLoc->GetSize() / 86400.f;
158  }
159  }
160  }
161  }

164 int MUSK_CH::Execute() {
165  InitialOutputs();
167  PointSourceLoading();
168  for (auto it = m_rteLyrs.begin(); it != m_rteLyrs.end(); ++it) {
169  // There are not any flow relationship within each routing layer.
170  // So parallelization can be done here.
171  int reachNum = CVT_INT(it->second.size());
172  size_t errCount = 0;
173  // the size of m_rteLyrs (map) is equal to the maximum stream order
174 //#pragma omp parallel for reduction(+:errCount)
175  for (int i = 0; i < reachNum; i++) {
176  int reachIndex = it->second[i]; // index in the array, i.e., subbasinID
177  if (m_inputSubbsnID == 0 || m_inputSubbsnID == reachIndex) {
178  // for OpenMP version, all reaches will be executed,
179  // for MPI version, only the current reach will be executed.
180  if (!ChannelFlow(reachIndex)) {
181  errCount++;
182  }
183  }
184  }
185  if (errCount > 0) {
186  throw ModelException(MID_MUSK_CH, "Execute", "Error occurred!");
187  }
188  }
189  return 0;
190 }

423 void MUSK_CH::updateWaterWidthDepth(const int i) {
425  float crossArea = m_chStorage[i] / m_chLen[i];
426  m_chWTdepth[i] = (sqrt(m_chBtmWidth[i] * m_chBtmWidth[i] + 4.f * m_chSideSlope[i] * crossArea) -
427  m_chBtmWidth[i]) * 0.5f / m_chSideSlope[i];
428  if (m_chWTdepth[i] < UTIL_ZERO) {
429  m_chWTWidth[i] = m_chBtmWidth[i];
430  } else {
431  m_chWTWidth[i] = m_chBtmWidth[i] + 2.f * m_chSideSlope[i] * m_chWTdepth[i];
432  }

434 void MUSK_CH::updateChannleBottomWidth(const int i) {
435  // Code from ttcoef.f of SWAT source code.
436  m_chBtmWidth[i] = m_chWidth[i] - 2.f * m_chSideSlope[i] * m_chDepth[i];
437  if (m_chBtmWidth[i] <= UTIL_ZERO) {
438  m_chBtmWidth[i] = 0.5f * m_chWidth[i];
439  m_chSideSlope[i] = (m_chWidth[i] - m_chBtmWidth[i]) * 0.5f / m_chDepth[i];
440  }

443 bool MUSK_CH::ChannelFlow(const int i) {
444  // 1. first add all the inflow water
445  // 1.1. water from this subbasin
446  float qiSub = 0.f;
447  if (nullptr != m_qiSub && m_qiSub[i] >= 0.f) {
448  qiSub = m_qiSub[i];
449  }
450  float qgSub = 0.f;
451  if (nullptr != m_qgSub && m_qgSub[i] >= 0.f) {
452  qgSub = m_qgSub[i];
453  }
454  float ptSub = 0.f;
455  if (nullptr != m_ptSub && m_ptSub[i] >= 0.f) {
456  ptSub = m_ptSub[i];
457  }
458  float qIn = m_qsSub[i] + qiSub + qgSub + ptSub + m_deepGroundwater;
459  // 1.2. water from upstream reaches
460  float qsUp = 0.f;
461  float qiUp = 0.f;
462  float qgUp = 0.f;
463  for (auto upRchID = m_reachUpStream.at(i).begin(); upRchID != m_reachUpStream.at(i).end(); ++upRchID) {
464  if (m_qsCh[*upRchID] != m_qsCh[*upRchID]) {
465  cout << "DayOfYear: " << m_dayOfYear << ", rchID: " << i << ", upRchID: " << *upRchID <<
466  ", surface part illegal!" << endl;
467  return false;
468  }
469  if (m_qiCh[*upRchID] != m_qiCh[*upRchID]) {
470  cout << "DayOfYear: " << m_dayOfYear << ", rchID: " << i << ", upRchID: " << *upRchID <<
471  ", subsurface part illegal!" << endl;
472  return false;
473  }
474  if (m_qgCh[*upRchID] != m_qgCh[*upRchID]) {
475  cout << "DayOfYear: " << m_dayOfYear << ", rchID: " << i << ", upRchID: " << *upRchID <<
476  ", groundwater part illegal!" << endl;
477  return false;
478  }
479  if (m_qsCh[*upRchID] > 0.f) qsUp += m_qsCh[*upRchID];
480  if (m_qiCh[*upRchID] > 0.f) qiUp += m_qiCh[*upRchID];
481  if (m_qgCh[*upRchID] > 0.f) qgUp += m_qgCh[*upRchID];
482  }
483  qIn += qsUp + qiUp + qgUp;
484 #ifdef PRINT_DEBUG
485  cout << "ID: " << i << ", surfaceQ: " << m_qsSub[i] << ", subsurfaceQ: " << qiSub <<
486  ", groundQ: " << qgSub << ", pointQ: " << ptSub <<
487  ", UPsurfaceQ: " << qsUp << ", UPsubsurface: " << qiUp << ", UPground: " << qgUp << endl;
488 #endif
489  // 1.3. water from bank storage
490  float bankOut = m_bankStorage[i] * (1.f - exp(-m_aBank));
491  m_bankStorage[i] -= bankOut;
492  qIn += bankOut / m_dt;
493 
494  // add inflow water to storage
495  float st0 = m_chStorage[i];
496  m_chStorage[i] += qIn * m_dt;
498  updateWaterWidthDepth(i);
499 
500  // Compute storage time constant for reach
501  // The storage time constant calculated for the reach segment with bankfull flows(s).
502  float k_bankfull = 0.f;
503  // The storage time constant calculated for the reach segment with one-tenth of the bankfull flows(s).
504  float k_bankfull2 = 0.f;
505  // Compute flow and travel time at bankfull depth, from ttcoef.f of SWAT source code.
506  // TODO, these two calculation should be moved to common_algorithm lib, which will be used when channel changes.
507  float d = m_chDepth[i];
508  float p = m_chBtmWidth[i] + 2.f * d * sqrt(m_chSideSlope[i] * m_chSideSlope[i] + 1.f);
509  float a = m_chBtmWidth[i] * d + m_chSideSlope[i] * d * d;
510  float rh = a / p;
511  k_bankfull = m_chLen[i] / (manningQ(1.f, rh, m_chMan[i], m_chSlope[i]) * 5.f / 3.f) / 3600.f;
512  // compute flow and travel time at 0.1 bankfull depth
513  d = 0.1f * m_chDepth[i];
514  p = m_chBtmWidth[i] + 2.f * d * sqrt(m_chSideSlope[i] * m_chSideSlope[i] + 1.f);
515  a = m_chBtmWidth[i] * d + m_chSideSlope[i] * d * d;
516  rh = a / p;
517  k_bankfull2 = m_chLen[i] / (manningQ(1.f, rh, m_chMan[i], m_chSlope[i]) * 5.f / 3.f) / 3600.f; // sec->hour
518  float xkm = k_bankfull * m_mskCoef1 + k_bankfull2 * m_mskCoef2; // Eq. 7:1.4.9 in SWAT Theory 2009.
519  // Check Muskingum numerical stability
520  float detmax = 2.f * xkm * (1.f - m_mskX);
521  float detmin = 2.f * xkm * m_mskX;
522  // Discretize time interval to meet the stability criterion
523  float det = 24.f; // hours
524  int nn = 0;
525  if (det > detmax) {
526  if (det / 2.f <= detmax) {
527  det = 12.f;
528  nn = 2;
529  } else if (det / 4.f <= detmax) {
530  det = 6.f;
531  nn = 4;
532  } else {
533  det = 1;
534  nn = 24;
535  }
536  } else {
537  det = 24;
538  nn = 1;
539  }
540  // get coefficients of Muskingum
541  float temp = detmax + det;
542  float c1 = (det - detmin) / temp;
543  float c2 = (det + detmin) / temp;
544  float c3 = (detmax - det) / temp;
545  // make sure any coefficient is positive
546  //if (c1 < 0) {
547  // c2 += c1;
548  // c1 = 0.f;
549  //}
550  //if (c3 < 0) {
551  // c2 += c1;
552  // c3 = 0.f;
553  //}
554 
556  // 2. Then subtract all the outflow water
557  // 2.1. transmission losses to deep aquifer, which is lost from the system
558  // the unit of kchb is mm/hr, 1. / 1000. / 3600. = 2.7777777777777776e-07
559  float seepage = m_Kchb[i] * 2.7777777777777776e-07f * m_chBtmWidth[i] * m_chLen[i] * m_dt;
560  if (qgSub < UTIL_ZERO) {
561  if (m_chStorage[i] > seepage) {
562  m_seepage[i] = seepage;
563  m_chStorage[i] -= seepage;
564  } else {
565  m_seepage[i] = m_chStorage[i];
566  m_chStorage[i] = 0.f;
567  }
568  } else {
569  m_seepage[i] = 0.f;
570  }
571 
572  // 2.2. calculate transmission losses to bank storage
573  float dch = m_chWTdepth[i];
574  float bankLen = dch * sqrt(1.f + m_chSideSlope[i] * m_chSideSlope[i]);
575  float bankInLoss = 2.f * m_Kbank[i] * 2.7777777777777776e-07f * bankLen * m_chLen[i] * m_dt; // m^3
576  if (m_chStorage[i] > bankInLoss) {
577  m_chStorage[i] -= bankInLoss;
578  } else {
579  bankInLoss = m_chStorage[i];
580  m_chStorage[i] = 0.f;
581  }
582  // water balance of the bank storage
583  // loss the water from bank storage to the adjacent unsaturated zone and groundwater storage
584  float bankOutGw = m_bankStorage[i] * (1.f - exp(-m_bBank));
585  m_bankStorage[i] += bankInLoss - bankOutGw;
586  if (nullptr != m_gwStorage) {
587  m_gwStorage[i] += bankOutGw / m_area[i] * 1000.f;
588  } // updated groundwater storage
589 
590  // 2.3. evaporation losses
591  float et = 0.f;
592  if (nullptr != m_petCh) {
593  et = m_Epch * m_petCh[i] * 0.001f * m_chWTWidth[i] * m_chLen[i]; //m3
594  if (m_chStorage[i] > et) {
595  m_chStorage[i] -= et;
596  } else {
597  et = m_chStorage[i];
598  m_chStorage[i] = 0.f;
599  }
600  }
601  if (FloatEqual(m_chStorage[i], 0.f)) {
602  m_qRchOut[i] = 0.f;
603  m_qsCh[i] = 0.f;
604  m_qiCh[i] = 0.f;
605  m_qgCh[i] = 0.f;
606  m_chWTdepth[i] = 0.f;
607  m_chWTWidth[i] = 0.f;
608  return true;
609  }
610 #ifdef PRINT_DEBUG
611  cout << " chStorage before routing " << m_chStorage[i] << endl;
612 #endif
613  // routing, there are water in the channel after inflow and transmission loss
615  float totalLoss = m_seepage[i] + bankInLoss + et;
616 
617  m_preChStorage[i] = m_chStorage[i];
618  m_preChWTDepth[i] = m_chWTdepth[i];
619  m_chStorage[i] = st0;
620 
621  // calculate coefficients
622  //MuskWeights wt;
623  //if (FloatEqual(m_chVel[i], 0.f)) m_chVel[i] = MIN_SLOPE;
624  //GetCoefficients(m_chLen[i], m_chVel[i], wt);
625  //int n = wt.n;
626  float q = 0.f;
627  for (int j = 0; j < nn; j++) {
628  m_qRchOut[i] = c1 * qIn + c2 * m_qIn[i] + c3 * m_qRchOut[i];
629  m_qIn[i] = qIn; // m^3/s
630  float tmp = m_chStorage[i] + (qIn - totalLoss / m_dt - m_qRchOut[i]) * det * 3600.f;
631  if (tmp < 0.f) {
632  m_qRchOut[i] = m_chStorage[i] / (det * 3600.f) + qIn;
633  m_chStorage[i] = 0.f;
634  } else {
635  m_chStorage[i] = tmp;
636  }
637  q += m_qRchOut[i];
638  }
639  m_qRchOut[i] = q / nn;
640 
641  float qInSum = m_qsSub[i] + qiSub + qgSub + qsUp + qiUp + qgUp;
642  if (qInSum < UTIL_ZERO) {
643  // In case of divided by zero.
644  m_qsCh[i] = 0.f;
645  m_qiCh[i] = 0.f;
646  m_qgCh[i] = 0.f;
647  m_qRchOut[i] = 0.f;
648  } else {
649  m_qsCh[i] = m_qRchOut[i] * (m_qsSub[i] + qsUp) / qInSum;
650  m_qiCh[i] = m_qRchOut[i] * (qiSub + qiUp) / qInSum;
651  m_qgCh[i] = m_qRchOut[i] * (qgSub + qgUp) / qInSum;
652  }
653 
654  // set variables for next time step
655  m_qIn[i] = qIn;
656  updateWaterWidthDepth(i);
657 #ifdef PRINT_DEBUG
658  cout << " chStorage after routing " << m_chStorage[i] << endl;
659  cout << " surfq: " << m_qsCh[i] << ", ifluq: " << m_qiCh[i] << ", groudq: " << m_qgCh[i] << endl;
660 #endif
661  return true;
662 }
