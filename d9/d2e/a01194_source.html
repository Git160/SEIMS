60 clsSubbasins::clsSubbasins(MongoGridFS* spatialData, map<string, clsRasterData<float> *> &rsMap,
61  int prefixID): m_nSubbasins(-1) {
62  m_subbasinIDs.clear();
63  m_subbasinsInfo.clear();
64 
65 // read subbasin data
66  int nCells = -1;
67  float *subbasinData = NULL;
68  float cellWidth = NODATA_VALUE;
69  ostringstream oss;
70  oss << prefixID << "_" << VAR_SUBBSN;
71  string subbasinFileName = GetUpper(oss.str());
72  oss.str("");
73 
74  rsMap[subbasinFileName]->getRasterData(&nCells, &subbasinData);
75 
76  m_nSubbasins = 0;
77 // valid cell indexes of each subbasin, key is subbasin ID, value is vector of cell's index
78  map<int, vector<int> *>cellListMap;
79  for (int i = 0; i < nCells; i++) {
80  int subID = int(subbasinData[i]);
81  if (cellListMap.find(subID) == cellListMap.end())
82  cellListMap[subID] = new vector<int>;
83  cellListMap[subID]->push_back(i);
84  }
85  m_nSubbasins = cellListMap.size();
86  for (map<int, vector<int> *>::iterator it = cellListMap.begin(); it != cellListMap.end(); it++) {
87 // swap for saving memory
88  vector<int>(*it->second).swap(*it->second);
89  int subID = it->first;
90  m_subbasinIDs.push_back(subID);
91  Subbasin *newSub = new Subbasin(subID);
92  int nCellsTmp = it->second->size();
93  int *tmp = new int[nCellsTmp];
94  for (int j = 0; j < nCellsTmp; j++)
95  tmp[j] = it->second->at(j);
96  newSub->setCellList(nCellsTmp, tmp);
97  newSub->setArea(cellWidth * cellWidth * nCellsTmp);
98  m_subbasinsInfo[subID] = newSub;
99  }
100  vector<int>(m_subbasinIDs).swap(m_subbasinIDs);
101 
102  // release cellListMap to save memory
103  for (map<int, vector<int> *>::iterator it = cellListMap.begin(); it != cellListMap.end();) {
104  if (it->second != NULL)
105  delete it->second;
106  cellListMap.erase(it++);
107  }
108  cellListMap.clear();
109 }

160 float clsSubbasins::subbasin2basin(string key) {
161  float temp = 0.f;
162  int totalCount = 0;
163  int nCount;
164  Subbasin *sub = NULL;
165  for (vector<int>::iterator it = m_subbasinIDs.begin(); it != m_subbasinIDs.end(); it++) {
166  sub = m_subbasinsInfo[*it];
167  nCount = sub->getCellCount();
168  if (StringMatch(key, VAR_SLOPE)) {
169  temp += atan(sub->getSlope()) * nCount;
170  } else if (StringMatch(key, VAR_PET)) {
171  temp += sub->getPET() * nCount;
172  } else if (StringMatch(key, VAR_PERCO)) {
173  temp += sub->getPerco() * nCount;
174  } else if (StringMatch(key, VAR_REVAP)) {
175  temp += sub->getEG() * nCount;
176  } else if (StringMatch(key, VAR_PERDE)) {
177  temp += sub->getPerde() * nCount;
178  } else if (StringMatch(key, VAR_RG)) {
179  temp += sub->getRG() * nCount;
180  } else if (StringMatch(key, VAR_QG)) {
181  temp += sub->getQG();
182  } else if (StringMatch(key, VAR_GW_Q)) {
183  temp += sub->getGW() * nCount;
184  }
185 
186  totalCount += nCount;
187  }
188  if (StringMatch(key, VAR_QG)) {
189  return temp;
190  } // basin sum
191  if (StringMatch(key, VAR_SLOPE))
192  return tan(temp / totalCount);
193  return temp / totalCount; // basin average
194 }
