96 int clsPI_MCS::Execute() {
97  //check input data
98  CheckInputData();
100  InitialOutputs();
101 
102  int julian = JulianDay(m_date);
103 #pragma omp parallel for
104  for (int i = 0; i < m_nCells; i++) {
105  if (m_pcp[i] > 0.f) {
106 #ifdef STORM_MODE
107  m_P[i] = m_P[i] * m_hilldt * 0.0002777777777777778f * cos(atan(m_slope[i]));
110 #endif // STORM_MODE
111  //interception storage capacity, 1. / 365. = 0.0027397260273972603
112  float degree = 2.f * PI * (julian - 87.f) * 0.0027397260273972603f;
114  float min = m_minIntcpStoCap[i];
115  float max = m_maxIntcpStoCap[i];
116  float capacity = min + (max - min) * pow(0.5f + 0.5f * sin(degree), m_intcpStoCapExp);
117 
118  //interception, currently, m_st[i] is storage of (t-1) time step
119  float availableSpace = capacity - m_canSto[i];
120  if (availableSpace < 0) {
121  availableSpace = 0.f;
122  }
123 
124  if (availableSpace < m_pcp[i]) {
125  m_intcpLoss[i] = availableSpace;
126  //if the cell is paddy, by default 15% part of pcp will be allocated to embankment area
127  if (CVT_INT(m_landUse[i]) == LANDUSE_ID_PADDY) {
128  //water added into ditches from low embankment, should be added to somewhere else.
129  float pcp2canal = m_pcp[i] * m_pcp2CanalFr * m_embnkFr;
130  m_netPcp[i] = m_pcp[i] - m_intcpLoss[i] - pcp2canal;
131  } else {
132  //net precipitation
133  m_netPcp[i] = m_pcp[i] - m_intcpLoss[i];
134  }
135  } else {
136  m_intcpLoss[i] = m_pcp[i];
137  m_netPcp[i] = 0.f;
138  }
139 
140  m_canSto[i] += m_intcpLoss[i];
141  } else {
142  m_intcpLoss[i] = 0.f;
143  m_netPcp[i] = 0.f;
144  }
145 #ifndef STORM_MODE
146  //evaporation
147  if (m_canSto[i] > m_pet[i]) {
148  m_IntcpET[i] = m_pet[i];
149  } else {
150  m_IntcpET[i] = m_canSto[i];
151  }
152  m_canSto[i] -= m_IntcpET[i];
153 #endif
154  }
155  return 0;
156 }
