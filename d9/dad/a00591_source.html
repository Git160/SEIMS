58 int DepressionFS::Execute() {
59  InitialOutputs();
60  if (m_checkInput) {
61  CheckInputData();
62  m_checkInput = false;
63  }
64 
65 #pragma omp parallel for
66  for (int i = 0; i < m_nCells; ++i) {
67  // sr is temporarily used to stored the water depth including the depression storage
68  float hWater = m_sr[i];
69  if (hWater <= m_depCap[i]) {
70  m_sd[i] = hWater;
71  m_sr[i] = 0.f;
72  } else {
73  m_sd[i] = m_depCap[i];
74  m_sr[i] = hWater - m_depCap[i];
75  }
76  m_storageCapSurplus[i] = m_depCap[i] - m_sd[i];
77  }
78  return 0;
79 }
