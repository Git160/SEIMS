49 void MUSLE_AS::InitialOutputs() {
50  CHECK_POSITIVE(MID_MUSLE_AS, m_nCells);
51  if (nullptr == m_eroSed) Initialize1DArray(m_nCells, m_eroSed, 0.f);
52  if (nullptr == m_eroSand) Initialize1DArray(m_nCells, m_eroSand, 0.f);
53  if (nullptr == m_eroSilt) Initialize1DArray(m_nCells, m_eroSilt, 0.f);
54  if (nullptr == m_eroClay) Initialize1DArray(m_nCells, m_eroClay, 0.f);
55  if (nullptr == m_eroSmAgg) Initialize1DArray(m_nCells, m_eroSmAgg, 0.f);
56  if (nullptr == m_eroLgAgg) Initialize1DArray(m_nCells, m_eroLgAgg, 0.f);
57  if (nullptr == m_usleLS) {
58  float constant = 3.451378310759016f; // pow(22.13f, 0.4f);
59  m_usleLS = new(nothrow) float[m_nCells];
60  m_slopeForPq = new(nothrow) float[m_nCells];
61 #pragma omp parallel for
62  for (int i = 0; i < m_nCells; i++) {
63  if (m_usleC[i] < 0.001f) {
64  m_usleC[i] = 0.001f;
65  }
66  if (m_usleC[i] > 1.0f) {
67  m_usleC[i] = 1.0f;
68  }
69  float lambda_i1 = m_flowAccm[i] * m_cellWth; // m
70  float lambda_i = lambda_i1 + m_cellWth;
71  float L = pow(lambda_i, 1.4f) - pow(lambda_i1, 1.4f); // m^1.4
72  L /= m_cellWth * constant;
73  float S = pow(sin(atan(m_slope[i])) * 11.160714285714286f, 1.3f); // 1. / 0.0896 = 11.160714285714286
74 
75  m_usleLS[i] = L * S; // LS factor
76  m_slopeForPq[i] = pow(m_slope[i] * 1000.f, 0.16f);
77  }
78  }
79  if (FloatEqual(m_cellAreaKM, NODATA_VALUE)) {
80  m_cellAreaKM = m_cellWth * m_cellWth * 0.000001f; // pow(m_cellWidth / 1000.f, 2.f);
81  m_cellAreaKM1 = 3.79f * pow(m_cellAreaKM, 0.7f);
82  m_cellAreaKM2 = 0.903f * pow(m_cellAreaKM, 0.017f);
83  }
84 }

86 int MUSLE_AS::Execute() {
87  CheckInputData();
88  InitialOutputs();
89 #pragma omp parallel for
90  for (int i = 0; i < m_nCells; i++) {
91  if (m_surfRf[i] < 0.0001f || m_rchID[i] > 0) {
92  m_eroSed[i] = 0.f;
93  m_eroSand[i] = 0.f;
94  m_eroSilt[i] = 0.f;
95  m_eroClay[i] = 0.f;
96  m_eroSmAgg[i] = 0.f;
97  m_eroLgAgg[i] = 0.f;
98  continue;
99  }
100  // peak flow, 1. / 25.4 = 0.03937007874015748
101  float q = m_cellAreaKM1 * m_slopeForPq[i] * pow(m_surfRf[i] * 0.03937007874015748f, m_cellAreaKM2);
102  // sediment yield
103  float Y = 11.8f * pow(m_surfRf[i] * m_cellAreaKM * 1000.0f * q,
104  0.56f) * m_usleK[i][0] * m_usleLS[i] * m_usleC[i] * m_usleP[i];
105  // the snow pack effect
106  if (m_snowAccum[i] > 0.0001f) {
107  Y /= exp(3.f * m_snowAccum[i] * 0.03937007874015748f);
108  }
109  m_eroSed[i] = Y * 1000.f;
110 
112  m_eroSand[i] = m_eroSed[i] * m_detSand[i];
113  m_eroSilt[i] = m_eroSed[i] * m_detSilt[i];
114  m_eroClay[i] = m_eroSed[i] * m_detClay[i];
115  m_eroSmAgg[i] = m_eroSed[i] * m_detSmAgg[i];
116  m_eroLgAgg[i] = m_eroSed[i] * m_detLgAgg[i];
117  }
118  return 0;
119 }
