66  unique_ptr<MongoCollection> collection(new MongoCollection(conn->getCollection(bmpDBName, m_arealSrcMgtTab)));
67  mongoc_cursor_t* cursor = collection->ExecuteQuery(b);
68 
69  bson_iter_t iter;
70  const bson_t *bsonTable;
71 
73  int count = 1;
74  while (mongoc_cursor_next(cursor, &bsonTable)) {
75  m_arealSrcMgtSeqs.push_back(count);
76  m_arealSrcMgtMap[count] = new ArealSourceMgtParams(bsonTable, iter);

83 void BMPArealSrcFactory::ReadArealSourceLocations(MongoClient* conn, string &bmpDBName) {
84  bson_t *b = bson_new();
85  bson_t *child1 = bson_new();
86  BSON_APPEND_DOCUMENT_BEGIN(b, "$query", child1);
87  BSON_APPEND_INT32(child1, BMP_ARSRC_FLD_CODE, m_arealSrc);
88  bson_append_document_end(b, child1);
89  bson_destroy(child1);
90 
91  unique_ptr<MongoCollection> collection(new MongoCollection(conn->getCollection(bmpDBName, m_arealSrcDistTab)));
92  mongoc_cursor_t* cursor = collection->ExecuteQuery(b);
93 
94  bson_iter_t iter;
95  const bson_t *bsonTable;
96 
97  while (mongoc_cursor_next(cursor, &bsonTable)) {
98  ArealSourceLocations *curArSrcLoc = new ArealSourceLocations(bsonTable, iter);
99  int curArSrcID = curArSrcLoc->GetArealSourceID();
100  if (ValueInVector(curArSrcID, m_arealSrcIDs)) {
101  m_arealSrcLocsMap[curArSrcID] = curArSrcLoc;
102  } else {
103  RemoveValueInVector(curArSrcID, m_arealSrcIDs);
104  }
105  }
106  bson_destroy(b);
107  mongoc_cursor_destroy(cursor);
