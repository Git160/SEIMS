102  @staticmethod
103  def run(function_name, in_files, in_params=None, out_files=None, mpi_params=None,
104  log_params=None):
105  """
106  Run TauDEM function.
107  The command will not execute if:
108  1) any input file does not exist;
109  2) all output files exist and were last modified no earlier than any input file.
110  An error will be detected after running the TauDEM command if:
111  1) any output file does not exist ;
112  2) any output file was last modified earlier than any input file.
113 
114  Args:
115  function_name: TauDEM function name (with or without path) as string
116  in_files: Required. Dict of pairs of parameter id (string) and file path (string)
117  for input files.
118  in_params: Optional. Dict of pairs of parameter id (string) and value (or None for a
119  flag parameter without a value) for input parameters.
120  out_files: Optional. Dict of pairs of parameter id (string) and file path (string) for
121  output files.
122  mpi_params: Optional. Dict of pairs of parameter id (string) and value or path for
123  MPI setting.
124  e.g. {'mpipath':'/soft/bin','hostfile':'/soft/bin/cluster.node','n':4} or
125  {'mpipath':'/soft/bin', 'n':4} or {'n':4}
126  log_params: Optional. Dict of pairs of parameter id (string) and value or path for
127  runtime and log output parameters. e.g. {'logfile': '/home/user/log.txt'}
128 
129  Returns:
130  True if TauDEM run successfully, otherwise False.
131  """
132  # check the log parameter
133  log_file = None
134  if log_params is not None:
135  if not isinstance(log_params, dict):
136  TauDEM.error("The log parameter must be a dict!\n")
137  if 'logfile' in log_params and log_params['logfile'] is not None:
138  log_file = log_params['logfile']
139 
140  # check input files
141  if in_files is None:
142  TauDEM.error("Input files parameter is required!\n", log_file)
143  if not isinstance(in_files, dict):
144  TauDEM.error("The input files parameter must be a dict!\n")
145  for (pid, path) in in_files.items():
146  if path is not None and not os.path.exists(path):
147  TauDEM.error("Input files parameter %s: %s is not existed!\n" % (pid, path),
148  log_file)
149 
150  # check output files to figure out if current run is necessary.
151  # need_run = True
152  # basedatetime = 0.
153  # for (pid, path) in in_files.items():
154  # if path is not None and os.path.getmtime(path) > basedatetime:
155  # basedatetime = os.path.getmtime(path)
156  # if out_files is not None:
157  # for (pid, out_file) in out_files.items():
158  # if out_file is not None and FileClass.is_up_to_date(out_file, basedatetime):
159  # need_run = False
160  # break
161  # if not need_run:
162  # return True
163 
164  # remove out_files to avoid any file IO related error
165  if out_files is not None:
166  if not isinstance(out_files, dict):
167  TauDEM.error("The output files parameter must be a dict!\n")
168  for (pid, out_file) in out_files.items():
169  if out_file is not None:
170  FileClass.remove_files(out_file)
171 
172  # concatenate command line
173  commands = []
174  # MPI header
175  if mpi_params is not None:
176  if not isinstance(mpi_params, dict):
177  TauDEM.error("The MPI settings parameter must be a dict!\n")
178  if 'mpipath' in mpi_params and mpi_params['mpipath'] is not None:
179  commands.append(mpi_params['mpipath'] + os.sep + 'mpiexec')
180  else:
181  commands.append('mpiexec')
182  if 'hostfile' in mpi_params and mpi_params['hostfile'] is not None:
183  commands.append('-f')
184  commands.append(mpi_params['hostfile'])
185  if 'n' in mpi_params and mpi_params['n'] > 1:
186  commands.append('-n')
187  commands.append(str(mpi_params['n']))
188  else: # If number of processor is less equal than 1, then do not call mpiexec.
189  commands = []
190  # append TauDEM function name, which can be full path or just one name
191  commands.append(function_name)
192  # append input files
193  for (pid, infile) in in_files.items():
194  if infile is None:
195  continue
196  if pid[0] != '-':
197  pid = '-' + pid
198  commands.append(pid)
199  commands.append(infile)
200  # append input parameters
201  if in_params is not None:
202  if not isinstance(in_params, dict):
203  TauDEM.error("The input parameters must be a dict!\n")
204  for (pid, v) in in_params.items():
205  if pid[0] != '-':
206  pid = '-' + pid
207  commands.append(pid)
208  # allow for parameter which is an flag without value
209  if v != '' and v is not None:
210  if MathClass.isnumerical(v):
211  commands.append(str(v))
212  else:
213  commands.append(v)
214  # append output parameters
215  if out_files is not None:
216  for (pid, outfile) in out_files.items():
217  if outfile is None:
218  continue
219  if pid[0] != '-':
220  pid = '-' + pid
221  commands.append(pid)
222  commands.append(outfile)
223  # run command
224  # print (commands)
225  runmsg = UtilClass.run_command(commands)
226  TauDEM.log(runmsg, log_file)
227  return True

393  @staticmethod
394  def watershed_delineation(bin_dir, mpi_bin, np, dem, outlet_file, thresh, d8_down_method,
395  namecfg, logfile=None):
396  """Watershed Delineation."""
397  # 1. Check directories
398  tau_dir = namecfg.workspace
399  UtilClass.mkdir(tau_dir)
400  # 2. Check log file
401  if logfile is not None and FileClass.is_file_exists(logfile):
402  os.remove(logfile)
403  # 3. Get predefined intermediate file names
404  filled_dem = namecfg.filldem
405  flow_dir = namecfg.d8flow
406  slope = namecfg.slp
407  flow_dir_dinf = namecfg.dinf
408  slope_dinf = namecfg.dinf_slp
409  dir_code_dinf = namecfg.dinf_d8dir
410  weight_dinf = namecfg.dinf_weight
411  acc = namecfg.d8acc
412  stream_raster = namecfg.stream_raster
413  modified_outlet = namecfg.outlet_m
414  stream_skeleton = namecfg.stream_pd
415  acc_with_weight = namecfg.d8acc_weight
416  stream_order = namecfg.stream_order
417  ch_network = namecfg.channel_net
418  ch_coord = namecfg.channel_coord
419  stream_net = namecfg.streamnet_shp
420  subbasin = namecfg.subbsn
421  dist2_stream_d8 = namecfg.dist2stream_d8
422 
423  # 4. perform calculation
424  UtilClass.writelog(logfile, "[Output] %d..., %s" % (10, "fill DEM..."), 'a')
425  TauDEM.fill(np, tau_dir, dem, filled_dem, mpi_bin, bin_dir)
426  UtilClass.writelog(logfile, "[Output] %d..., %s" %
427  (20, "Calculating D8 and Dinf flow direction..."), 'a')
428  TauDEM.d8flowdir(np, tau_dir, filled_dem, flow_dir, slope, mpi_bin, bin_dir)
429  TauDEM.dinfflowdir(np, tau_dir, filled_dem, flow_dir_dinf, slope_dinf, mpi_bin, bin_dir)
430  DinfUtil.output_compressed_dinf(flow_dir_dinf, dir_code_dinf, weight_dinf)
431  UtilClass.writelog(logfile, "[Output] %d..., %s" % (30, "D8 flow accumulation..."), 'a')
432  TauDEM.aread8(np, tau_dir, flow_dir, acc, None, None, mpi_bin, bin_dir)
433  UtilClass.writelog(logfile, "[Output] %d..., %s" %
434  (40, "Generating stream raster initially..."), 'a')
435  min_accum, max_accum, mean_accum, std_accum = RasterUtilClass.raster_statistics(acc)
436  TauDEM.threshold(np, tau_dir, acc, stream_raster, mean_accum, mpi_bin, bin_dir)
437  UtilClass.writelog(logfile, "[Output] %d..., %s" % (50, "Moving outlet to stream..."), 'a')
438  TauDEM.moveoutletstostrm(np, tau_dir, flow_dir, stream_raster, outlet_file,
439  modified_outlet, mpi_bin, bin_dir)
440  UtilClass.writelog(logfile, "[Output] %d..., %s" %
441  (60, "Generating stream skeleton..."), 'a')
442  TauDEM.peukerdouglas(np, tau_dir, filled_dem, stream_skeleton, mpi_bin, bin_dir)
443  UtilClass.writelog(logfile, "[Output] %d..., %s" %
444  (70, "Flow accumulation with outlet..."), 'a')
445  TauDEM.aread8(np, tau_dir, flow_dir, acc_with_weight, modified_outlet, stream_skeleton,
446  mpi_bin, bin_dir)
447 
448  if thresh <= 0: # find the optimal threshold using dropanalysis function
449  UtilClass.writelog(logfile, "[Output] %d..., %s" %
450  (75, "Drop analysis to select optimal threshold..."), 'a')
451  min_accum, max_accum, mean_accum, std_accum = \
452  RasterUtilClass.raster_statistics(acc_with_weight)
453  if mean_accum - std_accum < 0:
454  minthresh = mean_accum
455  else:
456  minthresh = mean_accum - std_accum
457  maxthresh = mean_accum + std_accum
458  numthresh = 20
459  logspace = 'true'
460  drp_file = namecfg.drptxt
461  TauDEM.dropanalysis(np, tau_dir, filled_dem, flow_dir, acc_with_weight,
462  acc_with_weight, modified_outlet, minthresh, maxthresh,
463  numthresh, logspace, drp_file, mpi_bin, bin_dir)
464  if not FileClass.is_file_exists(drp_file):
465  raise RuntimeError("Dropanalysis failed and drp.txt was not created!")
466  drpf = open(drp_file, "r")
467  temp_contents = drpf.read()
468  (beg, thresh) = temp_contents.rsplit(' ', 1)
469  print (thresh)
470  drpf.close()
471  UtilClass.writelog(logfile, "[Output] %d..., %s" % (80, "Generating stream raster..."), 'a')
472  TauDEM.threshold(np, tau_dir, acc_with_weight, stream_raster, float(thresh),
473  mpi_bin, bin_dir)
474  UtilClass.writelog(logfile, "[Output] %d..., %s" % (90, "Generating stream net..."), 'a')
475  TauDEM.streamnet(np, tau_dir, filled_dem, flow_dir, acc_with_weight, stream_raster,
476  modified_outlet, stream_order, ch_network,
477  ch_coord, stream_net, subbasin, mpi_bin, bin_dir)
478  UtilClass.writelog(logfile, "[Output] %d..., %s" %
479  (95, "Calculating distance to stream (D8)..."), 'a')
480  TauDEM.d8distdowntostream(np, tau_dir, flow_dir, filled_dem, stream_raster,
481  dist2_stream_d8, d8_down_method, 1, mpi_bin, bin_dir)
482  UtilClass.writelog(logfile, "[Output] %d.., %s" %
483  (100, "Original subbasin delineation is finished!"), 'a')
