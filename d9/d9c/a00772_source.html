67 int IUH_OL::Execute() {
68  this->CheckInputData();
69  this->initialOutputs();
70  // delete value of last time step
71  for (int n = 0; n <= m_nSubbasins; n++) {
72  m_Q_SBOF[n] = 0.f;
73  }
74 
75 #pragma omp parallel for
76  for (int i = 0; i < m_nCells; i++) {
77  //forward one time step
78  for (int j = 0; j < m_cellFlowCols; j++) {
79  if (j != m_cellFlowCols - 1) {
80  m_cellFlow[i][j] = m_cellFlow[i][j + 1];
81  } else {
82  m_cellFlow[i][j] = 0.f;
83  }
84  }
85 
86  float v_rs = m_rs[i];
87  if (v_rs > 0.f) {
88  int min = int(this->m_iuhCell[i][0]);
89  int max = int(this->m_iuhCell[i][1]);
90  int col = 2;
91  for (int k = min; k <= max; k++) {
92  m_cellFlow[i][k] += v_rs / 1000.f * m_iuhCell[i][col] * m_cellArea / m_TimeStep;
93  col++;
94  }
95  }
96  }
97 #pragma omp parallel for
98  for (int i = 0; i < m_nCells; i++) {
99  //add today's flow
100  int subi = (int) m_subbasin[i];
101  if (m_nSubbasins == 1) {
102  subi = 1;
103  } else if (subi >= m_nSubbasins + 1) {
104  throw ModelException(MID_IUH_OL, "Execute", "The subbasin " + ValueToString(subi) + " is invalid.");
105  }
106 
107  m_Q_SBOF[subi] += m_cellFlow[i][0]; //get new value
108 
109  m_OL_Flow[i] = m_cellFlow[i][0];
110  m_OL_Flow[i] = m_OL_Flow[i] * m_TimeStep * 1000.f / m_cellArea; // m3/s -> mm
111  //if(i == 1000) cout << m_OL_Flow[i] << endl;
112  }
113 
114  //float tmp = 0.f;
115 //#pragma omp parallel for reduction(+:tmp) // there is no need to use omp
116  for (int n = 1; n <= m_nSubbasins; n++) {
117  //get overland flow routing for entire watershed.
118  //tmp += m_Q_SBOF[n];
119  m_Q_SBOF[0] += m_Q_SBOF[n];
120  }
121 
122  return 0;
123 }
