302 int NutrientMovementViaWater::Execute() {
303  CheckInputData();
304  InitialOutputs();
305  // reset variables for the new timestep
306 #pragma omp parallel for
307  for (int i = 0; i < m_nCells; i++) {
308  m_latNO3[i] = 0.f;
309  }
310 
311  for (int ilyr = 0; ilyr < m_nRteLyrs; ilyr++) {
312  // There are not any flow relationship within each routing layer.
313  // So parallelization can be done here.
314  int ncells = CVT_INT(m_rteLyrs[ilyr][0]);
315 #pragma omp parallel for
316  for (int icell = 1; icell <= ncells; icell++) {
317  int i = CVT_INT(m_rteLyrs[ilyr][icell]); // cell ID
318  if (m_rchID[i] > 0) continue; // Skip the reach (stream) cells
319  NitrateLoss(i);
320  PhosphorusLoss(i);
321  SubbasinWaterQuality(i); // compute chl-a, CBOD and dissolved oxygen loadings
322  }
323  }
324  SumBySubbasin();
325  return 0;
326 }

328 void NutrientMovementViaWater::NitrateLoss(const int i) {
329  float percnlyr = 0.f;
330  for (int k = 0; k < CVT_INT(m_nSoilLyrs[i]); k++) {
331  // add nitrate leached from layer above (kg/ha)
332  m_soilNO3[i][k] += percnlyr;
333  if (m_soilNO3[i][k] < 1.e-6f) {
334  m_soilNO3[i][k] = 0.f;
335  }
336  // determine concentration of nitrate in mobile water
337  float sro = k == 0 ? m_surfRf[i] : 0.f; // surface runoff generated
338  float mw = 0.f; // amount of mobile water in the layer (vv in swat nlch.f)
339  float vno3 = 0.f; // kg/ha
340  float con = 0.f; // 100 mg/L
341  float ww = 0.f;
342  // Calculate the concentration of nitrate in the mobile water (con),
343  // equation 4:2.1.2, 4:2.1.3 and 4:2.1.4 in SWAT Theory 2009, p269
344  mw = m_soilPerco[i][k] + sro + m_subSurfRf[i][k] + 1.e-10f;
345  if (m_drainLyr[i] == k) mw += m_qtile;
346  float satportion = (1.f - m_anionExclFr[i]) * m_soilSat[i][k];
347  ww = -mw / satportion;
348  vno3 = m_soilNO3[i][k] * (1.f - exp(ww)); // kg/ha
349  if (mw > 1.e-10f) con = Max(vno3 / mw, 0.f);
350 
351  // calculate nitrate in surface runoff
352  // concentration of nitrate in surface runoff (cosurf)
353  float cosurf = 0.f;
354  if (m_isep_opt == 2)
355  cosurf = 1.f * con; // N percolation does not apply to failing septic place;
356  else
357  cosurf = m_nperco * con;
358  if (k == 0) {
359  m_surfRfNO3[i] = m_surfRf[i] * cosurf; // kg/ha
360  m_surfRfNO3[i] = Min(m_surfRfNO3[i], m_soilNO3[i][k]);
361  m_soilNO3[i][k] -= m_surfRfNO3[i];
362  }
363  // TODO: calculate nitrate in tile flow
364  if (m_drainLyr[i] == k) {
365  }
366  // nitrate moved with subsuface flow (kg/ha)
367  float ssfnlyr = k == 0 ? cosurf * m_subSurfRf[i][k] : con * m_subSurfRf[i][k]; // kg/ha
368  // calculate nitrate in lateral flow
369  // nitrate transported in lateral flow from layer (ssfnlyr)
370  ssfnlyr = Min(ssfnlyr, m_soilNO3[i][k]);
371  m_latNO3[i] += ssfnlyr;
372  // move the lateral no3 flow to the downslope cell (routing considered)
373  m_soilNO3[i][k] -= ssfnlyr;
374  int id_downstream = CVT_INT(m_flowOutIdxD8[i]);
375  if (id_downstream >= 0) m_soilNO3[id_downstream][k] += m_latNO3[i];
377 
378  // calculate nitrate in percolate
379  percnlyr = con * m_soilPerco[i][k];
380  percnlyr = Min(percnlyr, m_soilNO3[i][k]);
381  m_soilNO3[i][k] -= percnlyr;
382  }
383  // calculate nitrate leaching from soil profile
384  m_percoN[i] = percnlyr; // percolation of the last soil layer, kg/ha
385

387  // have been added to it's downslope cell. by LJ

393 void NutrientMovementViaWater::PhosphorusLoss(const int i) {
394  // amount of P leached from soil layer, kg/ha (vap)
395  float vap = 0.f;
396  float vap_tile = 0.f;
397  // compute soluble P lost in surface runoff
398  // variable to hold intermediate calculation result
399  float xx = m_soilBD[i][0] * m_soilDepth[i][0] * m_phoskd;
400  // units ==> surqsolp = [kg/ha * mm] / [t/m^3 * mm * m^3/t] = kg/ha
401  m_surfRfSolP[i] = m_soilSolP[i][0] * m_surfRf[i] / xx;
402  m_surfRfSolP[i] = Min(m_surfRfSolP[i], m_soilSolP[i][0]);
403  m_surfRfSolP[i] = Max(m_surfRfSolP[i], 0.f);
404  m_soilSolP[i][0] = m_soilSolP[i][0] - m_surfRfSolP[i];
405 
406  // compute soluble P leaching
407  vap = m_soilSolP[i][0] * m_soilPerco[i][0] / (m_cvtWt[i][0] / 1000.f * m_pperco);
408  vap = Min(vap, 0.5f * m_soilSolP[i][0]);
409  m_soilSolP[i][0] -= vap;
410 
411  // estimate soluble p in tiles due to crack flow
412  if (m_drainLyr[i] > 0) {
413  xx = Min(1.f, m_soilCrk[i] / 3.f);
414  vap_tile = xx * vap;
415  vap = vap - vap_tile;
416  }
417  if (m_nSoilLyrs[i] >= 2) {
418  m_soilSolP[i][1] += vap;
419  }
420  for (int k = 1; k < CVT_INT(m_nSoilLyrs[i]); k++) {
421  vap = m_soilSolP[i][k] * m_soilPerco[i][k] / (m_cvtWt[i][k] * 0.001f * m_pperco);
422  vap = Min(vap, 0.2f * m_soilSolP[i][k]);
423  m_soilSolP[i][k] -= vap;
424  if (k < m_nSoilLyrs[i] - 1) {
425  // this is slightly different with swat code in solp.f
426  m_soilSolP[i][k + 1] += vap; //leach to next layer
427  } else {
428  m_percoP[i] = vap;
429  } //leach to groundwater
430  }

433 void NutrientMovementViaWater::SubbasinWaterQuality(const int i) {
437 
438  // total amount of water entering main channel on current day, mm
439  float qdr = m_surfRf[i] + m_subSurfRf[i][0] + m_qtile;
440  if (qdr > 1.e-4f) {
441  // kilo moles of phosphorus in nutrient loading to main channel (tp)
442  float tp = 100.f * (m_sedorgn[i] + m_surfRfNO3[i]) / qdr; //100*kg/ha/mm = ppm
443  // regional adjustment on sub chla_a loading, the default value is 40
444  float chla_subco = 40.f;
445  m_surfRfChlA[i] = chla_subco * tp;
446  m_surfRfChlA[i] *= 0.001f; // um/L to mg/L
447 
448  // calculate enrichment ratio
449  if (m_olWtrEroSed[i] < 1e-4) m_olWtrEroSed[i] = 0.f;
450  float enratio = CalEnrichmentRatio(m_olWtrEroSed[i], m_surfRf[i], m_cellArea);
451 
452  // calculate organic carbon loading to main channel
453  float org_c = 0.f;
454  if (m_cbnModel == 2) {
455  org_c = m_sedLossCbn[i] * m_cellArea;
456  } else {
457  org_c = m_soilCbn[i][0] * 0.01f * enratio * (m_olWtrEroSed[i] * 0.001f) * 1000.f;
458  }
459  // calculate carbonaceous biological oxygen demand (CBOD) and COD(transform from CBOD)
460  float cbodu = 2.7f * org_c / (qdr * m_cellWth * m_cellWth * 1.e-6f); // mg/L
461  // convert cbod to cod
462  // The translation relationship is combined Wang Cai-Qin et al. (2014) with
463  // Guo and Long (1994); Xie et al. (2000); Jin et al. (2005).
464  float cod = m_cod_n * (cbodu * (1.f - exp(-5.f * m_cod_k)));
465  m_surfRfCod[i] = m_surfRf[i] * 0.001f * cod * 10.f; // mg/L converted to kg/ha
466  } else {
467  m_surfRfChlA[i] = 0.f;
468  m_surfRfCod[i] = 0.f;
469  }
470 }
