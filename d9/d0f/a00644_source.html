497 void Nutrient_Transformation::MineralizationStaticCarbonMethod(const int i) {
498  for (int k = 0; k < CVT_INT(m_nSoilLyrs[i]); k++) {
499  //soil layer used to compute soil water and soil temperature factors
500  int kk = k == 0 ? 1 : k;
501  float sut = 0.f;
502  // mineralization can occur only if temp above 0 deg
503  if (m_soilTemp[i] > 0) {
504  // compute soil water factor (sut)
505  if (m_soilWtrSto[i][kk] < 0) {
506  m_soilWtrSto[i][kk] = 0.0000001f;
507  }
508  sut = 0.1f + 0.9f * sqrt(m_soilWtrSto[i][kk] / m_soilFC[i][kk]);
509  sut = Max(0.05f, sut);
510 
511  //compute soil temperature factor
512  //variable to hold intermediate calculation result (xx)
513  float xx = 0.f;
514  //soil temperature factor (cdg)
515  float cdg = 0.f;
516  xx = m_soilTemp[i];
517  cdg = 0.9f * xx / (xx + exp(9.93f - 0.312f * xx)) + 0.1f;
518  cdg = Max(0.1f, cdg);
519 
520  // compute combined factor
521  xx = 0.f;
522  //combined temperature/soil water factor (csf)
523  float csf = 0.f;
524  xx = cdg * sut;
525  if (xx < 0) {
526  xx = 0.f;
527  }
528  if (xx > 1000000) {
529  xx = 1000000.f;
530  }
531  csf = sqrt(xx);
532 
533  // compute flow from active to stable pools
534  //amount of nitrogen moving from active organic to stable organic pool in layer (rwn)
535  float rwn = 0.f;
536  rwn = 0.1e-4f * (m_soilActvOrgN[i][k] * (1.f / m_orgNFrActN - 1.f) - m_soilStabOrgN[i][k]);
537  if (rwn > 0.f) {
538  rwn = Min(rwn, m_soilActvOrgN[i][k]);
539  } else {
540  rwn = -(Min(Abs(rwn), m_soilStabOrgN[i][k]));
541  }
542  m_soilStabOrgN[i][k] = Max(1.e-6f, m_soilStabOrgN[i][k] + rwn);
543  m_soilActvOrgN[i][k] = Max(1.e-6f, m_soilActvOrgN[i][k] - rwn);
544 
545  // compute humus mineralization on active organic n
546  //amount of nitrogen moving from active organic nitrogen pool to nitrate pool in layer (hmn)
547  float hmn = 0.f;
548  hmn = m_minrlCoef * csf * m_soilActvOrgN[i][k];
549  hmn = Min(hmn, m_soilActvOrgN[i][k]);
550  // compute humus mineralization on active organic p
551  xx = 0.f;
552  //amount of phosphorus moving from the organic pool to the labile pool in layer (hmp)
553  float hmp = 0.f;
554  xx = m_soilStabOrgN[i][k] + m_soilActvOrgN[i][k];
555  if (xx > 1.e-6f) {
556  hmp = 1.4f * hmn * m_soilHumOrgP[i][k] / xx;
557  } else {
558  hmp = 0.f;
559  }
560  hmp = Min(hmp, m_soilHumOrgP[i][k]);
561 
562  // move mineralized nutrients between pools;
563  m_soilActvOrgN[i][k] = Max(1.e-6f, m_soilActvOrgN[i][k] - hmn);
564  m_soilNO3[i][k] = m_soilNO3[i][k] + hmn;
565  m_soilHumOrgP[i][k] = m_soilHumOrgP[i][k] - hmp;
566  m_soilSolP[i][k] = m_soilSolP[i][k] + hmp;
567 
568  // compute residue decomp and mineralization of
569  // fresh organic n and p (upper two layers only)
570  //amount of nitrogen moving from fresh organic to nitrate(80%) and active organic(20%) pools in layer (rmn1)
571  float rmn1 = 0.f;
572  //amount of phosphorus moving from fresh organic to labile(80%) and organic(20%) pools in layer (rmp)
573  float rmp = 0.f;
574  if (k <= 2) {
575  //the C:N ratio (cnr)
576  float cnr = 0.f;
577  //the C:P ratio (cnr)
578  float cpr = 0.f;
579  //the nutrient cycling water factor for layer (ca)
580  float ca = 0.f;
581  float cnrf = 0.f;
582  float cprf = 0.f;
583  if (m_soilFrshOrgN[i][k] + m_soilNO3[i][k] > 1e-4f) {
584  //Calculate cnr, equation 3:1.2.5 in SWAT Theory 2009, p189
585  cnr = 0.58f * m_soilRsd[i][k] / (m_soilFrshOrgN[i][k] + m_soilNO3[i][k]);
586  if (cnr > 500) {
587  cnr = 500.f;
588  }
589  cnrf = exp(-0.693f * (cnr - 25.f) * 0.04f);
590  } else {
591  cnrf = 1.f;
592  }
593  if (m_soilFrshOrgP[i][k] + m_soilSolP[i][k] > 1e-4f) {
594  cpr = 0.58f * m_soilRsd[i][k] / (m_soilFrshOrgP[i][k] + m_soilSolP[i][k]);
595  if (cpr > 5000) {
596  cpr = 5000.f;
597  }
598  cprf = 0.f;
599  cprf = exp(-0.693f * (cpr - 200.f) * 0.005f);
600  } else {
601  cprf = 1.f;
602  }
603  //decay rate constant (decr)
604  float decr = 0.f;
605 
606  float rdc = 0.f;
607  //Calculate ca, equation 3:1.2.8 in SWAT Theory 2009, p190
608  ca = Min(Min(cnrf, cprf), 1.f);
609  //if (m_landcover[i] > 0)
610  //{
611  // decr = m_rsdco_pl[(int) m_landcover[i]] * ca * csf;
612  //} }else{
613  //{
614  // decr = 0.05f;
615  //}
616  if (m_pltRsdDecCoef[i] < 0.f) {
617  decr = 0.05f;
618  } else {
619  decr = m_pltRsdDecCoef[i] * ca * csf;
620  }
621  decr = Min(decr, 1.f);
622  m_soilRsd[i][k] = Max(1.e-6f, m_soilRsd[i][k]);
623  rdc = decr * m_soilRsd[i][k];
624  m_soilRsd[i][k] = m_soilRsd[i][k] - rdc;
625  if (m_soilRsd[i][k] < 0)m_soilRsd[i][k] = 0.f;
626  rmn1 = decr * m_soilFrshOrgN[i][k];
627  m_soilFrshOrgP[i][k] = Max(1.e-6f, m_soilFrshOrgP[i][k]);
628  rmp = decr * m_soilFrshOrgP[i][k];
629 
630  m_soilFrshOrgP[i][k] = m_soilFrshOrgP[i][k] - rmp;
631  m_soilFrshOrgN[i][k] = Max(1.e-6f, m_soilFrshOrgN[i][k]) - rmn1;
632 
633  //Calculate no3, aorgn, solp, orgp, equation 3:1.2.9 in SWAT Theory 2009, p190
634  m_soilNO3[i][k] = m_soilNO3[i][k] + 0.8f * rmn1;
635  m_soilActvOrgN[i][k] = m_soilActvOrgN[i][k] + 0.2f * rmn1;
636  m_soilSolP[i][k] = m_soilSolP[i][k] + 0.8f * rmp;
637  m_soilHumOrgP[i][k] = m_soilHumOrgP[i][k] + 0.2f * rmp;
638  }
639  // compute denitrification
640  //amount of nitrogen lost from nitrate pool in layer due to denitrification
641  float wdn = 0.f;
642  //Calculate wdn, equation 3:1.4.1 and 3:1.4.2 in SWAT Theory 2009, p194
643  if (sut >= m_denitThres) {
644  wdn = m_soilNO3[i][k] * (1.f - exp(-m_denitCoef * cdg * m_soilCbn[i][k]));
645  } else {
646  wdn = 0.f;
647  }
648  m_soilNO3[i][k] = m_soilNO3[i][k] - wdn;
649 
650  // Summary calculations
651  m_wshd_hmn = m_wshd_hmn + hmn * m_cellAreaFr;
652  m_wshd_rwn = m_wshd_rwn + rwn * m_cellAreaFr;
653  m_wshd_hmp = m_wshd_hmp + hmp * m_cellAreaFr;
654  m_wshd_rmn = m_wshd_rmn + rmn1 * m_cellAreaFr;
655  m_wshd_rmp = m_wshd_rmp + rmp * m_cellAreaFr;
656  m_wshd_dnit = m_wshd_dnit + wdn * m_cellAreaFr;
657  m_hmntl[i] = hmn;
658  m_rwntl[i] = rwn;
659  m_hmptl[i] = hmp;
660  m_rmn2tl[i] = rmn1;
661  m_rmptl[i] = rmp;
662  m_wdntl[i] = wdn;
663  }

671 void Nutrient_Transformation::Volatilization(const int i) {
672  float kk = 0.f;
673  for (int k = 0; k < CVT_INT(m_nSoilLyrs[i]); k++) {
674  //nitrification/volatilization temperature factor (nvtf)
675  float nvtf = 0.f;
676  //Calculate nvtf, equation 3:1.3.1 in SWAT Theory 2009, p192
677  nvtf = 0.41f * (m_soilTemp[i] - 5.f) * 0.1f;
678  if (m_soilNH4[i][k] > 0.f && nvtf >= 0.001f) {
679  float sw25 = 0.f;
680  float swwp = 0.f;
681  //nitrification soil water factor (swf)
682  float swf = 0.f;
683  //Calculate nvtf, equation 3:1.3.2 and 3:1.3.3 in SWAT Theory 2009, p192
684  sw25 = m_soilWP[i][k] + 0.25f * m_soilFC[i][k];
685  swwp = m_soilWP[i][k] + m_soilWtrSto[i][k];
686  if (swwp < sw25) {
687  swf = (swwp - m_soilWP[i][k]) / (sw25 - m_soilWP[i][k]);
688  } else {
689  swf = 1.f;
690  }
691  kk = k == 0 ? 0.f : m_soilDepth[i][k - 1];
692  //depth from the soil surface to the middle of the layer (dmidl/mm)
693  float dmidl = 0.f;
694  //volatilization depth factor (dpf)
695  float dpf = 0.f;
696  //nitrification regulator (akn)
697  float akn = 0.f;
698  //volatilization regulator (akn)
699  float akv = 0.f;
700  //amount of ammonium converted via nitrification and volatilization in layer (rnv)
701  float rnv = 0.f;
702  //amount of nitrogen moving from the NH3 to the NO3 pool (nitrification) in the layer (rnit)
703  float rnit = 0.f;
704  //amount of nitrogen lost from the NH3 pool due to volatilization (rvol)
705  float rvol = 0.f;
706  //volatilization CEC factor (cecf)
707  float cecf = 0.15f;
708  //Calculate nvtf, equation 3:1.3.2 and 3:1.3.3 in SWAT Theory 2009, p192
709  dmidl = (m_soilDepth[i][k] + kk) * 0.5f;
710  dpf = 1.f - dmidl / (dmidl + exp(4.706f - 0.0305f * dmidl));
711  //Calculate rnv, equation 3:1.3.6, 3:1.3.7 and 3:1.3.8 in SWAT Theory 2009, p193
712  akn = nvtf * swf;
713  akv = nvtf * dpf * cecf;
714  rnv = m_soilNH4[i][k] * (1.f - exp(-akn - akv));
715  //Calculate rnit, equation 3:1.3.9 in SWAT Theory 2009, p193
716  rnit = 1.f - exp(-akn);
717  //Calculate rvol, equation 3:1.3.10 in SWAT Theory 2009, p193
718  rvol = 1.f - exp(-akv);
719 
720  //calculate nitrification (NH3 => NO3)
721  if (rvol + rnit > 1.e-6f) {
722  //Calculate the amount of nitrogen removed from the ammonium pool by nitrification,
723  //equation 3:1.3.11 in SWAT Theory 2009, p193
724  rvol = rnv * rvol / (rvol + rnit);
725  //Calculate the amount of nitrogen removed from the ammonium pool by volatilization,
726  //equation 3:1.3.12 in SWAT Theory 2009, p194
727  rnit = rnv - rvol;
728  if (rnit < 0)rnit = 0.f;
729  m_soilNH4[i][k] = Max(1e-6f, m_soilNH4[i][k] - rnit);
730  }
731  if (m_soilNH4[i][k] < 0.f) {
732  rnit = rnit + m_soilNH4[i][k];
733  m_soilNH4[i][k] = 0.f;
734  }
735  //debug
736  //if (rnit != rnit)
737  // cout<<"cellid: "<<i<<", lyr: "<<k<<", akn: "<<akn<<", nvtf:"<<
738  // nvtf<<", swf: "<<swf<<", dpf: "<<dpf<<", cecf: "<<cecf<<endl;
739  m_soilNO3[i][k] = m_soilNO3[i][k] + rnit;
740  //calculate ammonia volatilization
741  m_soilNH4[i][k] = Max(1e-6f, m_soilNH4[i][k] - rvol);
742  if (m_soilNH4[i][k] < 0.f) {
743  rvol = rvol + m_soilNH4[i][k];
744  m_soilNH4[i][k] = 0.f;
745  }
746  //summary calculations
747  m_wshd_voln += rvol * m_cellAreaFr;
748  m_wshd_nitn += rnit * m_cellAreaFr;
749  }
750  }

753 void Nutrient_Transformation::CalculatePflux(const int i) {
754  for (int k = 0; k < CVT_INT(m_nSoilLyrs[i]); k++) {
755  // make sure that no zero or negative pool values come in
756  if (m_soilSolP[i][k] <= 1.e-6) m_soilSolP[i][k] = 1.e-6f;
757  if (m_soilActvMinP[i][k] <= 1.e-6) m_soilActvMinP[i][k] = 1.e-6f;
758  if (m_soilStabMinP[i][k] <= 1.e-6) m_soilStabMinP[i][k] = 1.e-6f;
759 
760  // Convert kg/ha to ppm so that it is more meaningful to compare between soil layers
761  float solp = 0.f;
762  float actp = 0.f;
763  float stap = 0.f;
764  solp = m_soilSolP[i][k] / m_conv_wt[i][k] * 1000000.f;
765  actp = m_soilActvMinP[i][k] / m_conv_wt[i][k] * 1000000.f;
766  stap = m_soilStabMinP[i][k] / m_conv_wt[i][k] * 1000000.f;
767 
768  // ***************Soluble - Active Transformations***************
769  // Dynamic PSP Ratio
770  float psp = 0.f;
771  //cout << psp << endl;
772  //PSP = -0.045*log (% clay) + 0.001*(Solution P, mg kg-1) - 0.035*(% Organic C) + 0.43
773  if (m_soilClay[i][k] > 0.f) {
774  psp = -0.045f * log(m_soilClay[i][k]) + 0.001f * solp;
775  psp = psp - 0.035f * m_soilCbn[i][k] + 0.43f;
776  } else {
777  psp = 0.4f;
778  }
779  // Limit PSP range
780  if (psp < .1f) psp = 0.1f;
781  if (psp > 0.7f) psp = 0.7f;
782 
783  // Calculate smoothed PSP average
784  if (m_phpSorpIdxBsn > 0.f) psp = (m_phpSorpIdxBsn * 29.f + psp * 1.f) * 0.03333333333333333f; // 1. / 30.
785  // Store PSP for tomorrow's smoothing calculation
786  m_phpSorpIdxBsn = psp;
787 
788  //***************Dynamic Active/Soluble Transformation Coeff******************
789  // Calculate P balance
790  float rto = psp / (1.f - psp);
791  float rmn1 = 0.f;
792  rmn1 = m_soilSolP[i][k] - m_soilActvMinP[i][k] * rto; // P imbalance
793  // Move P between the soluble and active pools based on vadas et al., 2006
794  if (rmn1 >= 0.f) {
795  // Net movement from soluble to active
796  rmn1 = Max(rmn1, (-1 * m_soilSolP[i][k]));
797  // Calculate dynamic coefficient
798  float vara = 0.918f * exp(-4.603f * psp);
799  float varb = -0.238f * log(vara) - 1.126f;
800  float arate = 0.f;
801  if (m_phpApldDays[i] > 0) {
802  arate = vara * pow(m_phpApldDays[i], varb);
803  } else {

1236  DF2 = LMNTA;
