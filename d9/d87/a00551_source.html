202 template<typename T>
203 void GetNumericFromBsonIterator(bson_iter_t *iter, T &numericvalue) {
204  try {
205  const bson_value_t *vv = bson_iter_value(iter);
206  if (vv->value_type == BSON_TYPE_INT32) {
207  numericvalue = (T) vv->value.v_int32;
208  } else if (vv->value_type == BSON_TYPE_INT64) {
209  numericvalue = (T) vv->value.v_int64;
210  } else if (vv->value_type == BSON_TYPE_DOUBLE) {
211  numericvalue = (T) vv->value.v_double;
212  } else if (vv->value_type == BSON_TYPE_UTF8) {
213  string tmp = vv->value.v_utf8.str;
214  numericvalue = (T) atof(tmp.c_str());
215  } else {
216  throw ModelException("MongoUtil", "GetNumericFromBsonIterator",
217  "bson iterator isn't or not contains a numeric value.\n");
218  }
219  }
220  catch (ModelException e) {
221  cout << e.toString() << endl;
222  exit(EXIT_FAILURE);
223  }
224 }
