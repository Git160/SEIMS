44  @staticmethod

46  """solar radiation (MJ/m2), n is sunshine duration (hour)"""
47  lat = lat * math.pi / 180.
48  a = 0.25
49  b = 0.5
50  d = HydroClimateUtilClass.dec(doy)
51  w = HydroClimateUtilClass.ws(lat, d)
52  nn = 24. * w / math.pi
53  # Extraterrestrial radiation for daily periods
54  ra = (24. * 60. * 0.082 * HydroClimateUtilClass.dr(doy) / math.pi) * \
55  (w * math.sin(lat) * math.sin(d) + math.cos(lat) * math.cos(d) * math.sin(w))
56  return (a + b * n / nn) * ra

92  time_sys = 'UTCTIME'
93  time_zone = 0
94  break
95  if str_line.lower().find('local') >= 0:
96  line_list = StringClass.split_string(str_line, [','])

98  time_zone = -1 * int(line_list[1])
99  break
