62  }
63 
64  // sum by subbasin
65  for (int i = 0; i < m_nCells; i++) {
66  //add today's flow

367 void NutrientMovementViaWater::NitrateLoss() {
369 // float tmpPercN = NODATA_VALUE;
370 // int tmpIdx = -1;
371  for (int iLayer = 0; iLayer < m_nRoutingLayers; iLayer++) {
372  // There are not any flow relationship within each routing layer.
373  // So parallelization can be done here.
374  int nCells = (int) m_routingLayers[iLayer][0];
375 #pragma omp parallel for
376  for (int iCell = 1; iCell <= nCells; iCell++) {
377  int i = (int) m_routingLayers[iLayer][iCell]; // cell ID
378  float percnlyr = 0.f;
379  m_latno3[i] = 0.f;
380  for (int k = 0; k < (int) m_nSoilLayers[i]; k++) {
381  // add nitrate leached from layer above (kg/ha)
382  //float tmpSolNo3 = m_sol_no3[i][k];
383  m_sol_no3[i][k] = m_sol_no3[i][k] + percnlyr;
384  //percnlyr = 0.f;
385  if (m_sol_no3[i][k] < 1.e-6f) {
386  m_sol_no3[i][k] = 0.f;
387  }
388  // determine concentration of nitrate in mobile water
389  float sro = 0.f;// surface runoff generated (sro)
390  float mw = 0.f;// amount of mobile water in the layer (mw)
391  float vno3 = 0.f; // kg/ha
392  float con = 0.f; // 100 mg/L
393  float ww = 0.f;
394 
395  if (k == 0) {
396  sro = m_surfr[i];
397  } else {
398  sro = 0.f;
399  }
400  if (m_ldrain[i] == k) {
401  mw += m_qtile;
402  }
403 
404  // Calculate the concentration of nitrate in the mobile water (con),
405  // equation 4:2.1.2, 4:2.1.3 and 4:2.1.4 in SWAT Theory 2009, p269
406  mw = m_sol_perco[i][k] + sro + m_flat[i][k] + 1.e-10f;
407  //if (i == 918)
408  // cout<<"sol_perco, k: "<<k<<", "<<m_sol_perco[i][k]<<", flat: "
409  // <<m_flat[i][k]<<endl;
410  float satportion = ((1.f - m_anion_excl[i]) * m_sol_wsatur[i][k]);
411  //if (mw > satportion) mw = satportion;
412  ww = -mw / satportion;
413  vno3 = m_sol_no3[i][k] * (1.f - exp(ww)); // kg/ha
414  if (mw > 1.e-10f) {
415  con = max(vno3 / mw, 0.f);
416  } // kg/ha/mm = 100 mg/L
417  //if (con > 0.1) con = 0.1;
418  //if (con != con)
419  //{
420  // cout<<"cellid: "<<i<<", layer: "<<k<<", perco water: "<<m_sol_perco[i][k]<<", satportion: "<<satportion<<

501  //float wt1 = m_sol_bd[i][0] * m_sol_thick[i][0] / 100.f; // mg/kg => kg/ha

503 
504  // amount of P leached from soil layer (vap)
505  float vap = 0.f;
506  float vap_tile = 0.f;
507  // compute soluble P lost in surface runoff
508  float xx = 0.f; // variable to hold intermediate calculation result
509  xx = m_sol_bd[i][0] * m_sol_z[i][0] * m_phoskd;
510  // units ==> surqsolp = [kg/ha * mm] / [t/m^3 * mm * m^3/t] = kg/ha
511  m_surqsolp[i] = m_sol_solp[i][0] * m_surfr[i] / xx;
512  m_surqsolp[i] = min(m_surqsolp[i], m_sol_solp[i][0]);
513  m_surqsolp[i] = max(m_surqsolp[i], 0.f);
514  m_sol_solp[i][0] = m_sol_solp[i][0] - m_surqsolp[i];
515 
516  // compute soluble P leaching
517  vap = m_sol_solp[i][0] * m_sol_perco[i][0] / ((m_conv_wt[i][0] / 1000.f) * m_pperco);
518  vap = min(vap, 0.5f * m_sol_solp[i][0]);
519  m_sol_solp[i][0] = m_sol_solp[i][0] - vap;
520 
521  // estimate soluble p in tiles due to crack flow
522  if (m_ldrain[i] > 0) {
523  xx = min(1.f, m_sol_crk[i] / 3.f);
524  vap_tile = xx * vap;
525  vap = vap - vap_tile;
526  }
527  if (m_nSoilLayers[i] >= 2) {
528  m_sol_solp[i][1] = m_sol_solp[i][1] + vap;
529  }
530  for (int k = 1; k < m_nSoilLayers[i]; k++) {
531  vap = 0.f;
532  //if (k != m_i_sep[i]) { // soil layer where biozone exists (m_i_sep)
533  vap = m_sol_solp[i][k] * m_sol_perco[i][k] / ((m_conv_wt[i][k] / 1000.f) * m_pperco);
534 
535  vap = min(vap, 0.2f * m_sol_solp[i][k]);
536  m_sol_solp[i][k] = m_sol_solp[i][k] - vap;
537 
538  if (k < m_nSoilLayers[i] - 1) {
539  m_sol_solp[i][k + 1] += vap;//leach to next layer

541  m_perco_p[i] = vap;

560 void NutrientMovementViaWater::SubbasinWaterQuality() {
561 #pragma omp parallel for
562  for (int i = 0; i < m_nCells; i++) {
566 
567  // total amount of water entering main channel on current day, mm
568  float qdr = 0.f;
569  qdr = m_surfr[i] + m_flat[i][0] + m_qtile;
570  if (qdr > 1.e-4f) {
571  // kilo moles of phosphorus in nutrient loading to main channel (tp)
572  float tp = 0.f;
573  tp = 100.f * (m_sedorgn[i] + m_surqno3[i]) / qdr; //100*kg/ha/mm = ppm
574  // regional adjustment on sub chla_a loading, the default value is 40
575  float chla_subco = 40.f;
576  m_surchl_a[i] = chla_subco * tp;
577  m_surchl_a[i] = m_surchl_a[i] / 1000.f; // um/L to mg/L
578 
579  // calculate enrichment ratio
580  if (m_sedimentYield[i] < 1e-4)m_sedimentYield[i] = 0.f;
581  float enratio = NutrCommon::CalEnrichmentRatio(m_sedimentYield[i], m_surfr[i], m_cellArea);
582 
583  // calculate organic carbon loading to main channel
584  float org_c = 0.f;
585  if (m_CbnModel == 2) {
586  org_c = m_sedc_d[i] * m_cellArea;
587  } else {
588  org_c = (m_sol_cbn[i][0] / 100.f) * enratio * (m_sedimentYield[i] / 1000.f) * 1000.f;
589  }
590  // calculate carbonaceous biological oxygen demand (CBOD) and COD(transform from CBOD)
591  float cbodu = 2.7f * org_c / (qdr * m_cellWidth * m_cellWidth * 1.e-6f); // mg/L
592  // convert cbod to cod
593  // The translation relationship is combined Wang Cai-Qin et al. (2014) with
594  // Guo and Long (1994); Xie et al. (2000); Jin et al. (2005).
595  float cod = m_cod_n * (cbodu * (1.f - exp(-5.f * m_cod_k)));
596  m_surcod[i] = m_surfr[i] / 1000.f * cod * 10.f; // mg/L converted to kg/ha
597  } else {
598  m_surchl_a[i] = 0.f;
599  m_surcod[i] = 0.f;
600  }
601  }
