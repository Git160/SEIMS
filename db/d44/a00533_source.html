28 int Interpolate::Execute() {
29  CheckInputData();
30  if (nullptr == m_itpOutput) { Initialize1DArray(m_nCells, m_itpOutput, 0.f); }
31  size_t err_count = 0;
32 #pragma omp parallel for reduction(+: err_count)
33  for (int i = 0; i < m_nCells; i++) {
34  int index = 0;
35  float value = 0.f;
36  for (int j = 0; j < m_nStatioins; j++) {
37  index = i * m_nStatioins + j;
38  value += m_stationData[j] * m_itpWeights[index];
39  if (value != value) {
40  err_count++;
41  cout << "CELL:" << i << ", Site: " << j << ", Weight: " << m_itpWeights[index] <<
42  ", siteData: " << m_stationData[j] << ", Value:" << value << ";" << endl;
43  }
44  if (m_itpVertical) {
45  float delta = m_dem[i] - m_hStations[j];
46  float factor = m_lapseRate[m_month - 1][m_dataType];
47  float adjust = m_itpWeights[index] * delta * factor * 0.01f;
48  value += adjust;
49  }
50  }
51  m_itpOutput[i] = value;
52  }
53  if (err_count > 0) {
54  throw ModelException(MID_ITP, "Execute", "Error occurred in weight data!");
55  }
56  return true;
57 }
