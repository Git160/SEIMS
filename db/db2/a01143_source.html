22 void SSR_DA::FlowInSoil(int id) {
23  float s0 = max(m_slope[id], 0.01f);
24  float flowWidth = m_CellWidth;
25  // there is no land in this cell
26  if (m_streamLink[id] > 0) {
27  flowWidth -= m_chWidth[id];
28  }
29  // initialize for current cell of current timestep
30  for (int j = 0; j < (int) m_soilLayers[id]; j++) {
31  m_qi[id][j] = 0.f;
32  m_qiVol[id][j] = 0.f;
33  }
34  // return with initial values if flowWidth is less than 0
35  if (flowWidth <= 0) return;
36  // number of flow-in cells
37  int nUpstream = (int) m_flowInIndex[id][0];
38  m_soilStorageProfile[id] = 0.f; // update soil storage on profile
39  for (int j = 0; j < (int) m_soilLayers[id]; j++) {
40  float smOld = m_soilStorage[id][j];
41  //sum the upstream subsurface flow, m3
42  float qUp = 0.f;
43  for (int upIndex = 1; upIndex <= nUpstream; ++upIndex) {
44  int flowInID = (int) m_flowInIndex[id][upIndex];
45  //cout << id << "\t" << flowInID << "\t" << m_nCells << endl;
46  if (m_qi[flowInID][j] > 0.f) {
47  qUp += m_qi[flowInID][j];
48  }// * m_flowInPercentage[id][upIndex];
49  //cout << id << "\t" << flowInID << "\t" << m_nCells << "\t" << m_qi[flowInID][j] << endl;
50  }
51  // add upstream water to the current cell
52  if (qUp < 0.f) qUp = 0.f;
53  m_soilStorage[id][j] += qUp; // mm
54  //TEST
55  if (m_soilStorage[id][j] != m_soilStorage[id][j] || m_soilStorage[id][j] < 0.f) {
56  ostringstream oss;
57  oss << "cell id: " << id << "\t layer: " << j << "\nmoisture is less than zero: " << m_soilStorage[id][j]
58  << "\t"
59  << smOld << "\nqUp: " << qUp << "\t depth:" << m_soilThick[id][j] << endl;
60  cout << oss.str() << endl;
61  throw ModelException(MID_SSR_DA, "Execute:FlowInSoil", oss.str());
62  }
63 
64  // if soil moisture is below the field capacity, no interflow will be generated, otherwise:
65  if (m_soilStorage[id][j] > m_fcmm[id][j]) {
66  // for the upper two layers, soil may be frozen
67  // also check if there are upstream inflow
68  if (j == 0 && m_soilT[id] <= m_frozenT && qUp <= 0.f) {
69  continue;
70  }
71 
72  float k = 0.f, maxSoilWater = 0.f, soilWater = 0.f, fcSoilWater = 0.f;
73  soilWater = m_soilStorage[id][j];
74  maxSoilWater = m_satmm[id][j];
75  fcSoilWater = m_fcmm[id][j];
76  //the moisture content can exceed the porosity in the way the algorithm is implemented
77  if (m_soilStorage[id][j] > maxSoilWater) {
78  k = m_ks[id][j];
79  } else {
81  float dcIndex = 2.f * m_poreIndex[id][j] + 3.f; // pore disconnectedness index
82  k = m_ks[id][j] * pow(m_soilStorage[id][j] / maxSoilWater, dcIndex);
83  if (k < 0.f) k = 0.f;
84  }
85  m_qi[id][j] = m_ki * s0 * k * m_dt / 3600.f * m_soilThick[id][j] / 1000.f / flowWidth; // the unit is mm
86 
87  if (soilWater - m_qi[id][j] > maxSoilWater) {
88  m_qi[id][j] = soilWater - maxSoilWater;
89  } else if (soilWater - m_qi[id][j] < fcSoilWater) {
90  m_qi[id][j] = soilWater - fcSoilWater;
91  }
92  m_qi[id][j] = max(0.f, m_qi[id][j]);
93 
94  m_qiVol[id][j] = m_qi[id][j] / 1000.f * m_CellWidth * flowWidth; //m3
95  m_qiVol[id][j] = max(0.f, m_qiVol[id][j]);
96  //Adjust the moisture content in the current layer, and the layer immediately below it
97  m_soilStorage[id][j] -= m_qi[id][j];
98  m_soilStorageProfile[id] += m_soilStorage[id][j];
99  if (m_soilStorage[id][j] != m_soilStorage[id][j] || m_soilStorage[id][j] < 0.f) {
100  ostringstream oss;
101  oss << id << "\t" << j << "\nmoisture is less than zero: " << m_soilStorage[id][j] << "\t"
102  << m_soilThick[id][j];
103  throw ModelException(MID_SSR_DA, "Execute", oss.str());
104  }
105  }
106  }

110 int SSR_DA::Execute() {
111  CheckInputData();
112  initialOutputs();
113 
114  for (int iLayer = 0; iLayer < m_nRoutingLayers; ++iLayer) {
115  // There are not any flow relationship within each routing layer.
116  // So parallelization can be done here.
117  int nCells = (int) m_routingLayers[iLayer][0];
118 
119 #pragma omp parallel for
120  for (int iCell = 1; iCell <= nCells; ++iCell) {
121  int id = (int) m_routingLayers[iLayer][iCell];
122  FlowInSoil(id);
123  }
124  }
125 
126  //cout << "end flowinsoil" << endl;
127  for (int i = 0; i < m_nSubbasin; i++) {
128  m_qiSubbasin[i] = 0.f;
129  }
130 
131 #pragma omp parallel for
132  for (int i = 0; i < m_nCells; i++) {
133  float qiAllLayers = 0.f;
134  if (m_streamLink[i] > 0) {
135  qiAllLayers = 0.f;
136  for (int j = 0; j < (int) m_soilLayers[i]; j++) {
137  if (m_qiVol[i][j] > UTIL_ZERO) {
138  qiAllLayers += m_qiVol[i][j] / m_dt;
139  }
140  }
141  //cout << m_nSubbasin << "\tsubbasin:" << m_subbasin[i] << "\t" << qiAllLayers << endl;
142  //if (m_nSubbasin > 1)
143  // m_qiSubbasin[int(m_subbasin[i])] += qiAllLayers;
144  //else
145  // m_qiSubbasin[1] += qiAllLayers;
146  m_qiSubbasin[int(m_subbasin[i])] += qiAllLayers;
147  }
148  }
149 
150  for (int i = 1; i <= m_nSubbasin; i++) {
151  //cout<<", "<<i<<": "<<m_qiSubbasin[i];
152  m_qiSubbasin[0] += m_qiSubbasin[i];
153  }
154  //cout<<endl;
155  return 0;
156 }
