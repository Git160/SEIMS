140 void clsReaches::Update(const map<string, ParamInfo *> &caliparams_map) {
141  for (int i = 0; i < REACH_PARAM_NUM; i++) {
142  auto it = caliparams_map.find(REACH_PARAM_NAME[i]);
143  if (it != caliparams_map.end()) {
144  ParamInfo *tmpParam = it->second;
145  if ((StringMatch(tmpParam->Change, PARAM_CHANGE_RC) && FloatEqual(tmpParam->Impact, 1.f)) ||
146  (StringMatch(tmpParam->Change, PARAM_CHANGE_AC) && FloatEqual(tmpParam->Impact, 0.f)) ||
147  (StringMatch(tmpParam->Change, PARAM_CHANGE_VC) && FloatEqual(tmpParam->Impact, NODATA_VALUE))) {
148  continue;
149  }
150  for (auto it2 = m_reachesMap.begin(); it2 != m_reachesMap.end(); it2++) {
151  float pre_value = it2->second->Get(REACH_PARAM_NAME[i]);
152  if (FloatEqual(pre_value, NODATA_VALUE)) continue;
153  float new_value = it->second->GetAdjustedValue(pre_value);
154  it2->second->Set(REACH_PARAM_NAME[i], new_value);
155  }
156  }
