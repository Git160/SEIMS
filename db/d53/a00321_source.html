290 def get_potential_bmps(suitbmps, sptag, up_sid, up_gvalue, down_sid, down_gvalue, method=1):
291  bmps = suitbmps[sptag][:]
292  bmps = list(set(bmps))
293  if 0 not in bmps:
294  bmps.append(0)
295  if method == 1: # Without any special rule
296  pass
297  elif method == 2:
298  # If not bottom slppos and the downslope unit is configured BMP, then remove 0
299  if down_sid > 0 and down_gvalue == 0:
300  bmps.remove(0)
301  elif method == 3:
302  new_bmps = list()
303  if up_sid < 0 and down_gvalue > 0: # the top slppos, and downslope with BMP
304  for _bid in bmps:
305  if _bid <= down_gvalue:
306  new_bmps.append(_bid)
307  elif down_sid < 0 and up_gvalue > 0: # the bottom slppos, and upslope with BMP
308  for _bid in bmps:
309  if up_gvalue <= _bid:
310  new_bmps.append(_bid)
311  elif down_sid > 0 and up_sid > 0: # middle slppos
312  for _bid in bmps:
313  if down_gvalue == 0 and up_gvalue <= _bid:
314  new_bmps.append(_bid)
315  elif up_gvalue <= _bid <= down_gvalue:
316  new_bmps.append(_bid)
317  else: # Do nothing
318  pass
319  if len(new_bmps) > 0:
320  bmps = list(set(new_bmps))
321  return bmps
