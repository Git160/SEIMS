161 int ExcessRunoff::Execute(void) {
162  CheckInputData();
163 
164  // allocate the output variable
165  if (this->m_pe == NULL) {
166  this->m_pe = new float[this->m_nCells];
167  this->m_infil = new float[this->m_nCells];
168  m_infilCapacitySurplus = new float[m_nCells];
169  m_soilMoisture = new float[m_nCells];
170 #pragma omp parallel for
171  for (int i = 0; i < m_nCells; i++) {
172  m_initSoilMoisture[i] = m_initSoilMoisture[i] * m_fieldCap[i];
173  m_soilMoisture[i] = m_initSoilMoisture[i];
174  }
175  }
176 
177  // allocate intermediate variables
178  if (this->m_accumuDepth == NULL) {
179  this->m_accumuDepth = new float[this->m_nCells];
180 #pragma omp parallel for
181  for (int i = 0; i < m_nCells; ++i) {
182  m_accumuDepth[i] = 0.0f;
183  }
184  }
185 
186 #pragma omp parallel for
187  for (int i = 0; i < m_nCells; i++) {
188  float pNet = m_pNet[i];
189  //account for the effects of snow melt and soil temperature

207  if (pNet > 0) {
208  // for frozen soil
209  //if (m_soilTemp[i] <= m_tSoilFrozen && m_soilMoisture[i] >= m_sFrozen*m_porosity[i])
210  //{

220  //else

223 
224  if (m_soilMoisture[i] >= m_porosity[i]) {
225  m_infil[i] = 0.f;
226  } else if (m_soilMoisture[i] >= m_fieldCap[i]) {
227  float limitContent = m_rootDepth[i] * (m_porosity[i] - m_soilMoisture[i]);
228  m_infil[i] = min(ks * m_dt, limitContent);

231  }
232 
233  //check if the infiltration potential exceeds the available water
234  if (m_infil[i] > pNet) {
235  //limit infiltration rate to available water supply

237  m_infilCapacitySurplus[i] = m_infil[i] - pNet;
238  } else {
239  m_infilCapacitySurplus[i] = 0.f;
240  }
241 
242  //Compute the cumulative depth of infitration

247  m_pe[i] = m_pNet[i] - m_infil[i];
