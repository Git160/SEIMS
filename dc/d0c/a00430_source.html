75 void MUSLE_AS::initialOutputs() {
76  if (m_nCells <= 0) {
77  throw ModelException(MID_MUSLE_AS, "CheckInputData",
78  "The dimension of the input data can not be less than zero.");
79  }
80 
81  if (m_sedimentYield == NULL) Initialize1DArray(m_nCells, m_sedimentYield, 0.f);
82  if (m_sandYield == NULL) Initialize1DArray(m_nCells, m_sandYield, 0.f);
83  if (m_siltYield == NULL) Initialize1DArray(m_nCells, m_siltYield, 0.f);
84  if (m_clayYield == NULL) Initialize1DArray(m_nCells, m_clayYield, 0.f);
85  if (m_smaggreYield == NULL) Initialize1DArray(m_nCells, m_smaggreYield, 0.f);
86  if (m_lgaggreYield == NULL) Initialize1DArray(m_nCells, m_lgaggreYield, 0.f);
87  if (m_usle_ls == NULL) {
88  float constant = pow(22.13f, 0.4f);
89  m_usle_ls = new float[m_nCells];
90  m_slopeForPq = new float[m_nCells];
91 #pragma omp parallel for
92  for (int i = 0; i < m_nCells; i++) {
93  if (m_usle_c[i] < 0.001f) {
94  m_usle_c[i] = 0.001f;
95  }
96  if (m_usle_c[i] > 1.0f) {
97  m_usle_c[i] = 1.0f;
98  }
99  float lambda_i1 = m_flowacc[i] * m_cellWidth; // m
100  float lambda_i = lambda_i1 + m_cellWidth;
101  float L = pow(lambda_i, 1.4f) - pow(lambda_i1, 1.4f); // m^1.4
102  L /= m_cellWidth * constant;
103 
104  float S = pow(sin(atan(m_slope[i])) / 0.0896f, 1.3f);
105 
106  m_usle_ls[i] = L * S;//equation 3 in memo, LS factor
107 
108  m_slopeForPq[i] = pow(m_slope[i] * 1000.f, 0.16f);
109  }
110  }
111  if (FloatEqual(m_cellAreaKM, NODATA_VALUE)) {
112  m_cellAreaKM = pow(m_cellWidth / 1000.f, 2.f);
113  m_cellAreaKM1 = 3.79f * pow(m_cellAreaKM, 0.7f);
114  m_cellAreaKM2 = 0.903f * pow(m_cellAreaKM, 0.017f);
115  }
116 }

118 float MUSLE_AS::getPeakRunoffRate(int cell) {
119  if (m_surfaceRunoff[cell] < 0.01f) {
120  return 0.f;
121  } else {
122  return m_cellAreaKM1 * m_slopeForPq[cell] *
123  pow(m_surfaceRunoff[cell] / 25.4f, m_cellAreaKM2);
124  } //equation 2 in memo, peak flow

127 int MUSLE_AS::Execute() {
128  CheckInputData();
129  initialOutputs();
130 #pragma omp parallel for
131  for (int i = 0; i < m_nCells; i++) {
132  if (m_surfaceRunoff[i] < 0.0001f || m_streamLink[i] > 0) {
133  m_sedimentYield[i] = 0.f;
134  } else {
135  float q = getPeakRunoffRate(i); //equation 2 in memo, peak flow
136  float Y = 11.8f * pow(m_surfaceRunoff[i] * m_cellAreaKM * 1000.0f * q, 0.56f)
137  * m_usle_k[i][0] * m_usle_ls[i] * m_usle_c[i] *
138  m_usle_p[i]; //equation 1 in memo, sediment yield
139 
140  if (m_snowAccumulation[i] > 0.0001f) {
141  Y /= exp(3.f * m_snowAccumulation[i] / 25.4f);
142  } //equation 4 in memo, the snow pack effect
143  m_sedimentYield[i] = Y * 1000.f;
144  }
145  //if(i == 1000) cout << m_sedimentYield[i] << "," << m_surfaceRunoff[i] << endl;
147  m_sandYield[i] = m_sedimentYield[i] * m_detachSand[i];
148  m_siltYield[i] = m_sedimentYield[i] * m_detachSilt[i];
149  m_clayYield[i] = m_sedimentYield[i] * m_detachClay[i];
150  m_smaggreYield[i] = m_sedimentYield[i] * m_detachSmAggre[i];
151  m_lgaggreYield[i] = m_sedimentYield[i] * m_detachLgAggre[i];
152  }
153  return 0;
154 }
