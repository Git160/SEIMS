25 int FindBoundingBox(clsRasterData<int> &rsSubbasin, map<int, SubBasin> &bboxMap) {
26  int nXSize = rsSubbasin.getCols();
27  int nYSize = rsSubbasin.getRows();
28 
29  int *pData = rsSubbasin.getRasterDataPointer();
30  int noDataValue = rsSubbasin.getNoDataValue();
31  // find bounding box for each subbasin
32  for (int i = 0; i < nYSize; i++) {
33  for (int j = 0; j < nXSize; j++) {
34  int id = pData[i * nXSize + j];
35  if (noDataValue != id) {
36  if (bboxMap.find(id) == bboxMap.end()) {
37  bboxMap[id] = SubBasin(j, i, j, i);
38  } else {
39  if (j < bboxMap[id].xMin) {
40  bboxMap[id].xMin = j;
41  } else if (j > bboxMap[id].xMax) {
42  bboxMap[id].xMax = j;
43  }
44 
45  if (i > bboxMap[id].yMax) {
46  bboxMap[id].yMax = i;
47  }
48  }
49  bboxMap[id].cellCount += 1;
50  }
51  }
52  }
53  return 0;
54 }

252  map<int, SubBasin>::iterator it;

254  for (it = bboxMap.begin(); it != bboxMap.end(); it++) {
255  int id = it->first;

262  int subYSize = subbasin.yMax - subbasin.yMin + 1;

267  //cout << subbasinFile << endl;
268  // create a new raster for the subbasin
269  char **papszOptions = NULL;

271  1, GDT_Float32, papszOptions);
272 
273  float *subData = (float *) CPLMalloc(sizeof(float) * subXSize * subYSize);
274 #pragma omp parallel for
275  for (int i = subbasin.yMin; i <= subbasin.yMax; i++) {
276  for (int j = subbasin.xMin; j <= subbasin.xMax; j++) {
277  int index = i * nXSize + j;
278  int subIndex = (i - subbasin.yMin) * subXSize + (j - subbasin.xMin);
279  if (subbasinData[index] == subbasinID) {
280  subData[subIndex] = rsData[index];
281  } else {
282  subData[subIndex] = noDataValue;
283  }
284  }
285  }
286 
287  //write the data to new file
288  GDALRasterBand *poDstBand = poDstDS->GetRasterBand(1);
289  poDstBand->RasterIO(GF_Write, 0, 0, subXSize, subYSize,
290  subData, subXSize, subYSize, GDT_Float32, 0, 0);
291  poDstBand->SetNoDataValue(noDataValue);
292 
293  double geoTrans[6];
294  float cellSize = rs.getCellWidth();
295  geoTrans[0] = rs.getXllCenter() + (subbasin.xMin - 0.5f) * cellSize;
296  geoTrans[1] = cellSize;
297  geoTrans[2] = 0;
298  geoTrans[3] = rs.getYllCenter() + (rs.getRows() - subbasin.yMin - 0.5f) * cellSize;
299  geoTrans[4] = 0;
300  geoTrans[5] = -cellSize;
301  poDstDS->SetGeoTransform(geoTrans);
302 
303  CPLFree(subData);
304  GDALClose(poDstDS);
305  }
