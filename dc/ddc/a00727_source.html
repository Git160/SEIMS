922  float up2 = 0.f;
923  float uapd = 0.f;
924  float upmx = 0.f;
925  float uapl = 0.f;
926  float gx = 0.f;
927  up2 = m_frPlantP[i] * m_biomass[i];
928  if (up2 < m_plantP[i]) up2 = m_plantP[i];
929  uapd = up2 - m_plantP[i];

931  m_frStrsP[i] = 1.f;
932  int ir = 0;
933  if (uapd < UTIL_ZERO) return;
934  for (int l = 0; l < m_nSoilLayers[i]; l++) {
935  if (ir > 0) break;
936  if (m_soilRD[i] <= m_soilDepth[i][l]) {
937  gx = m_soilRD[i];
938  ir = 1;
939  } else {
940  gx = m_soilDepth[i][l];
941  }
942  upmx = uapd * (1.f - exp(-m_PUpDis * gx / m_soilRD[i])) / uobp;
943  uapl = min(upmx - m_plantUpTkP[i], m_soilPsol[i][l]);
944  m_plantUpTkP[i] += uapl;
945  m_soilPsol[i][l] -= uapl;
946  }
947  if (m_plantUpTkP[i] < 0.f) m_plantUpTkP[i] = 0.f;
948  m_plantP[i] += m_plantUpTkP[i];
950  PGCommon::calPlantStressByLimitedNP(m_plantP[i], up2, &m_frStrsP[i]);
