204  """Generate potential runoff coefficient."""
205  # read landuselookup table from MongoDB

230  def coef_cal(lu_id, soil_texture, slope):
231  """Calculate runoff coefficient by landuse, soil texture and slope."""
232  if abs(lu_id - nodata_value1) < UTIL_ZERO or int(lu_id) < 0:
233  return nodata_value2
234  if int(lu_id) not in runoff_c0.keys():
235  if int(lu_id) not in id_omited:
236  print ('The landuse ID: %d does not exist.' % int(lu_id))
237  id_omited.append(int(lu_id))
238  stid = int(soil_texture) - 1
239  c0 = runoff_c0[int(lu_id)][stid]
240  s0 = runoff_s0[int(lu_id)][stid] / 100.
241  slp = slope
242 
243  if slp + s0 < 0.0001:
244  return c0
245  coef1 = (1 - c0) * slp / (slp + s0)
246  coef2 = c0 + coef1
247  # TODO, Check if it is (lu_id >= 98), by lj
248  if int(lu_id) == 106 or int(lu_id) == 107 or int(lu_id) == 105:
249  return coef2 * (1 - imper_perc) + imper_perc
250  else:
251  return coef2

254  coef = coef_cal_numpy(landu_data, soil_texture_array, slo_data)
255
