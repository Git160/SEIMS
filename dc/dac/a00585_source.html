92 void SEDR_SBAGNOLD::PointSourceLoading() {
94  for (auto it = m_ptSrcFactory.begin(); it != m_ptSrcFactory.end(); ++it) {
95  //cout<<"unique Point Source Factory ID: "<<it->first<<endl;
96  vector<int>& m_ptSrcMgtSeqs = it->second->GetPointSrcMgtSeqs();
97  map<int, PointSourceMgtParams *>& m_pointSrcMgtMap = it->second->GetPointSrcMgtMap();
98  vector<int>& m_ptSrcIDs = it->second->GetPointSrcIDs();
99  map<int, PointSourceLocations *>& m_pointSrcLocsMap = it->second->GetPointSrcLocsMap();
100  // 1. looking for management operations from m_pointSrcMgtMap
101  for (auto seqIter = m_ptSrcMgtSeqs.begin(); seqIter != m_ptSrcMgtSeqs.end(); ++seqIter) {
102  PointSourceMgtParams* curPtMgt = m_pointSrcMgtMap.at(*seqIter);
103  // 1.1 If current day is beyond the date range, then continue to next management
104  if (curPtMgt->GetStartDate() != 0 && curPtMgt->GetEndDate() != 0) {
105  if (m_date < curPtMgt->GetStartDate() || m_date > curPtMgt->GetEndDate()) {
106  continue;
107  }
108  }
109  // 1.2 Otherwise, get the water volume
110  float per_sed = curPtMgt->GetSedment();
111  // 1.3 Sum up all point sources
112  for (auto locIter = m_ptSrcIDs.begin(); locIter != m_ptSrcIDs.end(); ++locIter) {
113  if (m_pointSrcLocsMap.find(*locIter) != m_pointSrcLocsMap.end()) {
114  PointSourceLocations* curPtLoc = m_pointSrcLocsMap.at(*locIter);
115  int curSubID = curPtLoc->GetSubbasinID();
117  m_ptSub[curSubID] += per_sed * curPtLoc->GetSize() * 1000.f * m_dt * 1.1574074074074073e-05f;
118  }
119  }
120  }
121  }

124 int SEDR_SBAGNOLD::Execute() {
125  CheckInputData();
126  InitialOutputs();
128  PointSourceLoading();
129  for (auto it = m_reachLayers.begin(); it != m_reachLayers.end(); ++it) {
130  // There are not any flow relationship within each routing layer.
131  // So parallelization can be done here.
132  int nReaches = CVT_INT(it->second.size());
133 #pragma omp parallel for
134  for (int i = 0; i < nReaches; i++) {
135  int reachIndex = it->second[i]; // index in the array, which is equal to reach ID
136  if (m_inputSubbsnID == 0 || m_inputSubbsnID == reachIndex) {
137  // for OpenMP version, all reaches will be executed,
138  // for MPI version, only the current reach will be executed.
139  SedChannelRouting(reachIndex);
140  // compute changes in channel dimensions caused by downcutting and widening
141  if (m_vcd) {
142  DoChannelDowncuttingAndWidening(reachIndex);
143  }
144  }
145  }
146  }
147  return 0;
148 }

273 void SEDR_SBAGNOLD::SedChannelRouting(const int i) {
274  float qOutV = 0.f; // water volume (m^3) of flow out
275  float allWater = 0.f; // // water in reach during time step, qdin in SWAT
276  // initialize sediment in reach during time step
278  float allSediment = 0.f; // all sediment in reach, kg
279  float sedUp = 0.f; // sediment from upstream channels, kg
280  for (size_t j = 0; j < m_reachUpStream[i].size(); j++) {
281  int upReachId = m_reachUpStream[i][j];
282  sedUp += m_sedRchOut[upReachId];
283  }
284  allSediment = sedUp + m_sedtoCh[i] + m_sedStorage[i];
286  if (nullptr != m_ptSub && m_ptSub[i] > 0.f) {
287  allSediment += m_ptSub[i];
288  }
289  // initialize water in reach during time step
290  qOutV = m_qRchOut[i] * m_dt; // m^3
291  allWater = m_preChStorage[i];
292  if (((m_qRchOut[i] < UTIL_ZERO) && (m_chWtrDepth[i] < UTIL_ZERO)) || (allWater < 0.01f)) {
296  m_sedDeg[i] = 0.f;
297  m_sedDep[i] = 0.f;
298  m_sedRchOut[i] = 0.f;
299  m_sedConcRchOut[i] = 0.f;
300  m_rchSand[i] = 0.f;
301  m_rchSilt[i] = 0.f;
302  m_rchClay[i] = 0.f;
303  m_rchSag[i] = 0.f;
304  m_rchLag[i] = 0.f;
305  m_rchGra[i] = 0.f;
306  m_rchBankEro[i] = 0.f;
307  m_rchDeg[i] = 0.f;
308  m_rchDep[i] = 0.f;
309  m_fldPlainDep[i] = 0.f;
310  m_sedStorage[i] = allSediment;
311  return;
312  }
313  // initialize reach peak runoff rate and calculate flow velocity. SWAT Theory 2009, p.448
314  float peakFlowRate = m_qRchOut[i] * m_peakRateAdj; // unit: m3/s
315  float crossarea = allWater / m_chLen[i]; // unit: m2, p.432, eq.7:1.2.3
316  float peakVelocity = 0.f;
317  if (m_preChWtrDepth[i] < 0.01f) {
318  peakVelocity = 0.01f;
319  } else {
320  peakVelocity = peakFlowRate / crossarea;
321  }
322  if (peakVelocity > 5.f) {
323  // cout << "subbasin id: " << i <<", peakVelocity: " << peakVelocity << endl;
324  peakVelocity = 5.f;
325  }
326  float tbase = m_chLen[i] / (m_dt * peakVelocity);
327  if (tbase > 1.f) tbase = 1.f;
328 
329  //if (i == 4) cout << "qchOut: " << m_qchOut[i] << ", allwater: " << allWater <<
330  // ", chLen: " << m_chLen[i] << ", peakVelocity: " << peakVelocity <<
331  // ", m_preChWTDepth: " << m_preChWTDepth[i] << ", tbase: "<< tbase << endl;
332 
334  float initCon = 0.f; // cyin
335  float maxCon = 0.f; // cych
336  float sedDeposition = 0.f; // depnet, and dep
337  float sedDegradation = 0.f; // deg
338  float sedDegradation1 = 0.f; // deg1
339  float sedDegradation2 = 0.f; // deg2
340 
341  //deposition and degradation
342  initCon = allSediment / allWater; // kg/m^3
343  //max concentration
344  maxCon = m_sedTransEqCoef * pow(peakVelocity, m_sedTransEqExp) * 1000.f; // kg/m^3
345  //if (i == 4) cout<<"iniCon: "<<initCon<<", maxCon: "<<maxCon<<endl;
346  //initial concentration,mix sufficiently
347  sedDeposition = allWater * (initCon - maxCon); // kg
348  if (peakVelocity < m_critVelSedDep) {
349  sedDeposition = 0.f;
350  }
351 
352  if (sedDeposition < 0.f) {
353  //degradation
354  //cout << "initCon < maxCon, subbasin id: " << i << endl;
355  sedDegradation = -sedDeposition * tbase;
356  // first the deposited material will be degraded before channel bed
357  if (sedDegradation >= m_sedDep[i]) {
358  sedDegradation1 = m_sedDep[i];
359  sedDegradation2 = (sedDegradation - sedDegradation1) * m_chErod[i] * m_chCover[i];
360  } else {
361  sedDegradation1 = sedDegradation;
362  sedDegradation2 = 0.f;
363  }
364  sedDeposition = 0.f;
365  } else {
366  sedDegradation = 0.f;
367  sedDegradation1 = 0.f;
368  sedDegradation2 = 0.f;
369  }
370  //update sed deposition
371  m_sedDep[i] += sedDeposition - sedDegradation1;
372  if (m_sedDep[i] < UTIL_ZERO) m_sedDep[i] = 0.f;
373  m_sedDeg[i] += sedDegradation1 + sedDegradation2;
374 
375  //get sediment after deposition and degradation
376  allSediment += sedDegradation1 + sedDegradation2 - sedDeposition;
377  if (allSediment < UTIL_ZERO) allSediment = 0.f;
378  //get out flow water fraction
379  float outFraction = qOutV / allWater;
380  if (outFraction > 1.f) outFraction = 1.f;
381  m_sedRchOut[i] = allSediment * outFraction;
382  if (m_sedRchOut[i] < UTIL_ZERO) m_sedRchOut[i] = 0.f;
383  // update sediment storage
384  m_sedStorage[i] = allSediment - m_sedRchOut[i];
385  if (m_sedStorage[i] < UTIL_ZERO) m_sedStorage[i] = 0.f;
386 
387  // get final sediment in water, cannot large than 0.848 ton/m3
388  float maxSedinWt = 0.848f * qOutV * 1000.f;
389  if (m_sedRchOut[i] > maxSedinWt) {
390  m_sedDep[i] += m_sedRchOut[i] - maxSedinWt;
391  m_sedRchOut[i] = maxSedinWt;
392  }
394  m_sedConcRchOut[i] = m_sedRchOut[i] / qOutV;
395  m_rchSand[i] = 0.f;
397  m_rchSilt[i] = m_sedRchOut[i]; // the sediments is assumed to be silt for mass conservation
398  m_rchClay[i] = 0.f;
399  m_rchSag[i] = 0.f;
400  m_rchLag[i] = 0.f;
401  m_rchGra[i] = 0.f;
402 
403  m_rchBankEro[i] = 0.f;
404  m_rchDeg[i] = sedDegradation2;
405  m_rchDep[i] = sedDeposition;
406  m_fldPlainDep[i] = 0.f;
407 }

411  float depdeg = m_preChWtrDepth[id] - m_chWtrDepth[id]; // depth of degradation/deposition from original
412  if (depdeg < m_chSlope[id] * m_chLen[id]) {
413  //float storage = m_chStorage[id];
414  //float vout = m_qchOut[id] * m_dt;
415  if (m_preChStorage[id] > 1.4e6f) {
417  float cutDepth = 358.6f * m_chWtrDepth[id] * m_chSlope[id] * m_chErod[id];
418  m_chWtrDepth[id] += cutDepth;
419  m_chWtrWth[id] = m_chWtrDepth[id] * (m_chWidth[id] / m_chDepth[id]);
420 
421  m_chSlope[id] -= cutDepth / m_chLen[id];
422  m_chSlope[id] = Max(0.0001f, m_chSlope[id]);
