67 void BMPArealSrcFactory::ReadArealSourceManagements(MongoClient* conn, const string& bmpDBName) {
68  bson_t* b = bson_new();
69  bson_t *child1 = bson_new(), *child2 = bson_new();
70  BSON_APPEND_DOCUMENT_BEGIN(b, "$query", child1);

72  bson_append_document_end(b, child1);

76  bson_destroy(child1);

78 
79  std::unique_ptr<MongoCollection> collection(new MongoCollection(conn->GetCollection(bmpDBName, m_arealSrcMgtTab)));
80  mongoc_cursor_t* cursor = collection->ExecuteQuery(b);
81 
82  bson_iter_t iter;
83  const bson_t* bsonTable;
84 
86  int count = 1;
87  while (mongoc_cursor_next(cursor, &bsonTable)) {
88  m_arealSrcMgtSeqs.emplace_back(count);
89  m_arealSrcMgtMap[count] = new ArealSourceMgtParams(bsonTable, iter);
90  count++;

96 void BMPArealSrcFactory::ReadArealSourceLocations(MongoClient* conn, const string& bmpDBName) {
97  bson_t* b = bson_new();
98  bson_t* child1 = bson_new();
99  BSON_APPEND_DOCUMENT_BEGIN(b, "$query", child1);
100  BSON_APPEND_INT32(child1, BMP_ARSRC_FLD_CODE, m_arealSrc);
101  bson_append_document_end(b, child1);
102  bson_destroy(child1);
103 
104  std::unique_ptr<MongoCollection> collection(new MongoCollection(conn->GetCollection(bmpDBName, m_arealSrcDistTab)));
105  mongoc_cursor_t* cursor = collection->ExecuteQuery(b);
106 
107  bson_iter_t iter;
108  const bson_t* bsonTable;
109 
110  while (mongoc_cursor_next(cursor, &bsonTable)) {
111  ArealSourceLocations* curArSrcLoc = new ArealSourceLocations(bsonTable, iter);
112  int curArSrcID = curArSrcLoc->GetArealSourceID();
113  if (ValueInVector(curArSrcID, m_arealSrcIDs)) {
114 #ifdef HAS_VARIADIC_TEMPLATES
115  m_arealSrcLocsMap.emplace(curArSrcID, curArSrcLoc);
116 #else
117  m_arealSrcLocsMap.insert(make_pair(curArSrcID, curArSrcLoc));
118 #endif
119  } else {
120  RemoveValueInVector(curArSrcID, m_arealSrcIDs);
121  delete curArSrcLoc;
122  }
123  }
124  bson_destroy(b);
125  mongoc_cursor_destroy(cursor);
126 }
