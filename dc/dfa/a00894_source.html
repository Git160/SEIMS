57  """Compress dinf flow direction to D8 direction with weight
58  Args:
59  angle: D-inf flow direction angle
60  nodata: NoData value
61 
62  Returns:
63  Compressed flow direction and weight of the first direction
64  """
65  if MathClass.floatequal(angle, nodata):
66  return DEFAULT_NODATA, DEFAULT_NODATA
67  d = DinfUtil.check_orthogonal(angle)
68  if d is not None:
69  return d, 1
70  if angle < FlowModelConst.ne:
71  a1 = angle
72  d = 129 # 1+128
73  elif angle < FlowModelConst.n:
74  a1 = angle - FlowModelConst.ne
75  d = 192 # 128+64
76  elif angle < FlowModelConst.nw:
77  a1 = angle - FlowModelConst.n
78  d = 96 # 64+32
79  elif angle < FlowModelConst.w:
80  a1 = angle - FlowModelConst.nw
81  d = 48 # 32+16
82  elif angle < FlowModelConst.sw:
83  a1 = angle - FlowModelConst.w
84  d = 24 # 16+8
85  elif angle < FlowModelConst.s:
86  a1 = angle - FlowModelConst.sw
87  d = 12 # 8+4
88  elif angle < FlowModelConst.se:
89  a1 = angle - FlowModelConst.s
90  d = 6 # 4+2
91  else:
92  a1 = angle - FlowModelConst.se
93  d = 3 # 2+1
