311 float IKW_REACH::GetNewQ(float qIn, float qLast, float surplus, float alpha, float dt, float dx) {
312  /* Using Newton-Raphson Method */
313  float ab_pQ, dtX, C; //auxillary vars
314  int count;
315  float Qkx; //iterated discharge, becomes Qnew
316  float fQkx; //function
317  float dfQkx; //derivative
318  const float _epsilon = 1e-12f;
319  const float beta = 0.6f;
320 
321  /* if no input then output = 0 */
322  if ((qIn + qLast) <= -surplus * dx)//0)
323  {
324  //itercount = -1;
325  return (0);
326  }
327 
328  /* common terms */
329  ab_pQ = alpha * beta * pow(((qLast + qIn) / 2), beta - 1);
330  // derivative of diagonal average (space-time)
331 
332  dtX = dt / dx;
333  C = dtX * qIn + alpha * pow(qLast, beta) + dt * surplus;
334  //dt/dx*Q = m3/s*s/m=m2; a*Q^b = A = m2; surplus*dt = s*m2/s = m2
335  //C is unit volume of water
336  // first gues Qkx
337  Qkx = (dtX * qIn + qLast * ab_pQ + dt * surplus) / (dtX + ab_pQ);
338 
339  // VJ 050704, 060830 infil so big all flux is gone
340  //VJ 110114 without this de iteration cannot be solved for very small values
341  if (Qkx < MIN_FLUX) {
342  //itercount = -2;
343  return (0);
344  }
345 
346  Qkx = max(Qkx, MIN_FLUX);
347 
348  count = 0;
349  do {
350  fQkx = dtX * Qkx + alpha * pow(Qkx, beta) - C; /* Current k */
351  dfQkx = dtX + alpha * beta * pow(Qkx, beta - 1); /* Current k */
352  Qkx -= fQkx / dfQkx; /* Next k */
353  Qkx = max(Qkx, MIN_FLUX);
354  count++;
355  //qDebug() << count << fQkx << Qkx;
356  } while (fabs(fQkx) > _epsilon && count < MAX_ITERS_KW);
357 
358  if (Qkx != Qkx) {
359  throw ModelException("IKW_OL", "GetNewQ", "Error in iteration!");
360  }
361 
362  //itercount = count;
363  return Qkx;
364 }

366 void IKW_REACH::ChannelFlow(int i) {
367  float st0 = m_chStorage[i];
368 
369  float qiSub = 0.f;
370  if (nullptr != m_qiSub) {
371  qiSub = m_qiSub[i];
372  }
373  float qgSub = 0.f;
374  if (nullptr != m_qgSub) {
375  qgSub = m_qgSub[i];
376  }
377 
379  // first add all the inflow water
380  // 1. water from this subbasin
381  float qIn = m_qsSub[i] + qiSub + qgSub;
382 
383  // 2. water from upstream reaches
384  float qsUp = 0.f;
385  float qiUp = 0.f;
386  float qgUp = 0.f;
387  for (size_t j = 0; j < m_reachUpStream[i].size(); ++j) {
388  int upReachId = m_reachUpStream[i][j];
389  qsUp += m_qsCh[upReachId];
390  qiUp += m_qiCh[upReachId];
391  qgUp += m_qgCh[upReachId];
392  }
393  qIn += qsUp + qiUp + qgUp;
394  qIn += m_qUpReach;
395  // m_qUpReach is zero for not-parallel program and qsUp, qiUp and qgUp are zero for parallel computing
396 
397  // 3. water from bank storage
398  float bankOut = m_bankStorage[i] * (1 - exp(-m_aBank));
399 
400  m_bankStorage[i] -= bankOut;
401  qIn += bankOut / m_dt;
402 
403  // add inflow water to storage
404  m_chStorage[i] += qIn * m_dt;
405 
407  // then subtract all the outflow water
408  // 1. transmission losses to deep aquifer, which is lost from the system
409  // the unit of kchb is mm/hr
410  float seepage = m_Kchb[i] / 1000.f / 3600.f * m_chWidth[i] * m_chLen[i] * m_dt;
411  if (qgSub < 0.001f) {
412  if (m_chStorage[i] > seepage) {
413  m_seepage[i] = seepage;
414  m_chStorage[i] -= seepage;
415  } else {
416  m_seepage[i] = m_chStorage[i];
417  m_chStorage[i] = 0.f;
418  m_qOut[i] = 0.f;
419  m_qsCh[i] = 0.f;
420  m_qiCh[i] = 0.f;
421  m_qgCh[i] = 0.f;
422  return;
423  }
424  } else {
425  m_seepage[i] = 0.f;
426  }
427 
428  // 2. calculate transmission losses to bank storage
429  float dch = m_chStorage[i] / (m_chWidth[i] * m_chLen[i]);
430  float bankInLoss = 2 * m_Kbank[i] / 1000.f / 3600.f * dch * m_chLen[i] * m_dt; // m3/s
431  bankInLoss = 0.f;
432  if (m_chStorage[i] > bankInLoss) {
433  m_chStorage[i] -= bankInLoss;
434  } else {
435  bankInLoss = m_chStorage[i];
436  m_chStorage[i] = 0.f;
437  }
438  // water balance of the bank storage
439  // loss the water from bank storage to the adjacent unsaturated zone and groundwater storage
440  float bankOutGw = m_bankStorage[i] * (1 - exp(-m_bBank));
441  bankOutGw = 0.f;
442  m_bankStorage[i] = m_bankStorage[i] + bankInLoss - bankOutGw;
443  if (nullptr != m_gwStorage) {
444  m_gwStorage[i] += bankOutGw / m_area[i] * 1000.f;
445  } // updated groundwater storage
446 
447  if (m_chStorage[i] <= 0.f) {
448  m_qOut[i] = 0.f;
449  m_qsCh[i] = 0.f;
450  m_qiCh[i] = 0.f;
451  m_qgCh[i] = 0.f;
452  return;
453  }
454 
455  // 3. evaporation losses
456  float et = 0.f;
457  if (nullptr != m_petCh) {
458  et = m_Epch * m_petCh[i] / 1000.0f * m_chWidth[i] * m_chLen[i]; //m3
459  if (m_chStorage[i] > et) {
460  m_chStorage[i] -= et;
461  } else {
462  et = m_chStorage[i];
463  m_chStorage[i] = 0.f;
464  m_qOut[i] = 0.f;
465  m_qsCh[i] = 0.f;
466  m_qiCh[i] = 0.f;
467  m_qgCh[i] = 0.f;
468  return;
469  }
470  }
471 
473  // routing, there are water in the channel after inflow and transmission loss
474  float totalLoss = m_seepage[i] + bankInLoss + et;
475 
476  if (m_chStorage[i] >= 0.f) {
477  //qIn -= totalLoss/m_dt;
478  m_chStorage[i] = st0;
479 
480  float h = m_chStorage[i] / m_chWidth[i] / m_chLen[i];
481  float Perim = 2.f * h + m_chWidth[i];
482  float sSin = sqrt(sin(m_chSlope[i]));
483  float alpha = pow(m_chManning[i] / sSin * pow(Perim, _2div3), 0.6f);
484 
485  float lossRate = -totalLoss / m_dt / m_chWidth[i];
486  //lossRate = 0.f;
487  m_qOut[i] = GetNewQ(qIn, m_qOut[i], lossRate, alpha, m_dt, m_chLen[i]);
488 
489  float hNew = (alpha * pow(m_qOut[i], 0.6f)) / m_chWidth[i]; // unit m
490 
491  m_chStorage[i] += (qIn - m_qOut[i]) * m_dt;
492  //float hTest = h + (qIn-m_qOut[i])*m_dt/m_chWidth[i]/m_chLen[i];
493 
494  if (m_chStorage[i] < 0.f) {
495  m_qOut[i] = qIn;
496  m_chStorage[i] = 0.f;
497  }
498  } else {
499  m_qOut[i] = 0.f;
500  m_chStorage[i] = 0.f;
501  qIn = 0.f;
502  }
503 
504  float qInSum = m_qsSub[i] + qiSub + qgSub + qsUp + qiUp + qgUp;
505  m_qsCh[i] = m_qOut[i] * (m_qsSub[i] + qsUp) / qInSum;
506  m_qiCh[i] = m_qOut[i] * (qiSub + qiUp) / qInSum;
507  m_qgCh[i] = m_qOut[i] * (qgSub + qgUp) / qInSum;
508 
509  // set variables for next time step
510  m_qIn[i] = qIn;
511 
512  m_chWTdepth[i] = dch;
513 }
