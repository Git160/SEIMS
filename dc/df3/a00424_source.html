47 void ItpWeightData::ReadFromMongoDB(MongoGridFs* gfs, const string& filename) {
48  string wfilename = filename;
49  vector<string> gfilenames;
50  gfs->GetFileNames(gfilenames);
51  if (!ValueInVector(filename, gfilenames)) {
52  size_t index = filename.find_last_of('_');
53  string type = filename.substr(index + 1);
54  if (StringMatch(type, DataType_PotentialEvapotranspiration) || StringMatch(type, DataType_SolarRadiation)
55  || StringMatch(type, DataType_RelativeAirMoisture) || StringMatch(type, DataType_MeanTemperature)
56  || StringMatch(type, DataType_MaximumTemperature) || StringMatch(type, DataType_MinimumTemperature)) {
57  wfilename = filename.substr(0, index + 1) + DataType_Meteorology;
58  }
59  }
60  char* databuf;
61  size_t datalength;
62  gfs->GetStreamData(wfilename, databuf, datalength);
63  itp_weight_data_ = reinterpret_cast<float *>(databuf); // deprecate C-style: (float *) databuf
65  bson_t* md = gfs->GetFileMetadata(wfilename);
67  GetNumericFromBson(md, MONG_GRIDFS_WEIGHT_CELLS, n_rows_);

69 }
