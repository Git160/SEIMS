26 def get_psa_config():

38  psa_group.add_argument('-fast', action='store_true', help='Run FAST variant-based method')
39  # parse arguments

42  psa_mtd = 'morris' # Default
43  if args.fast:
44  psa_mtd = 'fast'
45  elif args.morris:
46  psa_mtd = 'morris'
47  if not FileClass.is_file_exists(ini_file):
48  raise ImportError('Configuration file is not existed: %s' % ini_file)
49  cf = ConfigParser()
50  cf.read(ini_file)
51  return cf, psa_mtd
