117 void NutrientinGroundwater::initialOutputs() {
118  CHECK_POSITIVE(MID_NUTRGW, m_nCells);
119  // allocate the output variables
120  if (nullptr == m_no3GwToCh) {
121  Initialize1DArray(m_nSubbasins + 1, m_no3GwToCh, 0.f);
122  Initialize1DArray(m_nSubbasins + 1, m_solpGwToCh, 0.f);
123  }
124  if (nullptr == m_gwNO3) {
125  m_gwNO3 = new float[m_nSubbasins + 1];
126  m_gwSolP = new float[m_nSubbasins + 1];
127  for (auto it = m_subbasinIDs.begin(); it != m_subbasinIDs.end(); it++) {
128  Subbasin *subbasin = m_subbasinsInfo->GetSubbasinByID(*it);
129  float subArea = subbasin->getCellCount() * m_cellWidth * m_cellWidth; //m2
130  m_gwNO3[*it] = m_gw0 * m_gwno3Con[*it] * subArea / 1000000.f;
131  m_gwSolP[*it] = m_gw0 * m_gwSolPCon[*it] * subArea / 1000000.f;
132  }
133  }

136 int NutrientinGroundwater::Execute() {
137  CheckInputData();
138  initialOutputs();
139  for (auto it = m_subbasinIDs.begin(); it != m_subbasinIDs.end(); it++) {
140  int id = *it;
141  Subbasin *subbasin = m_subbasinsInfo->GetSubbasinByID(id);
142  int nCells = subbasin->getCellCount();
143  float subArea = nCells * m_cellWidth * m_cellWidth; // m^2
144  float revap = subbasin->getEG();
147  float gwqVol = m_gw_q[id] * m_TimeStep; // m^3, water volume flow out
148  float reVapVol = revap * subArea / 1000.f; // m^3
149  float tmpGwStorage = m_gwStor[id] * subArea / 1000.f + gwqVol + reVapVol;
151  m_gwNO3[id] += m_perco_no3_gw[id];
152  m_gwSolP[id] += m_perco_solp_gw[id];
153  m_gwno3Con[id] = m_gwNO3[id] / tmpGwStorage * 1000.f; // kg / m^3 * 1000. = mg/L
154  m_gwSolPCon[id] = m_gwSolP[id] / tmpGwStorage * 1000.f;
156  //cout<<"subID: "<<id<<", gwQ: "<<m_gw_q[id] << ", ";
157  m_no3GwToCh[id] = m_gwno3Con[id] * gwqVol / 1000.f; // g/m3 * m3 / 1000 = kg
158  m_solpGwToCh[id] = m_gwSolPCon[id] * gwqVol / 1000.f;
159  //cout<<"subID: "<<id<<", gwno3Con: "<<m_gwno3Con[id] << ", no3gwToCh: "<<m_no3gwToCh[id] << ", ";
161  float no3ToSoil = revap / 1000.f * m_gwno3Con[id] * 10.f; // kg/ha (m*10*g/m3=kg/ha)
162  float solpToSoil = revap / 1000.f * m_gwSolPCon[id] * 10.f;
163  float no3ToSoil_kg = no3ToSoil * subArea / 10000.f;
164  float solpToSoil_kg = solpToSoil * subArea / 10000.f;
165  int *cells = subbasin->getCells();
166  int index = 0;
167  for (int i = 0; i < nCells; i++) {
168  index = cells[i];
169  m_sol_no3[index][(int) m_soilLayers[index] - 1] += no3ToSoil;
170  m_sol_solp[index][(int) m_soilLayers[index] - 1] += solpToSoil;
171  }
173  m_gwNO3[id] -= (m_no3GwToCh[id] + no3ToSoil_kg);
174  m_gwSolP[id] -= (m_solpGwToCh[id] + solpToSoil_kg);
175  }
176  return 0;
177 }
