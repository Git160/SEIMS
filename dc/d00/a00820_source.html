32  def match_subbasin(subbsn_file, site_dict, maindb):
33  """
34  Match the ID of subbasin
35  1. Read the coordinates of each subbasin's outlet, and
36  the outlet ID of the whole basin (not finished yet)
37  2. If the isOutlet field equals to
38  2.1 - 0, then return the subbasin_id of the site's location
39  2.2 - 1, then return the outlet ID of the whole basiin
40  2.3 - 2, then return the outlet ID of nearest subbasin
41  2.4 - 3, then return the outlet IDs of the conjunct subbasins
42  """
43  subbasin_raster = RasterUtilClass.read_raster(subbsn_file)
44  localx = site_dict.get(StationFields.x)
45  localy = site_dict.get(StationFields.y)
46  site_type = site_dict.get(StationFields.outlet)
47  subbasin_id = subbasin_raster.get_value_by_xy(localx, localy)
48  if subbasin_id is None and site_type != 1:
49  # the site is not inside the basin and not the outlet either.
50  return False, None
51  if site_type == 0:
52  return True, [subbasin_id]
53  elif site_type == 1:
54  outid = int(MongoQuery.get_init_parameter_value(maindb, SubbsnStatsName.outlet))
55  return True, [outid]
56  elif site_type == 2:
57  return True, [subbasin_id] # TODO

99  elif StringClass.string_match(site_flds[j], StationFields.elev):

251  """
252  This function mainly to import measurement data to MongoDB
253  data type may include Q (discharge, m3/s), SED (mg/L), tn (mg/L), tp (mg/L), etc.
254  the required parameters that defined in configuration file (*.ini)
255  """
256  if not cfg.use_observed:
257  return False
258  c_list = climdb.collection_names()
259  if not StringClass.string_in_list(DBTableNames.observes, c_list):
260  climdb.create_collection(DBTableNames.observes)
261  else:
262  climdb.drop_collection(DBTableNames.observes)
263  if not StringClass.string_in_list(DBTableNames.sites, c_list):
264  climdb.create_collection(DBTableNames.sites)
265  if not StringClass.string_in_list(DBTableNames.var_desc, c_list):
266  climdb.create_collection(DBTableNames.var_desc)
267 
268  file_list = FileClass.get_full_filename_by_suffixes(cfg.observe_dir, ['.txt'])
269  meas_file_list = []
270  site_loc = []
271  for fl in file_list:
272  if StringClass.is_substring('observed_', fl):
273  meas_file_list.append(fl)
274  else:
275  site_loc.append(fl)
276  ImportObservedData.data_from_txt(maindb, climdb, meas_file_list, site_loc,
277  cfg.spatials.subbsn)
