43  mongoc_cursor_t *cursor;

47  collection = mongoc_client_get_collection(conn, bmpDBName.c_str(), m_bmpCollection.c_str());
48  if (collection == NULL) {
49  throw ModelException("BMPs Scenario", "Read Plant Management Operations",
50  "Failed to get collection: " + m_bmpCollection + ".\n");
51  }
52  cursor = mongoc_collection_find(collection, MONGOC_QUERY_NONE, 0, 0, 0, b, NULL, NULL);
53  bson_iter_t itertor;
54  int paramNum = 10;
55  m_parameters = new float[paramNum];
56  int count = 1;

58  //int seqNo;
59  int mgtCode = -1, year = -9999, month = -9999, day = -9999;
60  float husc = 0.f;
61  bool usebaseHU = true;
62  if (bson_iter_init_find(&itertor, bsonTable, BMP_FLD_NAME)) {
63  m_name = GetStringFromBsonIterator(&itertor);
64  }
65  if (bson_iter_init_find(&itertor, bsonTable, BMP_PLTOP_FLD_LUCC)) {
66  GetNumericFromBsonIterator(&itertor, m_luccID);
67  }
68  if (m_luccID < 0) m_luccID = 1; // AGRL Agricultural Land-Generic
69  if (bson_iter_init_find(&itertor, bsonTable, BMP_PLTOP_FLD_MGTOP)) {
70  GetNumericFromBsonIterator(&itertor, mgtCode);

72  //if (bson_iter_init_find(&itertor,bsonTable,BMP_FLD_SEQUENCE))
73  // GetNumericFromBsonIterator(&itertor, seqNo);
74  if (bson_iter_init_find(&itertor, bsonTable, BMP_PLTOP_FLD_YEAR)) {
75  GetNumericFromBsonIterator(&itertor, year);
76  }
77  if (bson_iter_init_find(&itertor, bsonTable, BMP_PLTOP_FLD_MONTH)) {
78  GetNumericFromBsonIterator(&itertor, month);
79  }
80  if (bson_iter_init_find(&itertor, bsonTable, BMP_PLTOP_FLD_DAY)) {
81  GetNumericFromBsonIterator(&itertor, day);
82  }
83  if (bson_iter_init_find(&itertor, bsonTable, BMP_PLTOP_FLD_BASEHU)) {
84  usebaseHU = GetBoolFromBsonIterator(&itertor);
85  }
86  if (bson_iter_init_find(&itertor, bsonTable, BMP_PLTOP_FLD_HUSC)) {
87  GetNumericFromBsonIterator(&itertor, husc);
88  }
89  for (int i = 0; i < paramNum; i++) {
90  ostringstream oss;
91  oss << BMP_PLTOP_FLD_MGT_PRE << (i + 1);
92  if (bson_iter_init_find(&itertor, bsonTable, oss.str().c_str())) {
93  GetNumericFromBsonIterator(&itertor, m_parameters[i]);
94  }
95  }
96  int uniqueMgtCode = count * 1000 + mgtCode;
97  m_bmpSequence.push_back(uniqueMgtCode);
98  switch (mgtCode) {
99  case BMP_PLTOP_Plant:
100  m_bmpPlantOps[uniqueMgtCode] = new PlantOperation(mgtCode, usebaseHU, husc, year, month, day,
101  m_parameters);
102  break;
103  case BMP_PLTOP_Irrigation:
104  m_bmpPlantOps[uniqueMgtCode] = new IrrigationOperation(mgtCode, usebaseHU, husc, year, month, day,
105  m_parameters);
106  break;
107  case BMP_PLTOP_Fertilizer:
108  m_bmpPlantOps[uniqueMgtCode] = new FertilizerOperation(mgtCode, usebaseHU, husc, year, month, day,
109  m_parameters);
110  break;
111  case BMP_PLTOP_Pesticide:
112  m_bmpPlantOps[uniqueMgtCode] = new PesticideOperation(mgtCode, usebaseHU, husc, year, month, day,
113  m_parameters);
114  break;
115  case BMP_PLTOP_HarvestKill:
116  m_bmpPlantOps[uniqueMgtCode] = new HarvestKillOperation(mgtCode, usebaseHU, husc, year, month, day,
117  m_parameters);
118  break;
119  case BMP_PLTOP_Tillage:
120  m_bmpPlantOps[uniqueMgtCode] = new TillageOperation(mgtCode, usebaseHU, husc, year, month, day,
121  m_parameters);
122  break;
123  case BMP_PLTOP_Harvest:
124  m_bmpPlantOps[uniqueMgtCode] = new HarvestOnlyOperation(mgtCode, usebaseHU, husc, year, month, day,
125  m_parameters);
126  break;
127  case BMP_PLTOP_Kill:

138  break;
