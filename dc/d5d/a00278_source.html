50 void BMPPointSrcFactory::ReadPointSourceManagements(mongoc_client_t *conn, string &bmpDBName) {
51  bson_t *b = bson_new();
52  bson_t *child1 = bson_new(), *child2 = bson_new();
53  BSON_APPEND_DOCUMENT_BEGIN(b, "$query", child1);
54  BSON_APPEND_INT32(child1, BMP_FLD_SUB, m_subScenarioId);
55  bson_append_document_end(b, child1);
56  BSON_APPEND_DOCUMENT_BEGIN(b, "$orderby", child2);
57  BSON_APPEND_INT32(child2, BMP_FLD_SEQUENCE, 1);
58  bson_append_document_end(b, child2);
59  bson_destroy(child1);
60  bson_destroy(child2);
61 
62  mongoc_cursor_t *cursor;
63  const bson_t *bsonTable;
64  mongoc_collection_t *collection = NULL;
65  bson_iter_t iter;
66 
67  collection = mongoc_client_get_collection(conn, bmpDBName.c_str(), m_pointSrcMgtTab.c_str());
68  if (collection == NULL) {
69  throw ModelException("BMPs Scenario", "Read Point Source Management",
70  "Failed to get collection: " + m_pointSrcMgtTab + ".\n");
71  }
72  cursor = mongoc_collection_find(collection, MONGOC_QUERY_NONE, 0, 0, 0, b, NULL, NULL);
74  int count = 1;
75  while (mongoc_cursor_next(cursor, &bsonTable)) {
76  m_pointSrcMgtSeqs.push_back(count);
77  m_pointSrcMgtMap[count] = new PointSourceMgtParams(bsonTable, iter);
78  count++;
79  }
80  bson_destroy(b);
81  mongoc_collection_destroy(collection);
82  mongoc_cursor_destroy(cursor);

85 void BMPPointSrcFactory::ReadPointSourceLocations(mongoc_client_t *conn, string &bmpDBName) {
86  bson_t *b = bson_new();
87  bson_t *child1 = bson_new();
88  BSON_APPEND_DOCUMENT_BEGIN(b, "$query", child1);
89  BSON_APPEND_INT32(child1, BMP_PTSRC_FLD_CODE, m_pointSrc);
90  bson_append_document_end(b, child1);
91  bson_destroy(child1);
92 
93  mongoc_cursor_t *cursor;
94  const bson_t *bsonTable;
95  mongoc_collection_t *collection = NULL;
96  bson_iter_t iter;
97 
98  collection = mongoc_client_get_collection(conn, bmpDBName.c_str(), m_pointSrcDistTab.c_str());
99  if (collection == NULL) {
100  throw ModelException("BMPs Scenario", "Read Point Source Locations",
101  "Failed to get collection: " + m_pointSrcDistTab + ".\n");
102  }
103  cursor = mongoc_collection_find(collection, MONGOC_QUERY_NONE, 0, 0, 0, b, NULL, NULL);
104 
105  while (mongoc_cursor_next(cursor, &bsonTable)) {
106  PointSourceLocations *curPtSrcLoc = new PointSourceLocations(bsonTable, iter);
107  int curPtSrcID = curPtSrcLoc->GetPointSourceID();
108  if (ValueInVector(curPtSrcID, m_pointSrcIDs)) {
109  m_pointSrcLocsMap[curPtSrcID] = curPtSrcLoc;
110  } else {
111  RemoveValueInVector(curPtSrcID, m_pointSrcIDs);
112  }
113  }
114  bson_destroy(b);
115  mongoc_collection_destroy(collection);
116  mongoc_cursor_destroy(cursor);
