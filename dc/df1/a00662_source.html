39 void SUR_MR::InitialOutputs() {
40  CHECK_POSITIVE(MID_SUR_MR, m_nCells);
41  // allocate the output variables
42  if (nullptr == m_exsPcp) {
43  Initialize1DArray(m_nCells, m_exsPcp, 0.f);
44  Initialize1DArray(m_nCells, m_infil, 0.f);
45  Initialize1DArray(m_nCells, m_soilWtrStoPrfl, 0.f);
46  Initialize2DArray(m_nCells, m_maxSoilLyrs, m_soilWtrSto, NODATA_VALUE);
47 #pragma omp parallel for
48  for (int i = 0; i < m_nCells; i++) {
49  for (int j = 0; j < CVT_INT(m_nSoilLyrs[i]); j++) {
50  if (m_initSoilWtrStoRatio[i] >= 0.f && m_initSoilWtrStoRatio[i] <= 1.f && m_soilFC[i][j] >= 0.f) {
51  m_soilWtrSto[i][j] = m_initSoilWtrStoRatio[i] * m_soilFC[i][j];
52  } else {
53  m_soilWtrSto[i][j] = 0.f;
54  }
55  m_soilWtrStoPrfl[i] += m_soilWtrSto[i][j];
56  }
57  }
58  }
60  if (nullptr == m_soilSumSat && m_soilSat != nullptr) {
61  m_soilSumSat = new(nothrow) float[m_nCells];
62 #pragma omp parallel for
63  for (int i = 0; i < m_nCells; i++) {
64  m_soilSumSat[i] = 0.f;
65  for (int j = 0; j < CVT_INT(m_nSoilLyrs[i]); j++) {
66  m_soilSumSat[i] += m_soilSat[i][j];
67  }
68  }
69  }

72 int SUR_MR::Execute() {
73  CheckInputData();
74  InitialOutputs();
75  m_maxPcpRf *= m_dt * 1.1574074074074073e-05f;
76 #pragma omp parallel for
77  for (int i = 0; i < m_nCells; i++) {
78  float hWater = 0.f;
79  hWater = m_netPcp[i] + m_deprSto[i];
80  if (hWater > 0.f) {
82  m_soilWtrStoPrfl[i] = 0.f;
83  for (int ly = 0; ly < CVT_INT(m_nSoilLyrs[i]); ly++) {
84  m_soilWtrStoPrfl[i] += m_soilWtrSto[i][ly];
85  }
86  float smFraction = Min(m_soilWtrStoPrfl[i] / m_soilSumSat[i], 1.f);
87  // for frozen soil, no infiltration will occur
88  if (m_soilTemp[i] <= m_soilFrozenTemp && smFraction >= m_soilFrozenWtrRatio) {
89  m_exsPcp[i] = m_netPcp[i];
90  m_infil[i] = 0.f;
91  } else {
92  float alpha = m_rfExp - (m_rfExp - 1.f) * hWater / m_maxPcpRf;
93  if (hWater >= m_maxPcpRf) {
94  alpha = 1.f;
95  }
96 
97  //runoff percentage
98  float runoffPercentage;
99  if (m_potRfCoef[i] > 0.99f) {
100  runoffPercentage = 1.f;
101  } else {
102  runoffPercentage = m_potRfCoef[i] * pow(smFraction, alpha);
103  }
104 
105  float surfq = hWater * runoffPercentage;
106  if (surfq > hWater) surfq = hWater;
107  m_infil[i] = hWater - surfq;
108  m_exsPcp[i] = surfq;
109 
113  }
114  } else {
115  m_exsPcp[i] = 0.f;
116  m_infil[i] = 0.f;
117  }
118  if (m_infil[i] > 0.f) {
119  m_soilWtrSto[i][0] += m_infil[i];
120  }
121  }
122  return 0;
123 }
