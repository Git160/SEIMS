28 def crossover_slppos(tagnames, ind1, ind2):
29  """Crossover operator based on slope position units.
30  Args:
31  tagnames(list): slope position tags and names, from up to bottom of hillslope.
32  The format is [(tag, name),...].
33  ind1: The first individual participating in the crossover.
34  ind2: The second individual participating in the crossover.
35 
36  Returns:
37  A tuple of two individuals.
38  """
39  sp_num = len(tagnames)
40  size = min(len(ind1), len(ind2))
41  assert (size > sp_num * 2)
42 
43  while True:
44  cxpoint1 = random.randint(0, size - 1)
45  cxpoint2 = random.randint(1, size)
46  if cxpoint2 < cxpoint1: # Swap the two cx points
47  cxpoint1, cxpoint2 = cxpoint2, cxpoint1
48  # print(cxpoint1, cxpoint2)
49  cs1 = cxpoint1 / sp_num
50  cs2 = cxpoint2 / sp_num
51  # print(cs1, cs2)
52  cxpoint1 = cs1 * sp_num
53  if cxpoint2 % sp_num != 0:
54  cxpoint2 = sp_num * (cs2 + 1)
55  if cxpoint1 == cxpoint2:
56  cxpoint2 += sp_num
57  # print(cxpoint1, cxpoint2)
58  if not (cxpoint1 == 0 and cxpoint2 == size):
59  break # avoid change the entire genes
60  ind1[cxpoint1:cxpoint2], ind2[cxpoint1:cxpoint2] \
61  = ind2[cxpoint1:cxpoint2], ind1[cxpoint1:cxpoint2]
62 
63  # Check the validation according to spatial configuration rules.
64 
65  return ind1, ind2

99  perc, indpb, method=1):
100  """
101  Mutation Gene values based on slope position rules.

106  gene2unit(dict): Gene index to slope position unit ID.

111  individual(list or tuple): Individual to be mutated.
112  perc(float): percent of gene length for mutate, default is 0.02
113  indpb(float): Independent probability for each attribute to be mutated.
114  method(int): Domain knowledge based rule method.
115 
116  Returns:
117  A tuple of one individual.
118  """
119  if perc > 0.5:
120  perc = 0.5
121  elif perc < 0.01:
122  perc = 0.01
123  try:
124  mut_num = random.randint(1, int(len(individual) * perc))
125  except ValueError or Exception:
126  return individual
127  # print('Max mutate num: %d' % mut_num)
128  muted = list()
129  for m in range(mut_num):
130  if random.random() > indpb:
131  continue
132  # mutate will happen
133  mpoint = random.randint(0, len(individual) - 1)
134  while mpoint in muted:
135  mpoint = random.randint(0, len(individual) - 1)
136 
137  oldgenev = individual[mpoint]
138  unitid = gene2unit[mpoint]

143  down_gid = -1 # gene index

183 def mutate_rdm(bmps_mut_target, individual, perc, indpb):
184  """
185  Mutation Gene values randomly, old gene value is excluded from target values.
186  Args:
187  bmps_mut_target(list or tuple): All available gene values.
188  individual(list or tuple): Individual to be mutated.
189  perc(float): percent of gene length for mutate, default is 0.02

191 
192  Returns:
193  A tuple of one individual.
194  """
195  if perc > 0.5:
196  perc = 0.5
197  elif perc < 0.01:
198  perc = 0.01
199  if 0 not in bmps_mut_target:
200  bmps_mut_target.append(0)
201  try:
202  mut_num = random.randint(1, int(len(individual) * perc))
203  except ValueError or Exception:
204  return individual
