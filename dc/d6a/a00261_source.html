101  def Precipitation(self, subbsn_id, start_time, end_time):
102  """
103  The precipitation is read according to the subbasin ID.
104  Especially when plot a specific subbasin (such as ID 3).
105  For the whole basin, the subbasin ID is 0.
106  Returns:
107  Precipitation data list with the first element as datetime.
108  [[Datetime1, value1], [Datetime2, value2], ..., [Datetimen, valuen]]
109  """
110  pcp_date_value = list()
111  sitelist_tab = self.maindb[DBTableNames.main_sitelist]
112  findsites = sitelist_tab.find_one({FieldNames.subbasin_id: subbsn_id,
113  FieldNames.mode: self.Mode})
114  if findsites is not None:
115  site_liststr = findsites[FieldNames.site_p]
116  else:
117  raise RuntimeError('Cannot find precipitation site for subbasin %d.' % subbsn_id)
118  site_list = StringClass.extract_numeric_values_from_string(site_liststr)
119  site_list = [int(v) for v in site_list]
120  if len(site_list) == 0:
121  raise RuntimeError('Cannot find precipitation site for subbasin %d.' % subbsn_id)
122 
123  pcp_dict = OrderedDict()
124 
125  for pdata in self.climatedb[DBTableNames.data_values].find(
126  {DataValueFields.utc: {"$gte": start_time, '$lte': end_time},
127  DataValueFields.type: DataType.p,
128  DataValueFields.id: {"$in": site_list}}).sort([(DataValueFields.utc, 1)]):
129  curt = pdata[DataValueFields.utc]
130  curv = pdata[DataValueFields.value]
131  if curt not in pcp_dict:
132  pcp_dict[curt] = 0.
133  pcp_dict[curt] += curv
134  # average
135  if len(site_list) > 1:
136  for t in pcp_dict:
137  pcp_dict[t] /= len(site_list)
138  for t, v in pcp_dict.items():
139  # print(str(t), v)
140  pcp_date_value.append([t, v])
141  print('Read precipitation from %s to %s done.' % (start_time.strftime('%c'),
142  end_time.strftime('%c')))
143  return pcp_date_value
