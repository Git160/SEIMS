98  def read_param_ranges(self):
99  """Read param_rng.def file
100 
101  name,lower_bound,upper_bound,group,dist
102  (group and dist are optional)
103 
104  e.g.,
105  Param1,0,1[,Group1][,dist1]
106  Param2,0,1[,Group2][,dist2]
107  Param3,0,1[,Group3][,dist3]
108 
109  Returns:
110  a dictionary containing:
111  - names - the names of the parameters
112  - bounds - a list of lists of lower and upper bounds
113  - num_vars - a scalar indicating the number of variables
114  (the length of names)
115  - groups - a list of group names (strings) for each variable
116  - dists - a list of distributions for the problem,
117  None if not specified or all uniform
118  """
119  # read param_defs.json if already existed
120  if not self.param_defs:
121  if FileClass.is_file_exists(self.cfg.outfiles.param_defs_json):
122  with open(self.cfg.outfiles.param_defs_json, 'r') as f:
123  self.param_defs = UtilClass.decode_strs_in_dict(json.load(f))
124  return
125  # read param_range_def file and output to json file
126  client = ConnectMongoDB(self.cfg.hostname, self.cfg.port)
127  conn = client.get_conn()
128  db = conn[self.cfg.spatial_db]
129  collection = db['PARAMETERS']
130 
131  names = list()
132  bounds = list()
133  groups = list()
134  dists = list()
135  num_vars = 0
136  items = read_data_items_from_txt(self.cfg.param_range_def)
137  for item in items:
138  if len(item) < 3:
139  continue
140  # find parameter name, print warning message if not existed
141  cursor = collection.find({'NAME': item[0]}, no_cursor_timeout=True)
142  if not cursor.count():
143  print('WARNING: parameter %s is not existed!' % item[0])
144  continue
145  num_vars += 1
146  names.append(item[0])
147  bounds.append([float(item[1]), float(item[2])])
148  # If the fourth column does not contain a group name, use
149  # the parameter name
150  if len(item) >= 4:
151  groups.append(item[3])
152  else:
153  groups.append(item[0])
154  if len(item) >= 5:
155  dists.append(item[4])
156  else:
157  dists.append('unif')
158  if groups == names:
159  groups = None
160  elif len(set(groups)) == 1:
161  raise ValueError('Only one group defined, results will not bemeaningful')
162 
163  # setting dists to none if all are uniform
164  # because non-uniform scaling is not needed
165  if all([d == 'unif' for d in dists]):
166  dists = None
167 
168  self.param_defs = {'names': names, 'bounds': bounds,
169  'num_vars': num_vars, 'groups': groups, 'dists': dists}
170 
171  # Save as json, which can be loaded by json.load()
172  json_data = json.dumps(self.param_defs, indent=4, cls=SpecialJsonEncoder)
173  with open(self.cfg.outfiles.param_defs_json, 'w') as f:
174  f.write(json_data)

220  """
221  if self.output_values is None or len(self.output_values) == 0:
222  if FileClass.is_file_exists(self.cfg.outfiles.output_values_txt):
223  self.output_values = numpy.loadtxt(self.cfg.outfiles.output_values_txt)
224  return
225  assert (self.run_count > 0)
226  # model configurations
227  model_cfg_dict = {'bin_dir': self.cfg.seims_bin, 'model_dir': self.cfg.model_dir,
228  'nthread': self.cfg.seims_nthread, 'lyrmethod': self.cfg.seims_lyrmethod,
229  'hostname': self.cfg.hostname, 'port': self.cfg.port,
230  'scenario_id': 0}
231 
232  # split tasks if needed
233  task_num = self.run_count / 1000
234  if task_num == 0:
235  split_seqs = list(range(self.run_count))
236  else:
237  split_seqs = numpy.array_split(numpy.arange(self.run_count), task_num)
238  split_seqs = [a.tolist() for a in split_seqs]
239 
240  # PSA evaluation period
241  psa_period = '%s,%s' % (self.cfg.psa_stime.strftime('%Y-%m-%d %H:%M:%S'),
242  self.cfg.psa_etime.strftime('%Y-%m-%d %H:%M:%S'))
243  for idx, cali_seqs in enumerate(split_seqs):
244  cur_out_file = self.cfg.outfiles.output_values_dir + os.path.sep + 'outputs_%d.txt' % idx
245  if FileClass.is_file_exists(cur_out_file):
246  continue
247  try: # parallel on multiprocesor or clusters using SCOOP
248  from scoop import futures
249  temp_output_values = list(futures.map(evaluate_model_response,
250  [model_cfg_dict] * len(cali_seqs),
251  cali_seqs, [psa_period] * len(cali_seqs)))
252  except ImportError or ImportWarning: # serial
253  temp_output_values = list(map(evaluate_model_response,
254  [model_cfg_dict] * len(cali_seqs), cali_seqs,
255  [psa_period] * len(cali_seqs)))
256  if not isinstance(temp_output_values, numpy.ndarray):
257  temp_output_values = numpy.array(temp_output_values)
258  numpy.savetxt(cur_out_file, temp_output_values, delimiter=' ', fmt='%.4e')
259  # load the first part of output values
260  self.output_values = numpy.loadtxt(self.cfg.outfiles.output_values_dir + os.path.sep +
261  'outputs_0.txt')
262  if task_num == 0:
263  import shutil
264  shutil.move(self.cfg.outfiles.output_values_dir + os.path.sep + 'outputs_0.txt',
265  self.cfg.outfiles.output_values_txt)
266  shutil.rmtree(self.cfg.outfiles.output_values_dir)
267  return
268  for idx in range(1, task_num):
269  tmp_outputs = numpy.loadtxt(self.cfg.outfiles.output_values_dir + os.path.sep +
270  'outputs_%d.txt' % idx)
271  self.output_values = numpy.concatenate((self.output_values, tmp_outputs))
272  numpy.savetxt(self.cfg.outfiles.output_values_txt,
273  self.output_values, delimiter=' ', fmt='%.4e')

275  def calculate_sensitivity(self):
276  """Calculate Morris elementary effects.
277  It is worth to be noticed that evaluate_models() allows to return
278  several output variables, hence we should calculate each of them separately.
279  """
280  if not self.psa_si:

282  with open(self.cfg.outfiles.psa_si_json, 'r') as f:
283  self.psa_si = UtilClass.decode_strs_in_dict(json.load(f))
284  return
285  if self.output_values is None or len(self.output_values) == 0:
286  self.evaluate_models()
287  if self.param_values is None or len(self.param_values) == 0:
288  self.generate_samples()
289  if not self.param_defs:
290  self.read_param_ranges()
291  row, col = self.output_values.shape
292  assert (row == self.run_count)
293  for i in range(col):
294  if self.cfg.method == 'morris':
295  tmp_Si = morris_alz(self.param_defs,
296  self.param_values,
297  self.output_values[:, i],
298  conf_level=0.95, print_to_console=True,
299  num_levels=self.cfg.morris.num_levels,
300  grid_jump=self.cfg.morris.grid_jump)
301  elif self.cfg.method == 'fast':
302  tmp_Si = fast_alz(self.param_defs, self.output_values[:, i],
303  print_to_console=True)
304  else:
305  raise ValueError('%s method is not supported now!' % self.cfg.method)
306  self.psa_si[i] = tmp_Si
307  # print(self.psa_si)
308  # Save as json, which can be loaded by json.load()
309  json_data = json.dumps(self.psa_si, indent=4, cls=SpecialJsonEncoder)

335  continue

337  sortidx = numpy.argsort(numpy.argsort(mtx, axis=0), axis=0)

340  output_str += header

343  output_str += '\n'
344  output_str += '%s-Sorted\n' % outvar
345  output_str += header
346  for i, v in enumerate(sortidx):
347  output_str += param_names[i] + ',' + ','.join('{}'.format(i) for i in v) + '\n'
348  output_str += '\n'
