46 int ReservoirMethod::Execute() {
47  if (!CheckInputData()) return -1;
48  initialOutputs();
49  float QGConvert = 1.f * m_CellWidth * m_CellWidth / (m_TimeStep) / 1000.f; // mm ==> m3/s
50  for (vector<int>::iterator it = m_subbasinIDs.begin(); it != m_subbasinIDs.end(); it++) {
51  int subID = *it;
52  Subbasin *curSub = m_subbasinsInfo->GetSubbasinByID(subID);
53 
54  // get percolation from the bottom soil layer at the subbasin scale
55  int curCellsNum = curSub->getCellCount();
56  int *curCells = curSub->getCells();
57  float perco = 0.f;
58 #pragma omp parallel for reduction(+:perco)
59  for (int i = 0; i < curCellsNum; i++) {
60  int index = 0;
61  index = curCells[i];
62  perco += m_perc[index][(int) m_soilLayers[index] - 1];
63  }
64  perco /= curCellsNum; // mean mm
68  float ratio2gw = 1.f;
69  perco *= ratio2gw;
70  curSub->setPerco(perco);
71 
72  //if (perco > 0.f)
73  //{
74  // cout << "subID: "<<subID<<", perco mean: "<<perco << endl;
75  //}
76 
77  //calculate EG, i.e. Revap
78  float revap = 0.f;
79  float fPET = 0.f;
80  float fEI = 0.f;
81  float fED = 0.f;
82  float fES = 0.f;
83  float plantEP = 0.f;
84  fPET = Sum(curCellsNum, curCells, m_D_PET) / curCellsNum;
85  fEI = Sum(curCellsNum, curCells, m_D_EI) / curCellsNum;
86  fED = Sum(curCellsNum, curCells, m_D_ED) / curCellsNum;
87  fES = Sum(curCellsNum, curCells, m_D_ES) / curCellsNum;
88  plantEP = Sum(curCellsNum, curCells, m_plantEP) / curCellsNum;
89 
90  curSub->setPET(fPET);
91 
92  //if percolation < 0.01, EG will be 0. if percolation >= 0.01, EG will be calculated by equation (why? this is not used currently. Junzhi Liu 2016-08-14).
93  //if (perco >= 0.01f)
94  //{
95  revap = (fPET - fEI - fED - fES - plantEP) * m_gwStore[subID] / m_GWMAX;
96  if (revap != revap) {
97  cout << "fPET: " << fPET << ", fEI: " << fEI << ", fED: " << fED << ", fES: " << fES << ", plantEP: "
98  << plantEP << ", " << " subbasin ID: " << subID << ", gwStore: " << m_gwStore[subID] << endl;
99  throw ModelException(MID_GWA_RE, "Execute", "revap calculation wrong!");
100  }
101  revap = max(revap, 0.f);
102  revap = min(revap, perco);
103  //}
104  //float prevRevap = curSub->getEG();
105  //if (prevRevap != revap)
106  //{
107  // curSub->setEG(revap);
108  // curSub->setIsRevapChanged(true);
109  //}
110  //else
111  // curSub->setIsRevapChanged(false);
112  curSub->setEG(revap);
113 
114  //deep percolation
115  float percoDeep = perco * m_dp_co;
116  curSub->setPerde(percoDeep);
117 
118  // groundwater runoff (mm)
119  float slopeCoef = curSub->getSlopeCoef();
120  float kg = m_Kg * slopeCoef;
121  float groundRunoff = kg * pow(m_gwStore[subID], m_Base_ex); //mm
122  if (groundRunoff != groundRunoff) {
123  cout << groundRunoff;
124  }
125 
126  float groundQ = groundRunoff * curCellsNum * QGConvert; // groundwater discharge (m3/s)
127 
128  float groundStorage = m_gwStore[subID];
129  groundStorage += (perco - revap - percoDeep - groundRunoff);
130 
131  //add the ground water from bank storage, 2011-3-14
132  float gwBank = 0.f;
133  // at the first time step m_VgroundwaterFromBankStorage is NULL
134  if (m_VgroundwaterFromBankStorage != NULL) {
135  gwBank = m_VgroundwaterFromBankStorage[subID];
136  }
137  groundStorage += gwBank / curSub->getArea() * 1000.f;
138 
139  groundStorage = max(groundStorage, 0.f);
140  if (groundStorage > m_GWMAX) {
141  groundRunoff += (groundStorage - m_GWMAX);
142  groundQ = groundRunoff * curCellsNum * QGConvert; // groundwater discharge (m3/s)
143  groundStorage = m_GWMAX;
144  }
145  curSub->setRG(groundRunoff);
146  curSub->setGW(groundStorage);
147  curSub->setQG(groundQ);
148  if (groundStorage != groundStorage) {
149  ostringstream oss;
150  oss << perco << "\t" << revap << "\t" << percoDeep << "\t" << groundRunoff << "\t" << m_gwStore[subID]
151  << "\t" << m_Kg << "\t" <<
152  m_Base_ex << "\t" << slopeCoef << endl;
153  throw ModelException("Subbasin", "setInputs", oss.str());
154  }
155  m_T_Perco[subID] = curSub->getPerco();
156  m_T_Revap[subID] = curSub->getEG();
157  m_T_PerDep[subID] = curSub->getPerde();
158  m_T_RG[subID] = curSub->getRG(); //get rg of specific subbasin
159  m_T_QG[subID] = curSub->getQG(); //get qg of specific subbasin
160  m_petSubbasin[subID] = curSub->getPET();
161  m_gwStore[subID] = curSub->getGW();
162  }
163 
164  m_T_Perco[0] = m_subbasinsInfo->subbasin2basin(VAR_PERCO);
165  m_T_Revap[0] = m_subbasinsInfo->subbasin2basin(VAR_REVAP);
166  m_T_PerDep[0] = m_subbasinsInfo->subbasin2basin(VAR_PERDE);
167  m_T_RG[0] = m_subbasinsInfo->subbasin2basin(VAR_RG); // get rg of entire watershed
168  m_gwStore[0] = m_subbasinsInfo->subbasin2basin(VAR_GW_Q);
169  m_T_QG[0] = m_subbasinsInfo->subbasin2basin(VAR_QG); // get qg of entire watershed
170 
171  // output to GWWB
172  for (int i = 0; i <= m_nSubbasins; i++) {
173  m_T_GWWB[i][0] = m_T_Perco[i];
174  m_T_GWWB[i][1] = m_T_Revap[i];
175  m_T_GWWB[i][2] = m_T_PerDep[i];
176  m_T_GWWB[i][3] = m_T_RG[i];
177  m_T_GWWB[i][4] = m_gwStore[i];
178  m_T_GWWB[i][5] = m_T_QG[i];
179  }
180 
181  // update soil moisture
182  for (vector<int>::iterator it = m_subbasinIDs.begin(); it != m_subbasinIDs.end(); it++) {
183  Subbasin *sub = m_subbasinsInfo->GetSubbasinByID(*it);
184  int *cells = sub->getCells();
185  int nCells = sub->getCellCount();
186  int index = 0;
187  for (int i = 0; i < nCells; i++) {
188  index = cells[i];
189  m_soilStorage[index][(int) m_soilLayers[index] - 1] += sub->getEG();
190  // TODO: Is it need to allocate revap to each soil layers??? By LJ
191  }
192  }
193  return 0;
