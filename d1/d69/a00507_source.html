31 void SSR_DA::FlowInSoil(int id) {
32  float s0 = max(m_slope[id], 0.01f);
33  float flowWidth = m_CellWidth;
34  // there is no land in this cell
35  if (m_streamLink[id] > 0) {
36  flowWidth -= m_chWidth[id];
37  }
38  // initialize for current cell of current timestep
39  for (int j = 0; j < (int) m_soilLayers[id]; j++) {
40  m_qi[id][j] = 0.f;
41  m_qiVol[id][j] = 0.f;
42  }
43  // return with initial values if flowWidth is less than 0
44  if (flowWidth <= 0) return;
45  // number of flow-in cells
46  int nUpstream = (int) m_flowInIndex[id][0];
47  m_soilStorageProfile[id] = 0.f; // update soil storage on profile
48  for (int j = 0; j < (int) m_soilLayers[id]; j++) {
49  float smOld = m_soilStorage[id][j];
50  //sum the upstream subsurface flow, m3
51  float qUp = 0.f;
52  for (int upIndex = 1; upIndex <= nUpstream; ++upIndex) {
53  int flowInID = (int) m_flowInIndex[id][upIndex];
54  //cout << id << "\t" << flowInID << "\t" << m_nCells << endl;
55  if (m_qi[flowInID][j] > 0.f) {
56  qUp += m_qi[flowInID][j];
57  }// * m_flowInPercentage[id][upIndex];
58  //cout << id << "\t" << flowInID << "\t" << m_nCells << "\t" << m_qi[flowInID][j] << endl;
59  }
60  // add upstream water to the current cell
61  if (qUp < 0.f) qUp = 0.f;
62  m_soilStorage[id][j] += qUp; // mm
63  //TEST
64  if (m_soilStorage[id][j] != m_soilStorage[id][j] || m_soilStorage[id][j] < 0.f) {
65  ostringstream oss;
66  oss << "cell id: " << id << "\t layer: " << j << "\nmoisture is less than zero: " << m_soilStorage[id][j]
67  << "\t"
68  << smOld << "\nqUp: " << qUp << "\t depth:" << m_soilThick[id][j] << endl;
69  cout << oss.str() << endl;
70  throw ModelException(MID_SSR_DA, "Execute:FlowInSoil", oss.str());
71  }
72 
73  // if soil moisture is below the field capacity, no interflow will be generated, otherwise:
74  if (m_soilStorage[id][j] > m_fcmm[id][j]) {
75  // for the upper two layers, soil may be frozen
76  // also check if there are upstream inflow
77  if (j == 0 && m_soilT[id] <= m_frozenT && qUp <= 0.f) {
78  continue;
79  }
80 
81  float k = 0.f, maxSoilWater = 0.f, soilWater = 0.f, fcSoilWater = 0.f;
82  soilWater = m_soilStorage[id][j];
83  maxSoilWater = m_satmm[id][j];
84  fcSoilWater = m_fcmm[id][j];
85  //the moisture content can exceed the porosity in the way the algorithm is implemented
86  if (m_soilStorage[id][j] > maxSoilWater) {
87  k = m_ks[id][j];
88  } else {
90  float dcIndex = 2.f * m_poreIndex[id][j] + 3.f; // pore disconnectedness index
91  k = m_ks[id][j] * pow(m_soilStorage[id][j] / maxSoilWater, dcIndex);
92  if (k < 0.f) k = 0.f;
93  }
94  m_qi[id][j] = m_ki * s0 * k * m_dt / 3600.f * m_soilThick[id][j] / 1000.f / flowWidth; // the unit is mm
95 
96  if (soilWater - m_qi[id][j] > maxSoilWater) {
97  m_qi[id][j] = soilWater - maxSoilWater;
98  } else if (soilWater - m_qi[id][j] < fcSoilWater) {
99  m_qi[id][j] = soilWater - fcSoilWater;
100  }
101  m_qi[id][j] = max(0.f, m_qi[id][j]);
102 
103  m_qiVol[id][j] = m_qi[id][j] / 1000.f * m_CellWidth * flowWidth; //m3
104  m_qiVol[id][j] = max(0.f, m_qiVol[id][j]);
105  //Adjust the moisture content in the current layer, and the layer immediately below it
106  m_soilStorage[id][j] -= m_qi[id][j];
107  m_soilStorageProfile[id] += m_soilStorage[id][j];
108  if (m_soilStorage[id][j] != m_soilStorage[id][j] || m_soilStorage[id][j] < 0.f) {
109  ostringstream oss;
110  oss << id << "\t" << j << "\nmoisture is less than zero: " << m_soilStorage[id][j] << "\t"
111  << m_soilThick[id][j];
112  throw ModelException(MID_SSR_DA, "Execute", oss.str());
113  }
114  }
115  }

119 int SSR_DA::Execute() {
120  CheckInputData();
121  initialOutputs();
122 
123  for (int iLayer = 0; iLayer < m_nRoutingLayers; ++iLayer) {
124  // There are not any flow relationship within each routing layer.
125  // So parallelization can be done here.
126  int nCells = (int) m_routingLayers[iLayer][0];
127 
128 #pragma omp parallel for
129  for (int iCell = 1; iCell <= nCells; ++iCell) {
130  int id = (int) m_routingLayers[iLayer][iCell];
131  FlowInSoil(id);
132  }
133  }
134 
135  //cout << "end flowinsoil" << endl;
136  for (int i = 0; i < m_nSubbasin; i++) {
137  m_qiSubbasin[i] = 0.f;
138  }
139 
140 #pragma omp parallel for
141  for (int i = 0; i < m_nCells; i++) {
142  float qiAllLayers = 0.f;
143  if (m_streamLink[i] > 0) {
144  qiAllLayers = 0.f;
145  for (int j = 0; j < (int) m_soilLayers[i]; j++) {
146  if (m_qiVol[i][j] > UTIL_ZERO) {
147  qiAllLayers += m_qiVol[i][j] / m_dt;
148  }
149  }
150  //cout << m_nSubbasin << "\tsubbasin:" << m_subbasin[i] << "\t" << qiAllLayers << endl;
151  //if (m_nSubbasin > 1)
152  // m_qiSubbasin[int(m_subbasin[i])] += qiAllLayers;
153  //else
154  // m_qiSubbasin[1] += qiAllLayers;
155  m_qiSubbasin[int(m_subbasin[i])] += qiAllLayers;
156  }
157  }
158 
159  for (int i = 1; i <= m_nSubbasin; i++) {
160  //cout<<", "<<i<<": "<<m_qiSubbasin[i];
161  m_qiSubbasin[0] += m_qiSubbasin[i];
162  }
163  //cout<<endl;
164  return 0;
