153 bool ModuleFactory::LoadParseLibrary(const string& module_path, vector<string>& moduleIDs,
154  map<string, SEIMSModuleSetting *>& moduleSettings,
155  vector<DLLINSTANCE>& dllHandles,
156  map<string, InstanceFunction>& instanceFuncs,
157  map<string, MetadataFunction>& metadataFuncs,
158  map<string, vector<ParamInfo *> >& moduleParameters,
159  map<string, vector<ParamInfo *> >& moduleInputs,
160  map<string, vector<ParamInfo *> >& moduleOutputs,
161  map<string, vector<ParamInfo *> >& moduleInOutputs,
162  vector<ParamInfo*>& tfValueInputs) {
163  size_t n = moduleIDs.size();
164  // read all the .dll or .so and create objects
165  for (size_t i = 0; i < n; i++) {
166  string id = moduleIDs[i];
167  string dllID = id;
168  // for ITP modules, the input ids are ITP_T, ITP_P and ITP should be used as ID name
169  // The same to TSD_RD.
170  if (id.find(MID_ITP) != string::npos) {
171 #ifdef MSVC
172  dllID = MID_ITP;
173 #else
174  dllID = LIBPREFIX + string(MID_ITP);
175 #endif /* MSVC */
176  dllID += POSTFIX;
177  } else if (id.find(MID_TSD_RD) != string::npos) {
178 #ifdef MSVC
179  dllID = MID_TSD_RD;
180 #else
181  dllID = LIBPREFIX + string(MID_TSD_RD);
182 #endif /* MSVC */
183  dllID += POSTFIX;
184  }
185 
186  // load function pointers from DLL
187  ReadDLL(module_path, id, dllID, dllHandles, instanceFuncs, metadataFuncs);
188 
189  // load metadata
190  MetadataFunction metadataInfo = metadataFuncs[id];
191  const char* current_metadata = metadataInfo();
192  // parse the metadata
193  TiXmlDocument doc;
194  doc.Parse(current_metadata);
195  ReadParameterSetting(id, doc, moduleSettings[id], moduleParameters);
196  ReadIOSetting(id, doc, moduleSettings[id], TagInputs, TagInputVariable, moduleInputs);
197  ReadIOSetting(id, doc, moduleSettings[id], TagOutputs, TagOutputVariable, moduleOutputs);
198  ReadIOSetting(id, doc, moduleSettings[id], TagInOutputs, TagInOutputVariable, moduleInOutputs);
199  }
200  map<string, vector<ParamInfo *> >(moduleParameters).swap(moduleParameters);
201  map<string, vector<ParamInfo *> >(moduleInputs).swap(moduleInputs);
202  map<string, vector<ParamInfo *> >(moduleOutputs).swap(moduleOutputs);
203  map<string, vector<ParamInfo *> >(moduleInOutputs).swap(moduleInOutputs);
204  // set the connections among objects
205  for (auto it = moduleIDs.begin(); it != moduleIDs.end(); ++it) {
206  for (auto itInput = moduleInputs[*it].begin(); itInput != moduleInputs[*it].end(); ++itInput) {
207  if (StringMatch((*itInput)->Source, Source_Module) ||
208  StringMatch((*itInput)->Source, Source_Module_Optional)) {
209  (*itInput)->DependPara = FindDependentParam(*itInput, moduleIDs, moduleOutputs);
210  if ((*itInput)->Transfer == TF_SingleValue) {
211  // Check and append Inputs need to be transferred
212  if ((*itInput)->DependPara != nullptr) {
213  tfValueInputs.push_back((*itInput));
214  } else {
215  throw ModelException("ModelFactory", "LoadParseLibrary",
216  "Couldn't find dependent output for " + (*itInput)->Name);
217  }
218  }
219  }
220  }
221  }
222  for (auto it = moduleInOutputs.begin(); it != moduleInOutputs.end(); ++it) {
223  if (it->second.empty()) continue;
224  for (auto itParam = it->second.begin(); itParam != it->second.end(); ++itParam) {
225  tfValueInputs.push_back(*itParam);
226  }
227  }
228  return true;
229 }

253 ParamInfo* ModuleFactory::FindDependentParam(ParamInfo* paramInfo, vector<string>& moduleIDs,
254  map<string, vector<ParamInfo *> >& moduleOutputs) {
255  string paraName = GetComparableName(paramInfo->Name);
256  dimensionTypes paraType = paramInfo->Dimension;
257  transferTypes tfType = paramInfo->Transfer;
258  for (auto it = moduleIDs.rbegin(); it != moduleIDs.rend(); ++it) {
259  // loop from the last module
260  for (auto itOut = moduleOutputs[*it].begin(); itOut != moduleOutputs[*it].end(); ++itOut) {
261  string compareName = GetComparableName((*itOut)->Name);
262  if (!StringMatch(paraName, compareName)) continue;
263  if ((*itOut)->Dimension == paraType && // normal
264  (tfType == (*itOut)->Transfer || tfType == TF_Whole) && // specified handling for mpi version
265  !StringMatch(*it, paramInfo->ModuleID)) {
266  // Avoid to dependent on the module itself
267  (*itOut)->OutputToOthers = true;
268  return *itOut;
269  }
270  }
271  }
272  if (!StringMatch(paramInfo->Source, Source_Module_Optional)) {
273  throw ModelException("ModuleFactory", "FindDependentParam",
274  "Can not find input for " + paraName + " of " + paramInfo->ModuleID +
275  " from other Modules.\n");
276  }
277  return nullptr;
278 }

280 void ModuleFactory::ReadDLL(const string& module_path, const string& id, const string& dllID,
281  vector<DLLINSTANCE>& dllHandles,
282  map<string, InstanceFunction>& instanceFuncs,
283  map<string, MetadataFunction>& metadataFuncs) {
284  // the dll file is already read, return
285  if (instanceFuncs.find(id) != instanceFuncs.end()) {
286  return;
287  }
288  // check if the dll file exists
289  string moduleFileName = module_path + SEP + dllID + LIBSUFFIX;
290  if (!FileExists(moduleFileName)) {
291  throw ModelException("ModuleFactory", "ReadDLL", moduleFileName + " does not exist or has no read permission!");
292  }
293  //load library
294 #ifdef WIN32
295  HINSTANCE handle = LoadLibrary(TEXT(moduleFileName.c_str()));
296  if (handle == nullptr) throw ModelException("ModuleFactory", "ReadDLL", "Could not load " + moduleFileName);
297  instanceFuncs[id] = InstanceFunction(GetProcAddress(HMODULE(handle), "GetInstance"));
298  metadataFuncs[id] = MetadataFunction(GetProcAddress(HMODULE(handle), "MetadataInformation"));
299 #else
300  void *handle = dlopen(moduleFileName.c_str(), RTLD_LAZY);
301  if (handle == nullptr) {
302  cout << dlerror() << endl;
303  throw ModelException("ModuleFactory", "ReadDLL", "Could not load " + moduleFileName);
304  }
305  instanceFuncs[id] = InstanceFunction(dlsym(handle, "GetInstance"));
306  metadataFuncs[id] = MetadataFunction(dlsym(handle, "MetadataInformation"));
307 #endif /* WIN32 */
308  dllHandles.push_back(handle);
309  if (instanceFuncs[id] == nullptr) {
310  throw ModelException("ModuleFactory", "ReadDLL",
311  moduleFileName + " does not implement API function: GetInstance");
312  }
313  if (metadataFuncs[id] == nullptr) {
314  throw ModelException("ModuleFactory", "ReadDLL",
315  moduleFileName + " does not implement API function: MetadataInformation");
316  }
317  StatusMessage(("Read DLL: " + moduleFileName).c_str());
318 }

347 void ModuleFactory::ReadParameterSetting(string& moduleID, TiXmlDocument& doc, SEIMSModuleSetting* setting,
348  map<string, vector<ParamInfo *> >& moduleParameters) {
349  moduleParameters.insert(make_pair(moduleID, vector<ParamInfo *>()));
350  vector<ParamInfo *>& vecPara = moduleParameters.at(moduleID);
351  TiXmlElement* eleMetadata = doc.FirstChildElement(TagMetadata.c_str());
352  // start getting the parameters
353  TiXmlElement* eleParams = eleMetadata->FirstChildElement(TagParameters.c_str());
354  if (eleParams != nullptr) {
355  TiXmlElement* eleParam = eleParams->FirstChildElement(TagParameter.c_str());
356  while (eleParam != nullptr) {
357  // clear the object
358  ParamInfo* param = new ParamInfo();
359  // set the module id
360  param->ModuleID = moduleID;
361  // get the parameter name
362  TiXmlElement* elItm = eleParam->FirstChildElement(TagVariableName.c_str());
363  if (elItm != nullptr) {
364  if (elItm->GetText() != nullptr) {
365  param->Name = GetUpper(string(elItm->GetText()));
366  param->BasicName = param->Name;
367  param->ClimateType = setting->dataTypeString();
368 
369  //set climate data type got from config.fig
370  //this is used for interpolation module
371  if (StringMatch(param->Name, Tag_DataType)) param->Value = setting->dataType();
372 
373  //special process for interpolation modules
374  if (StringMatch(param->Name, Tag_Weight)) {
375  if (setting->dataTypeString().length() == 0) {
376  throw ModelException("ModuleFactory", "ReadParameterSetting",
377  "The parameter " + string(Tag_Weight) +
378  " should have corresponding data type in module " + moduleID);
379  }
380  if (StringMatch(setting->dataTypeString(), DataType_MeanTemperature) ||
381  StringMatch(setting->dataTypeString(), DataType_MinimumTemperature) ||
382  StringMatch(setting->dataTypeString(), DataType_MaximumTemperature)) {
383  //The weight coefficient file is same for TMEAN, TMIN and TMAX,
384  // so just need to read one file named "Weight_M"
385  param->Name += "_M";
386 
387  } else {
388  // Combine weight and data type. e.g. Weight + PET = Weight_PET,
389  // this combined string must be the same with the parameter column
390  // in the climate table of parameter database.
391  param->Name += "_" + setting->dataTypeString();
392  }
393  }
394 
395  //special process for interpolation modules
396  if (StringMatch(param->Name, Tag_StationElevation)) {
397  if (setting->dataTypeString().length() == 0) {
398  throw ModelException("ModuleFactory", "readParameterSetting",
399  "The parameter " + string(Tag_StationElevation) +
400  " should have corresponding data type in module " + moduleID);
401  }
402  if (StringMatch(setting->dataTypeString(), DataType_Precipitation)) {
403  param->BasicName += "_P";
404  param->Name += "_P";
405  } else {
406  param->BasicName += "_M";
407  param->Name += "_M";
408  }
409  }
410 
411  if (StringMatch(param->Name, Tag_VerticalInterpolation)) //if do the vertical interpolation
412  {
413  param->Value = (setting->needDoVerticalInterpolation() ? 1.0f : 0.0f);
414  }
415  }
416  }
417  // get the parameter description
418  elItm = eleParam->FirstChildElement(TagVariableDescription.c_str());
419  if (elItm != nullptr) {
420  if (elItm->GetText() != nullptr) {
421  param->Description = elItm->GetText();
422  }
423  }
424  // get the parameter units
425  elItm = eleParam->FirstChildElement(TagVariableUnits.c_str());
426  if (elItm != nullptr) {
427  if (elItm->GetText() != nullptr) {
428  param->Units = elItm->GetText();
429  }
430  }
431  // get the parameter source
432  elItm = eleParam->FirstChildElement(TagVariableSource.c_str());
433  if (elItm != nullptr) {
434  if (elItm->GetText() != nullptr) {
435  param->Source = elItm->GetText();
436  }
437  }
438  // get the parameter dimension
439  elItm = eleParam->FirstChildElement(TagVariableDimension.c_str());
440  if (elItm != nullptr) {
441  if (elItm->GetText() != nullptr) {
442  param->Dimension = MatchType(string(elItm->GetText()));
443  }
444  }
445  // cleanup
446  elItm = nullptr;
447  // parameter must have these values
448  if (param->Name.empty()) {
449  throw ModelException("ModuleFactory", "ReadParameterSetting",
450  "Some parameters have not name in metadata!");
451  }
452  if (param->Source.empty()) {
453  string name = param->Name;
454  throw ModelException("ModuleFactory", "ReadParameterSetting",
455  "parameter " + name + " does not have source!");
456  }
457  if (param->Dimension == DT_Unknown) {
458  string name = param->Name;
459  throw ModelException("ModuleFactory", "ReadParameterSetting",
460  "parameter " + name + " does not have dimension!");
461  }
462  // add to the list
463  vecPara.push_back(param);
464  // get the next parameter if it exists
465  eleParam = eleParam->NextSiblingElement();
466  } // while
467  }
468 }

475 void ModuleFactory::ReadIOSetting(string& moduleID, TiXmlDocument& doc, SEIMSModuleSetting* setting,
476  const string& header,
477  const string& title, map<string, vector<ParamInfo *> >& variables) {
478  TiXmlElement* eleMetadata = doc.FirstChildElement(TagMetadata.c_str());
479  TiXmlElement* eleVariables = eleMetadata->FirstChildElement(header.c_str());
480  if (nullptr == eleVariables) return;
481  variables.insert(make_pair(moduleID, vector<ParamInfo *>()));
482  vector<ParamInfo *>& vecPara = variables.at(moduleID);
483  TiXmlElement* eleVar = eleVariables->FirstChildElement(title.c_str());
484  while (eleVar != nullptr) {
485  ParamInfo* param = new ParamInfo();
486  // set the module id
487  param->ModuleID = moduleID;
488  param->ClimateType = setting->dataTypeString();
489  // get the input variable name
490  TiXmlElement* elItm = eleVar->FirstChildElement(TagVariableName.c_str());
491  if (elItm != nullptr && elItm->GetText() != nullptr) {
492  param->Name = GetUpper(string(elItm->GetText()));
493  param->BasicName = param->Name;
494  param->IsConstant = IsConstantInputFromName(param->Name);
495  if (setting->dataTypeString().length() > 0) {
496  param->Name = param->Name + "_" + setting->dataTypeString();
497  }
498  }
499  // get the input variable units(
500  elItm = eleVar->FirstChildElement(TagVariableUnits.c_str());
501  if (elItm != nullptr && elItm->GetText() != nullptr) {
502  param->Units = elItm->GetText();
503  }
504  // get the input variable description
505  elItm = eleVar->FirstChildElement(TagVariableDescription.c_str());
506  if (elItm != nullptr && elItm->GetText() != nullptr) {
507  param->Description = elItm->GetText();
508  }
509  // get the input variable source
510  elItm = eleVar->FirstChildElement(TagVariableSource.c_str());
511  if (elItm != nullptr && elItm->GetText() != nullptr) {
512  param->Source = elItm->GetText();
513  }
514  // get the input variable dimension
515  elItm = eleVar->FirstChildElement(TagVariableDimension.c_str());
516  if (elItm != nullptr && elItm->GetText() != nullptr) {
517  param->Dimension = MatchType(string(elItm->GetText()));
518  }
519  // get the input variable transfer type
520  elItm = eleVar->FirstChildElement(TagVariableTransfer.c_str());
521  if (elItm != nullptr && elItm->GetText() != nullptr) {
522  param->Transfer = MatchTransferType(string(elItm->GetText()));
523  }
524  elItm = nullptr;
525  if (header.find("output", 0) != string::npos) {
526  param->IsOutput = true;
527  }
528  // input must have these values
529  if (param->Name.empty()) {
530  delete param;
531  throw ModelException("SEIMSModule", "ReadIOSetting", "Some variables have no name in metadata!");
532  }
533  if (param->Source.empty() && !param->IsOutput) {
534  string name = param->Name;
535  delete param;
536  throw ModelException("SEIMSModule", "ReadIOSetting", "Variable " + name + " does not have source!");
537  }
538  if (param->Dimension == DT_Unknown) {
539  string name = param->Name;
540  delete param;
541  throw ModelException("SEIMSModule", "ReadIOSetting", "Variable " + name + " does not have dimension!");
542  }
543  vecPara.push_back(param);
544  // get the next input if it exists
545  eleVar = eleVar->NextSiblingElement();
546  }
547 }

555  DataType_Precipitation, DataType_MeanTemperature, DataType_MaximumTemperature,

558  };

565  //TrimSpaces(tokens[i]);
566  tokens[i] = Trim(tokens[i]);
567  }
568  if (tokens.empty()) continue;
569  // is there anything in the first item? Or, the size of tokens greater than 4?
570  if (tokens[0].empty() || tokens.size() < 4) continue;
571  // there is something to add so resize the header list to append it
572  size_t sz = settings.size(); // get the current number of rows
573  if (tokens[3].find(MID_ITP) != string::npos ||
574  tokens[3].find(MID_TSD_RD) != string::npos) {
575  settings.resize(sz + 7);
576  for (size_t j = 0; j < 7; j++) {
577  vector<string> tokensTemp(tokens);
578  tokensTemp[1] += "_" + T_variables[j];
579  if (tokens[3].find(MID_ITP) != string::npos) {
580  vector<string> ITPProperty = SplitString(*iter, '_');
581  if (ITPProperty.size() == 2) {

583  errno = 0;

586  tokensTemp[1] += "_1";
587  } else {
588  tokensTemp[1] += "_0";
589  }
590  }
591  }
592  settings[sz + j] = tokensTemp;
593  }
594  } else {
595  settings.resize(sz + 1); // resize with one more row
596  settings[sz] = tokens;
597  }
598  }
599  return true;

647  noInputsFromOthers = false;

651  modules[iModule]->SetInputsDone(true);
652  return;
653  }
654 
655  for (size_t j = 0; j < inputs.size(); j++) {
656  ParamInfo* dependParam = inputs[j]->DependPara;
657  if (dependParam == nullptr) continue;
658 
659  int k = 0;

662  break;

667  GetValueFromDependencyModule(k, modules);

669  string compareName = GetComparableName(dependParam->Name);

674  modules[iModule]->Set1DData(inputs[j]->Name.c_str(), dataLen, data);
