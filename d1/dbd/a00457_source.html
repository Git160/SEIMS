337 bool DataCenterMongoDB::ReadParametersInDB() {
338  bson_t* filter = bson_new();
339  std::unique_ptr<MongoCollection>

342 
343  bson_error_t* err = NULL;

348  bson_destroy(filter);

351  }
352  while (mongoc_cursor_more(cursor) && mongoc_cursor_next(cursor, &info)) {
353  ParamInfo* p = new ParamInfo();
354  bson_iter_t iter;
355  if (bson_iter_init_find(&iter, info, PARAM_FLD_NAME)) {
356  p->Name = GetStringFromBsonIterator(&iter);
357  }
358  if (bson_iter_init_find(&iter, info, PARAM_FLD_UNIT)) {
359  p->Units = GetStringFromBsonIterator(&iter);
360  }
361  if (bson_iter_init_find(&iter, info, PARAM_FLD_VALUE)) {
362  GetNumericFromBsonIterator(&iter, p->Value);
363  }
364  if (bson_iter_init_find(&iter, info, PARAM_FLD_CHANGE)) {
365  p->Change = GetStringFromBsonIterator(&iter);
366  }
367  if (bson_iter_init_find(&iter, info, PARAM_FLD_IMPACT)) {
368  GetNumericFromBsonIterator(&iter, p->Impact);
369  }
370  if (bson_iter_init_find(&iter, info, PARAM_FLD_MAX)) {
371  GetNumericFromBsonIterator(&iter, p->Maximum);
372  }
373  if (bson_iter_init_find(&iter, info, PARAM_FLD_MIN)) {
374  GetNumericFromBsonIterator(&iter, p->Minimun);
375  }
376  if (bson_iter_init_find(&iter, info, PARAM_CALI_VALUES) && calibration_id_ >= 0) {
377  // Overwrite p->Impact according to calibration ID
378  string cali_values_str = GetStringFromBsonIterator(&iter);
379  vector<float> cali_values;
380  SplitStringForValues(cali_values_str, ',', cali_values);
381  if (calibration_id_ < CVT_INT(cali_values.size())) {
382  p->Impact = cali_values[calibration_id_];
383  }
384  }
385 #ifdef HAS_VARIADIC_TEMPLATES
386  if (!init_params_.emplace(GetUpper(p->Name), p).second) {
387 #else
388  if (!init_params_.insert(make_pair(GetUpper(p->Name), p)).second) {
389 #endif
390  cout << "ERROR: Load parameter: " << GetUpper(p->Name) << " failed!" << endl;
391  return false;
392  }
395  if (StringMatch(p->Name, VAR_SW_CAP)) {
396  for (int si = 0; si < SOILWATER_VARS_NUM; si++) {
397  ParamInfo* tmpp = new ParamInfo(*p);
398  tmpp->Name = SOILWATER_VARS[si];
399 #ifdef HAS_VARIADIC_TEMPLATES
400  init_params_.emplace(GetUpper(tmpp->Name), tmpp);
401 #else
402  init_params_.insert(make_pair(GetUpper(tmpp->Name), tmpp));
403 #endif
404  }
405  }
406  }

409  return true;
