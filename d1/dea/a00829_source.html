81  """Get cell number of each subbasin.
82  Args:
83  subbsn_file: subbasin raster file.
84 
85  Returns:
86  subbasin cell count dict and cell width
87  """
88  num_dic = dict()
89  wtsd_raster = RasterUtilClass.read_raster(subbsn_file)
90  data = wtsd_raster.data
91  xsize = wtsd_raster.nCols
92  ysize = wtsd_raster.nRows
93  dx = wtsd_raster.dx
94  nodata_value = wtsd_raster.noDataValue
95  for i in range(ysize):
96  for j in range(xsize):
97  k = int(data[i][j])
98  if abs(k - nodata_value) > UTIL_ZERO:
99  num_dic[k] = num_dic.setdefault(k, 0) + 1
100  return num_dic, dx

116  def down_stream(reach_shp, is_taudem=True):
117  """Construct stream order layers etc.
118  Args:
119  reach_shp: reach ESRI shapefile.
120  is_taudem: is TauDEM or not, true is default.
121 
122  Returns:
123  down_stream_dic: the key is stream id, and value is its downstream id
124  downstream_up_order_dic: from outlet up stream dict
125  upstream_down_order_dic: from source down stream dict
126  depth_dic: stream depth dict
127  slope_dic: stream slope dict
128  width_dic: stream width dict
129  len_dic: stream length dict
130  """
131  down_stream_dic = dict()
132  depth_dic = dict()
133  slope_dic = dict()
134  width_dic = dict()
135  len_dic = dict()
136  ds_reach = ogr_Open(reach_shp)
137  layer_reach = ds_reach.GetLayer(0)
138  layer_def = layer_reach.GetLayerDefn()
139  if not is_taudem: # For ArcSWAT
140  ImportReaches2Mongo._LINKNO = 'FROM_NODE'
141  ImportReaches2Mongo._DSLINKNO = 'TO_NODE'
142  ImportReaches2Mongo._SLOPE = 'Slo2' # TauDEM: Slope (tan); ArcSWAT: Slo2 (100*tan)
143  ImportReaches2Mongo._LENGTH = 'Len2' # TauDEM: Length; ArcSWAT: Len2
144  i_from = layer_def.GetFieldIndex(ImportReaches2Mongo._LINKNO)
145  i_to = layer_def.GetFieldIndex(ImportReaches2Mongo._DSLINKNO)
146  i_depth = layer_def.GetFieldIndex(ImportReaches2Mongo._DEPTH)
147  i_slope = layer_def.GetFieldIndex(ImportReaches2Mongo._SLOPE)
148  i_width = layer_def.GetFieldIndex(ImportReaches2Mongo._WIDTH)
149  i_len = layer_def.GetFieldIndex(ImportReaches2Mongo._LENGTH)
150 
151  g = nx.DiGraph()
152  ft = layer_reach.GetNextFeature()
153  while ft is not None:
154  node_from = ft.GetFieldAsInteger(i_from)
155  node_to = ft.GetFieldAsInteger(i_to)
156  if i_depth > -1:
157  depth_dic[node_from] = ft.GetFieldAsDouble(i_depth)
158  else:
159  depth_dic[node_from] = 1
160 
161  if i_depth > -1:
162  slope_dic[node_from] = ft.GetFieldAsDouble(i_slope)
163  if slope_dic[node_from] < MINI_SLOPE:
164  slope_dic[node_from] = MINI_SLOPE
165  else:
166  slope_dic[node_from] = MINI_SLOPE
167 
168  if i_width > -1:
169  width_dic[node_from] = ft.GetFieldAsDouble(i_width)
170  else:
171  width_dic[node_from] = 10
172 
173  len_dic[node_from] = ft.GetFieldAsDouble(i_len)
174  down_stream_dic[node_from] = node_to
175  if node_to > 0:
176  # print node_from, node_to
177  g.add_edge(node_from, node_to)
178  ft = layer_reach.GetNextFeature()
179 
180  # find outlet subbasin
181  outlet = -1
182  for node in g.nodes():
183  if g.out_degree(node) == 0:
184  outlet = node
185  if outlet < 0:
186  raise ValueError("Can't find outlet subbasin ID, please check the "
187  "threshold for stream extraction!")
188  print ('outlet subbasin:%d' % outlet)
189 
190  # assign order from outlet to upstream subbasins
191  downstream_up_order_dic = dict()
192  ImportReaches2Mongo.stream_orders_from_outlet_up(downstream_up_order_dic, g, outlet, 1)
193  # find the maximum order number
194  max_order = 0
195  for k, v in downstream_up_order_dic.items():
196  if v > max_order:
197  max_order = v
198  # reserve the order number
199  for k, v in downstream_up_order_dic.items():
200  downstream_up_order_dic[k] = max_order - v + 1
201 
202  # assign order from the source subbasins
203  upstream_down_order_dic = dict()
204  order_num = 1
205  nodelist = g.nodes()
206  while len(nodelist) != 0:
207  nodelist = g.nodes()
208  del_list = list()
209  for node in nodelist:
210  if g.in_degree(node) == 0:
211  upstream_down_order_dic[node] = order_num
212  del_list.append(node)
213  for item in del_list:
214  g.remove_node(item)
215  order_num += 1
216 
217  return (down_stream_dic, downstream_up_order_dic, upstream_down_order_dic,
218  depth_dic, slope_dic, width_dic, len_dic)
