146 void SEDR_SBAGNOLD::PointSourceLoading() {
148  for (map<int, BMPPointSrcFactory *>::iterator it = m_ptSrcFactory.begin(); it != m_ptSrcFactory.end(); it++) {
149  //cout<<"unique Point Source Factory ID: "<<it->first<<endl;
150  vector<int> m_ptSrcMgtSeqs = it->second->GetPointSrcMgtSeqs();
151  map < int, PointSourceMgtParams * > m_pointSrcMgtMap = it->second->GetPointSrcMgtMap();
152  vector<int> m_ptSrcIDs = it->second->GetPointSrcIDs();
153  map < int, PointSourceLocations * > m_pointSrcLocsMap = it->second->GetPointSrcLocsMap();
154  // 1. looking for management operations from m_pointSrcMgtMap
155  for (vector<int>::iterator seqIter = m_ptSrcMgtSeqs.begin(); seqIter != m_ptSrcMgtSeqs.end(); seqIter++) {
156  PointSourceMgtParams *curPtMgt = m_pointSrcMgtMap.at(*seqIter);
157  // 1.1 If current day is beyond the date range, then continue to next management
158  if (curPtMgt->GetStartDate() != 0 && curPtMgt->GetEndDate() != 0) {
159  if (m_date < curPtMgt->GetStartDate() || m_date > curPtMgt->GetEndDate()) {
160  continue;
161  }
162  }
163  // 1.2 Otherwise, get the water volume
164  float per_sed = curPtMgt->GetSedment();
165  // 1.3 Sum up all point sources
166  for (vector<int>::iterator locIter = m_ptSrcIDs.begin(); locIter != m_ptSrcIDs.end(); locIter++) {
167  if (m_pointSrcLocsMap.find(*locIter) != m_pointSrcLocsMap.end()) {
168  PointSourceLocations *curPtLoc = m_pointSrcLocsMap.at(*locIter);
169  int curSubID = curPtLoc->GetSubbasinID();
171  m_ptSub[curSubID] += per_sed * curPtLoc->GetSize() * 1000.f * m_dt / 86400.f;
172  }
173  }
174  }

373 void SEDR_SBAGNOLD::SedChannelRouting(int i) {
374  float qOutV = 0.f; // water volume (m^3) of flow out
375  float allWater = 0.f; // // water in reach during time step, qdin in SWAT
376  // initialize sediment in reach during time step
378  float allSediment = 0.f; // all sediment in reach, kg
379  float sedUp = 0.f; // sediment from upstream channels, kg
380  for (size_t j = 0; j < m_reachUpStream[i].size(); ++j) {
381  int upReachId = m_reachUpStream[i][j];
382  sedUp += m_sedOut[upReachId];
383  }
384  allSediment = sedUp + m_sedtoCh[i] + m_sedStorage[i];
386  if (m_ptSub != NULL && m_ptSub[i] > 0.f) {
387  allSediment += m_ptSub[i];
388  }
389  //if (i == 12) cout<<"\tall sed: "<<allSediment<<", sed up: "<<sedUp<<", sed to ch: "<<m_sedtoCh[i]<<", strg: "<<m_sedStorage[i]<<endl;
390 
391  // initialize water in reach during time step
392  qOutV = m_qchOut[i] * m_dt; // m^3
393  //allWater = m_chStorage[i] + qOutV;
394  allWater = m_preChStorage[i];
395  if ((m_qchOut[i] < UTIL_ZERO && m_chWTdepth[i] < UTIL_ZERO) || allWater < 0.01f) {
399  m_sedDeg[i] = 0.f;
400  m_sedDep[i] = 0.f;
401  m_sedOut[i] = 0.f;
402  m_sedConc[i] = 0.f;
403  m_rchSand[i] = 0.f;
404  m_rchSilt[i] = 0.f;
405  m_rchClay[i] = 0.f;
406  m_rchSag[i] = 0.f;
407  m_rchLag[i] = 0.f;
408  m_rchGra[i] = 0.f;
409  m_rchBankEro[i] = 0.f;
410  m_rchDeg[i] = 0.f;
411  m_rchDep[i] = 0.f;
412  m_flplainDep[i] = 0.f;
413  m_sedStorage[i] = allSediment;
414  return;
415  }
416  // initialize reach peak runoff rate and calculate flow velocity
417  float peakFlowRate = m_qchOut[i] * m_prf;
418  float crossarea = allWater / m_chLen[i]; // SWAT, eq. 7:1.1.7
419  float peakVelocity = 0.f;
420  if (m_preChWTDepth[i] < 0.01f) {
421  peakVelocity = 0.01f;
422  } else {
423  peakVelocity = peakFlowRate / crossarea;
424  }
425  if (peakVelocity > 5.f) peakVelocity = 5.f;
426 
428  float tbase = m_chLen[i] / (m_dt * peakVelocity);
429  if (tbase > 1.f) tbase = 1.f;
430 
432  float initCon = 0.f; // cyin
433  float maxCon = 0.f; // cych
434  float sedDeposition = 0.f; // depnet, and dep
435  float sedDegradation = 0.f; // deg
436  float sedDegradation1 = 0.f; // deg1
437  float sedDegradation2 = 0.f; // deg2
438 
439  //deposition and degradation
440  initCon = allSediment / allWater; // kg/m^3
441  //max concentration
442  maxCon = m_spcon * pow(peakVelocity, m_spexp) * 1000.f; // kg/m^3
443  //if (i == 12) cout<<"iniCon: "<<initCon<<", maxCon: "<<maxCon<<endl;
444  //initial concentration,mix sufficiently
445  sedDeposition = allWater * (initCon - maxCon); // kg
446  if (peakVelocity < m_vcrit) {
447  sedDeposition = 0.f;
448  }
449 
450  if (sedDeposition < 0.f) //degradation
451  {
452  sedDegradation = -sedDeposition * tbase;
453  // first the deposited material will be degraded before channel bed
454  if (sedDegradation >= m_sedDep[i]) {
455  sedDegradation1 = m_sedDep[i];
456  sedDegradation2 = (sedDegradation - sedDegradation1) * m_chErod[i] * m_chCover[i];
457  } else {
458  sedDegradation1 = sedDegradation;
459  sedDegradation2 = 0.f;
460  }
461  sedDeposition = 0.f;
462  } else {
463  sedDegradation = 0.f;
464  sedDegradation1 = 0.f;
465  sedDegradation2 = 0.f;
466  }
467  //update sed deposition
468  m_sedDep[i] += sedDeposition - sedDegradation1;
469  if (m_sedDep[i] < UTIL_ZERO) m_sedDep[i] = 0.f;
470  m_sedDeg[i] += sedDegradation1 + sedDegradation2;
471 
472  //get sediment after deposition and degradation
473  allSediment += sedDegradation1 + sedDegradation2 - sedDeposition;
474  if (allSediment < UTIL_ZERO) allSediment = 0.f;
475  //get out flow water fraction
476  float outFraction = qOutV / allWater;
477  if (outFraction > 1.f) outFraction = 1.f;
478  m_sedOut[i] = allSediment * outFraction;
479  if (m_sedOut[i] < UTIL_ZERO) m_sedOut[i] = 0.f;
480  // update sediment storage
481  m_sedStorage[i] = allSediment - m_sedOut[i];
482  if (m_sedStorage[i] < UTIL_ZERO) m_sedStorage[i] = 0.f;
483
