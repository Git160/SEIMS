76  def read_simulation_timerange(self):
77  """Read simulation time range from MongoDB."""
78  client = ConnectMongoDB(self.hostname, self.port)
79  conn = client.get_conn()
80  db = conn[self.main_db]
81  collection = db['FILE_IN']
82  try:
83  stime_str = collection.find_one({'TAG': 'STARTTIME'}, no_cursor_timeout=True)['VALUE']
84  etime_str = collection.find_one({'TAG': 'ENDTIME'}, no_cursor_timeout=True)['VALUE']
85  stime = StringClass.get_datetime(stime_str)
86  etime = StringClass.get_datetime(etime_str)
87  dlt = etime - stime + timedelta(seconds=1)
88  self.timerange = (dlt.days * 86400. + dlt.seconds) / 86400. / 365.
89  except NetworkTimeout or Exception:
90  # In case of unexpected raise
91  self.timerange = 1. # set default
92  pass
93  client.close()

116  def export_to_mongodb(self):
117  """Export current scenario to MongoDB.
118  Delete the same ScenarioID if existed.
119  """
120  client = ConnectMongoDB(self.hostname, self.port)
121  conn = client.get_conn()
122  db = conn[self.scenario_db]
123  collection = db['BMP_SCENARIOS']
124  try:
125  # find ScenarioID, remove if existed.
126  if collection.find({'ID': self.ID}, no_cursor_timeout=True).count():
127  collection.remove({'ID': self.ID})
128  except NetworkTimeout or Exception:
129  # In case of unexpected raise
130  pass
131  for objid, bmp_item in self.bmp_items.iteritems():
132  bmp_item['_id'] = ObjectId()
133  collection.insert_one(bmp_item)
134  client.close()

136  def export_to_txt(self):

147  outfile.write('Scenario items:\n')
148  if len(self.bmp_items) > 0:
149  for obj, item in self.bmp_items.iteritems():
150  header = item.keys()

152  outfile.write('\t'.join(header))
153  outfile.write('\n')

155  outfile.write('\t'.join(str(v) for v in item.values()))
156  outfile.write('\n')
