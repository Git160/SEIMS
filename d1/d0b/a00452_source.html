48 void Subbasin::SetSlope(float* slope) {
49  float slope_sum = 0.f;
50 #pragma omp parallel for reduction(+: slope_sum)
51  for (int i = 0; i < n_cells_; i++) {
52  slope_sum += atan(slope[cells_[i]]); // radian
53  }
54  slope_sum /= n_cells_;

141  cout << "Subbasin data loaded failed!" << endl;
142  return nullptr;

144 #ifdef HAS_VARIADIC_TEMPLATES
145  rs_map.emplace(*it, tmp_raster);

167 float clsSubbasins::Subbasin2Basin(const string& key) {
168  float temp = 0.f;
169  int total_count = 0;
170  Subbasin* sub = nullptr;
171  for (auto it = subbasin_ids_.begin(); it != subbasin_ids_.end(); ++it) {
172  sub = subbasin_objs_[*it];
173  int n_count = sub->GetCellCount();
174  if (StringMatch(key, VAR_SLOPE)) {
175  temp += atan(sub->GetSlope()) * n_count;
176  } else if (StringMatch(key, VAR_PET)) {
177  temp += sub->GetPet() * n_count;
178  } else if (StringMatch(key, VAR_PERCO)) {
179  temp += sub->GetPerco() * n_count;
180  } else if (StringMatch(key, VAR_REVAP)) {
181  temp += sub->GetEg() * n_count;
182  } else if (StringMatch(key, VAR_PERDE)) {
183  temp += sub->GetPerde() * n_count;
184  } else if (StringMatch(key, VAR_RG)) {
185  temp += sub->GetRg() * n_count;
186  } else if (StringMatch(key, VAR_QG)) {
187  temp += sub->GetQg();
188  } else if (StringMatch(key, VAR_GW_Q)) {
189  temp += sub->GetGw() * n_count;
190  }
191 
192  total_count += n_count;
193  }
194  if (StringMatch(key, VAR_QG)) {
195  return temp;
196  } // basin sum
197  if (StringMatch(key, VAR_SLOPE))
198  return tan(temp / total_count);
199  return temp / total_count; // basin average
200 }

202 void clsSubbasins::SetSlopeCoefficient(float* rs_slope) {
203  for (auto it = subbasin_objs_.begin(); it != subbasin_objs_.end(); ++it) {
204  if (it->second->GetSlope() <= 0.f) {
205  it->second->SetSlope(rs_slope);

209  for (auto it = subbasin_objs_.begin(); it != subbasin_objs_.end(); ++it) {
210  if (basin_slope <= 0.f) {
211  cout << "WARNING: Mean slope of the whole basin is absent, the Groundwater modules"
212  " may be problematic, please contact the developers!" << endl;
213  it->second->SetSlopeCoefofBasin(1.f);
214  } else {
215  float slp_coef = it->second->GetSlope() / basin_slope;
216  it->second->SetSlopeCoefofBasin(slp_coef);
217  }
218  }
