94 int NPS_Management::Execute() {

96  if (m_cellArea < 0.f) m_cellArea = m_cellWidth * m_cellWidth;
97  for (map<int, BMPArealSrcFactory *>::iterator it = m_arealSrcFactory.begin(); it != m_arealSrcFactory.end(); it++) {
99  if (!it->second->GetLocationLoadStatus()) {
100  it->second->SetArealSrcLocsMap(m_nCells, m_mgtFields);
101  }
103  vector<int> tmpArealFieldIDs = it->second->GetArealSrcIDs();
104  vector<int> tmpMgtSeqs = it->second->GetArealSrcMgtSeqs();
105  for (vector<int>::iterator mgtSeqIter = tmpMgtSeqs.begin(); mgtSeqIter != tmpMgtSeqs.end(); mgtSeqIter++) {
106  ArealSourceMgtParams *tmpMgtParams = it->second->GetArealSrcMgtMap().at(*mgtSeqIter);
107  if (tmpMgtParams->GetStartDate() != 0 && tmpMgtParams->GetEndDate() != 0) {
108  if (m_date < tmpMgtParams->GetStartDate() || m_date > tmpMgtParams->GetEndDate()) {
109  continue;
110  }
111  }
112  for (vector<int>::iterator fIDIter = tmpArealFieldIDs.begin(); fIDIter != tmpArealFieldIDs.end();
113  fIDIter++) {
114  float deltaWtrMM = 0.f, deltaNH4 = 0.f, deltaNO3 = 0.f, deltaOrgN = 0.f;
115  float deltaMinP = 0.f, deltaOrgP = 0.f;
116  map < int, ArealSourceLocations * > tmpLocMap = it->second->GetArealSrcLocsMap();
117  if (tmpLocMap.find(*fIDIter) == tmpLocMap.end()) {
118  continue;
119  }
120  ArealSourceLocations *tmpLoc = tmpLocMap.at(*fIDIter);
121  float tmpSize = tmpLoc->GetSize();
122  int tmpNCells = tmpLoc->GetValidCells();
123  if (tmpMgtParams->GetWaterVolume() > 0.f) {
124  deltaWtrMM =

126  1000.f;

129  tmpSize * m_timestep / 86400.f / tmpNCells / m_cellArea * 10000.f;
130  if (tmpMgtParams->GetNH4() > 0.f) {
131  deltaNH4 = cvt * tmpMgtParams->GetNH4();
132  }
133  if (tmpMgtParams->GetNO3() > 0.f) {
134  deltaNO3 = cvt * tmpMgtParams->GetNO3();
135  }
136  if (tmpMgtParams->GetOrgN() > 0.f) {
137  deltaOrgN = cvt * tmpMgtParams->GetOrgN();
138  }
139  if (tmpMgtParams->GetMinP() > 0.f) {
140  deltaMinP = cvt * tmpMgtParams->GetMinP();
141  }
142  if (tmpMgtParams->GetOrgP() > 0.f) {
143  deltaOrgP = cvt * tmpMgtParams->GetOrgP();
144  }
145  vector<int> tmpCellIdx = tmpLoc->GetCellsIndex();
146  for (vector<int>::iterator idxIter = tmpCellIdx.begin(); idxIter != tmpCellIdx.end(); idxIter++) {
148  if (deltaWtrMM > 0.f && m_soilStorage != NULL) {
149  m_soilStorage[*idxIter][0] += deltaWtrMM;
150  }
151  if (deltaNO3 > 0.f && m_sol_no3 != NULL) {
152  m_sol_no3[*idxIter][0] += deltaNO3;
153  }
154  if (deltaNH4 > 0.f && m_sol_nh4 != NULL) {
