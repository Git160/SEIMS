239 int NutrOL_IUH::Execute() {
240  if (!this->CheckInputData()) {
241  return false;
242  }
243  this->initialOutputs();
244 
245 #pragma omp parallel for
246  for (int n = 0; n < m_nsub + 1; n++) {
247  // delete value of last time step
248  m_surqno3ToCh[n] = 0.f;
249  m_latno3ToCh[n] = 0.f;
250  m_no3gwToCh[n] = 0.f;
251  m_surqsolpToCh[n] = 0.f;
252  m_minpgwToCh[n] = 0.f;
253  m_sedorgnToCh[n] = 0.f;
254  m_sedorgpToCh[n] = 0.f;
255  m_sedminpaToCh[n] = 0.f;
256  m_sedminpsToCh[n] = 0.f;
257  m_ammoToCh[n] = 0.f;
258  m_nitriteToCh[n] = 0.f;
259  m_codToCh[n] = 0.f;
260  }
261 
262  float area = m_cellWidth * m_cellWidth;
263 
264 #pragma omp parallel for
265  for (int i = 0; i < m_nCells; i++) {
266  // get the ratio that flow and sediment into channel
267  float sedFract, flowFract;
268  sedFract = CalculateSedinFlowFraction(i);
269  flowFract = CalculateFlowFraction(i);
270 
271  // calculate new nutrient concentration
272  m_cellsurqno3 = GetnewNutrient(i, m_surqno3, m_cellsurqno3, flowFract);
273  m_celllatno3 = GetnewNutrient(i, m_latno3, m_celllatno3, 1.f);
274  m_cellno3gw = GetnewNutrient(i, m_no3gw, m_cellno3gw, 1.f);
275  m_cellsurqsolp = GetnewNutrient(i, m_surqsolp, m_cellsurqsolp, flowFract);
276  m_cellminpgw = GetnewNutrient(i, m_minpgw, m_cellminpgw, 1.f);
277  m_cellsedorgn = GetnewNutrient(i, m_sedorgn, m_cellsedorgn, sedFract);
278  m_cellsedorgp = GetnewNutrient(i, m_sedorgp, m_cellsedorgp, sedFract);
279  m_cellsedminpa = GetnewNutrient(i, m_sedminpa, m_cellsedminpa, sedFract);
280  m_cellsedminps = GetnewNutrient(i, m_sedminps, m_cellsedminps, sedFract);
281  //m_cellammo = GetnewNutrient(i, m_ammo, m_cellammo, 1.f);
282  //m_cellnitrite = GetnewNutrient(i, m_nitrite, m_cellnitrite, 1.f);
283  m_cellcod = GetnewNutrient(i, m_cod, m_cellcod, flowFract);
284 
285  }
286 
287  for (int i = 0; i < m_nCells; i++) {
288  //add today's nutrient
289  int subi = (int) m_subbasin[i];
290  if (m_nsub == 1) {
291  subi = 1;
292  } else if (subi >= m_nsub + 1) {
293  ostringstream oss;
294  oss << subi;
295  throw ModelException("NutrOL_IUH", "Execute", "The subbasin " + oss.str() + " is invalid.");
296  }
297  m_surqno3ToCh[subi] += m_cellsurqno3[i][0];
298  m_latno3ToCh[subi] += m_celllatno3[i][0];
299  m_no3gwToCh[subi] += m_cellno3gw[i][0];
300  m_surqsolpToCh[subi] += m_cellsurqsolp[i][0];
301  m_minpgwToCh[subi] += m_cellminpgw[i][0];
302  m_sedorgnToCh[subi] += m_cellsedorgn[i][0];
303  m_sedorgpToCh[subi] += m_cellsedorgp[i][0];
304  m_sedminpaToCh[subi] += m_cellsedminpa[i][0];
305  m_sedminpsToCh[subi] += m_cellsedminps[i][0];
306  m_ammoToCh[subi] += 0.f; // No calculation before
307  m_nitriteToCh[subi] += 0.f; //No calculation before
308  m_codToCh[subi] += m_cellcod[i][0];
309  }
310  // sum the nutrient in entire watershed.
311  m_surqno3ToCh[0] = GettotalNutr(m_surqno3ToCh);
312  m_latno3ToCh[0] = GettotalNutr(m_surqno3ToCh);
313  m_no3gwToCh[0] = GettotalNutr(m_no3gwToCh);
314  m_surqsolpToCh[0] = GettotalNutr(m_surqsolpToCh);
315  m_minpgwToCh[0] = GettotalNutr(m_minpgwToCh);
316  m_sedorgnToCh[0] = GettotalNutr(m_sedorgnToCh);
317  m_sedorgpToCh[0] = GettotalNutr(m_sedorgpToCh);
318  m_sedminpaToCh[0] = GettotalNutr(m_sedminpaToCh);
319  m_sedminpsToCh[0] = GettotalNutr(m_sedminpsToCh);
320  m_ammoToCh[0] = 0.f; // No calculation before
321  m_nitriteToCh[0] = 0.f; //No calculation before
322  m_codToCh[0] = GettotalNutr(m_codToCh);
323 
324  return 0;
325 }

351 float **NutrOL_IUH::GetnewNutrient(int id, float *nutr, float **cellNutr, float ratio) {
352  float cellArea = m_cellWidth * m_cellWidth;
353  //forward one time step
354  for (int j = 0; j < m_cellNutrCols; j++) {
355  if (j != m_cellNutrCols - 1) {
356  cellNutr[id][j] = cellNutr[id][j + 1];
357  } else {
358  cellNutr[id][j] = 0.f;
359  }
360  }
361  if (nutr[id] > 0.f) {
362  int min = int(this->m_iuhCell[id][0]);
363  int max = int(this->m_iuhCell[id][1]);
364  int col = 2;
365  for (int k = min; k <= max; k++) {
366  cellNutr[id][k] += nutr[id] * m_iuhCell[id][col] * cellArea / 10000.f * ratio; // unit: kg
367  col++;
368  }
369  }
370  return cellNutr;
371 }
