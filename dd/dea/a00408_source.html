299 bool GridLayering::GridLayeringFromSource() {
300  int n = n_rows_ * n_cols_;
301  Initialize1DArray(n, layers_updown_, out_nodata_);
302 
303  int* last_layer = nullptr;
304  Initialize1DArray(n, last_layer, out_nodata_);
305  int num_last_layer = 0; // the number of cells of last layer
306 
307  int index = 0;
308  vector<vector<int> > layerCells_updown;
309  // 1. find source grids, i.e., the first layer
310  for (int i = 0; i < n; i++) {
311  if (flow_in_num_[i] == 0) { last_layer[num_last_layer++] = i; }
312  }
313  // 2. loop and layer
314  int num_next_layer = 0;
315  int cur_num = 0; //the layering number of the current layer. In the end, it is the total number of layers.
316  int* next_layer = nullptr;
317  Initialize1DArray(n, next_layer, out_nodata_);
318  int* tmp = nullptr;
319  vector<int> lyr_cells;
320  while (num_last_layer > 0) {
321  cur_num++;
322  num_next_layer = 0;
323  for (int i_in_layer = 0; i_in_layer < num_last_layer; i_in_layer++) {
324  index = last_layer[i_in_layer];
325  lyr_cells.emplace_back(pos_index_[index]);
326  layers_updown_[index] = cur_num;
327 
328  int i = index / n_cols_;
329  int j = index % n_cols_;
330 
331  int dir = flowdir_matrix_[index];
332  if (dir & 1 && j != n_cols_ - 1) {
333  if (--flow_in_num_[index + 1] == 0) {
334  next_layer[num_next_layer++] = index + 1;
335  }
336  }
337  if (dir & 2 && i != n_rows_ - 1 && j != n_cols_ - 1) {
338  if (--flow_in_num_[index + n_cols_ + 1] == 0) {
339  next_layer[num_next_layer++] = index + n_cols_ + 1;
340  }
341  }
342  if (dir & 4 && i != n_rows_ - 1) {
343  if (--flow_in_num_[index + n_cols_] == 0) {
344  next_layer[num_next_layer++] = index + n_cols_;
345  }
346  }
347  if (dir & 8 && i != n_rows_ - 1 && j != 0) {
348  if (--flow_in_num_[index + n_cols_ - 1] == 0) {
349  next_layer[num_next_layer++] = index + n_cols_ - 1;
350  }
351  }
352  if (dir & 16 && j != 0) {
353  if (--flow_in_num_[index - 1] == 0) {
354  next_layer[num_next_layer++] = index - 1;
355  }
356  }
357  if (dir & 32 && i != 0 && j != 0) {
358  if (--flow_in_num_[index - n_cols_ - 1] == 0) {
359  next_layer[num_next_layer++] = index - n_cols_ - 1;
360  }
361  }
362  if (dir & 64 && i != 0) {
363  if (--flow_in_num_[index - n_cols_] == 0) {
364  next_layer[num_next_layer++] = index - n_cols_;
365  }
366  }
367  if (dir & 128 && i != 0 && j != n_cols_ - 1) {
368  if (--flow_in_num_[index - n_cols_ + 1] == 0) {
369  next_layer[num_next_layer++] = index - n_cols_ + 1;
370  }
371  }
372  }
373  layerCells_updown.emplace_back(vector<int>(lyr_cells));
374  lyr_cells.clear();
375 
376  num_last_layer = num_next_layer;
377  tmp = last_layer;
378  last_layer = next_layer;
379  next_layer = tmp;
380  }
381  Release1DArray(last_layer);
382  Release1DArray(next_layer);
383 
384  int layer_num = CVT_INT(layerCells_updown.size());
385  int length = n_valid_cells_ + layer_num + 1;
386  Initialize1DArray(length, layer_cells_updown_, 0.f);
387  layer_cells_updown_[0] = CVT_FLT(layer_num);
388  index = 1;
389  for (auto it = layerCells_updown.begin(); it != layerCells_updown.end(); ++it) {
390  layer_cells_updown_[index++] = CVT_FLT((*it).size());
391  for (auto it2 = it->begin(); it2 != it->end(); ++it2) {
392  layer_cells_updown_[index++] = CVT_FLT(*it2);
393  }
394  }
395  return OutputGridLayering(layering_updown_name_, length,
396  layers_updown_, layer_cells_updown_);
397 }

406 
407  int cur_num = 0;

442  }
443  }
444  index2 = (i + 1) * n_cols_ + j;

446  if (--flow_out_num_[index2] == 0) {
447  next_layer[num_next_layer++] = index2;
448  }
449  }
450  index2 = (i + 1) * n_cols_ + j - 1;
451  if (i != n_rows_ - 1 && j != 0 && flowdir_matrix_[index2] & 128) {
452  if (--flow_out_num_[index2] == 0) {
453  next_layer[num_next_layer++] = index2;
454  }
455  }
456  index2 = i * n_cols_ + j - 1;

459  next_layer[num_next_layer++] = index2;
460  }
461  }
462  index2 = (i - 1) * n_cols_ + j - 1;
463  if (i != 0 && j != 0 && flowdir_matrix_[index2] & 2) {
464  if (--flow_out_num_[index2] == 0) {
465  next_layer[num_next_layer++] = index2;
466  }
467  }

470  if (--flow_out_num_[index2] == 0) {
471  next_layer[num_next_layer++] = index2;
472  }
473  }
474  index2 = (i - 1) * n_cols_ + j + 1;
475  if (i != 0 && j != n_cols_ - 1 && flowdir_matrix_[index2] & 8) {

493  layers_downup_[i] = cur_num - layers_downup_[i] + 1;
494  }
495  }
496  Release1DArray(last_layer);
497  Release1DArray(next_layer);
498 
499  int layer_num = CVT_INT(layer_cells_downup.size());

501  Initialize1DArray(length, layer_cells_downup_, 0.f);
502  layer_cells_downup_[0] = CVT_FLT(layer_num);
503  int index = 1;
504  for (auto it = layer_cells_downup.end() - 1; it != layer_cells_downup.begin(); --it) {
505  layer_cells_downup_[index++] = CVT_FLT((*it).size());
506  for (auto it2 = it->begin(); it2 != it->end(); ++it2) {
507  layer_cells_downup_[index++] = CVT_FLT(*it2);
508  }
509  }
