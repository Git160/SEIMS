33  def addItem(self, itemDict):

38  raise ValueError("The hydroClimate dict must have mean temperature!")
39  curY = itemDict[Tag_DT_Year.upper()]
40  curTmp = itemDict[DataType_MeanTemperature.upper()]
41  if curY not in self.Count.keys():
42  self.Count[curY] = 1
43  self.MeanTmp[curY] = curTmp
44  if curTmp > T_base:
45  self.PHUTOT[curY] = curTmp
46  else:
47  self.PHUTOT[curY] = 0.
48  else:
49  self.Count[curY] += 1
50  self.MeanTmp[curY] += curTmp
51  if curTmp > T_base:

54  def annualStats(self):
55  """Calculate annual statistics."""
56  for Y in self.Count.keys():
57  self.MeanTmp[Y] = round(self.MeanTmp[Y] / self.Count[Y], 1)
58  self.MeanTmp0 += self.MeanTmp[Y]
59  self.PHUTOT[Y] = round(self.PHUTOT[Y], 1)
60  self.PHU0 += self.PHUTOT[Y]
61  self.PHU0 = round(self.PHU0 / len(self.Count.keys()), 1)
62  self.MeanTmp0 = round(self.MeanTmp0 / len(self.Count.keys()), 1)
