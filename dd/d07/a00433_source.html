64 void BMPPointSrcFactory::ReadPointSourceManagements(MongoClient* conn, const string& bmpDBName) {
65  bson_t* b = bson_new();
66  bson_t *child1 = bson_new(), *child2 = bson_new();
67  BSON_APPEND_DOCUMENT_BEGIN(b, "$query", child1);
68  BSON_APPEND_INT32(child1, BMP_FLD_SUB, m_subScenarioId);
69  bson_append_document_end(b, child1);
70  BSON_APPEND_DOCUMENT_BEGIN(b, "$orderby", child2);
71  BSON_APPEND_INT32(child2, BMP_FLD_SEQUENCE, 1);
72  bson_append_document_end(b, child2);
73  bson_destroy(child1);
74  bson_destroy(child2);
75 
76  std::unique_ptr<MongoCollection> collection(new MongoCollection(conn->GetCollection(bmpDBName, m_pointSrcMgtTab)));
77  mongoc_cursor_t* cursor = collection->ExecuteQuery(b);
78 
79  bson_iter_t iter;
80  const bson_t* bsonTable;
81 
83  int count = 1;
84  while (mongoc_cursor_next(cursor, &bsonTable)) {
85  m_pointSrcMgtSeqs.emplace_back(count);
86 #ifdef HAS_VARIADIC_TEMPLATES
87  m_pointSrcMgtMap.emplace(count, new PointSourceMgtParams(bsonTable, iter));
88 #else
89  m_pointSrcMgtMap.insert(make_pair(count, new PointSourceMgtParams(bsonTable, iter)));
90 #endif
91  count++;
92  }
93  bson_destroy(b);
94  mongoc_cursor_destroy(cursor);
95 }

97 void BMPPointSrcFactory::ReadPointSourceLocations(MongoClient* conn, const string& bmpDBName) {
98  bson_t* b = bson_new();
99  bson_t* child1 = bson_new();
100  BSON_APPEND_DOCUMENT_BEGIN(b, "$query", child1);
101  BSON_APPEND_INT32(child1, BMP_PTSRC_FLD_CODE, m_pointSrc);
102  bson_append_document_end(b, child1);
103  bson_destroy(child1);
104 
105  std::unique_ptr<MongoCollection> collection(new MongoCollection(conn->GetCollection(bmpDBName, m_pointSrcDistTab)));
106  mongoc_cursor_t* cursor = collection->ExecuteQuery(b);
107 
108  bson_iter_t iter;
109  const bson_t* bsonTable;
110 
111  while (mongoc_cursor_next(cursor, &bsonTable)) {
112  PointSourceLocations* curPtSrcLoc = new PointSourceLocations(bsonTable, iter);
113  int curPtSrcID = curPtSrcLoc->GetPointSourceID();
114  if (ValueInVector(curPtSrcID, m_pointSrcIDs)) {
115 #ifdef HAS_VARIADIC_TEMPLATES
116  m_pointSrcLocsMap.emplace(curPtSrcID, curPtSrcLoc);
117 #else
118  m_pointSrcLocsMap.insert(make_pair(curPtSrcID, curPtSrcLoc));
119 #endif
120  } else {
121  RemoveValueInVector(curPtSrcID, m_pointSrcIDs);
122  delete curPtSrcLoc;
123  }
124  }
125  bson_destroy(b);
126  mongoc_cursor_destroy(cursor);
127 }
