106  def get_value_by_row_col(self, row, col):
107  """Get raster value by (row, col)
108  Args:
109  row: row number
110  col: col number
111 
112  Returns:
113  raster value, None if the input are invalid
114  """

125  def get_value_by_xy(self, x, y):
126  """Get raster value by xy coordinates
127  Args:
128  x: X Coordinate
129  y: Y Coordinate
130 
131  Returns:
132  raster value, None if the input are invalid
133  """
134  if x < self.xMin or x > self.xMax or y < self.yMin or y > self.yMax:
135  return None
136  # raise ValueError("The x or y value must be within the Min and Max!")
137  else:
138  row = self.nRows - int(numpy.ceil((y - self.yMin) / self.dx))
139  col = int(numpy.floor((x - self.xMin) / self.dx))
140  value = self.data[row][col]
141  if value == self.noDataValue:
142  return None
143  else:
144  return value
