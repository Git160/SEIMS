230 def generate_lat_raster():
231  geodata2dbdir = WORKING_DIR + os.sep + DIR_NAME_GEODATA2DB
232  dem_file = geodata2dbdir + os.sep + filleddemM
233  ds = RasterUtilClass.ReadRaster(dem_file)
234  src_srs = ds.srs
235  if not src_srs.ExportToProj4():
236  raise ValueError("The source raster %s has not coordinate, which is required!" % dem_file)
237  dst_srs = osr.SpatialReference()
238  dst_srs.ImportFromEPSG(4326) # WGS84
239  # dst_wkt = dst_srs.ExportToWkt()
240  transform = osr.CoordinateTransformation(src_srs, dst_srs)
241 
242  point_ll = ogr.CreateGeometryFromWkt("POINT (%f %f)" % (ds.xMin, ds.yMin))
243  point_ur = ogr.CreateGeometryFromWkt("POINT (%f %f)" % (ds.xMax, ds.yMax))
244 
245  point_ll.Transform(transform)
246  point_ur.Transform(transform)
247 
248  lowerLat = point_ll.GetY()
249  upLat = point_ur.GetY()
250 
251  rows = ds.nRows
252  cols = ds.nCols
253  deltaLat = (upLat - lowerLat) / float(rows)
254 
255  def cal_cell_lat(row, col):
256  return upLat - (row + 0.5) * deltaLat
257 
258  dataLat = numpy.fromfunction(cal_cell_lat, (rows, cols))
259  dataLat = numpy.where(ds.validZone, dataLat, ds.data)
260  RasterUtilClass.WriteGTiffFile(geodata2dbdir + os.sep + cellLat, rows, cols, dataLat, ds.geotrans, ds.srs,
261  ds.noDataValue, GDT_Float32)
