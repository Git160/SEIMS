238 template <typename T>
239 bool GetNumericFromBsonIterator(bson_iter_t* iter, T& numericvalue) {
240  const bson_value_t* vv = bson_iter_value(iter);
241  if (vv->value_type == BSON_TYPE_INT32) {
242  numericvalue = static_cast<T>(vv->value.v_int32);
243  } else if (vv->value_type == BSON_TYPE_INT64) {
244  numericvalue = static_cast<T>(vv->value.v_int64);
245  } else if (vv->value_type == BSON_TYPE_DOUBLE) {
246  numericvalue = static_cast<T>(vv->value.v_double);
247  } else if (vv->value_type == BSON_TYPE_UTF8) {
248  string tmp = vv->value.v_utf8.str;
249  if (tmp.find_first_of("0123456789") == string::npos) {
250  cout << "bson iterator isn't or not contains a numeric value." << endl;
251  return false;
252  }
253  char* end = nullptr;
254  numericvalue = static_cast<T>(strtod(tmp.c_str(), &end));
255  } else {
256  cout << "bson iterator isn't or not contains a numeric value." << endl;
257  return false;
258  }
259  return true;
260 }
