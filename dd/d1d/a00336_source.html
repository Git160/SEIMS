290  def ParseTimespan(self, items):
291  for item in items:
292  if 'TIMESPAN' not in item:
293  continue
294  item = item.split('\n')[0]
295  values = StringClass.extract_numeric_values_from_string(item)
296  if values is None or len(values) != 1:
297  continue
298  time = values[0]
299  titles = item.replace('[', '').split(']')[:-1]
300  if len(titles) < 3:
301  continue
302  titles = [title.strip() for title in titles]
303  if titles[1] not in self.timespan:
304  self.timespan[titles[1]] = dict()
305  if len(titles) > 3:
306  if titles[2] not in self.timespan[titles[1]]:
307  self.timespan[titles[1]][titles[2]] = dict()
308  if titles[3] not in self.timespan[titles[1]][titles[2]]:
309  self.timespan[titles[1]][titles[2]][titles[3]] = list()
310  self.timespan[titles[1]][titles[2]][titles[3]].append(time)
311  else:
312  if titles[2] not in self.timespan[titles[1]]:
313  self.timespan[titles[1]][titles[2]] = list()
314  self.timespan[titles[1]][titles[2]].append(time)
