38 void Settings::SetSettingTagStrings(vector<string>& stringvector) {
39  assert(stringvector.size() > 0);
40  for (vector<string>::iterator iter = stringvector.begin(); iter != stringvector.end(); ++iter) {
41  // parse the line into separate items
42  vector<string> tokens = SplitString(*iter, '|');
43  // is there anything in the token list?
44  if (tokens.size() > 0) {
45  for (size_t i = 0; i < tokens.size(); i++) {
46  tokens[i] = trim(tokens[i]);
47  }
48  // is there anything in the first item?
49  if (tokens[0].size() > 0) {
50  // there is something to add so resize the header list to append it
51  int sz = m_Settings.size(); // get the current number of rows
52  m_Settings.resize(sz + 1); // resize with one more row
53  m_Settings[sz] = tokens;
54  } // if there is nothing in the first item of the token list there is nothing to add to the header list
55  }
