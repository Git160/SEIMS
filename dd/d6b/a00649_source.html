37 void SOL_WB::SetValueToSubbasins() {
38  if (m_subbasinsInfo != nullptr) {
39  for (auto it = m_subbasinIDs.begin(); it != m_subbasinIDs.end(); ++it) {
40  Subbasin* curSub = m_subbasinsInfo->GetSubbasinByID(*it);
41  int* cells = curSub->GetCells();
42  int cellsNum = curSub->GetCellCount();
43  float ratio = 1.f / CVT_FLT(cellsNum);
44  float ri = 0.f; // total subsurface runoff of soil profile (mm)
45  float sm = 0.f; // total soil moisture of soil profile (mm)
46  float pcp = 0.f, netPcp = 0.f, depET = 0.f, infil = 0.f;
47  float itpET = 0.f, netPerc = 0.f, r = 0.f, revap = 0.f;
48  float rs = 0.f, meanT = 0.f, soilT = 0.f, es = 0.f, totalET = 0.f;
49 
50  for (int i = 0; i < cellsNum; i++) {
51  // loop cells of current subbasin
52  int cell = cells[i];
53  ri = 0.f;
54  sm = 0.f;
55  for (int j = 0; j < CVT_INT(m_nSoilLyrs[cell]); j++) {
56  ri += m_subSurfRf[cell][j] * ratio;
57  sm += m_soilWtrSto[cell][j] * ratio;
58  }
59  pcp += m_PCP[cell] * ratio;
60  meanT += m_meanTemp[cell] * ratio;
61  soilT += m_soilTemp[cell] * ratio;
62  netPcp += m_netPcp[cell] * ratio;
63  itpET += m_IntcpET[cell] * ratio;
64  depET += m_deprStoET[cell] * ratio;
65  infil += m_infil[cell] * ratio;
66  totalET += (m_IntcpET[cell] + m_deprStoET[cell] + m_soilET[cell]) * ratio; // add plant et?
67  es += m_soilET[cell] * ratio;
68  netPerc += (m_soilPerco[cell][CVT_INT(m_nSoilLyrs[cell]) - 1] - m_Revap[cell]) * ratio;
69  revap += m_Revap[cell] * ratio;
70  rs += m_surfRf[cell] * ratio;
71  r += (m_surfRf[cell] + ri) * ratio;
72  }
73  float rg = m_RG[*it];
74  r += rg;
75 
76  m_soilWtrBal[*it][0] = pcp;
77  m_soilWtrBal[*it][1] = meanT;
78  m_soilWtrBal[*it][2] = soilT;
79  m_soilWtrBal[*it][3] = netPcp;
80  m_soilWtrBal[*it][4] = itpET;
81  m_soilWtrBal[*it][5] = depET;
82  m_soilWtrBal[*it][6] = infil;
83  m_soilWtrBal[*it][7] = totalET;
84  m_soilWtrBal[*it][8] = es;
85  m_soilWtrBal[*it][9] = netPerc;
86  m_soilWtrBal[*it][10] = revap;
87  m_soilWtrBal[*it][11] = rs;
88  m_soilWtrBal[*it][12] = ri;
89  m_soilWtrBal[*it][13] = rg;
90  m_soilWtrBal[*it][14] = r;
91  m_soilWtrBal[*it][15] = sm;
92  }
93  }
