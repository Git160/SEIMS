31 void Scenario::loadScenarioName() {
32  vector<string>::iterator it = find(m_bmpCollections.begin(), m_bmpCollections.end(), string(TAB_BMP_SCENARIO));
33  if (it == m_bmpCollections.end()) {
34  throw ModelException("BMP Scenario", "loadScenarioName", "The BMP database '" + m_bmpDBName +
35  "' does not exist or there is not a table named '" +
36  TAB_BMP_SCENARIO + "' in BMP database.");
37  }
38  mongoc_collection_t *sceCollection;
39  sceCollection = mongoc_client_get_collection(m_conn, m_bmpDBName.c_str(), TAB_BMP_SCENARIO);
41  bson_t *query = bson_new(), *reply = bson_new();
42  query = BCON_NEW("distinct", BCON_UTF8(TAB_BMP_SCENARIO), "key", FLD_SCENARIO_NAME,
43  "query", "{", FLD_SCENARIO_ID, BCON_INT32(m_id), "}");
44  bson_iter_t iter, sub_iter;
45  bson_error_t *err = NULL;
46  if (mongoc_collection_command_simple(sceCollection, query, NULL, reply, err)) {
47  //cout<<bson_as_json(reply,NULL)<<endl;
48  if (bson_iter_init_find(&iter, reply, "values") &&
49  (BSON_ITER_HOLDS_DOCUMENT(&iter) || BSON_ITER_HOLDS_ARRAY(&iter)) &&
50  bson_iter_recurse(&iter, &sub_iter)) {
51  while (bson_iter_next(&sub_iter)) {
52  m_name = GetStringFromBsonIterator(&sub_iter);
53  break;
54  }
55  } else {
56  throw ModelException("BMP Scenario", "loadScenarioName",
57  "There is not scenario existed with the ID: " + ValueToString(m_id)
58  + " in " + TAB_BMP_SCENARIO + " table in BMP database.");
59  }
60  }
61  bson_destroy(query);
62  bson_destroy(reply);
63  mongoc_collection_destroy(sceCollection);
64 }

66 void Scenario::loadBMPs() {
67  vector<string>::iterator it = find(m_bmpCollections.begin(), m_bmpCollections.end(), string(TAB_BMP_INDEX));
68  if (it == m_bmpCollections.end()) {
69  throw ModelException("BMP Scenario", "loadScenarioName", "The BMP database '" + m_bmpDBName +
70  "' does not exist or there is not a table named '" +
71  TAB_BMP_INDEX + "' in BMP database.");
72  }
73  bson_t *query = bson_new();
74  BSON_APPEND_INT32(query, FLD_SCENARIO_ID, m_id);
75  //cout<<bson_as_json(query, NULL)<<endl;
76  mongoc_collection_t *collection = mongoc_client_get_collection(m_conn, m_bmpDBName.c_str(), TAB_BMP_SCENARIO);
77  mongoc_collection_t *collbmpidx = mongoc_client_get_collection(m_conn, m_bmpDBName.c_str(), TAB_BMP_INDEX);
78  mongoc_cursor_t *cursor = mongoc_collection_find(collection, MONGOC_QUERY_NONE, 0, 0, 0, query, NULL, NULL);
79  bson_error_t *err = NULL;
80  if (mongoc_cursor_error(cursor, err)) {
81  throw ModelException("BMP Scenario", "loadBMPs",
82  "There are no record with scenario ID: " + ValueToString(m_id));
83  }
84  const bson_t *info;
85  while (mongoc_cursor_more(cursor) && mongoc_cursor_next(cursor, &info)) {
86  //cout<<bson_as_json(info,0)<<endl;
87  bson_iter_t iter;
88  int BMPID = -1;
89  int subScenario = -1;
90  string distribution = "";
91  string collectionName = "";
92  string location = "";
93  if (bson_iter_init_find(&iter, info, FLD_SCENARIO_BMPID)) GetNumericFromBsonIterator(&iter, BMPID);
94  if (bson_iter_init_find(&iter, info, FLD_SCENARIO_SUB)) GetNumericFromBsonIterator(&iter, subScenario);
95  if (bson_iter_init_find(&iter, info, FLD_SCENARIO_DIST)) distribution = GetStringFromBsonIterator(&iter);
96  if (bson_iter_init_find(&iter, info, FLD_SCENARIO_TABLE)) collectionName = GetStringFromBsonIterator(&iter);
97  if (bson_iter_init_find(&iter, info, FLD_SCENARIO_LOCATION)) location = GetStringFromBsonIterator(&iter);
98 
99  int BMPType = -1;
100  int BMPPriority = -1;
101  bson_t *queryBMP = bson_new();
102  BSON_APPEND_INT32(queryBMP, FLD_BMP_ID, BMPID);
103  mongoc_cursor_t *cursor2 = mongoc_collection_find(collbmpidx, MONGOC_QUERY_NONE, 0, 0, 0, queryBMP, NULL,
104  NULL);
105  if (mongoc_cursor_error(cursor2, err)) {
106  throw ModelException("BMP Scenario", "loadBMPs",
107  "There are no record with BMP ID: " + ValueToString(BMPID));
108  }
109  const bson_t *info2;
110  while (mongoc_cursor_more(cursor2) && mongoc_cursor_next(cursor2, &info2)) {
111  bson_iter_t sub_iter;
112  if (bson_iter_init_find(&sub_iter, info2, FLD_BMP_TYPE)) GetNumericFromBsonIterator(&sub_iter, BMPType);
113  if (bson_iter_init_find(&sub_iter, info2, FLD_BMP_PRIORITY)) {
114  GetNumericFromBsonIterator(&sub_iter, BMPPriority);
115  }
116  }
117  //cout<<BMPID<<","<<BMPType<<","<<distribution<<","<<parameter<<endl;
118  bson_destroy(queryBMP);
119  mongoc_cursor_destroy(cursor2);
121  int uniqueBMPID = BMPID * 100000 + subScenario;
122  if (this->m_bmpFactories.find(uniqueBMPID) == this->m_bmpFactories.end()) {
123  if (BMPID == BMP_TYPE_POINTSOURCE) {
124  this->m_bmpFactories[uniqueBMPID] = new BMPPointSrcFactory(m_id, BMPID, subScenario, BMPType,
125  BMPPriority, distribution,
126  collectionName, location);
127  }
128  if (BMPID == BMP_TYPE_PLANT_MGT) {
129  this->m_bmpFactories[uniqueBMPID] = new BMPPlantMgtFactory(m_id, BMPID, subScenario, BMPType,
130  BMPPriority, distribution,
131  collectionName, location);
132  }
133  if (BMPID == BMP_TYPE_AREALSOURCE) {
134  this->m_bmpFactories[uniqueBMPID] = new BMPArealSrcFactory(m_id, BMPID, subScenario, BMPType,
135  BMPPriority, distribution,
136  collectionName, location);
137  }
138  }
139  }
140  bson_destroy(query);
141  mongoc_cursor_destroy(cursor);
142  mongoc_collection_destroy(collection);
143  mongoc_collection_destroy(collbmpidx);
144 }
