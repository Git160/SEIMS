292 void DataCenter::Set1DData(const string& para_name, const string& remote_filename,
293  SimulationModule* p_module, const bool vertital_itp) {
294  int n;

299  p_module->Set1DData(para_name.c_str(), array1d_len_map_.at(remote_filename), data);

301  }
302  if (weight_data_map_.find(remote_filename) != weight_data_map_.end()) {

304  weight_data->GetWeightData(&n, &data);
305  p_module->Set1DData(para_name.c_str(), n, data);

316  }
317  }
319  else if (StringMatch(para_name, Tag_Weight)) {
320  ReadItpWeightData(remote_filename, n, data);
321  }
323  else if (StringMatch(para_name, Tag_Elevation_Meteorology)) {
324  if (vertital_itp) {
325  n = clim_station_->NumberOfSites(DataType_Meteorology);
326  data = clim_station_->GetElevation(DataType_Meteorology);
327  } else {
328  return;
329  }
330  }
332  else if (StringMatch(para_name, Tag_Elevation_Precipitation)) {
333  if (vertital_itp) {
334  n = clim_station_->NumberOfSites(DataType_Precipitation);
335  data = clim_station_->GetElevation(DataType_Precipitation);
336  } else {
337  return;
338  }
339  }
341  else if (StringMatch(para_name, Tag_Latitude_Meteorology)) {
342  if (vertital_itp) {
343  n = clim_station_->NumberOfSites(DataType_Meteorology);
344  data = clim_station_->GetLatitude(DataType_Meteorology);
345  } else {
346  return;
347  }
348  }
350  else {
351  Read1DArrayData(para_name, remote_filename, n, data);
352  }
353  if (nullptr == data) {
354  throw ModelException("ModuleFactory", "Set1DData", "Failed reading file " + remote_filename);
355  }
356  p_module->Set1DData(para_name.c_str(), n, data);

398 void DataCenter::SetRaster(const string& para_name, const string& remote_filename, SimulationModule* p_module) {
399  int n, lyrs;
400  float* data = nullptr;
401  float** data2d = nullptr;
402  FloatRaster* raster = nullptr;
403  if (rs_map_.find(remote_filename) == rs_map_.end()) {
404  raster = ReadRasterData(remote_filename);
405  if (nullptr == raster) {
406  throw ModelException("DataCenter", "SetRaster", "Load " + remote_filename + " failed!");
407  }
408  string upper_name = GetUpper(para_name);
409  auto find_iter = init_params_.find(upper_name);
410  bool adjust_data = false;
411  if (find_iter != init_params_.end()) {
412  ParamInfo* tmp_param = find_iter->second;
413  if ((StringMatch(tmp_param->Change, PARAM_CHANGE_RC) && !FloatEqual(tmp_param->Impact, 1.f)) ||
414  (StringMatch(tmp_param->Change, PARAM_CHANGE_AC) && !FloatEqual(tmp_param->Impact, 0.f)) ||
415  (StringMatch(tmp_param->Change, PARAM_CHANGE_VC) && !FloatEqual(tmp_param->Impact, NODATA_VALUE)) ||
416  StringMatch(tmp_param->Change, PARAM_CHANGE_NC)) {
417  adjust_data = true;
418  }
419  }
421  if (raster->Is2DRaster()) {
422  if (!raster->Get2DRasterData(&n, &lyrs, &data2d)) {
423  throw ModelException("DataCenter", "SetRaster", "Load " + remote_filename + " failed!");
424  }
425  if (nullptr != data2d && adjust_data) {
426  init_params_[upper_name]->Adjust2DRaster(n, raster->GetLayers(), data2d);
427  }
428  } else {
429  if (!raster->GetRasterData(&n, &data)) {
430  throw ModelException("DataCenter", "SetRaster", "Load " + remote_filename + " failed!");
431  }
432  if (nullptr != data && adjust_data) {
433  init_params_[upper_name]->Adjust1DRaster(n, data);
434  }
435  }
436  } else {
437  raster = rs_map_.at(remote_filename);
438  //cout << remoteFileName << endl;
439  }
440  if (raster->Is2DRaster()) {
441  if (!raster->Get2DRasterData(&n, &lyrs, &data2d)) {
442  throw ModelException("DataCenter", "SetRaster", "Load " + remote_filename + " failed!");
443  }
444  p_module->Set2DData(para_name.c_str(), n, lyrs, data2d);
445  } else {
446  if (!raster->GetRasterData(&n, &data)) {
447  throw ModelException("DataCenter", "SetRaster", "Load " + remote_filename + " failed!");
448  }
449  p_module->Set1DData(para_name.c_str(), n, data);
450  }
