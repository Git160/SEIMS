99  """Read cali_param_rng.def file
100 
101  name,lower_bound,upper_bound
102 
103  e.g.,
104  Param1,0,1
105  Param2,0.5,1.2
106  Param3,-1.0,1.0
107 
108  Returns:
109  a dictionary containing:
110  - names - the names of the parameters
111  - bounds - a list of lists of lower and upper bounds
112  - num_vars - a scalar indicating the number of variables
113  (the length of names)
114  """
115  # read param_defs.json if already existed
116  if self.param_defs:
117  return self.param_defs
118  # read param_range_def file and output to json file
119  client = ConnectMongoDB(self.cfg.hostname, self.cfg.port)
120  conn = client.get_conn()
121  db = conn[self.cfg.spatial_db]
122  collection = db['PARAMETERS']
123 
124  names = list()
125  bounds = list()
126  num_vars = 0
127  if not FileClass.is_file_exists(self.cfg.param_range_def):
128  raise ValueError('Parameters definition file: %s is not'
129  ' existed!' % self.cfg.param_range_def)
130  items = read_data_items_from_txt(self.cfg.param_range_def)
131  for item in items:
132  if len(item) < 3:
133  continue
134  # find parameter name, print warning message if not existed
135  cursor = collection.find({'NAME': item[0]}, no_cursor_timeout=True)
136  if not cursor.count():
137  print('WARNING: parameter %s is not existed!' % item[0])
138  continue
139  num_vars += 1
140  names.append(item[0])
141  bounds.append([float(item[1]), float(item[2])])
142  self.param_defs = {'names': names, 'bounds': bounds, 'num_vars': num_vars}
143  return self.param_defs

164  param_num = self.ParamDefs['num_vars']
