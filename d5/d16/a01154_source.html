47  m_qsTemp[reachId] += m_qs[i];
48  if (m_qi != NULL) {
49  m_qiTemp[reachId] += m_qi[i];
50  }
51  }
52  }
53 
54  m_qsInput[0] = 0.f;
55  m_qiInput[0] = 0.f;
56  for (int i = 1; i <= m_nReaches; i++) {
57  m_qsInput[i] = (m_qsInput[i] * (m_tsCounter - 1) + m_qsTemp[i]) / m_tsCounter;
58  m_qiInput[i] = (m_qiInput[i] * (m_tsCounter - 1) + m_qiTemp[i]) / m_tsCounter;
59 
60  m_qsInput[0] += m_qsInput[i];
61  m_qiInput[0] += m_qiInput[i];
62  }
63 
64  m_tsCounter++;
