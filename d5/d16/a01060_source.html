128  for (time_t t = startTime; t < endTime; t += m_dtCh) {

131  int curYear = GetYear(t);
132  int yearIdx = curYear - startYear;
133  nHs = int(m_dtCh / m_dtHs);
134  for (int i = 0; i < nHs; i++) {
135  StepHillSlope(t + i * m_dtHs, yearIdx, i);
136  }
137  StepChannel(t, yearIdx);
138  AppendOutputData(t);
139  }
140  StepOverall(startTime, endTime);

187 void ModelMain::AppendOutputData(const time_t time) {
188  vector<PrintInfo *>::iterator it;
189  for (it = m_output->m_printInfos.begin(); it < m_output->m_printInfos.end(); it++) {
190  int iModule = (*it)->m_moduleIndex;
191 
192  //find the corresponding output variable and module
193  ParamInfo *param = (*it)->m_param;
194  if (param == NULL) {
195  throw ModelException("ModelMain", "Output",
196  "Output id " + (*it)->getOutputID() + " does not have corresponding output variable.");
197  }
198 
199  SimulationModule *module = m_simulationModules[iModule];
200  if (module == NULL) {
201  throw ModelException("ModelMain", "Output",
202  "Output id " + (*it)->getOutputID() + " does not have corresponding module.");
203  }
204 
205  //process every output file
206  vector<PrintInfoItem *>::iterator itemIt;
207  for (itemIt = (*it)->m_PrintItems.begin(); itemIt < (*it)->m_PrintItems.end(); itemIt++) {
208  PrintInfoItem *item = *itemIt;
209  const char *keyName = param->Name.c_str();
210  //time_t t1 = item->getStartTime();
211  //time_t t2 = item->getEndTime();
212  if (time >= item->getStartTime() && time <= item->getEndTime()) {
213  if (param->Dimension == DT_Single) {
214  float value;
215  module->GetValue(keyName, &value);
216  item->TimeSeriesData[time] = value;
217  }
218  //time series data for sites or some time series data for subbasins, such as T_SBOF,T_SBIF
219  else if (param->Dimension == DT_Array1D) {
220  int index = item->SubbasinID;
221  //time series data for some time series data for subbasins
222  if (index < 0) index = 0;
223 
224  int n;
225  float *data;
226  module->Get1DData(keyName, &n, &data);
227  item->TimeSeriesData[time] = data[index];
228  } else if (param->Dimension == DT_Array2D) //time series data for subbasins
229  {
230  //some modules will calculate result for all subbasins or all reaches,
231  //regardless of whether they are output to file or not. In this case,
232  //the 2-D array will contain all the results and the subbasinid or reachid
233  //will be used to locate the result.
234  if (StringMatch(param->BasicName, "RECH") || //discharge of reach
235  StringMatch(param->BasicName, "WABA") || //channel water balance
236  StringMatch(param->BasicName, "RSWB") || //reservoir water balance
237  StringMatch(param->BasicName, "RESB") || //reservoir sediment balance
238  StringMatch(param->BasicName, "CHSB") ||
239  StringMatch(param->BasicName, VAR_GWWB) || // groundwater water balance
240  StringMatch(param->BasicName, VAR_SOWB) // soil water balance
241  )// TODO: more conditions will be added in the future.
242  {
243  //for modules in which only the results of output subbasins are calculated.
244  //In this case, the 2-D array just contain the results of selected subbasins in file.out.
245  //So, the index of Subbasin in file.out will be used to locate the result.
246  int subbasinIndex = item->SubbasinIndex;
247  if (subbasinIndex == -1) {
248  char s[20];
249  strprintf(s, 20, "%d", item->SubbasinID);
250  throw ModelException("ModelMain", "Output",
251  "Can't find subbasin " + string(s) + " in input sites.");
252  }
253  float **data;
254  int nRows, nCols;
255  module->Get2DData(param->BasicName.c_str(), &nRows, &nCols, &data);
256  item->add1DTimeSeriesResult(time, nCols, data[subbasinIndex]);
257  } else {
258  float **data;
259  int nRows, nCols;
260  module->Get2DData(param->BasicName.c_str(), &nRows, &nCols, &data);
261  item->AggregateData2D(time, nRows, nCols, data);
262  }
263  } else if (param->Dimension == DT_Raster1D) //spatial distribution, calculate average,sum,min or max
264  {
265  int n;
266  float *data;
267  //cout << keyName << " " << n << endl;
268  module->Get1DData(keyName, &n, &data);
269  item->AggregateData(time, n, data);
270  } else if (param->Dimension == DT_Raster2D) // spatial distribution with layers
271  {
272  int n, lyrs;
273  float **data;
274  module->Get2DData(keyName, &n, &lyrs, &data);
275  item->AggregateData2D(time, n, lyrs, data);
276  }
277  }
