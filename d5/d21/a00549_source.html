45 void MUSLE_AS::InitialOutputs() {
46  CHECK_POSITIVE(MID_MUSLE_AS, m_nCells);
47  if (nullptr == m_sedimentYield) Initialize1DArray(m_nCells, m_sedimentYield, 0.f);
48  if (nullptr == m_sandYield) Initialize1DArray(m_nCells, m_sandYield, 0.f);
49  if (nullptr == m_siltYield) Initialize1DArray(m_nCells, m_siltYield, 0.f);
50  if (nullptr == m_clayYield) Initialize1DArray(m_nCells, m_clayYield, 0.f);
51  if (nullptr == m_smaggreYield) Initialize1DArray(m_nCells, m_smaggreYield, 0.f);
52  if (nullptr == m_lgaggreYield) Initialize1DArray(m_nCells, m_lgaggreYield, 0.f);
53  if (nullptr == m_usle_ls) {
54  float constant = pow(22.13f, 0.4f);
55  m_usle_ls = new float[m_nCells];
56  m_slopeForPq = new float[m_nCells];
57 #pragma omp parallel for
58  for (int i = 0; i < m_nCells; i++) {
59  if (m_usle_c[i] < 0.001f) {
60  m_usle_c[i] = 0.001f;
61  }
62  if (m_usle_c[i] > 1.0f) {
63  m_usle_c[i] = 1.0f;
64  }
65  float lambda_i1 = m_flowacc[i] * m_cellWidth; // m
66  float lambda_i = lambda_i1 + m_cellWidth;
67  float L = pow(lambda_i, 1.4f) - pow(lambda_i1, 1.4f); // m^1.4
68  L /= m_cellWidth * constant;
69 
70  float S = pow(sin(atan(m_slope[i])) / 0.0896f, 1.3f);
71 
72  m_usle_ls[i] = L * S; // LS factor
73 
74  m_slopeForPq[i] = pow(m_slope[i] * 1000.f, 0.16f);
75  }
76  }
77  if (FloatEqual(m_cellAreaKM, NODATA_VALUE)) {
78  m_cellAreaKM = pow(m_cellWidth / 1000.f, 2.f);
79  m_cellAreaKM1 = 3.79f * pow(m_cellAreaKM, 0.7f);
80  m_cellAreaKM2 = 0.903f * pow(m_cellAreaKM, 0.017f);
81  }
82 }

84 int MUSLE_AS::Execute() {
85  CheckInputData();
86  InitialOutputs();
87 #pragma omp parallel for
88  for (int i = 0; i < m_nCells; i++) {
89  if (m_surfaceRunoff[i] < 0.0001f || m_streamLink[i] > 0) {
90  m_sedimentYield[i] = 0.f;
91  m_sandYield[i] = 0.f;
92  m_siltYield[i] = 0.f;
93  m_clayYield[i] = 0.f;
94  m_smaggreYield[i] = 0.f;
95  m_lgaggreYield[i] = 0.f;
96  continue;
97  }
98  // peak flow
99  float q = m_cellAreaKM1 * m_slopeForPq[i] * pow(m_surfaceRunoff[i] / 25.4f, m_cellAreaKM2);
100  // sediment yield
101  float Y = 11.8f * pow(m_surfaceRunoff[i] * m_cellAreaKM * 1000.0f * q,
102  0.56f) * m_usle_k[i][0] * m_usle_ls[i] * m_usle_c[i] * m_usle_p[i];
103  // the snow pack effect
104  if (m_snowAccumulation[i] > 0.0001f) {
105  Y /= exp(3.f * m_snowAccumulation[i] / 25.4f);
106  }
107  m_sedimentYield[i] = Y * 1000.f;
108 
110  m_sandYield[i] = m_sedimentYield[i] * m_detachSand[i];
111  m_siltYield[i] = m_sedimentYield[i] * m_detachSilt[i];
112  m_clayYield[i] = m_sedimentYield[i] * m_detachClay[i];
113  m_smaggreYield[i] = m_sedimentYield[i] * m_detachSmAggre[i];
114  m_lgaggreYield[i] = m_sedimentYield[i] * m_detachLgAggre[i];
115  }
116  return 0;
117 }
