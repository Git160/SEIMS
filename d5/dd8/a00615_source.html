396 void ModelMain::Output(time_t time) {
397  vector<PrintInfo *>::iterator it;
398  for (it = m_output->m_printInfos.begin(); it < m_output->m_printInfos.end(); it++) {
399  int iModule = (*it)->m_moduleIndex;
400 
401  //find the corresponding output variable and module
402  ParamInfo *param = (*it)->m_param;
403  if (param == NULL) {
404  throw ModelException("ModelMain", "Output",
405  "Output id " + (*it)->getOutputID() + " does not have corresponding output variable.");
406  }
407 
408  SimulationModule *module = m_simulationModules[iModule];
409  if (module == NULL) {
410  throw ModelException("ModelMain", "Output",
411  "Output id " + (*it)->getOutputID() + " does not have corresponding module.");
412  }
413 
414  //process every output file
415  vector<PrintInfoItem *>::iterator itemIt;
416  for (itemIt = (*it)->m_PrintItems.begin(); itemIt < (*it)->m_PrintItems.end(); itemIt++) {
417  PrintInfoItem *item = *itemIt;
418  const char *keyName = param->Name.c_str();
419  //time_t t1 = item->getStartTime();
420  //time_t t2 = item->getEndTime();
421  if (time >= item->getStartTime() && time <= item->getEndTime()) {
422  if (param->Dimension == DT_Single) {
423  float value;
424  module->GetValue(keyName, &value);
425  item->TimeSeriesData[time] = value;
426  }
427  //time series data for sites or some time series data for subbasins, such as T_SBOF,T_SBIF
428  else if (param->Dimension == DT_Array1D) {
429  int index = -1;
430  if (index < 0) {
431  index = item->SubbasinID;
432  } //time series data for some time series data for subbasins
433  if (index < 0)
434  index = 0;
435 
436  int n;
437  float *data;
438  module->Get1DData(keyName, &n, &data);
439  item->TimeSeriesData[time] = data[index];
440  } else if (param->Dimension == DT_Array2D) //time series data for subbasins
441  {
442  //some modules will calculate result for all subbasins or all reaches,
443  //regardless of whether they are output to file or not. In this case,
444  //the 2-D array will contain all the results and the subbasinid or reachid
445  //will be used to locate the result.
446  if (StringMatch(param->BasicName, "RECH") || //discharge of reach
447  StringMatch(param->BasicName, "WABA") || //channel water balance
448  StringMatch(param->BasicName, "RSWB") || //reservoir water balance
449  StringMatch(param->BasicName, "RESB") || //reservoir sediment balance
450  StringMatch(param->BasicName, "CHSB") ||
451  StringMatch(param->BasicName, VAR_GWWB) || // groundwater water balance
452  StringMatch(param->BasicName, VAR_SOWB) // soil water balance
453  )// TODO: more conditions will be added in the future.
454  {
455  //for modules in which only the results of output subbasins are calculated.
456  //In this case, the 2-D array just contain the results of selected subbasins in file.out.
457  //So, the index of Subbasin in file.out will be used to locate the result.
458  int subbasinIndex = item->SubbasinIndex;
459 
460  subbasinIndex = item->SubbasinID;
461 
462  if (subbasinIndex == -1) {
463  char s[20];
464  strprintf(s, 20, "%d", item->SubbasinID);
465  throw ModelException("SettingsConfig", "Output",
466  "Can't find subbasin " + string(s) + " in input sites.");
467  }
468 
469  float **data;
470  int nRows, nCols;
471  module->Get2DData(param->BasicName.c_str(), &nRows, &nCols, &data);
472  item->add1DTimeSeriesResult(time, nCols, data[subbasinIndex]);
473  } else {
474  float **data;
475  int nRows, nCols;
476  module->Get2DData(param->BasicName.c_str(), &nRows, &nCols, &data);
477  item->AggregateData2D(time, nRows, nCols, data);
478  }
479  } else if (param->Dimension == DT_Raster1D) //spatial distribution, calculate average,sum,min or max
480  {
481  int n;
482  float *data;
483  //cout << keyName << " " << n << endl;
484  module->Get1DData(keyName, &n, &data);
485  item->AggregateData(time, n, data);
486  } else if (param->Dimension == DT_Raster2D) // spatial distribution with layers
487  {
488  int n, lyrs;
489  float **data;
490  module->Get2DData(keyName, &n, &lyrs, &data);
491  item->AggregateData2D(time, n, lyrs, data);
492  }
493  }
494  }
495  }
