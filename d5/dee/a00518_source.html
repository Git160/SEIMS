157 void NutrientinGroundwater::initialOutputs() {
158  if (this->m_nCells <= 0) {
159  throw ModelException(MID_NUTRGW, "CheckInputData",
160  "The dimension of the input data can not be less than zero.");
161  }
162  // allocate the output variables
163  if (m_no3GwToCh == NULL) {
164  Initialize1DArray(m_nSubbasins + 1, m_no3GwToCh, 0.f);
165  Initialize1DArray(m_nSubbasins + 1, m_solpGwToCh, 0.f);
166  }
167  if (m_gwNO3 == NULL) {
168  m_gwNO3 = new float[m_nSubbasins + 1];
169  m_gwSolP = new float[m_nSubbasins + 1];
170  for (int i = 1; i <= m_nSubbasins; i++) {
171  Subbasin *subbasin = m_subbasinsInfo->GetSubbasinByID(i);
172  float subArea = subbasin->getCellCount() * m_cellWidth * m_cellWidth; //m2
173  m_gwNO3[i] = m_gw0 * m_gwno3Con[i] * subArea / 1000000.f;
174  m_gwSolP[i] = m_gw0 * m_gwSolCon[i] * subArea / 1000000.f;
175  }
176  }

186  //cout<<endl;
187  for (vector<int>::iterator iter = m_subbasinIDs.begin(); iter != m_subbasinIDs.end(); iter++) {
188  int id = *iter;
189  Subbasin *subbasin = m_subbasinsInfo->GetSubbasinByID(id);
190  int nCells = subbasin->getCellCount();
191  float subArea = nCells * m_cellWidth * m_cellWidth; // m^2
192  float revap = subbasin->getEG();
195  float gwqVol = m_gw_q[id] * m_TimeStep; // m^3, water volume flow out
196  float reVapVol = revap * subArea / 1000.f; // m^3
197  float tmpGwStorage = m_gwStor[id] * subArea / 1000.f + gwqVol + reVapVol;
199  m_gwNO3[id] += m_perco_no3_gw[id];
200  m_gwSolP[id] += m_perco_solp_gw[id];
201  m_gwno3Con[id] = m_gwNO3[id] / tmpGwStorage * 1000.f; // kg / m^3 * 1000. = mg/L
202  m_gwSolCon[id] = m_gwSolP[id] / tmpGwStorage * 1000.f;
204  //cout<<"subID: "<<id<<", gwQ: "<<m_gw_q[id] << ", ";
205  m_no3GwToCh[id] = m_gwno3Con[id] * gwqVol / 1000.f; // g/m3 * m3 / 1000 = kg
206  m_solpGwToCh[id] = m_gwSolCon[id] * gwqVol / 1000.f;
207  //cout<<"subID: "<<id<<", gwno3Con: "<<m_gwno3Con[id] << ", no3gwToCh: "<<m_no3gwToCh[id] << ", ";
209  float no3ToSoil = revap / 1000.f * m_gwno3Con[id] * 10.f; // kg/ha (m*10*g/m3=kg/ha)
210  float solpToSoil = revap / 1000.f * m_gwSolCon[id] * 10.f;
211  float no3ToSoil_kg = no3ToSoil * subArea / 10000.f;
212  float solpToSoil_kg = solpToSoil * subArea / 10000.f;
213  int *cells = subbasin->getCells();
214  int index = 0;
215  for (int i = 0; i < nCells; i++) {
216  index = cells[i];
217  //if (index == cellid)
218  // cout<<"revap: "<<revap<<", gwNo3: "<<m_gwNO3[id]<<", gwStorg: "<<
219  // tmpGwStorage<<", percoNo3Gw: "<<m_perco_no3_gw[id]<<
220  // ", gwNo3Con: "<<m_gwno3Con[id]<<", no3ToSoil: "<<no3ToSoil<<endl;
221  m_sol_no3[index][(int) m_soilLayers[index] - 1] += no3ToSoil;
222  m_sol_solp[index][(int) m_soilLayers[index] - 1] += solpToSoil;
223  }
225  m_gwNO3[id] -= (m_no3GwToCh[id] + no3ToSoil_kg);
226  m_gwSolP[id] -= (m_solpGwToCh[id] + solpToSoil_kg);
