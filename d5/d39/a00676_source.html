112 void NutrientinGroundwater::InitialOutputs() {
113  CHECK_POSITIVE(MID_NUTRGW, m_nCells);
114  // allocate the output variables
115  if (nullptr == m_gwNO3ToCh) {
116  Initialize1DArray(m_nSubbsns + 1, m_gwNO3ToCh, 0.f);
117  Initialize1DArray(m_nSubbsns + 1, m_gwSolPToCh, 0.f);
118  }
119  if (nullptr == m_gwNO3) {
121  m_gwNO3 = new float[m_nSubbsns + 1];
122  m_gwSolP = new float[m_nSubbsns + 1];
123  for (auto it = m_subbasinIDs.begin(); it != m_subbasinIDs.end(); ++it) {
124  Subbasin* subbasin = m_subbasinsInfo->GetSubbasinByID(*it);
125  float subArea = subbasin->GetArea(); //m^2
126  m_gwNO3[*it] = m_gw0 * m_gwNO3Conc[*it] * subArea * 0.000001f;
127  m_gwSolP[*it] = m_gw0 * m_gwSolPConc[*it] * subArea * 0.000001f;
128  }
129  }

132 int NutrientinGroundwater::Execute() {
133  CheckInputData();
134  InitialOutputs();
135  for (auto it = m_subbasinIDs.begin(); it != m_subbasinIDs.end(); ++it) {
136  int id = *it;
137  Subbasin* subbasin = m_subbasinsInfo->GetSubbasinByID(id);
138  int nCells = subbasin->GetCellCount();
139  float subArea = subbasin->GetArea(); // m^2
140  float revap = subbasin->GetEg();
143  float gwqVol = m_gw_q[id] * m_TimeStep; // m^3, water volume flow out
144  float reVapVol = revap * subArea * 0.001f; // m^3
145  float tmpGwStorage = m_gwStor[id] * subArea * 0.001f + gwqVol + reVapVol;
147  m_gwNO3[id] += m_perco_no3_gw[id];
148  m_gwSolP[id] += m_perco_solp_gw[id];
149  m_gwNO3Conc[id] = m_gwNO3[id] / tmpGwStorage * 1000.f; // kg / m^3 * 1000. = mg/L
150  m_gwSolPConc[id] = m_gwSolP[id] / tmpGwStorage * 1000.f;
152  //cout<<"subID: "<<id<<", gwQ: "<<m_gw_q[id] << ", ";
153  m_gwNO3ToCh[id] = m_gwNO3Conc[id] * gwqVol * 0.001f; // g/m3 * m3 / 1000 = kg
154  m_gwSolPToCh[id] = m_gwSolPConc[id] * gwqVol * 0.001f;
155  //cout<<"subID: "<<id<<", gwno3Con: "<<m_gwno3Con[id] << ", no3gwToCh: "<<m_no3gwToCh[id] << ", ";
157  float no3ToSoil = revap * 0.001f * m_gwNO3Conc[id] * 10.f; // kg/ha (m*10*g/m3=kg/ha)
158  float solpToSoil = revap * 0.001f * m_gwSolPConc[id] * 10.f;
159  float no3ToSoil_kg = no3ToSoil * subArea * 0.0001f;
160  float solpToSoil_kg = solpToSoil * subArea * 0.0001f;
161  int* cells = subbasin->GetCells();
162  int index = 0;
163  for (int i = 0; i < nCells; i++) {
164  index = cells[i];
165  m_soilNO3[index][CVT_INT(m_nSoilLyrs[index]) - 1] += no3ToSoil;
166  m_soilSolP[index][CVT_INT(m_nSoilLyrs[index]) - 1] += solpToSoil;
167  }
169  m_gwNO3[id] -= m_gwNO3ToCh[id] + no3ToSoil_kg;
170  m_gwSolP[id] -= m_gwSolPToCh[id] + solpToSoil_kg;
171  }
172  return 0;
173 }
