163  int ii = rowIndex + CellOrdering::m_d1[index];
164  int jj = colIndex + CellOrdering::m_d2[index];
165  if (ii < rsDEM.getRows() - 1 && jj < rsDEM.getCols() - 1) {
166  RowColCoor rc_new(ii, jj);

168  flag = false;

171  colIndex = jj;
172  }
173  } else {
174  flag = false;
175  }
176  }
