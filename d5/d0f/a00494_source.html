35 void SOL_WB::setValueToSubbasins() {
36  if (m_subbasinsInfo != NULL) {
37  vector<int>::iterator it;
38  for (it = m_subbasinIDs.begin(); it != m_subbasinIDs.end(); it++) {
39  Subbasin *curSub = m_subbasinsInfo->GetSubbasinByID(*it);
40  int *cells = curSub->getCells();
41  int cellsNum = curSub->getCellCount();
42  float ri = 0.f; // total subsurface runoff of soil profile (mm)
43  float sm = 0.f; // total soil moisture of soil profile (mm)
44  float pcp = 0.f, netPcp = 0.f, depET = 0.f, infil = 0.f;
45  float itpET = 0.f, netPerc = 0.f, r = 0.f, revap = 0.f;
46  float rs = 0.f, meanT = 0.f, soilT = 0.f, es = 0.f, totalET = 0.f;
47 
48  for (int i = 0; i < cellsNum; i++) // loop cells of current subbasin
49  {
50  int cell = cells[i];
51  ri = 0.f;
52  sm = 0.f;
53  for (int j = 0; j < (int) m_soilLayers[cell]; j++) {
54  ri += m_RI[cell][j] / float(cellsNum);
55  sm += m_soilStorage[cell][j] / float(cellsNum);
56  }
57  pcp += m_PCP[cell] / float(cellsNum);
58  meanT += m_tMean[cell] / float(cellsNum);
59  soilT += m_SoilT[cell] / float(cellsNum);
60  netPcp += m_pNet[cell] / float(cellsNum);
61  itpET += m_EI[cell] / float(cellsNum);
62  depET += m_ED[cell] / float(cellsNum);
63  infil += m_Infil[cell] / float(cellsNum);
64  totalET += (m_EI[cell] + m_ED[cell] + m_ES[cell]) / float(cellsNum); // add plant et?
65  es += m_ES[cell] / float(cellsNum);
66  netPerc += (m_Perco[cell][(int) m_soilLayers[cell] - 1] - m_Revap[cell]) / float(cellsNum);
67  revap += m_Revap[cell] / float(cellsNum);
68  rs += m_RS[cell] / float(cellsNum);
69  r += (m_RS[cell] + ri) / float(cellsNum);
70  }
71  float rg = m_RG[*it];
72  r += rg;
73 
74  m_soilWaterBalance[*it][0] = pcp;
75  m_soilWaterBalance[*it][1] = meanT;
76  m_soilWaterBalance[*it][2] = soilT;
77  m_soilWaterBalance[*it][3] = netPcp;
78  m_soilWaterBalance[*it][4] = itpET;
79  m_soilWaterBalance[*it][5] = depET;
80  m_soilWaterBalance[*it][6] = infil;
81  m_soilWaterBalance[*it][7] = totalET;
82  m_soilWaterBalance[*it][8] = es;
83  m_soilWaterBalance[*it][9] = netPerc;
84  m_soilWaterBalance[*it][10] = revap;
85  m_soilWaterBalance[*it][11] = rs;
86  m_soilWaterBalance[*it][12] = ri;
87  m_soilWaterBalance[*it][13] = rg;
88  m_soilWaterBalance[*it][14] = r;
89  m_soilWaterBalance[*it][15] = sm;
90  }
91  }
