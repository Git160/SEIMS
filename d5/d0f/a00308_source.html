30 class SoilProperty(object):
31  """
32  base class of Soil physical and general chemical properties
33  Attributes:
34  SOILLAYERS (int, None): (nly in SWAT model, the same below) number of soil layers
35  SOILDEPTH (float, mm): (sol_z) depth from the surface to bottom of soil layer
36  SOILTHICK (float, mm): soil thickness for calculation convenient, derived from `SOILDEPTH`
37  OM (float, %): organic matter content (weight percent)
38  SOL_CBN (float, %): (sol_cbn) percent organic carbon in soil layer, calculated by `OM`
39  SOL_N (float, %): (sol_n) used when using CSWAT = 1, i.e, C-FARM one carbon pool model, derived from `SOL_CBN`
40  CLAY (float, %): (sol_clay) percent clay content in soil material, diameter < 0.002 mm
41  SILT (float, %): (sol_silt) percent silt content in soil material,diameter between 0.002 mm and 0.05 mm
42  SAND (float, %): (sol_sand) percent sand content in soil material,diameter between 0.05 mm and 2 mm
43  ROCK (float, %): (sol_rock) percent of rock fragments content in soil material,diameter > 2 mm
44  SOIL_TEXTURE (int, None): soil texture code used in WetSpa and SWAT model based on soil texture triangle by USDA
45  HYDRO_GROUP (int, None): Hydrological soil group, 1, 2, 3, and 4 to represent A, B, C, and D
46  SOL_ZMX (float, mm): (sol_zmx) maximum rooting depth of soil profile
47  ANION_EXCL (float, None): (anion_excl) fraction of porosity from which anions are excluded, default is 0.5
48  SOL_CRK (float, None): (sol_crk) crack volume potential of soil expressed as a fraction of the total soil volume
49  DENSITY (float, Mg/m3 or g/cm3): (sol_bd) bulk density of each soil layer
50  SOL_AVBD (float, Mg/m3 or g/cm3): (sol_avbd) average bulk density for soil profile
51  CONDUCTIVITY (float, mm/hr): (sol_k) saturated hydraulic conductivity
52  SOL_HK (float, None): (sol_hk) beta coefficent to calculate hydraulic conductivity
53  WILTINGPOINT (float, mm H2O / mm soil): (sol_wp) water content of soil at -1.5 MPa (wilting point)
54  SOL_WPMM (float, mm H2O): (sol_wpmm) water content of soil at -1.5 MPa (wilting point)
55  SOL_SUMWP (float, mm H2O): (sol_sumwp) amount of water held in the soil profile at wilting point
56  FIELDCAP (float,mm H2O / mm soil): (sol_up) water content of soil at -0.033 MPa (field capacity)
57  AWC (float,mm H2O / mm soil): (sol_awc) available water capacity of soil layer
58  SOL_AWC (float,mm H2O): (sol_fc) amount of water available to plants in soil layer at field capacity (fc - wp)
59  SOL_SUMAWC (float,mm H2O): (sol_sumfc) amount of water held in soil profile at field capacity
60  POROSITY (float,None): (sol_por) total porosity of soil layer expressed as a fraction of the total volume
61  POREINDEX (float,None): pore size distribution index
62  SOL_AVPOR (float,None): (sol_avpor) average porosity for entire soil profile
63  SOL_UL (float,mm H2O): (sol_ul) amount of water held in the soil layer at saturation (sat - wp water)
64  SOL_SUMUL (float,mm H2O): (sol_sumul) amount of water held in soil profile at saturation
65  USLE_K (float,None): USLE K factor
66  SOL_ALB (float,None): albedo of top soil surface
67  WFSH (float,mm): wetting front matric potential (usde in Green-Ampt method)
68  ESCO (float,None): soil evaporation compensation factor
69  VWT (float,None): (vwt) variable water table factor, used in percolation modules
70  DET_SAND (float,None): (det_san) detached sediment size distribution, sand fraction
71  DET_SILT (float,None): (det_sil) detached sediment size distribution, silt fraction
72  DET_CLAY (float,None): (det_cla) detached sediment size distribution, clay fraction
73  DET_SMAGG (float,None): (det_sag) detached sediment size distribution, small aggregation fraction
74  DET_LGAGG (float,None): (det_lag) detached sediment size distribution, large aggregation fraction
75  CRDEP (float,mm): (crdep) maximum or potential crack volume
76  VOLCR (float,mm): (volcr) crack volume for soil layer, should be calculated in SEIMS, using moist_ini
77  SOL_NO3 (float,kg/ha): (sol_no3) concentration of nitrate in soil layers
78  SOL_NH4 (float,kg/ha): (sol_nh4) concentration of ammonium-N in soil layers
79  SOL_ORGN (float,kg/ha): (sol_orgn) organic N concentration in soil layers
80  SOL_ORGP (float,kg/ha): (sol_orgp) organic P concentration in soil layers
81  SOL_SOLP (float,kg/ha): (sol_solp) soluble P concentration in soil layers

260  self.SOL_N = list()
261  for i in range(self.SOILLAYERS):
262  self.SOL_N.append(self.SOL_CBN[i] / 11.)

264  raise IndexError("Clay content must have a size equal to NLAYERS and "

267  raise IndexError("Silt content must have a size equal to NLAYERS and "

270  raise IndexError("Sand content must have a size equal to NLAYERS and "

274  "should not include NODATA (-9999)!")

276  # temporary variables
277  tmp_fc = list()
278  tmp_sat = list()
279  tmp_wp = list()
280  for i in range(self.SOILLAYERS):
281  s = self.SAND[i] * 0.01 # % -> decimal
282  c = self.CLAY[i] * 0.01
283  om = self.OM[i]
284  wpt = -0.024 * s + 0.487 * c + 0.006 * om + 0.005 * s * om \
285  - 0.013 * c * om + 0.068 * s * c + 0.031
286  tmp_wp.append(1.14 * wpt - 0.02)
287  fct = -0.251 * s + 0.195 * c + 0.011 * om + 0.006 * s * om \
288  - 0.027 * c * om + 0.452 * s * c + 0.299
289  fc = fct + 1.283 * fct * fct - 0.374 * fct - 0.015
290  s33t = 0.278 * s + 0.034 * c + 0.022 * om - 0.018 * s * om \
291  - 0.027 * c * om - 0.584 * s * c + 0.078
292  s33 = 1.636 * s33t - 0.107
293  sat = fc + s33 - 0.097 * s + 0.043
294  tmp_fc.append(fc)
295  tmp_sat.append(sat)
296 
297  if self.WILTINGPOINT and len(self.WILTINGPOINT) != self.SOILLAYERS:
298  raise IndexError("Wilting point must have a size equal to soil layers number!")
299  elif not self.WILTINGPOINT:
300  self.WILTINGPOINT = tmp_wp[:]
301  elif DEFAULT_NODATA in self.WILTINGPOINT:
302  for i in range(self.SOILLAYERS):
303  if self.WILTINGPOINT[i] == DEFAULT_NODATA:
304  self.WILTINGPOINT[i] = tmp_wp[i]
305  if self.DENSITY and len(self.DENSITY) != self.SOILLAYERS:
306  raise IndexError("Bulk density must have a size equal to soil layers number!")
307  elif not self.DENSITY or DEFAULT_NODATA in self.DENSITY:
308  tmp_bd = list()
309  for i in range(self.SOILLAYERS):
310  sat = tmp_sat[i]
311  fc = tmp_fc[i]
312  if self.FIELDCAP and len(self.FIELDCAP) == self.SOILLAYERS:
313  sat = sat - fc + self.FIELDCAP[i]
314  tmp_bd.append(2.65 * (1.0 - sat))
315  if DEFAULT_NODATA in self.DENSITY:
316  for i in range(self.SOILLAYERS):
317  if self.DENSITY[i] == DEFAULT_NODATA:
318  self.DENSITY[i] = tmp_bd[i]
319  elif not self.DENSITY:
320  self.DENSITY = tmp_bd[:]
321  if self.FIELDCAP and len(self.FIELDCAP) != self.SOILLAYERS:
322  raise IndexError("Field capacity must have a size equal to soil layers number!")
323  elif not self.FIELDCAP or DEFAULT_NODATA in self.FIELDCAP:
324  tmp_fc_bdeffect = list()
325  for i in range(self.SOILLAYERS):
326  fc = tmp_fc[i]
327  sat = tmp_sat[i]
328  if len(self.DENSITY) != 0 and len(self.DENSITY) == self.SOILLAYERS:
329  p_df = self.DENSITY[i]
330  else:
331  p_df = 2.65 * (1. - sat)
332  sat_df = 1. - p_df / 2.65
333  tmp_fc_bdeffect.append(fc - 0.2 * (sat - sat_df))

337  self.FIELDCAP[i] = tmp_fc_bdeffect[i]

365  raise IndexError("Soil Porosity must have a size equal to soil layers number!")
366  elif not self.POROSITY:
367  for i in range(self.SOILLAYERS):
368  self.POROSITY.append(1. - self.DENSITY[i] / 2.65) # from the theory of swat

374  raise IndexError("Saturated conductivity must have a size equal to soil layers number!")

378  lamda = self.POREINDEX[i]
379  fc = tmp_fc[i]
380  sat = tmp_sat[i]
381  tmp_k.append(1930. * pow(sat - fc, 3. - lamda))
382  if not self.CONDUCTIVITY:
383  self.CONDUCTIVITY = tmp_k[:]
384  elif DEFAULT_NODATA in self.CONDUCTIVITY:
385  for i in range(self.SOILLAYERS):
386  if self.CONDUCTIVITY[i] == DEFAULT_NODATA:
387  self.CONDUCTIVITY[i] = tmp_k[i]

389  if not self.WILTINGPOINT:
390  for i in range(self.SOILLAYERS):
391  tmpwp = 0.4 * self.CLAY[i] * self.DENSITY[i] / 100.
392  if tmpwp <= 0.:
393  tmpwp = 0.005
394  self.WILTINGPOINT.append(tmpwp)
395  # calculate field capcity (sol_up)
396  if not self.FIELDCAP:
397  for i in range(self.SOILLAYERS):
398  self.FIELDCAP.append(self.WILTINGPOINT[i] + self.AWC[i])
399  # calculate porosity
400  if not self.POROSITY:
401  for i in range(self.SOILLAYERS):
402  self.POROSITY.append(1. - self.DENSITY[i] / 2.65)
403  if self.SOL_CRK == DEFAULT_NODATA:
404  self.SOL_CRK = numpy.mean(self.POROSITY)
405  for i in range(self.SOILLAYERS):
406  if self.FIELDCAP[i] >= self.POROSITY[i]:
407  self.FIELDCAP[i] = self.POROSITY[i] - 0.05
408  self.WILTINGPOINT[i] = self.FIELDCAP[i] - self.AWC[i]
409  if self.WILTINGPOINT[i] <= 0.:
410  self.FIELDCAP[i] = self.POROSITY[i] * 0.75
411  self.WILTINGPOINT[i] = self.POROSITY[i] * 0.25
412  # compute drainable porosity and variable water table factor
413  drpor = self.POROSITY[i] - self.FIELDCAP[i]
414  self.VWT.append((437.13 * drpor * drpor) - (95.08 * drpor) + 8.257)
415  sa = self.SAND[0] / 100.
416  cl = self.CLAY[0] / 100.
417  si = self.SILT[0] / 100.
418  # determine detached sediment size distribution typical for mid-western soils in USA
419  # (Foster et al., 1980) Based on SWRRB.
420  self.DET_SAND = 2.49 * sa * (1. - cl)
421  self.DET_SILT = 0.13 * si
422  self.DET_CLAY = 0.2 * cl
423  if cl < 0.25:
424  self.DET_SMAGG = 2.0 * cl
425  elif cl > 0.5:
426  self.DET_SMAGG = 0.57
427  else:
428  self.DET_SMAGG = 0.28 * (cl - 0.25) + 0.5
429  self.DET_LGAGG = 1. - self.DET_SAND - self.DET_SILT - self.DET_CLAY - self.DET_SMAGG
430  # Error check, may happen for soils with more sand. The fraction should not added up to 1.0
431  if self.DET_LGAGG < 0.:
432  self.DET_SAND /= (1. - self.DET_LGAGG)
433  self.DET_SILT /= (1. - self.DET_LGAGG)
434  self.DET_CLAY /= (1. - self.DET_LGAGG)
435  self.DET_SMAGG /= (1. - self.DET_LGAGG)
436  self.DET_LGAGG = 0.
437  # initialize water/drainage coefs for each soil layer
438  sumpor = 0.
439  self.SOL_SUMUL = 0.

516  @staticmethod
517  def usle_k_epic(sand, silt, clay, om):
518  """Calculate USLE_K factor according to EPIC (Erosion Productivity Impact Calculator).
519  (Williams et al., 1983)
520  """
521  cbn = om * 0.58
522  sn = 1. - sand * 0.01
523  a = (0.2 + 0.3 * math.exp(-0.256 * sand * (1. - silt * 0.01)))
524  b = math.pow(silt / (clay + silt), 0.3)
525  c = (1. - 0.25 * cbn / (cbn + math.exp(3.72 - 2.95 * cbn)))
526  d = (1. - 0.7 * sn / (sn + math.exp(-5.51 + 22.9 * sn)))
527  k = a * b * c * d
528  return k

569  def getsoiltexture_usda(clay, silt, sand):
570  """The soil texture code system is from WetSpa Extension and SWAT model which is
571  based on the soil texture triangle developed by USDA.
572  The unit is percentage, silt + sand + clay [+ Rock] = 100.
573  The corresponding default soil parameters (e.g. Ks, porosity) are stored in
574  `seims/database/SoilLookup.txt`.
575  Args:
576  clay: clay content percentage
577  silt: silt content percentage
578  sand: sand content percentage
579 
580  Returns:
581  [Soil texture ID, Hydrological soil group, USLE K factor]
582  """
583  if clay >= 40 >= silt and sand <= 45:
584  return [12, 1, 0.22] # clay / nian tu
585  elif clay >= 40 and silt >= 40:
586  return [11, 1, 0.26] # silt caly / fen nian tu
587  elif clay >= 35 and sand >= 45:
588  return [10, 1, 0.28] # sandy clay / sha nian tu
589  elif clay >= 25 and 20 <= sand <= 45:
590  return [9, 2, 0.3] # clay loam / nian rang tu
591  elif clay >= 25 and sand <= 20:
592  return [8, 2, 0.32] # silt clay loam / fen zhi nian rang tu
593  elif clay >= 20 and silt <= 30 and sand >= 45:
594  return [7, 2, 0.2] # sandy clay loam / sha zhi nian rang tu
595  elif clay >= 10 and 30 <= silt <= 50 and sand <= 50:
596  return [6, 3, 0.3] # loam / rang tu
597  elif 50 <= silt <= 80 or clay >= 15 and silt >= 80:
598  return [4, 3, 0.38] # silt loam / fen zhi rang tu
599  elif silt >= 80 and clay <= 15:
600  return [5, 3, 0.34] # silt / fen tu
601  elif clay <= 10 and sand <= 50 or 50 <= sand <= 80:
602  return [3, 4, 0.13] # sandy loam / sha zhi rang tu
603  elif sand <= 90:
604  return [2, 4, 0.04] # loamy sand / rang zhi sha tu
