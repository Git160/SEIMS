22  """standardize spatial database name"""
23  if is4cluster and ModelNameUtils.Cluster not in spatialdbname.lower():
24  spatialdbname = ModelNameUtils.Cluster + "_" + spatialdbname
25  if is4storm:
26  if not ModelNameUtils.Storm.lower() in spatialdbname.lower():

30  if is4cluster and (not ModelNameUtils.Cluster.lower() in spatialdbname.lower()):

32  return spatialdbname
