18 int SET_LM::Execute() {
19  CheckInputData();
20  InitialOutputs();
21 #pragma omp parallel for
22  for (int i = 0; i < m_nCells; i++) {
23  m_soilET[i] = 0.0f;
24  if (m_soilTemp[i] <= m_soilFrozenTemp) { continue; }
25 
26  float etDeficiency = m_pet[i] - m_IntcpET[i] - m_deprStoET[i] - m_maxPltET[i];
27  if (etDeficiency <= 0.f) continue;
28  for (int j = 0; j < CVT_INT(m_nSoilLyrs[i]); j++) {
29  float et2d = 0.f;
30  if (m_soilWtrSto[i][j] >= m_soilFC[i][j]) { et2d = etDeficiency; } else if (m_soilWtrSto[i][j] >= m_soilWP[i
31  ][j]) {
32  et2d = etDeficiency * (m_soilWtrSto[i][j] - m_soilWP[i][j]) / (m_soilFC[i][j] - m_soilWP[i][j]);
33  } else { et2d = 0.0f; }
34 
35  float sm0 = m_soilWtrSto[i][j];
36  float availableWater = (m_soilWtrSto[i][j] - m_soilWP[i][j]) * m_soilThk[i][j];
37  if (et2d > availableWater) {
38  et2d = availableWater;
39  m_soilWtrSto[i][j] = m_soilWP[i][j];
40  } else {
41  m_soilWtrSto[i][j] -= et2d / m_soilThk[i][j];
42  }
43 
44  if (m_soilWtrSto[i][j] < 0.f) {
45  cout << "SET_LM\t" << sm0 << "\t" << m_soilWP[i][j] << "\t" << m_soilWtrSto[i][j] << "\t"
46  << availableWater / m_soilThk[i][j] << "\t" << et2d << endl;
47  throw ModelException(MID_SET_LM, "Execute", "moisture is less than zero.");
48  }
49 
50  etDeficiency -= et2d;
51  m_soilET[i] += et2d;
52  }
53  }
54  return 0;
55 }
