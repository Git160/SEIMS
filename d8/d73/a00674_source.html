471 void NutrCH_QUAL2E::PointSourceLoading() {
473  if (nullptr == m_ptNO3ToCh) {
474  Initialize1DArray(m_nReaches + 1, m_ptNO3ToCh, 0.f);
475  Initialize1DArray(m_nReaches + 1, m_ptNH4ToCh, 0.f);
476  Initialize1DArray(m_nReaches + 1, m_ptOrgNToCh, 0.f);
477  Initialize1DArray(m_nReaches + 1, m_ptTNToCh, 0.f);
478  Initialize1DArray(m_nReaches + 1, m_ptSolPToCh, 0.f);
479  Initialize1DArray(m_nReaches + 1, m_ptOrgPToCh, 0.f);
480  Initialize1DArray(m_nReaches + 1, m_ptTPToCh, 0.f);
481  Initialize1DArray(m_nReaches + 1, m_ptCODToCh, 0.f);
482  } else {
484 #pragma omp parallel for
485  for (int i = 0; i <= m_nReaches; i++) {
486  m_ptNO3ToCh[i] = 0.f;
487  m_ptNH4ToCh[i] = 0.f;
488  m_ptOrgNToCh[i] = 0.f;
489  m_ptTNToCh[i] = 0.f;
490  m_ptSolPToCh[i] = 0.f;
491  m_ptOrgPToCh[i] = 0.f;
492  m_ptTPToCh[i] = 0.f;
493  m_ptCODToCh[i] = 0.f;
494  }
495  }
497  for (auto it = m_ptSrcFactory.begin(); it != m_ptSrcFactory.end(); ++it) {
498  //cout<<"unique Point Source Factory ID: "<<it->first<<endl;
499  vector<int>& m_ptSrcMgtSeqs = it->second->GetPointSrcMgtSeqs();
500  map<int, PointSourceMgtParams *>& m_pointSrcMgtMap = it->second->GetPointSrcMgtMap();
501  vector<int>& m_ptSrcIDs = it->second->GetPointSrcIDs();
502  map<int, PointSourceLocations *>& m_pointSrcLocsMap = it->second->GetPointSrcLocsMap();
503  // 1. looking for management operations from m_pointSrcMgtMap
504  for (auto seqIter = m_ptSrcMgtSeqs.begin(); seqIter != m_ptSrcMgtSeqs.end(); ++seqIter) {
505  PointSourceMgtParams* curPtMgt = m_pointSrcMgtMap.at(*seqIter);
506  // 1.1 If current day is beyond the date range, then continue to next management
507  if (curPtMgt->GetStartDate() != 0 && curPtMgt->GetEndDate() != 0) {
508  if (m_date < curPtMgt->GetStartDate() || m_date > curPtMgt->GetEndDate()) {
509  continue;
510  }
511  }
512  // 1.2 Otherwise, get the nutrient concentration, mg/L
513  float per_wtr = curPtMgt->GetWaterVolume();
514  float per_no3 = curPtMgt->GetNO3();
515  float per_nh4 = curPtMgt->GetNH4();
516  float per_orgn = curPtMgt->GetOrgN();
517  float per_solp = curPtMgt->GetSolP();
518  float per_orgP = curPtMgt->GetOrgP();
519  float per_cod = curPtMgt->GetCOD();
520  // 1.3 Sum up all point sources
521  for (auto locIter = m_ptSrcIDs.begin(); locIter != m_ptSrcIDs.end(); ++locIter) {
522  if (m_pointSrcLocsMap.find(*locIter) != m_pointSrcLocsMap.end()) {
523  PointSourceLocations* curPtLoc = m_pointSrcLocsMap.at(*locIter);
524  int curSubID = curPtLoc->GetSubbasinID();
525  float cvt = per_wtr * curPtLoc->GetSize() * 0.001f * m_dt * 1.1574074074074073e-05f;
527  m_ptNO3ToCh[curSubID] += per_no3 * cvt;
528  m_ptNH4ToCh[curSubID] += per_nh4 * cvt;
529  m_ptOrgNToCh[curSubID] += per_orgn * cvt;
530  m_ptOrgPToCh[curSubID] += per_orgP * cvt;
531  m_ptSolPToCh[curSubID] += per_solp * cvt;
532  m_ptCODToCh[curSubID] += per_cod * cvt;
533  m_ptTNToCh[curSubID] += (per_no3 + per_nh4 + per_orgn) * cvt;
534  m_ptTPToCh[curSubID] += (per_solp + per_orgP) * cvt;
535  }
536  }
537  }
538  }
539  // sum up point sources loadings of all subbasins
540  for (int i = 1; i <= m_nReaches; i++) {
541  m_ptTNToCh[0] += m_ptTNToCh[i];
542  m_ptTPToCh[0] += m_ptTPToCh[i];
543  m_ptCODToCh[0] += m_ptCODToCh[i];
544  }
