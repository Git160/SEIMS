96 int NPS_Management::Execute() {
97  CheckInputData();
98  if (m_cellArea < 0.f) m_cellArea = m_cellWidth * m_cellWidth;
99  for (map<int, BMPArealSrcFactory *>::iterator it = m_arealSrcFactory.begin(); it != m_arealSrcFactory.end(); it++) {
101  if (!it->second->GetLocationLoadStatus()) {
102  it->second->SetArealSrcLocsMap(m_nCells, m_mgtFields);
103  }
105  vector<int> tmpArealFieldIDs = it->second->GetArealSrcIDs();
106  vector<int> tmpMgtSeqs = it->second->GetArealSrcMgtSeqs();
107  for (vector<int>::iterator mgtSeqIter = tmpMgtSeqs.begin(); mgtSeqIter != tmpMgtSeqs.end(); mgtSeqIter++) {
108  ArealSourceMgtParams *tmpMgtParams = it->second->GetArealSrcMgtMap().at(*mgtSeqIter);
109  if (tmpMgtParams->GetStartDate() != 0 && tmpMgtParams->GetEndDate() != 0) {
110  if (m_date < tmpMgtParams->GetStartDate() || m_date > tmpMgtParams->GetEndDate()) {
111  continue;
112  }
113  }
114  for (vector<int>::iterator fIDIter = tmpArealFieldIDs.begin(); fIDIter != tmpArealFieldIDs.end();
115  fIDIter++) {
116  float deltaWtrMM = 0.f, deltaNH4 = 0.f, deltaNO3 = 0.f, deltaOrgN = 0.f;
117  float deltaMinP = 0.f, deltaOrgP = 0.f;
118  map < int, ArealSourceLocations * > tmpLocMap = it->second->GetArealSrcLocsMap();
119  if (tmpLocMap.find(*fIDIter) == tmpLocMap.end()) {
120  continue;
121  }
122  ArealSourceLocations *tmpLoc = tmpLocMap.at(*fIDIter);
123  float tmpSize = tmpLoc->GetSize();
124  int tmpNCells = tmpLoc->GetValidCells();
125  if (tmpMgtParams->GetWaterVolume() > 0.f) {
126  deltaWtrMM =
127  tmpMgtParams->GetWaterVolume() * tmpSize * m_timestep / 86400.f / tmpNCells / m_cellArea *
128  1000.f;
129  }
130  float cvt =
131  tmpSize * m_timestep / 86400.f / tmpNCells / m_cellArea * 10000.f;
132  if (tmpMgtParams->GetNH4() > 0.f) {
133  deltaNH4 = cvt * tmpMgtParams->GetNH4();
134  }
135  if (tmpMgtParams->GetNO3() > 0.f) {
136  deltaNO3 = cvt * tmpMgtParams->GetNO3();
137  }
138  if (tmpMgtParams->GetOrgN() > 0.f) {
139  deltaOrgN = cvt * tmpMgtParams->GetOrgN();
140  }
141  if (tmpMgtParams->GetMinP() > 0.f) {
142  deltaMinP = cvt * tmpMgtParams->GetMinP();
143  }
144  if (tmpMgtParams->GetOrgP() > 0.f) {
145  deltaOrgP = cvt * tmpMgtParams->GetOrgP();
146  }
147  vector<int> tmpCellIdx = tmpLoc->GetCellsIndex();
148  for (vector<int>::iterator idxIter = tmpCellIdx.begin(); idxIter != tmpCellIdx.end(); idxIter++) {
150  if (deltaWtrMM > 0.f && m_soilStorage != NULL) {
151  m_soilStorage[*idxIter][0] += deltaWtrMM;
152  }
153  if (deltaNO3 > 0.f && m_sol_no3 != NULL) {
154  m_sol_no3[*idxIter][0] += deltaNO3;
155  }
156  if (deltaNH4 > 0.f && m_sol_nh4 != NULL) {
157  m_sol_nh4[*idxIter][0] += deltaNH4;
158  }
159  if (deltaMinP > 0.f && m_sol_solp != NULL) {
160  m_sol_solp[*idxIter][0] += deltaMinP;
161  }
162  if (deltaOrgN > 0.f && m_sol_orgn != NULL) {
163  m_sol_orgn[*idxIter][0] += deltaOrgN;
164  }
165  if (deltaOrgP > 0.f && m_sol_orgp != NULL) {
166  m_sol_orgp[*idxIter][0] += deltaOrgP;
167  }
168  }
169  }
170  }
171  }
172  //cout<<"NPSMGT, cell id 5878, sol_no3[0]: "<<m_sol_no3[5878][0]<<endl;
173  //cout<<", new: "<<m_sol_solp[46364][0]<<endl;
174  return true;
