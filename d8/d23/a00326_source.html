47 def read_pareto_points_from_txt(txt_file, sce_name, xname, yname):
48  with open(txt_file) as f:
49  lines = f.readlines()
50  pareto_points = OrderedDict()
51  pareto_popnum = OrderedDict()
52  found = False
53  cur_gen = -1
54  iden_idx = -1
55  xidx = -1
56  yidx = -1
57  for line in lines:
58  str_line = line
59  for LF in LFs:
60  if LF in line:
61  str_line = line.split(LF)[0]
62  break
63  values = StringClass.extract_numeric_values_from_string(str_line)
64  # Check generation
65  if str_line[0] == '#' and 'Generation' in str_line:
66  if len(values) != 1:
67  continue
68  gen = int(values[0])
69  # if gen not in gens:
70  # found = False
71  # continue
72  found = True
73  cur_gen = gen
74  pareto_popnum[cur_gen] = list()
75  pareto_points[cur_gen] = {xname[0]: list(), yname[0]: list()}
76  continue
77  if not found:
78  continue
79  if values is None: # means header line
80  line_list = StringClass.split_string(str_line, ['\t'])
81  for idx, v in enumerate(line_list):
82  if StringClass.string_match(v, sce_name):
83  iden_idx = idx
84  break
85  for idx, v in enumerate(line_list):
86  if StringClass.string_match(v, xname[0]):
87  xidx = idx
88  if StringClass.string_match(v, yname[0]):
89  yidx = idx
90  continue
91  if xidx < 0 or yidx < 0 or iden_idx < 0:
92  continue
93  # now append the real Pareto front point data
94  pareto_points[cur_gen][xname[0]].append(values[xidx])
95  pareto_points[cur_gen][yname[0]].append(values[yidx])
96  pareto_popnum[cur_gen].append(int(values[iden_idx]))
97 
98  all_sceids = list()
99  acc_num = list()
100  genids = sorted(list(pareto_popnum.keys()))
101  for idx, genid in enumerate(genids):
102  for _id in pareto_popnum[genid]:
103  if _id not in all_sceids:
104  all_sceids.append(_id)
105  acc_num.append(len(all_sceids))
106 
107  return pareto_points, (genids, acc_num)
