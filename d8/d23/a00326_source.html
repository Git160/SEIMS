42 bool SettingsInput::readDate() {
43  //read start and end time
44  m_startDate = ConvertToTime2(Value(Tag_StartTime), "%d-%d-%d %d:%d:%d", true);
45  m_endDate = ConvertToTime2(Value(Tag_EndTime), "%d-%d-%d %d:%d:%d", true);
46 
47  if (m_startDate == -1 || m_endDate == -1) return false;
48 
49  // make sure the start and end times are in the proper order
50  if (m_endDate < m_startDate) {
51  time_t tmp = m_startDate;
52  m_startDate = m_endDate;
53  m_endDate = tmp;
54  }
55 
56  m_mode = GetUpper(Value(Tag_Mode));
57 
58  //read interval
59  vector <string> dtList = SplitString(Value(Tag_Interval), ',');
60  m_dtHs = atoi(dtList[0].c_str());
61  m_dtCh = m_dtHs;
62  if (dtList.size() > 1) {
63  m_dtCh = atoi(dtList[1].c_str());
64  }
65  // convert the time interval to seconds to conform to time_t struct
66  if (StringMatch(m_mode, Tag_Mode_Daily)) {
67  m_dtHs = 86400; // 86400 secs is 1 day
68  m_dtCh = 86400;
69  }
70  return true;
71 }

73 void SettingsInput::ReadSiteList() {
74  bson_t *query;
75  query = bson_new();
76  // subbasin id
77  BSON_APPEND_INT32(query, Tag_SubbasinId, m_subbasinID);
78  // mode
79  BSON_APPEND_UTF8(query, Tag_Mode, m_mode.c_str());
80 
81  //string siteListTable = DB_TAB_SITELIST;
82  bool stormMode = false;
83  if (StringMatch(m_mode, Tag_Mode_Storm)) {
84  stormMode = true;
85  }
86  mongoc_cursor_t *cursor;
87  mongoc_collection_t *collection;
88  const bson_t *doc;
89  collection = mongoc_client_get_collection(m_conn, m_dbName.c_str(), DB_TAB_SITELIST);
90  cursor = mongoc_collection_find(collection, MONGOC_QUERY_NONE, 0, 0, 0, query, NULL, NULL);
91  while (mongoc_cursor_next(cursor, &doc)) {
92  bson_iter_t iter;
93  if (bson_iter_init(&iter, doc) && bson_iter_find(&iter, MONG_SITELIST_DB)) {
94  m_dbHydro = GetStringFromBsonIterator(&iter);
95  } else {
96  throw ModelException("SettingsInput", "ReadSiteList", "The DB field does not exist in SiteList table.");
97  }
98  string siteList = "";
99  if (bson_iter_init(&iter, doc) && bson_iter_find(&iter, SITELIST_TABLE_M)) {
100  siteList = GetStringFromBsonIterator(&iter);
101  m_inputStation->ReadSitesData(m_dbHydro, siteList, DataType_MeanTemperature, m_startDate, m_endDate);
102  m_inputStation->ReadSitesData(m_dbHydro, siteList, DataType_MaximumTemperature, m_startDate, m_endDate);
103  m_inputStation->ReadSitesData(m_dbHydro, siteList, DataType_MinimumTemperature, m_startDate, m_endDate);
104  m_inputStation->ReadSitesData(m_dbHydro, siteList, DataType_WindSpeed, m_startDate, m_endDate);
105  m_inputStation->ReadSitesData(m_dbHydro, siteList, DataType_SolarRadiation, m_startDate, m_endDate);
106  m_inputStation->ReadSitesData(m_dbHydro, siteList, DataType_RelativeAirMoisture, m_startDate, m_endDate);
107  }
108 
109  if (bson_iter_init(&iter, doc) && bson_iter_find(&iter, SITELIST_TABLE_P)) {
110  siteList = GetStringFromBsonIterator(&iter);
111  m_inputStation->ReadSitesData(m_dbHydro, siteList, DataType_Precipitation, m_startDate, m_endDate,
112  stormMode);
113  }
114 
115  if (bson_iter_init(&iter, doc) && bson_iter_find(&iter, SITELIST_TABLE_PET)) {
116  siteList = GetStringFromBsonIterator(&iter);
117  m_inputStation->ReadSitesData(m_dbHydro, siteList, DataType_PotentialEvapotranspiration, m_startDate,
118  m_endDate, stormMode);
119  }
120  }
121  bson_destroy(query);
122  mongoc_cursor_destroy(cursor);
123  mongoc_collection_destroy(collection);
124 }

134 bool SettingsInput::LoadSettingsInputFromMongoDB() {
135  bson_t *b = bson_new();
136  bson_t *child1 = bson_new();
137  BSON_APPEND_DOCUMENT_BEGIN(b, "$query", child1);
138  bson_append_document_end(b, child1);
139  bson_destroy(child1);
140 
141  mongoc_cursor_t *cursor;
142  const bson_t *bsonTable;
143  mongoc_collection_t *collection;
144 
145  collection = mongoc_client_get_collection(m_conn, m_dbName.c_str(), DB_TAB_FILEIN);
146  cursor = mongoc_collection_find(collection, MONGOC_QUERY_NONE, 0, 0, 0, b, NULL, NULL);
147 
148  bson_iter_t itertor;
149  while (mongoc_cursor_more(cursor) && mongoc_cursor_next(cursor, &bsonTable)) {
150  vector <string> tokens(2);
151  if (bson_iter_init_find(&itertor, bsonTable, Tag_ConfTag)) {
152  tokens[0] = GetStringFromBsonIterator(&itertor);
153  }
154  if (bson_iter_init_find(&itertor, bsonTable, Tag_ConfValue)) {
155  tokens[1] = GetStringFromBsonIterator(&itertor);
156  }
157  int sz = m_Settings.size(); // get the current number of rows
158  m_Settings.resize(sz + 1); // resize with one more row
159  m_Settings[sz] = tokens;
160  }
161  bson_destroy(b);
162  mongoc_collection_destroy(collection);
163  mongoc_cursor_destroy(cursor);
164  return true;
165 }
