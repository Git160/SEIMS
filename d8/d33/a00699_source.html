128 int AET_PT_H::Execute() {
129  CheckInputData();
130  initialOutputs();
131 #pragma omp parallel for
132  for (int i = 0; i < m_nCells; i++) {
134  float esd = 0.f, etco = 0.f, effnup = 0.f;
135  float no3up = 0.f, es_max = 0.f, eos1 = 0.f, xx = 0.f;
136  float cej = 0.f, eaj = 0.f, pet = 0.f, esleft = 0.f;
137  float evzp = 0.f, eosl = 0.f, dep = 0.f, evz = 0.f, sev = 0.f;
138  pet = m_pet[i] - m_canEvp[i];
139  esd = 500.f;
140  etco = 0.8f;
141  effnup = 0.1f;
142 
143  if (pet < UTIL_ZERO) {
144  pet = 0.f;
145  m_ppt[i] = 0.f; // i.e., ep_max
146  es_max = 0.f;
147  } else {
149  if (m_lai[i] <= 3.f) {
150  m_ppt[i] = m_lai[i] * pet / 3.f;
151  } else {
152  m_ppt[i] = pet;
153  }
154  if (m_ppt[i] < 0.f) m_ppt[i] = 0.f;
156  cej = -5.e-5f;
157  eaj = 0.f;
158  es_max = 0.f;
159  eos1 = 0.f;
160  if (m_snowAcc[i] >= 0.5f) {
161  eaj = 0.5f;
162  } else {
163  eaj = exp(cej * (m_solCov[i] + 0.1f));
164  }
165  es_max = pet * eaj;
166  eos1 = pet / (es_max + m_ppt[i] + 1.e-10f);
167  eos1 = es_max * eos1;
168  es_max = min(es_max, eos1);
169  es_max = max(es_max, 0.f);
171  if (pet < es_max + m_ppt[i] && !FloatEqual(es_max + m_ppt[i], 0.f)) {
172  es_max = pet * es_max / (es_max + m_ppt[i]);
173  m_ppt[i] = pet * m_ppt[i] / (es_max + m_ppt[i]);
174  }
175  if (pet < es_max + m_ppt[i]) {
176  es_max = pet - m_ppt[i] - UTIL_ZERO;
177  }
178 
180  esleft = es_max;
182  if (m_tMean[i] > 0.f) {
183  if (m_snowAcc[i] >= esleft) {
185  m_snowAcc[i] -= esleft;
186  m_snowSB[i] += esleft;
187  esleft = 0.f;
188  } else {
190  esleft -= m_snowAcc[i];
191  m_snowSB[i] += m_snowAcc[i];
192  m_snowAcc[i] = 0.f;
193  }
194  }
195  // take soil evap from each soil layer
196  evzp = 0.f;
197  eosl = esleft;
198  for (int ly = 0; ly < (int) m_nSoilLayers[i]; ly++) {
199  dep = 0.f;
201  if (ly == 0) {
202  dep = m_soilDepth[i][ly];
203  } else {
204  dep = m_soilDepth[i][ly - 1];
205  }
206  if (dep < esd) {
208  evz = 0.f;
209  sev = 0.f;
210  xx = 0.f;
211  evz = eosl * m_soilDepth[i][ly] /
212  (m_soilDepth[i][ly] + exp(2.374f - 0.00713f * m_soilDepth[i][ly]));
213  sev = evz - evzp * m_esco[i];
214  evzp = evz;
215  if (m_soilStorage[i][ly] < m_solFC[i][ly]) {
216  xx = 2.5f * (m_soilStorage[i][ly] - m_solFC[i][ly]) / m_solFC[i][ly];
217  sev *= Expo(xx);
218  }
219  sev = min(sev, m_soilStorage[i][ly] * etco);
220  if (sev < 0.f || sev != sev) sev = 0.f;
221  if (sev > esleft) sev = esleft;
223  if (m_soilStorage[i][ly] > sev) {
224  esleft -= sev;
225  m_soilStorage[i][ly] = max(UTIL_ZERO, m_soilStorage[i][ly] - sev);
226  } else {
227  esleft -= m_soilStorage[i][ly];
228  m_soilStorage[i][ly] = 0.f;
229  }
230  }
232  if (ly == 1)
233  {
234  no3up = 0.f;
235  no3up = effnup * sev * m_solNo3[i][ly] / (m_soilStorage[i][ly] + UTIL_ZERO);
236  no3up = min(no3up, m_solNo3[i][ly]);
237  m_no3Up += no3up / m_nCells;
238  m_solNo3[i][ly] -= no3up;
239  m_solNo3[i][ly - 1] += no3up;
240  }
241  }
243  m_soilStorageProfile[i] = 0.f;
244  for (int ly = 0; ly < (int) m_nSoilLayers[i]; ly++) {
245  m_soilStorageProfile[i] += m_soilStorage[i][ly];
246  }
248  //if (esleft != esleft || es_max != es_max)
249  // cout<<"esleft: "<<esleft<<", es_max: "<<es_max<<endl;
250  if (es_max > esleft) {
251  m_soilESDay[i] = es_max - esleft;
252  } else {
253  m_soilESDay[i] = 0.f;
254  }
255  }
256  }
257  //cout<<"AET_PTH, cell id 5878, sol_no3[0]: "<<m_solNo3[5878][0]<<endl;
258  return true;
259 }
