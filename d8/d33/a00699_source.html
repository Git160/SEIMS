49  dic = {}

51  if MathClass.isnumerical(item[i]):
52  dic[fieldArray[i].upper()] = float(item[i])
53  else:
54  dic[fieldArray[i].upper()] = str(item[i]).upper()
55  if StringClass.stringinlist(Tag_ST_LocalX, dic.keys()) and StringClass.stringinlist(Tag_ST_LocalY,
56  dic.keys()):

58  dic[Tag_ST_LocalX.upper()], dic[Tag_ST_LocalY.upper()])
