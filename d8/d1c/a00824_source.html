16 int SET_LM::Execute() {
17  CheckInputData();
18  initialOutputs();
19 #pragma omp parallel for
20  for (int i = 0; i < m_nCells; i++) {
21  m_soilET[i] = 0.0f;
22  if (m_soilT[i] <= m_frozenT) { continue; }
23 
24  float etDeficiency = m_PET[i] - m_EI[i] - m_ED[i] - m_plantET[i];
25  if (etDeficiency <= 0.f) continue;
26  for (int j = 0; j < (int) m_soilLayers[i]; j++) {
27  float et2d = 0.f;
28  if (m_sm[i][j] >= m_fc[i][j]) { et2d = etDeficiency; }
29  else if (m_sm[i][j] >= m_wp[i][j]) {
30  et2d = etDeficiency * (m_sm[i][j] - m_wp[i][j]) / (m_fc[i][j] - m_wp[i][j]);
31  } else { et2d = 0.0f; }
32 
33  float sm0 = m_sm[i][j];
34  float availableWater = (m_sm[i][j] - m_wp[i][j]) * m_soilThick[i][j];
35  if (et2d > availableWater) {
36  et2d = availableWater;
37  m_sm[i][j] = m_wp[i][j];
38  } else {
39  m_sm[i][j] -= et2d / m_soilThick[i][j];
40  }
41 
42  if (m_sm[i][j] < 0.f) {
43  cout << "SET_LM\t" << sm0 << "\t" << m_wp[i][j] << "\t" << m_sm[i][j] << "\t"
44  << availableWater / m_soilThick[i][j] << "\t" << et2d << endl;
45  throw ModelException(MID_SET_LM, "Execute", "moisture is less than zero.");
46  }
47 
48  etDeficiency -= et2d;
49  m_soilET[i] += et2d;
50  }
51  }
52  return 0;
53 }
