21 int PER_STR::Execute() {
22  CheckInputData();
23  initialOutputs();
24 #pragma omp parallel for
25  for (int i = 0; i < m_nCells; i++) {
26  // Note that, infiltration, pothole seepage, irrigation etc. have been added to
27  // the first soil layer in other modules. By LJ
28  float excessWater = 0.f, maxSoilWater = 0.f, fcSoilWater = 0.f;
29  for (int j = 0; j < (int) m_soilLayers[i]; j++) {
30  excessWater = 0.f;
31  maxSoilWater = m_sat[i][j];
32  fcSoilWater = m_fc[i][j];
33  // determine gravity drained water in layer
34  excessWater += m_soilStorage[i][j] - fcSoilWater;
35  //if (i == 100)
36  // cout<<"lyr: "<<j<<", soil storage: "<<m_soilStorage[i][j]<<", fc: "<<fcSoilWater<<", excess: "<<excessWater<<endl;
37  // for the upper two layers, soil may be frozen
38  if (j == 0 && m_soilT[i] <= m_frozenT) {
39  continue;
40  }
41  m_perc[i][j] = 0.f;
42  // No movement if soil moisture is below field capacity
43  if (excessWater > 1.e-5f) {
44  float maxPerc = maxSoilWater - fcSoilWater;
45  if (maxPerc < 0.f) maxPerc = 0.1f;
46  float tt = 3600.f * maxPerc / m_ks[i][j]; // secs
47  m_perc[i][j] = excessWater * (1.f - exp(-m_dt / tt)); // secs
48 
49  if (m_perc[i][j] > maxPerc) {
50  m_perc[i][j] = maxPerc;
51  }
52  //Adjust the moisture content in the current layer, and the layer immediately below it
53  m_soilStorage[i][j] -= m_perc[i][j];
54  excessWater -= m_perc[i][j];
55  m_soilStorage[i][j] = max(UTIL_ZERO, m_soilStorage[i][j]);
56  // redistribute soil water if above field capacity (high water table), rewrite from sat_excess.f of SWAT
57  //float qlyr = m_soilStorage[i][j];
58  if (j < (int) m_soilLayers[i] - 1) {
59  m_soilStorage[i][j + 1] += m_perc[i][j];
60  if (m_soilStorage[i][j] - m_sat[i][j] > 1.e-4f) {
61  m_soilStorage[i][j + 1] += m_soilStorage[i][j] - m_sat[i][j];
62  m_soilStorage[i][j] = m_sat[i][j];
63  }
64  } else {
65  if (m_soilStorage[i][j] - m_sat[i][j] > 1.e-4f) {
66  float ul_excess = m_soilStorage[i][j] - m_sat[i][j];
67  m_soilStorage[i][j] = m_sat[i][j];
68  for (int ly = (int) m_soilLayers[i] - 2; ly >= 0; ly--) {
69  m_soilStorage[i][ly] += ul_excess;
70  if (m_soilStorage[i][ly] > m_sat[i][ly]) {
71  ul_excess = m_soilStorage[i][ly] - m_sat[i][ly];
72  m_soilStorage[i][ly] = m_sat[i][ly];
73  } else {
74  ul_excess = 0.f;
75  break;
76  }
77  if (ly == 0 && ul_excess > 0.f) {
78  // add ul_excess to depressional storage and then to surfq
79  if (m_potVol != NULL) {
80  m_potVol[i] += ul_excess;
81  } else {
82  m_surfQmm[i] += ul_excess;
83  }
84  }
85  }
86  }
87  }
88  } else {
89  m_perc[i][j] = 0.f;
90  }
91  }
93  m_soilStorageProfile[i] = 0.f;
94  for (int ly = 0; ly < (int) m_soilLayers[i]; ly++) {
95  m_soilStorageProfile[i] += m_soilStorage[i][ly];
96  }
97  }
98  // DEBUG
99  //cout << "PER_STR, cell id 14377, m_soilStorage: ";
100  //for (int i = 0; i < (int)m_soilLayers[14377]; i++)
101  // cout << m_soilStorage[14377][i] << ", ";
102  //cout << endl;
103  // END OF DEBUG
104  return 0;
105 }
