577 bool MGTOpt_SWAT::GetOperationCode(const int i, const int factoryID, vector<int>& nOps) {
579  vector<int>& tmpOpSeqences = m_mgtFactory[factoryID]->GetOperationSequence();
580  map<int, PltMgtOp *>& tmpOperations = m_mgtFactory[factoryID]->GetOperations();
581  // get the next should be done sequence number
582  int nextSeq = -1;
583  if (m_doneOpSequence[i] == -1 || m_doneOpSequence[i] == CVT_INT(tmpOpSeqences.size()) - 1) {
584  nextSeq = 0;
585  } else {
586  nextSeq = m_doneOpSequence[i] + 1;
587  }
588  int opCode = tmpOpSeqences[nextSeq];
589  // figure out the nextSeq is satisfied or not.
590  if (tmpOperations.find(opCode) == tmpOperations.end()) {
591  return false;
592  }
593  PltMgtOp* tmpOperation = tmpOperations.at(opCode);
594  bool dateDepent = false;
595  bool huscDepent = false;
597  if (m_month == tmpOperation->GetMonth() && m_day == tmpOperation->GetDay()) {
598  dateDepent = true;
599  }
601  if (tmpOperation->GetHUFraction() >= 0.f) {
602  float aphu = NODATA_VALUE;
603  if (!FloatEqual(m_dormFlag[i], 1.f)) {
604  if (tmpOperation->UseBaseHUSC() && FloatEqual(m_igro[i], 0.f)) {
605  // use base hu
606  aphu = m_phuBase[i];
607  if (aphu >= tmpOperation->GetHUFraction()) {
608  huscDepent = true;
609  }
610  } else {
611  // use accumulated plant hu
612  aphu = m_phuAccum[i];
613  if (aphu >= tmpOperation->GetHUFraction()) {
614  huscDepent = true;
615  }
616  }
617  }
618  }
621  if (dateDepent || huscDepent) {
622  nOps.emplace_back(opCode);
623  m_doneOpSequence[i] = nextSeq;
624  }
625  return !nOps.empty();

743 void MGTOpt_SWAT::ExecuteIrrigationOperation(const int i, const int factoryID, const int nOp) {
744  IrrOp* curOperation = static_cast<IrrOp *>(m_mgtFactory[factoryID]->GetOperations().at(nOp));
747  int m_irrSource = curOperation->IRRSource();
749  int m_irrNo = curOperation->IRRNo() <= 0 ? CVT_INT(m_subbsnID[i]) : curOperation->IRRNo();
751  float m_irrApplyDepth = curOperation->IRRApplyDepth();
754  float m_irrEfficiency = curOperation->IRREfficiency();
755 
756  m_irrFlag[i] = 1;
757  int tmpSubbsnID = CVT_INT(m_subbsnID[i]);
758  if (m_irrSource > IRR_SRC_RES) {
762  float vmma = 0.f;
763  float vmm = 0.f;
764  float cnv = 0.f;
765  float vmxi = 0.f;
766  float vol = 0.f;
767  float vmms = 0.f;
768  float vmmd = 0.f;
769  if (m_nCellsSubbsn.find(m_irrNo) == m_nCellsSubbsn.end()) {
771  m_irrNo = CVT_INT(m_subbsnID[i]);
772  }
773 
774  cnv = m_nAreaSubbsn[m_irrNo] * 10.f;
775  switch (m_irrSource) {
777  case IRR_SRC_SHALLOW:
778  if (m_shallowWaterDepth[tmpSubbsnID] < UTIL_ZERO) {
779  m_shallowWaterDepth[tmpSubbsnID] = 0.f;
780  }
781  vmma += m_shallowWaterDepth[tmpSubbsnID] * cnv * m_irrEfficiency;
782  vmms = vmma;
783  vmma /= m_nCellsSubbsn[m_irrNo];
784  vmm = Min(m_soilSumFC[i], vmma);
785  break;
786  case IRR_SRC_DEEP: vmma += m_deepWaterDepth[tmpSubbsnID] * cnv * m_irrEfficiency;
787  vmmd = vmma;
788  vmma /= m_nCellsSubbsn[m_irrNo];
789  vmm = Min(m_soilSumFC[i], vmma);
790  break;
791  case IRR_SRC_OUTWTSD:
792  vmm = m_soilSumFC[i];
793  break;
794  default: break;
795  }
797  if (vmm > 0.f) {
798  cnv = m_cellArea * 10.f;
799  vmxi = m_irrApplyDepth < UTIL_ZERO ? m_soilSumFC[i] : m_irrApplyDepth;
800  if (vmm > vmxi) vmm = vmxi;
801  vol = vmm * cnv;
802  float pot_fr = 0.f;
803  if (FloatEqual(m_impndTrig[i], 0.f) && m_potVol != nullptr) {
806  pot_fr = 1.f;
807  if (m_potArea != nullptr) {
808  m_potVol[i] += vol / (10.f * m_potArea[i]);
809  } else {
810  m_potVol[i] += vol / (10.f * m_cellArea);
811  }
812  m_irrWtrAmt[i] = vmm;
813  } else {
814  pot_fr = 0.f;
816  m_irrWtrAmt[i] = vmm * (1.f - curOperation->IRRSQfrac());
817  m_irrWtr2SurfqAmt[i] = vmm * curOperation->IRRSQfrac();
818  }
820  if (pot_fr > UTIL_ZERO) {
821  vol = m_irrWtrAmt[i] * cnv * m_irrEfficiency;
822  }
823  switch (m_irrSource) {
824  case IRR_SRC_SHALLOW: cnv = m_nAreaSubbsn[m_irrNo] * 10.f;
825  vmma = 0.f;
826  if (vmms > -0.01f) {
827  vmma = vol * m_shallowWaterDepth[tmpSubbsnID] * cnv / vmms;
828  }
829  vmma /= cnv;
830  m_shallowWaterDepth[tmpSubbsnID] -= vmma;
831  if (m_shallowWaterDepth[tmpSubbsnID] < 0.f) {
832  vmma += m_shallowWaterDepth[tmpSubbsnID];
833  m_shallowWaterDepth[tmpSubbsnID] = 0.f;
834  }
835  m_shallowIrrWater[i] += vmma;
836  break;
837  case IRR_SRC_DEEP: cnv = m_nAreaSubbsn[m_irrNo] * 10.f;
838  vmma = 0.f;
839  if (vmmd > 0.01f) {
840  vmma = vol * (m_deepWaterDepth[tmpSubbsnID] * cnv / vmmd);
841  }
842  vmma /= cnv;
843  m_deepWaterDepth[tmpSubbsnID] -= vmma;
844  if (m_deepWaterDepth[tmpSubbsnID] < 0.f) {
845  vmma += m_deepWaterDepth[tmpSubbsnID];
846  m_deepWaterDepth[tmpSubbsnID] = 0.f;
847  }
848  m_deepIrrWater[i] += vmma;
849  break;
850  default: break;
851  }
852  }

900  //float fertN = 0.f;
903  //float fertP = 0.f;
904  //float fertSolP = 0.f;
906  //float cFertN = 0.f;
908  //float cFertP = 0.f;
911  float rtof = 0.5f;
912  float xx = 0.f;
913  float gc = 0.f; //, gc1 = 0.f;
915  int lyrs = 2;
916  if (m_potVol != nullptr) {
917  if (FloatEqual(CVT_INT(m_landCover[i]), CROP_PADDYRICE) && fertype == 0 && m_potVol[i] > 0.f) {
918  lyrs = 1;

933  m_soilActvOrgN[i][l] += (1.f - rtof) * xx * fertilizerKgHa * fertOrgN;
934  m_soilFrshOrgP[i][l] += rtof * xx * fertilizerKgHa * fertOrgP;
935  m_soilHumOrgP[i][l] += (1.f - rtof) * xx * fertilizerKgHa * fertOrgP;

940  m_soilManP[i][l] += xx * fertilizerKgHa * fertOrgP;
941  } else if (m_cbnModel == 2) {

948  m_soilHumOrgP[i][l] += (1.f - rtof) * xx * fertilizerKgHa * fertOrgP;
950  m_soilHSN[i][l] += (1.f - rtof) * xx * fertilizerKgHa * fertOrgN;

959  X10 = 0.85f - 0.018f * RLN;
960  if (X10 < 0.01f) { X10 = 0.01f; } else if (X10 > 0.7f) X10 = 0.7f;
961 
963  XXX = X8 * X10;
964  m_soilLMC[i][l] += XXX;
966  YY = X1 * X10;
967  m_soilLM[i][l] += YY;
969  ZZ = X1 * rtof * fertOrgN * X10;
970  m_soilLMN[i][l] += ZZ;
971
