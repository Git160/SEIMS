37 void SUR_MR:: InitialOutputs() {
38  CHECK_POSITIVE(MID_SUR_MR, m_nCells);
39  // allocate the output variables
40  if (nullptr == m_pe) {
41  Initialize1DArray(m_nCells, m_pe, 0.f);
42  Initialize1DArray(m_nCells, m_infil, 0.f);
43  Initialize1DArray(m_nCells, m_soilStorageProfile, 0.f);
44  Initialize2DArray(m_nCells, m_nSoilLayers, m_soilStorage, NODATA_VALUE);
45 #pragma omp parallel for
46  for (int i = 0; i < m_nCells; i++) {
47  for (int j = 0; j < (int) m_soilLayers[i]; j++) {
48  if (m_initSoilStorage[i] >= 0.f && m_initSoilStorage[i] <= 1.f && m_sol_awc[i][j] >= 0.f) {
49  m_soilStorage[i][j] = m_initSoilStorage[i] * m_sol_awc[i][j];
50  } else {
51  m_soilStorage[i][j] = 0.f;
52  }
53  m_soilStorageProfile[i] += m_soilStorage[i][j];
54  }
55  }
56  }
58  if (m_sol_sumsat == NULL && m_sol_ul != NULL) {
59  m_sol_sumsat = new float[m_nCells];
60 #pragma omp parallel for
61  for (int i = 0; i < m_nCells; i++) {
62  m_sol_sumsat[i] = 0.f;
63  for (int j = 0; j < (int)m_soilLayers[i]; j++) {
64  m_sol_sumsat[i] += m_sol_ul[i][j];
65  }
66  }
67  }
68 }

70 int SUR_MR::Execute() {
71  CheckInputData();
72  InitialOutputs();
73  m_pMax = m_pMax * m_dt / 86400.f;
74 #pragma omp parallel for
75  for (int i = 0; i < m_nCells; i++) {
76  float hWater = 0.f;
77  hWater = m_pNet[i] + m_sd[i];
78  if (hWater > 0.f) {
80  m_soilStorageProfile[i] = 0.f;
81  for (int ly = 0; ly < (int) m_soilLayers[i]; ly++) {
82  m_soilStorageProfile[i] += m_soilStorage[i][ly];
83  }
84  float smFraction = Min(m_soilStorageProfile[i] / m_sol_sumsat[i], 1.f);
85  // for frozen soil, no infiltration will occur
86  if (m_soilTemp[i] <= m_tFrozen && smFraction >= m_sFrozen) {
87  m_pe[i] = m_pNet[i];
88  m_infil[i] = 0.f;
89  } else {
90  float alpha = m_kRunoff - (m_kRunoff - 1.f) * hWater / m_pMax;
91  if (hWater >= m_pMax) {
92  alpha = 1.f;
93  }
94 
95  //runoff percentage
96  float runoffPercentage;
97  if (m_runoffCo[i] > 0.99f) {
98  runoffPercentage = 1.f;
99  } else {
100  runoffPercentage = m_runoffCo[i] * pow(smFraction, alpha);
101  }
102 
103  float surfq = hWater * runoffPercentage;
104  if (surfq > hWater) surfq = hWater;
105  m_infil[i] = hWater - surfq;
106  m_pe[i] = surfq;
107 
111  }
112  } else {
113  m_pe[i] = 0.f;
114  m_infil[i] = 0.f;
115  }
116  if (m_infil[i] > 0.f) {
117  m_soilStorage[i][0] += m_infil[i];
118  }
119  }
120  return 0;
121 }
