278  @staticmethod
279  def calculate_latitude_dependent_parameters(lat_file, min_dayl_file, dormhr_file):
280  """
281  Calculate latitude dependent parameters, include:
282  1. minimum daylength (daylmn), 2. day length threshold for dormancy (dormhr)
283  """
284  # calculate minimum daylength, from readwgn.f of SWAT
285  # daylength=2*acos(-tan(sd)*tan(lat))/omega
286  # where solar declination, sd, = -23.5 degrees for minimum daylength in
287  # northern hemisphere and -tan(sd) = .4348
288  # absolute value is taken of tan(lat) to convert southern hemisphere
289  # values to northern hemisphere
290  # the angular velocity of the earth's rotation, omega, = 15 deg/hr or
291  # 0.2618 rad/hr and 2/0.2618 = 7.6394
292  cellLatR = RasterUtilClass.ReadRaster(lat_file)
293  latData = cellLatR.data
294  # daylmnData = cellLatR.data
295  zero = numpy.zeros((cellLatR.nRows, cellLatR.nCols))
296  # nodata = numpy.ones((cellLatR.nRows, cellLatR.nCols)) * cellLatR.noDataValue
297  # convert degrees to radians (2pi/360=1/57.296)
298  daylmnData = 0.4348 * numpy.abs(numpy.tan(latData / 57.296))
299  condition = daylmnData < 1.
300  daylmnData = numpy.where(condition, numpy.arccos(daylmnData), zero)
301  # condition2 = latData != cellLatR.noDataValue
302  daylmnData *= 7.6394
303  daylmnData = numpy.where(cellLatR.validZone, daylmnData, latData)
304  RasterUtilClass.WriteGTiffFile(min_dayl_file, cellLatR.nRows, cellLatR.nCols,
305  daylmnData, cellLatR.geotrans, cellLatR.srs, cellLatR.noDataValue, GDT_Float32)
306 
307  # calculate day length threshold for dormancy
308  def calDormHr(lat):
309  if lat == cellLatR.noDataValue:
310  return cellLatR.noDataValue
311  else:
312  if 20. <= lat <= 40.:
313  return (numpy.abs(lat - 20.)) / 20.
314  elif lat > 40.:
315  return 1.
316  elif lat < 20.:
317  return -1.
318 
319  calDormHr_numpy = numpy.frompyfunc(calDormHr, 1, 1)
320 
321  # dormhrData = numpy.copy(latData)
322  if dorm_hr < -UTIL_ZERO:
323  dormhrData = calDormHr_numpy(latData)
324  else:
325  dormhrData = numpy.where(cellLatR.validZone, numpy.ones((cellLatR.nRows, cellLatR.nCols)) * dorm_hr,
326  latData)
327  RasterUtilClass.WriteGTiffFile(dormhr_file, cellLatR.nRows, cellLatR.nCols, dormhrData, cellLatR.geotrans,
328  cellLatR.srs, cellLatR.noDataValue, GDT_Float32)
