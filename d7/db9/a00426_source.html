30 clsReach::clsReach(const bson_t*& bson_table) {
31  bson_iter_t iterator;
32  // read numerical parameters
33  for (int i = 0; i < REACH_PARAM_NUM; i++) {
34  float tmp_param;
35  if (bson_iter_init_find(&iterator, bson_table, REACH_PARAM_NAME[i])) {
36  if (GetNumericFromBsonIterator(&iterator, tmp_param)) {
37  if (REACH_PARAM_NAME[i] == REACH_BOD && tmp_param < 1.e-6f) tmp_param = 1.e-6f;
38  param_map_.insert(make_pair(REACH_PARAM_NAME[i], tmp_param));
39  }
40  }
41  }
42  // read group informations, DO NOT THROW EXCEPTION!
43  if (!bson_iter_init_find(&iterator, bson_table, REACH_GROUP)) {
44  cout << "No GROUP field found!" << endl;
45  return;
46  }
47  if (!SplitStringForValues(GetStringFromBsonIterator(&iterator), ',', group_number_)) {
48  cout << "Split for group numbers failed!" << endl;
49  return;
50  }
51  for (int i = 0; i < REACH_GROUP_METHOD_NUM; i++) {
52  if (!bson_iter_init_find(&iterator, bson_table, REACH_GROUP_NAME[i])) continue;
53  vector<int> g_idx;
54  if (!SplitStringForValues(GetStringFromBsonIterator(&iterator), ',', g_idx)) {
55  cout << "Split for group indexes failed!" << endl;
56  return;
57  }
58  if (group_number_.size() != g_idx.size()) {
59  cout << "The size of group method " << REACH_GROUP_NAME[i] << " is not equal to GROUP! " << endl;
60  continue;
61  }
62  map<int, int> group_index;
63  for (int j = 0; j < group_number_.size(); j++) {
64  group_index.insert(make_pair(group_number_[j], g_idx[j]));
65  }
66  if (group_index.empty()) continue;
67  group_index_.insert(make_pair(REACH_GROUP_NAME[i], group_index));
68  }
69 }

94 clsReaches::clsReaches(MongoClient* conn, const string& db_name,
95  const string& collection_name, const LayeringMethod mtd /* = UP_DOWN */) {
96  bson_t* b = bson_new();
97  bson_t* child1 = bson_new();
98  bson_t* child2 = bson_new();
99  BSON_APPEND_DOCUMENT_BEGIN(b, "$query", child1);
100  bson_append_document_end(b, child1);
101  BSON_APPEND_DOCUMENT_BEGIN(b, "$orderby", child2);
102  BSON_APPEND_INT32(child2, REACH_SUBBASIN, 1);
103  bson_append_document_end(b, child2);
104  bson_destroy(child1);
105  bson_destroy(child2);
106 
107  std::unique_ptr<MongoCollection> collection(new MongoCollection(conn->GetCollection(db_name, collection_name)));
108  reach_num_ = collection->QueryRecordsCount();
109  if (reach_num_ < 0) {
110  bson_destroy(b);
111  // TODO, No exception should be thrown during constructing a class, considering Init function. lj.
112  throw ModelException("clsReaches", "ReadAllReachInfo",
113  "Failed to get document number of collection: " + collection_name + ".\n");
114  }
115  reach_up_streams_.resize(reach_num_ + 1);
116 
117  mongoc_cursor_t* cursor = collection->ExecuteQuery(b);
118  const bson_t* bson_table;
119  while (mongoc_cursor_more(cursor) && mongoc_cursor_next(cursor, &bson_table)) {
120  clsReach* cur_reach = new clsReach(bson_table);
121  int sub_id = int(cur_reach->Get(REACH_SUBBASIN));
122  reaches_obj_.insert(make_pair(sub_id, cur_reach));
123  }
124  // In SEIMS, reach ID is the same as Index of array and vector.
125  for (int i = 1; i <= reach_num_; i++) {
126  int down_stream_id = int(reaches_obj_.at(i)->Get(REACH_DOWNSTREAM));
127  reach_down_stream_.insert(make_pair(i, down_stream_id));
128  if (down_stream_id <= 0 || down_stream_id > reach_num_) continue;
129  reach_up_streams_[down_stream_id].push_back(i);
130  }
131  // Build layers of reaches according to layering method
132  reach_layers_.clear();
133  for (int i = 1; i <= reach_num_; i++) {
134  int order = int(reaches_obj_.at(i)->Get(REACH_UPDOWN_ORDER));
135  if (mtd == DOWN_UP) { order = int(reaches_obj_.at(i)->Get(REACH_DOWNUP_ORDER)); }
136  if (reach_layers_.find(order) == reach_layers_.end()) {
137  reach_layers_.insert(make_pair(order, vector<int>()));
138  }
139  reach_layers_.at(order).push_back(i);
140  }
141 
142  bson_destroy(b);
143  mongoc_cursor_destroy(cursor);
144 }

184 void clsReaches::Update(const map<string, ParamInfo *>& caliparams_map) {
185  for (int i = 0; i < REACH_PARAM_NUM; i++) {
186  auto it = caliparams_map.find(REACH_PARAM_NAME[i]);
187  if (it != caliparams_map.end()) {
188  ParamInfo* tmp_param = it->second;
189  if (StringMatch(tmp_param->Change, PARAM_CHANGE_RC) && FloatEqual(tmp_param->Impact, 1.f) ||
190  StringMatch(tmp_param->Change, PARAM_CHANGE_AC) && FloatEqual(tmp_param->Impact, 0.f) ||
191  StringMatch(tmp_param->Change, PARAM_CHANGE_VC) && FloatEqual(tmp_param->Impact, NODATA_VALUE) ||
192  StringMatch(tmp_param->Change, PARAM_CHANGE_NC)) {
193  continue;
194  }
195  for (auto it2 = reaches_obj_.begin(); it2 != reaches_obj_.end(); ++it2) {
196  float pre_value = it2->second->Get(REACH_PARAM_NAME[i]);
197  if (FloatEqual(pre_value, NODATA_VALUE)) continue;
198  float new_value = it->second->GetAdjustedValue(pre_value);
199  it2->second->Set(REACH_PARAM_NAME[i], new_value);
