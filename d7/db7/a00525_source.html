108 int IUH_NUTR_OL::Execute() {
109  CheckInputData();
110 
111  initialOutputs();
112  // delete value of last time step
113 #pragma omp parallel for
114  for (int i = 0; i < m_nSubbasins + 1; i++) {
115  m_sedtoCh[i] = 0.f;
116  }
117 
118  int nt = 0;
119  float qs_cell = 0.f;
120  //float area = m_CellWidth * m_CellWidth;
121 
122 #pragma omp parallel for
123  for (int i = 0; i < m_nCells; i++) {
124  //forward one time step
125  for (int j = 0; j < m_cellFlowCols; j++) {
126  if (j != m_cellFlowCols - 1) {
127  m_cellSed[i][j] = m_cellSed[i][j + 1];
128  } else {
129  m_cellSed[i][j] = 0.f;
130  }
131  }
132 
133  if (m_sedYield[i] > 0.f) {
134  int min = int(this->m_iuhCell[i][0]);
135  int max = int(this->m_iuhCell[i][1]);
136  int col = 2;
137  for (int k = min; k <= max; k++) {
138  m_cellSed[i][k] += m_sedYield[i] * m_iuhCell[i][col];
139  col++;
140  }
141  }
142  }
143 
144  for (int i = 0; i < m_nCells; i++) {
145  //add today's flow
146  int subi = (int) m_subbasin[i];
147  if (m_nSubbasins == 1) {
148  subi = 1;
149  } else if (subi >= m_nSubbasins + 1) {
150  ostringstream oss;
151  oss << subi;
152  throw ModelException(MID_IUH_NUTR_OL, "Execute", "The subbasin " + oss.str() + " is invalid.");
153  }
154  m_sedtoCh[subi] += m_cellSed[i][0];
155 
156  m_sedOL[i] = m_cellSed[i][0];
157  }
158 
159  float tmp = 0.f;
160 #pragma omp parallel for reduction(+:tmp)
161  for (int i = 1; i < m_nSubbasins + 1; i++) {
162  tmp += m_sedtoCh[i]; //get overland flow routing for entire watershed.
163  }
164  m_sedtoCh[0] = tmp;
165 
166  return 0;
167 }
