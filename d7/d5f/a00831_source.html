291  def calculate_latitude_dependent_parameters(lat_file, min_dayl_file, dormhr_file, dorm_hr):
292  """
293  Calculate latitude dependent parameters, include:
294  1. minimum daylength (daylmn), 2. day length threshold for dormancy (dormhr)
295  """
296  # calculate minimum daylength, from readwgn.f of SWAT
297  # daylength=2*acos(-tan(sd)*tan(lat))/omega
298  # where solar declination, sd, = -23.5 degrees for minimum daylength in
299  # northern hemisphere and -tan(sd) = .4348
300  # absolute value is taken of tan(lat) to convert southern hemisphere
301  # values to northern hemisphere
302  # the angular velocity of the earth's rotation, omega, = 15 deg/hr or
303  # 0.2618 rad/hr and 2/0.2618 = 7.6394
304  cell_lat_r = RasterUtilClass.read_raster(lat_file)
305  lat_data = cell_lat_r.data
306  # daylmn_data = cell_lat_r.data
307  zero = numpy.zeros((cell_lat_r.nRows, cell_lat_r.nCols))
308  # nodata = numpy.ones((cell_lat_r.nRows, cell_lat_r.nCols)) * cell_lat_r.noDataValue
309  # convert degrees to radians (2pi/360=1/57.296)
310  daylmn_data = 0.4348 * numpy.abs(numpy.tan(lat_data / 57.296))
311  condition = daylmn_data < 1.
312  daylmn_data = numpy.where(condition, numpy.arccos(daylmn_data), zero)
313  # condition2 = lat_data != cell_lat_r.noDataValue
314  daylmn_data *= 7.6394
315  daylmn_data = numpy.where(cell_lat_r.validZone, daylmn_data, lat_data)
316  RasterUtilClass.write_gtiff_file(min_dayl_file, cell_lat_r.nRows, cell_lat_r.nCols,
317  daylmn_data, cell_lat_r.geotrans, cell_lat_r.srs,
318  cell_lat_r.noDataValue, GDT_Float32)
