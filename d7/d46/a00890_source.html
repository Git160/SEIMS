258 void CellOrdering::BuildTree(void) // tree of the land use
259 {
260  int *dir = m_dir->getRasterDataPointer();
261  int *LanduCode = m_landu->getRasterDataPointer();
262  for (int i = 0; i < m_nRows; ++i) {
263  for (int j = 0; j < m_nCols; ++j) {
264  if (m_mask->isNoData(RowColCoor(i, j))) {
265  continue;
266  }
267  int id = i * m_nCols + j;
268  // flow out
269  int k = dir[id];
270  int outIndex = m_dirToIndexMap[dir[id]];
271  int iOut = i + m_d1[outIndex];
272  int jOut = j + m_d2[outIndex];
273  int idOut = iOut * m_nCols + jOut;
274  m_cells[id]->SetID(id);
275  // set landuse information to cells
276  int landu = LanduCode[id];
277  m_cells[id]->SetLanduseCode(landu);
278 
279  if (iOut < 0 || iOut >= m_nRows || jOut < 0 || jOut >= m_nCols || m_cells[idOut] == NULL) {
280  m_cells[id]->SetOutCellID(-1);
281  } else {
282  m_cells[id]->SetOutCellID(idOut);
283  m_cells[idOut]->AddInCellID(id);
284  }
285  }
286  }
287  cout << "\t\t\tTotally " << m_cells.size() << " cells has been build!" << endl;
