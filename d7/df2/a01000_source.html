32 void Scenario::loadScenarioName() {
33  vector<string>::iterator it = find(m_bmpCollections.begin(), m_bmpCollections.end(),
34  string(TAB_BMP_SCENARIO));
35  if (it == m_bmpCollections.end()) {
36  throw ModelException("BMP Scenario", "loadScenarioName", "The BMP database '" + m_bmpDBName +
37  "' does not exist or there is not a table named '" +
38  TAB_BMP_SCENARIO + "' in BMP database.");
39  }
40  mongoc_collection_t *sceCollection = m_conn->getCollection(m_bmpDBName, TAB_BMP_SCENARIO);
42  bson_t *query = bson_new(), *reply = bson_new();
43  query = BCON_NEW("distinct", BCON_UTF8(TAB_BMP_SCENARIO), "key", FLD_SCENARIO_NAME,
44  "query", "{", FLD_SCENARIO_ID, BCON_INT32(m_sceneID), "}");
45  bson_iter_t iter, sub_iter;
46  bson_error_t *err = NULL;
47  if (mongoc_collection_command_simple(sceCollection, query, NULL, reply, err)) {
48  //cout<<bson_as_json(reply,NULL)<<endl;
49  if (bson_iter_init_find(&iter, reply, "values") &&
50  (BSON_ITER_HOLDS_DOCUMENT(&iter) || BSON_ITER_HOLDS_ARRAY(&iter)) &&
51  bson_iter_recurse(&iter, &sub_iter)) {
52  while (bson_iter_next(&sub_iter)) {
53  m_name = GetStringFromBsonIterator(&sub_iter);
54  break;
55  }
56  } else {
57  throw ModelException("BMP Scenario", "loadScenarioName",
58  "There is not scenario existed with the ID: " +
59  ValueToString(m_sceneID) + " in " + TAB_BMP_SCENARIO +
60  " table in BMP database.");
61  }
62  }
63  bson_destroy(query);
64  bson_destroy(reply);
65  mongoc_collection_destroy(sceCollection);
66 }

68 void Scenario::loadBMPs(void) {
69  vector<string>::iterator it = find(m_bmpCollections.begin(), m_bmpCollections.end(), string(TAB_BMP_INDEX));
70  if (it == m_bmpCollections.end()) {
71  throw ModelException("BMP Scenario", "loadScenarioName", "The BMP database '" + m_bmpDBName +
72  "' does not exist or there is not a table named '" +
73  TAB_BMP_INDEX + "' in BMP database.");
74  }
75  bson_t *query = bson_new();
76  BSON_APPEND_INT32(query, FLD_SCENARIO_ID, m_sceneID);
77  //cout<<bson_as_json(query, NULL)<<endl;
78  unique_ptr<MongoCollection> collection(new MongoCollection(m_conn->getCollection(m_bmpDBName, TAB_BMP_SCENARIO)));
79  unique_ptr<MongoCollection> collbmpidx(new MongoCollection(m_conn->getCollection(m_bmpDBName, TAB_BMP_INDEX)));
80  mongoc_cursor_t* cursor = collection->ExecuteQuery(query);
81 
82  bson_error_t *err = NULL;
83  if (mongoc_cursor_error(cursor, err)) {
84  throw ModelException("BMP Scenario", "loadBMPs",
85  "There are no record with scenario ID: " + ValueToString(m_sceneID));
86  }
87  const bson_t *info;
88  while (mongoc_cursor_more(cursor) && mongoc_cursor_next(cursor, &info)) {
89  //cout<<bson_as_json(info,0)<<endl;
90  bson_iter_t iter;
91  int BMPID = -1;
92  int subScenario = -1;
93  string distribution = "";
94  string collectionName = "";
95  string location = "";
96  if (bson_iter_init_find(&iter, info, FLD_SCENARIO_BMPID)) GetNumericFromBsonIterator(&iter, BMPID);
97  if (bson_iter_init_find(&iter, info, FLD_SCENARIO_SUB)) GetNumericFromBsonIterator(&iter, subScenario);
98  if (bson_iter_init_find(&iter, info, FLD_SCENARIO_DIST)) distribution = GetStringFromBsonIterator(&iter);
99  if (bson_iter_init_find(&iter, info, FLD_SCENARIO_TABLE)) collectionName = GetStringFromBsonIterator(&iter);
100  if (bson_iter_init_find(&iter, info, FLD_SCENARIO_LOCATION)) location = GetStringFromBsonIterator(&iter);
101 
103  vector<string> dist = SplitString(distribution, '|');
104  if (dist.size() >= 2 && StringMatch(dist[0], FLD_SCENARIO_DIST_RASTER)) {
105  string gridfs_name = ValueToString(m_subbsnID) + "_" + GetUpper(dist[1]);
106  if (m_sceneRsMap.find(gridfs_name) == m_sceneRsMap.end()) {
107  m_sceneRsMap.insert(make_pair(gridfs_name, nullptr));
108  }
109  dist[1] = gridfs_name;
110  }
111 
112  int BMPType = -1;
113  int BMPPriority = -1;
114  bson_t *queryBMP = bson_new();
115  BSON_APPEND_INT32(queryBMP, FLD_BMP_ID, BMPID);
116 
117  mongoc_cursor_t* cursor2 = collbmpidx->ExecuteQuery(queryBMP);
118 
119  if (mongoc_cursor_error(cursor2, err)) {
120  throw ModelException("BMP Scenario", "loadBMPs",
121  "There are no record with BMP ID: " + ValueToString(BMPID));
122  }
123  const bson_t *info2;
124  while (mongoc_cursor_more(cursor2) && mongoc_cursor_next(cursor2, &info2)) {
125  bson_iter_t sub_iter;
126  if (bson_iter_init_find(&sub_iter, info2, FLD_BMP_TYPE)) {
127  GetNumericFromBsonIterator(&sub_iter, BMPType);
128  }
129  if (bson_iter_init_find(&sub_iter, info2, FLD_BMP_PRIORITY)) {
130  GetNumericFromBsonIterator(&sub_iter, BMPPriority);
131  }
132  }
133  //cout<<BMPID<<","<<BMPType<<","<<distribution<<","<<parameter<<endl;
134  bson_destroy(queryBMP);
135  mongoc_cursor_destroy(cursor2);
137  int uniqueBMPID = BMPID * 100000 + subScenario;
138  if (this->m_bmpFactories.find(uniqueBMPID) == this->m_bmpFactories.end()) {
139  if (BMPID == BMP_TYPE_POINTSOURCE) {
140  this->m_bmpFactories[uniqueBMPID] = new BMPPointSrcFactory(m_sceneID, BMPID, subScenario, BMPType,
141  BMPPriority, dist,
142  collectionName, location);
143  }
144  if (BMPID == BMP_TYPE_PLANT_MGT) {
145  this->m_bmpFactories[uniqueBMPID] = new BMPPlantMgtFactory(m_sceneID, BMPID, subScenario, BMPType,
146  BMPPriority, dist,
147  collectionName, location);
148  }
149  if (BMPID == BMP_TYPE_AREALSOURCE) {
150  this->m_bmpFactories[uniqueBMPID] = new BMPArealSrcFactory(m_sceneID, BMPID, subScenario, BMPType,
151  BMPPriority, dist,
152  collectionName, location);
153  }
154  if (BMPID == BMP_TYPE_AREALSTRUCT) {
155  this->m_bmpFactories[uniqueBMPID] = new BMPArealStructFactory(m_sceneID, BMPID, subScenario, BMPType,
156  BMPPriority, dist,
157  collectionName, location);
158  }
159  }
160  }
161  bson_destroy(query);
162  mongoc_cursor_destroy(cursor);
163 }
