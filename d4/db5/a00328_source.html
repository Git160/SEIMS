1006 template <typename T, typename MASK_T>
1007 clsRasterData<T, MASK_T>::clsRasterData(vector<string>& filenames,
1008  const bool calc_pos /* = true */,
1009  clsRasterData<MASK_T>* mask /* = nullptr */,
1010  const bool use_mask_ext /* = true */,
1011  T default_value /* = static_cast<T>(NODATA_VALUE) */) {
1012  InitializeRasterClass();
1013  if (filenames.empty()) {
1015  StatusNoUnitTest("The filenames MUST have at least one raster file path!");
1016  return;
1017  }
1018  if (!CheckRasterFilesExist(filenames)) {
1020  StatusNoUnitTest("Please make sure all file path existed!");
1021  return;
1022  }
1023  if (filenames.size() == 1) {
1025  if (!ConstructFromSingleFile(filenames[0], calc_pos, mask,
1026  use_mask_ext, default_value)) {
1027  return;
1028  }
1029  } else {
1031  n_lyrs_ = CVT_INT(filenames.size());
1034  if (!ConstructFromSingleFile(filenames[0], calc_pos, mask,
1035  use_mask_ext, default_value)) {
1036  return;
1037  }
1039  core_name_ = SplitString(core_name_, '_').at(0);
1040  full_path_ = GetPathFromFullName(filenames[0]) + SEP + core_name_ + "_%d." + GetSuffix(filenames[0]);
1045  Initialize2DArray(n_cells_, n_lyrs_, raster_2d_, no_data_value_);
1046 #pragma omp parallel for
1047  for (int i = 0; i < n_cells_; i++) {
1048  raster_2d_[i][0] = raster_[i];
1049  }
1050  Release1DArray(raster_);
1052  for (int fileidx = 1; fileidx < CVT_INT(filenames.size()); fileidx++) {
1053  map<string, double> tmpheader;
1054  T* tmplyrdata = nullptr;
