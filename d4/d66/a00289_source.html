136  def metis_partition(g, weight, wp, bin_dir):
137  """Partition subbasins into multiple groups by METIS
138 
139  Args:
140  g: `NetworkX.DiGraph` object
141  weight: weight of each node, e.g., area of subbasin, {subbasinID: weight}
142  wp: output directory
143  bin_dir: directory of METIS package
144  Returns:
145  group_dict: {subbasinID: {'group': group_number_list,
146  'kmetis': group_ids_list_by_kmetis,
147  'pmetis': group_ids_list_by_pmetis}
148  }
149  """
150  group_dict = dict()
151  for subbsn_id in g.nodes():
152  group_dict[subbsn_id] = {'group': list(), 'kmetis': list(), 'pmetis': list()}
153 
154  metis_input = ImportReaches2Mongo.prepare_node_with_weight_for_metis(g, weight, wp)
155  # Creating group divided numbers
156  nlist = list(range(1, 129))
157  nlist.extend([192, 256, 384, 512, 768, 1536])
158  # nlist should be less than the number of subbasin, otherwise it will make nonsense.
159  ns = g.nodes()
160  nlist = [x for x in nlist if x <= max(ns)]
161  # Make directories for KMETIS and PMETIS
162  UtilClass.mkdir(wp + os.path.sep + 'kmetis')
163  UtilClass.mkdir(wp + os.path.sep + 'pmetis')
164  for n in nlist:
165  print('divide number: %d' % n)
166  if n <= 1:
167  for subbsn_id in g.nodes():
168  group_dict[subbsn_id]['group'].append(1)
169  group_dict[subbsn_id]['kmetis'].append(0)
170  group_dict[subbsn_id]['pmetis'].append(0)
171  continue
172  # kmetis, -ptype=kway, direct k-way partitioning (default)
173  str_command = '"%s/gpmetis" %s %d' % (bin_dir, metis_input, n)
174  result = UtilClass.run_command(str_command)
175  with open('%s/kmetis/kmetisResult%d.txt' % (wp, n), 'w') as f_metis_output:
176  for line in result:
177  f_metis_output.write(line)
178  metis_output = '%s.part.%d' % (metis_input, n)
179  with open(metis_output, 'r') as f:
180  lines = f.readlines()
181  group_kmetis = [int(item) for item in lines]
182  adjust_group_result(weight, group_kmetis, n)
183  shutil.move(metis_output, '%s/kmetis/metis.part.%d' % (wp, n))
184 
185  # pmetis, -ptype=rb, recursive bisectioning
186  str_command = '"%s/gpmetis" -ptype=rb %s %d' % (bin_dir, metis_input, n)
187  result = UtilClass.run_command(str_command)
188  with open('%s/pmetis/pmetisResult%d.txt' % (wp, n), 'w') as f_metis_output:
189  for line in result:
190  f_metis_output.write(line)
191  with open(metis_output, 'r') as f:
192  lines = f.readlines()
193  group_pmetis = [int(item) for item in lines]
194  adjust_group_result(weight, group_pmetis, n)
195  shutil.move(metis_output, '%s/pmetis/metis.part.%d' % (wp, n))
196 
197  for i, (gk, gp) in enumerate(zip(group_kmetis, group_pmetis)):
198  group_dict[i + 1]['group'].append(n)
199  group_dict[i + 1]['kmetis'].append(gk)
200  group_dict[i + 1]['pmetis'].append(gp)
201  return group_dict

249  @staticmethod
250  def construct_downup_order(g):
251  """
252 
253  Returns:
254  downstream_up_order_dic: from outlet up stream dict
255  upstream_down_order_dic: from source down stream dict
256  """
257  # find outlet subbasin
258  outlet = -1
259  for node in g.nodes():
260  if g.out_degree(node) == 0:
261  outlet = node
262  if outlet < 0:
263  raise ValueError('Cannot find outlet subbasin ID, please check the '
264  'threshold for stream extraction!')
265  print('outlet subbasin:%d' % outlet)
266 
267  # assign order from outlet to upstream subbasins from 1
268  downstream_up_order_dic = dict()
269  start_node = [outlet]
270  order_num = 0
271  while start_node:
272  tmp = list()
273  order_num += 1
274  for snode in start_node:
275  downstream_up_order_dic[snode] = order_num
276  for in_nodes in g.in_edges(snode):
277  tmp.append(in_nodes[0])
278  start_node = tmp[:]
279  # order_num now is the maximum order number from outlet
280  # reserve the order number
281  for k, v in downstream_up_order_dic.items():
282  downstream_up_order_dic[k] = order_num - v + 1
283  return downstream_up_order_dic

285  @staticmethod
286  def construct_updown_order(graph):
287  # assign order from the source subbasins
288  g = graph.copy()
289  upstream_down_order_dic = dict()
290  order_num = 1
291  nodelist = g.nodes()
292  while len(nodelist) != 0:
293  nodelist = g.nodes()
294  del_list = list()
295  for node in nodelist:
296  if g.in_degree(node) == 0:
297  upstream_down_order_dic[node] = order_num
298  del_list.append(node)
299  for item in del_list:
300  g.remove_node(item)
301  order_num += 1
302 
303  return upstream_down_order_dic

361 
362  Args:
363  cfg: configuration object

365  """

368

370  downup_order = ImportReaches2Mongo.construct_downup_order(g)
371  updown_order = ImportReaches2Mongo.construct_updown_order(g)
372 
373  # interpolation among different stream orders
374  min_manning = 0.035
375  max_manning = 0.075
376 
377  rch_orders = list(updown_order.values())
378  min_order = min(rch_orders)
379  max_order = max(rch_orders)

382  for tmpid in list(reach_dict.keys()):
383  reach_dict[tmpid]['manning'] = max_manning - a * (updown_order[tmpid] - min_order)
384

386  cfg.dirs.metis, cfg.seims_bin)

388  ImportReaches2Mongo.add_group_field(cfg.vecs.reach, ImportReaches2Mongo._LINKNO,
389  group_metis)
390  ImportReaches2Mongo.add_group_field(cfg.vecs.subbsn, ImportReaches2Mongo._SUBBASIN,
391  group_metis)

469 def adjust_group_result(weight_dic, group_list, n_groups):
470  """Adjust group result"""
471  group_dic = dict() # group tmpid from 0
472  group_weight_dic = dict() # subbasin tmpid form 1
473  n = len(weight_dic.keys())
474  for sub_id in range(1, n + 1):
475  group_id = group_list[sub_id - 1]
476  group_dic.setdefault(group_id, []).append(sub_id)
477  if group_id not in group_weight_dic:
478  group_weight_dic[group_id] = 0
479  group_weight_dic[group_id] += weight_dic[sub_id][ImportReaches2Mongo._NUMCELLS]
480  # get average weight for each group, which is the ideal aim of task scheduling
481  ave_eight = 0
482  for tmpid in group_weight_dic:
483  ave_eight += group_weight_dic[tmpid]
484  ave_eight /= n_groups
485 
486  for iGroup in range(n_groups):
487  if iGroup not in group_dic:
488  max_id, max_weight = get_max_weight(group_weight_dic, group_dic)
489  if max_id < 0:
490  continue
491  sub_list = group_dic[max_id]
492  sub_id = sub_list[0]
493  group_dic[iGroup] = [sub_id, ]
494  group_list[sub_id - 1] = iGroup
495  group_dic[max_id].remove(sub_id)
