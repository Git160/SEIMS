238 float ORYZA::CalHeatUnitDaily(int i) {
239  float tt = 0.f;
240  float hu = 0.f;
241  for (int k = 1; k <= 24; k++) {
242  float td = m_meanTemp[i] + 0.5f * std::abs(m_tMax[i] - m_tMin[i]) * cos(0.2618f * (k - 14.f));
243  if (td > m_tbd && td < m_tmd) {
244  if (td > m_tod) {
245  td = m_tod - (td - m_tod) * (m_tod - m_tbd) / (m_tmd - m_tod);
246  }
247  tt = tt + (td - m_tbd) / 24;
248  }
249  }
250  hu = tt;
251  return hu;
252 }

254 float ORYZA::CalDevelopmentRate(int i) {
255  float dl = 0.f, tstr = 0.f, ppfac = 0.f;
256  float dvr = 0.f;
257  float hu = CalHeatUnitDaily(i);

265  else

267  ppfac = Min(1.f, Max(0, ppfac));
268  dvr = m_dvri * hu * ppfac;
269  } else if (m_dvs[i] < 1.f) dvr = m_dvrp * hu;
270  else if (m_dvs[i] > 1.f) dvr = m_dvrr * hu;
271 
272  if (m_cropsta[i] == 3.f) tstr = m_ts[i];

275

280 void ORYZA::CalDayLengthAndSINB(int i) {
281  float DEGTRAD = 0.017453292f, zzcos, zzsin;
282  // float dayLenP = 0.f; // not used?
284 
285  float dec = -asin(sin(23.45f * DEGTRAD) * cos(2.f * PI * (m_dayOfYear + 10.f) / 365.f));
286  /*m_sinLD[i] = sin (DEGTRAD * 31.2f) * sin (dec);
287  m_cosLD[i] = cos (DEGTRAD * 31.2f) * cos (dec);*/
288 
289  m_sinLD[i] = sin(DEGTRAD * m_cellLat) * sin(dec);
290  m_cosLD[i] = cos(DEGTRAD * m_cellLat) * cos(dec);
291  float aob = m_sinLD[i] / m_cosLD[i];
292 
293  if (aob < -1) {
294  m_dayL[i] = 0;
295  zzcos = 0;
296  zzsin = 1;
297  } else if (aob > 1) {
298  m_dayL[i] = 24;
299  zzcos = 0;
300  zzsin = -1;
301  } else {
302  m_dayL[i] = 12.f * (1.f + 2.f * asin(aob) / PI);
303  //dayLenP = 12.f * (1.f + 2.0f * asin(aob) / PI);
304  float zza = PI * (12.f + m_dayL[i]) / 24.f;
305  zzcos = cos(zza);
306  zzsin = sin(zza);
307  }
308 
309  float dsinb = 2.f * 3600.f * (m_dayL[i] * 0.5f * m_sinLD[i] - 12.f * m_cosLD[i] * zzcos / PI);
310  m_dsinbe[i] = 2.f * 3600.f * (m_dayL[i] * (0.5f * m_sinLD[i] + 0.2f * pow(m_sinLD[i], 2.f) + 0.1f *
311  pow(m_cosLD[i], 2.f)) - (12.f * m_cosLD[i] * zzcos + 9.6f * m_sinLD[i] * m_cosLD[i] *
312  zzcos + 2.4f * pow(m_cosLD[i], 2.f) * zzcos * zzsin) / PI);
313 
314  m_solcon[i] = 1370.f * (1.f + 0.033f * cos(2.f * PI * m_dayOfYear / 365.f));
315 }

317 void ORYZA::CalDirectRadiation(int i) {
318  float atmtr = 0.f, frdif = 0.f;
319  m_sinb[i] = m_sinLD[i] + m_cosLD[i] * cos((m_hour - 12.f) * 0.2617993f);
320  if (m_sinb[i] > 0) {
322  // SR : convert from MJ/m2/day to J/m2/day
323  float tmpr1 = m_SR[i] * 1000000.f * m_sinb[i] * (1.f + 0.4f * m_sinb[i]) / m_dsinbe[i];
324  atmtr = tmpr1 / (m_solcon[i] * m_sinb[i]);
325  if (atmtr < 0.22) frdif = 1.f;
326  else if (atmtr > 0.22 && atmtr < 0.35) frdif = 1.f - 6.4f * pow(atmtr - 0.22f, 2.f);
327  else frdif = 1.47f - 1.66f * atmtr;
328 
329  float xx = 0.15f + 0.85f * (1.f - exp(-0.1f / m_sinb[i]));
330  frdif = Max(frdif, xx);
331 
332  m_rdpdf[i] = tmpr1 * m_frpar * frdif;
333  m_rdpdr[i] = tmpr1 * m_frpar * (1 - frdif);
334  }

337 void ORYZA::CalLeafAbsorbRadiation(int i) {
339  float cslv = 0.2f;
340  float tmpr1 = sqrt(1.f - cslv);
341  float rflh = (1.f - tmpr1) / (1.f + tmpr1);
342  float rfls = rflh * 2.f / (1.f + 2.f * m_sinb[i]);
343  //compute Extinction coefficient
344  float kdf = 0.f;
345  if (m_dvs[i] > 0.f && m_dvs[i] < 0.65f) kdf = 0.4f;
346  else if (m_dvs[i] < 1.f) kdf = 0.4f + 0.2f / 0.35f * (m_dvs[i] - 0.65f);
347  else kdf = 0.6f;
348  float ecpdf = kdf;
349 
350  float clustf = ecpdf / (0.8f * tmpr1);
351  float ecpbl = 0.5f / m_sinb[i] * clustf;
352  float ecptd = ecpbl * tmpr1;
353 
354  float rapdfl = (1.f - rflh) * m_rdpdf[i] * ecpdf * exp(-ecpdf * m_gaid[i]);

384 void ORYZA::Sgpl(int i) {
385  // Gauss weights for three point Gauss
386  float gsx[3] = {0.112702, 0.500000, 0.887298};
387  float gsw[3] = {0.277778, 0.444444, 0.277778};
388  float gpshl = 0.f, gpsll = 0.f, rapsll = 0.f;
389  // m_gai[i] = m_aLAI;
390 
391  CalLeafAbsorbRadiation(i);
392  // calculate redf and nflv
393  if (m_meanTemp[i] <= 10.f) m_redf = 0.f;
394  else if (m_meanTemp[i] <= 20.f) m_redf = 0.1f * (m_meanTemp[i] - 10.f);
395  else if (m_meanTemp[i] <= 37.f) m_redf = 1.f;
396  else m_redf = Max(0.f, (1.f - (m_meanTemp[i] - 37.f) * 0.2f));
397  if (m_dvs[i] <= 0.16f) m_nflv = 0.54f;
398  else if (m_dvs[i] <= 0.33f) m_nflv = 0.54f + (m_dvs[i] - 0.16f) * 6.25f;
399  else if (m_dvs[i] <= 0.65f) m_nflv = 1.53f - (m_dvs[i] - 0.33f);
400  else if (m_dvs[i] <= 0.8f) m_nflv = 1.222f + (m_dvs[i] - 0.65f) * 2.3f;
401  else if (m_dvs[i] <= 1.f) m_nflv = 1.56f - (m_dvs[i] - 0.8f) * 1.25f;
402  else if (m_dvs[i] <= 1.45f) m_nflv = 1.29f + (m_dvs[i] - 1.f) * 0.16f;
403  else m_nflv = 1.36f - (m_dvs[i] - 1.45f);
404

409

411  float tmpr1 = m_rapshl[i] + rapsll * gsx[k];
412  if (Amax2 > 0) {
413  gpsll = gpsll + Amax2 * (1.f - exp(-tmpr1 * m_eff / Amax2)) * gsw[k];
414  }
415  rapsll = rapsll + tmpr1 * gsw[k];
416  }
417  m_gpl[i] = m_fslla[i] * gpsll + (1.f - m_fslla[i]) * gpshl;
418  m_rapl[i] = m_fslla[i] * rapsll + (1.f - m_fslla[i]) * m_rapshl[i];
419 }

493 void ORYZA::LAI(int i) {
494  // Actual relative growth rate of leaves ((oCd)-1)
495  float rgrl = m_rgrlMX - (1.f - m_frStrsN[i]) * (m_rgrlMX - m_rgrlMN);
496  float x = 1.f, testSet = 0.0001f;
497  bool flag = false;

499  float hu = CalHeatUnitDaily(i);
500  // Temperature sum for leaf area development at transplanting
501  float tslvtr = 0.f, tshckl = 0.f;
502 
503  // for transplanted rice

505  // seedbed: no drought stress effects in seedbed
506  if (m_lai[i] < 1.f) {
507  m_gLai = m_lai[i] * rgrl * hu;

509  laiExs = m_lai[i];
510  } else {

513  if (flag) {

517  float gLai2 = (m_wlvg[i] + m_rwlvg[i]) * m_sla[i] - m_lai[i];

521  }

523  // Transplanting effects: dilution and shock-setting
524  tslvtr = m_ts[i];

530  else {
531  if (m_lai[i] < 0.f && m_dvs[i] < 1.f) {

534  laiExs = m_lai[i];
535  } else {
536  float test = std::abs(m_lai[i] / m_wlvg[i] - m_sla[i]) / m_sla[i];
537  if (test < testSet) flag = true;
538  if (flag) m_gLai = (m_wlvg[i] + m_rwlvg[i]) * m_sla[i] - m_lai[i];
539  else {
540  float gLai1 = (m_wlvg[i] + m_rwlvg[i] - wlvgExs) * m_sla[i] + laiExs - m_lai[i];
541  float gLai2 = (m_wlvg[i] + m_rwlvg[i]) * m_sla[i] - m_lai[i];
542  m_gLai = (gLai1 + gLai2) / (x + 1.f);
543  x += 1.f;
544  }
545  }
546  }

572 
573  // Leaf rolling under drought stress (only for photosynthesis)
574  float m_laiRol = m_lai[i] * (0.5f * m_frStrsWtr[i] + 0.5f);
575  float ssga = 0.f;
576  if (m_dvs[i] >= 0.f && m_dvs[i] < 0.9f) ssga = 0.0003f;
577  else ssga = 0.0003f - 0.00025f * (m_dvs[i] - 0.9f);

580 
581  CalDailyCanopyPhotosynthesisRate(i);
582

584  float dpari = m_rapcdt[i] / 1.e6f;
585  // compute the daily incoming photosynthetically active radiation
586  float dpar = m_frpar * m_SR[i];
587  // Unrolling of ALAI again
588  m_aLAI = m_lai[i] + 0.5f * m_sai;
589  // drought stress will decreases the rate
590  m_dtga[i] = m_dtga[i] * m_frStrsWtr[i];
591  //compute the fraction of dry matter to the shoot(FSH), leave(FLV), stems(FST), panicle(FSO), root(FRT), leaf death(DRLV)
592  if (m_dvs[i] > 0.f && m_dvs[i] <= 1.f) {
593  m_fsh = m_aFsh + m_bFsh * m_dvs[i];
594  if (m_dvs[i] < 0.5f) {
595  m_flv = m_aFlv;
596  } else {

598  }

600  m_fst = 1.f - m_flv;
601  m_fso = 0.f;
602  } else {
603  m_fst = 0.7f - (m_dvs[i] - 0.75f);
604  m_fso = 1.f - m_flv - m_fst;
605  }
606  m_drlv = 0.f;
607  } else if (m_dvs[i] < 1.2f) {
608  m_fsh = 1.f;
609  m_flv = 0.f;
610  m_fst = 1.f - m_aFso - m_bFso * (m_dvs[i] - 1.f);
611  m_fso = 1.f - m_fst;
612  m_drlv = m_aDrlv + m_bDrlv * (m_dvs[i] - 1.f);
613  } else {
614  m_fsh = 1.f;
615  m_flv = 0.f;
616  m_fst = 0.f;
617  m_fso = 1.f;
618  m_drlv = m_aDrlv + m_bDrlv * (m_dvs[i] - 1.f);

623  m_nsllv = 1.f;
624  } else if (m_pltN[i] < 2.f) {

626  } else {
627  m_nsllv = 1.5f;
628  }

630  float llv = m_nsllv * m_wlvg[i] * m_drlv;
631  if (m_dvs[i] > 1.f) {
632  m_lstr = m_wst[i] / m_tclstr;
633  } else {
634  m_lstr = 0.f;
635  }
636  // Maintenance requirements
637  m_teff = pow(m_q10, (m_tav - m_tref) *0.1f);
638  float mnDVS = m_wlvg[i] / (m_wlvg[i] + m_wlvd[i]);
639  float rmcr = (m_wlvg[i] * m_mainLV + m_wst[i] * m_mainST + m_wso[i] *
640  m_mainSO + m_wrt[i] * m_mainRT) * m_teff * mnDVS;
641  // Carbohydrate requirement for dry matter production
642  float crGCR = m_fsh * (m_crgLV * m_flv + m_crgST * m_fst * (1.f - m_fstr) +
643  m_crgSTR * m_fstr * m_fst + m_crgSO * m_fso) + m_crgRT * m_frt;
644  // Gross and net growth rate of crop (GCR, NGCR)
645  m_gcr[i] = (m_dtga[i] * 30.f / 44.f - rmcr + m_lstr * m_lrstr * m_fcSTR * 30.f / 12.f) / crGCR;
646  float xx = m_gcr[i] - m_lstr * m_lrstr * m_fcSTR * 30.f / 12.f;
647  float ngcr = Max(0, xx);
648  // Intermediate variable for planting density, used to calculate the reduction in net weight
649  float pltr = 1.f;
650  if (m_cropsta[i] == 3.f) {
651  pltr = m_nplh * m_nh / m_nplsb;
652  }
653  // Growth rates of crop organs at transplanting
654  float rwlvg1 = m_wlvg[i] * (1.f - pltr);
655  float gst1 = m_wsts[i] * (1.f - pltr);
656  float rwstr1 = m_wstr[i] * (1.f - pltr);
657  float grt1 = m_wrt[i] * (1.f - pltr);
658  // Growth rates of crop organs
659  float grt = m_gcr[i] * m_frt - grt1;
660  glv = m_gcr[i] * m_fsh * m_flv - rwlvg1;
661  float rwlvg = glv - llv;
662  gst = m_gcr[i] * m_fsh * m_fst * (1.f - m_fstr) - gst1;
663  float gstr = m_gcr[i] * m_fsh * m_fstr - rwstr1;
664  float rwstr = gstr - m_lstr;
665  gso = m_gcr[i] * m_fsh * m_fso;
666  float ggr = 0.f;
667  if (m_dvs[i] > 0.95) ggr = gso;
668 
669  CalSpikeletAndGrainRate(i);
670 
671  m_sla[i] = m_aSLA + m_bSLA * exp(m_cSLA * (m_dvs[i] - m_dSLA));
672  m_sla[i] = Min(m_slaMX, m_sla[i]);
673  LAI(i);
674 
675  // Growth respiration of the crop (RGCR)
676  float co2RT = 44.f / 12.f * (m_crgRT * 12.f / 30.f - m_fcRT);
677  float co2LV = 44.f / 12.f * (m_crgLV * 12.f / 30.f - m_fcLV);
678  float co2ST = 44.f / 12.f * (m_crgST * 12.f / 30.f - m_fcST);
679  float co2STR = 44.f / 12.f * (m_crgSTR * 12.f / 30.f - m_fcSTR);
680  float co2SO = 44.f / 12.f * (m_crgSO * 12.f / 30.f - m_fcSO);
681  float m_rgcr = (grt + grt1) * co2RT + (glv + rwlvg1) * co2LV + (gst + gst1) * co2ST + gso * co2SO +
682  (gstr + rwstr1) * co2STR + (1.f - m_lrstr) * m_lstr * m_fcSTR * 44.f / 12.f;
683  float ctrans = rwlvg1 * m_fcLV + gst1 * m_fcST + rwstr1 * m_fcSTR + grt1 * m_fcRT;
684  float rtnass = (m_dtga[i] * 30.f / 44.f - rmcr) * 44.f / 30.f - m_rgcr - ctrans * 44.f / 12.f;
685 
686  // update the state variables like dvs, lai, wso, wrr
687  // Integrate rate variables
688  m_ts[i] += hu;
689  m_wlvg[i] += rwlvg;
690  m_wlvd[i] += llv;
691  m_wsts[i] += gst;
692  m_wstr[i] += rwstr;
693  m_wso[i] += gso; // storage organs
694  m_wrt[i] += grt; // root
695  m_wrr[i] += ggr; // final yield
696  m_ngr[i] += m_gngr[i];
697  m_nsp[i] += gnsp;
698  m_tnass[i] += rtnass;
699  // Calculate sums of states
700  m_wst[i] = m_wstr[i] + m_wsts[i]; // stem
701  m_wlv[i] = m_wlvg[i] + m_wlvd[i]; // leaf
702  m_wagt[i] = m_wlv[i] + m_wst[i] + m_wso[i]; // dry weight above ground
703  m_biomass[i] = m_wagt[i] + m_wrt[i];
704  m_frRoot[i] = m_wrt[i] / m_biomass[i];
705  // Leaf area index and total area index (leaves + stems)
706  m_lai[i] += m_gLai;
707  m_aLAI = m_lai[i] + 0.5f * m_sai;
708  // Root length
709  m_zrt[i] += m_gzrt;
710  m_zrt[i] = Min(m_zrt[i], m_zrtMCD);
711  }

800 void ORYZA::CalPlantNUptake(int i) {
801  // compute N demand of crop
802  float nMinSO;
803  float nMaxL = m_dvs[i] < 0.4f ? 0.053f : 0.053f - m_nMaxL * (m_dvs[i] - 0.4f);
804  float nMinL = 0.025f - m_nMinL * m_dvs[i];
805  if (m_ancrf[i] < 50.f) {
806  nMinSO = 0.006f - m_anMinSO * m_ancrf[i];
807  } else if (m_ancrf[i] < 400.f) {
808  nMinSO = 0.0008f + m_bnMinSO * (m_ancrf[i] - 50.f);
809  } else {
810  nMinSO = 0.017f;
811  }
812  float LeafDemandN = nMaxL * (m_wlvg[i] + glv) - m_anlv[i];
813  float StemDemandN = nMaxL * 0.5f * (m_wst[i] + gst) - m_anst[i];
814  float SODemandNMax = m_nMaxSO * gso;
815  float SODemandNMin = nMinSO * gso;
816  // Calculate translocation of N from organs, No translocation before DVS = 0.95
817  float aTNlv, aTNst, aTN, ntso, aTnrt;
818  if (m_dvs[i] < 0.95f) {
819  aTNlv = 0.f;
820  aTNst = 0.f;
821  aTN = 1.f;
822  ntso = 0.f;
823  aTnrt = 0.f;
824  } else {
825  aTNlv = Max(0.f, m_anlv[i] - m_wlvg[i] * m_rfnlv);
826  aTNst = Max(0.f, m_anst[i] - m_wst[i] * m_rfnst);
827  aTnrt = (aTNlv + aTNst) * m_fntrt;
828  aTN = aTNlv + aTNst + aTnrt;
829  ntso = aTN / m_tcntrf;
830  ntso = Max(ntso, SODemandNMin);
831  ntso = Min(ntso, SODemandNMax);
832  }
833  // Actual N translocation rates from plant organs, ATN should not be 0
834  float ntlv = ntso * aTNlv / aTN;
835  float ntst = ntso * aTNst / aTN;
836  float ntrt = ntso * aTnrt / aTN;
837  // Calculate nitrogen uptake
838  // float n_reduc = 300.f; /// nitrogen uptake reduction factor (not currently used; defaulted 300.)
839  float tnsoil = 0.f;
840  for (int k = 0; k < m_nSoilLayers[i]; k++) {
841  tnsoil += m_soilNO3[i][k];
842  }
843  // Available N uptake is mimimum of soil supply and maximum crop uptake
844  float nupp = Min(8.f, tnsoil);
845  float totNDemand = LeafDemandN + ntlv + (StemDemandN + ntst) + (SODemandNMax - ntso);
846  float naLV = Max(0.f, Min(LeafDemandN + ntlv, nupp * (LeafDemandN + ntlv) / totNDemand));
847  float naST = Max(0.f, Min(StemDemandN + ntst, nupp * (StemDemandN + ntst) / totNDemand));
848  float naSO = Max(0.f, Min(SODemandNMax - ntso, nupp* (SODemandNMax - ntso) / totNDemand));
849  m_anst[i] += naST - ntst;
850  m_plantUpTkN[i] = naLV + naST + naSO;
851  if (m_plantUpTkN[i] < 0.f) m_plantUpTkN[i] = 0.f;
852  m_pltN[i] += m_plantUpTkN[i];
853  for (int l = 0; l < CVT_INT(m_nSoilLayers[i]); l++) {
854  float uno3l = Min(m_plantUpTkN[i], m_soilNO3[i][l]);
855  m_soilNO3[i][l] -= uno3l;
856  }
857  float nstan = 0.f, nlvan = 0.f;
858  if (m_dvs[i] < 1.f) {
859  nstan = naST - ntst;
860  nlvan = naLV - ntlv;
861  }
862  m_ancrf[i] += nstan + nlvan;
863  m_anlv[i] += naLV;
864  float fnlv = m_anlv[i] / m_wlvg[i];
865  m_frStrsN[i] = (fnlv - 0.9f * nMaxL) / (nMaxL - 0.9f * nMaxL);
866  m_frStrsN[i] = Max(m_frStrsN[i], 0);
867  m_frStrsN[i] = Min(m_frStrsN[i], 1.f);
868 }
