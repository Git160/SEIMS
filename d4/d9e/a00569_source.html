117 int AET_PT_H::Execute() {
118  CheckInputData();
119  InitialOutputs();
120 #pragma omp parallel for
121  for (int i = 0; i < m_nCells; i++) {
123  float esd = 0.f, etco = 0.f, effnup = 0.f;
124  float no3up = 0.f, es_max = 0.f, eos1 = 0.f, xx = 0.f;
125  float cej = 0.f, eaj = 0.f, pet = 0.f, esleft = 0.f;
126  float evzp = 0.f, eosl = 0.f, dep = 0.f, evz = 0.f, sev = 0.f;
127  pet = m_pet[i] - m_IntcpET[i];
128  esd = 500.f;
129  etco = 0.8f;
130  effnup = 0.1f;
131 
132  if (pet < UTIL_ZERO) {
133  pet = 0.f;
134  m_maxPltET[i] = 0.f; // i.e., ep_max
135  es_max = 0.f;
136  } else {
138  if (m_lai[i] <= 3.f) {
139  m_maxPltET[i] = m_lai[i] * pet * _1div3;
140  } else {
141  m_maxPltET[i] = pet;
142  }
143  if (m_maxPltET[i] < 0.f) m_maxPltET[i] = 0.f;
145  cej = -5.e-5f;
146  eaj = 0.f;
147  es_max = 0.f;
148  eos1 = 0.f;
149  if (m_snowAccum[i] >= 0.5f) {
150  eaj = 0.5f;
151  } else {
152  eaj = exp(cej * (m_rsdCovSoil[i] + 0.1f));
153  }
154  es_max = pet * eaj;
155  eos1 = pet / (es_max + m_maxPltET[i] + 1.e-10f);
156  eos1 = es_max * eos1;
157  es_max = Min(es_max, eos1);
158  es_max = Max(es_max, 0.f);
160  if (pet < es_max + m_maxPltET[i] && !FloatEqual(es_max + m_maxPltET[i], 0.f)) {
161  es_max = pet * es_max / (es_max + m_maxPltET[i]);
162  m_maxPltET[i] = pet * m_maxPltET[i] / (es_max + m_maxPltET[i]);
163  }
164  if (pet < es_max + m_maxPltET[i]) {
165  es_max = pet - m_maxPltET[i] - UTIL_ZERO;
166  }
167 
169  esleft = es_max;
171  if (m_tMean[i] > 0.f) {
172  if (m_snowAccum[i] >= esleft) {
174  m_snowAccum[i] -= esleft;
175  m_snowSublim[i] += esleft;
176  esleft = 0.f;
177  } else {
179  esleft -= m_snowAccum[i];
180  m_snowSublim[i] += m_snowAccum[i];
181  m_snowAccum[i] = 0.f;
182  }
183  }
184  // take soil evap from each soil layer
185  evzp = 0.f;
186  eosl = esleft;
187  for (int ly = 0; ly < CVT_INT(m_nSoilLyrs[i]); ly++) {
188  dep = 0.f;
190  if (ly == 0) {
191  dep = m_soilDepth[i][ly];
192  } else {
193  dep = m_soilDepth[i][ly - 1];
194  }
195  if (dep < esd) {
197  evz = 0.f;
198  sev = 0.f;
199  xx = 0.f;
200  evz = eosl * m_soilDepth[i][ly] /
201  (m_soilDepth[i][ly] + exp(2.374f - 0.00713f * m_soilDepth[i][ly]));
202  sev = evz - evzp * m_esco[i];
203  evzp = evz;
204  if (m_soilWtrSto[i][ly] < m_solFC[i][ly]) {
205  xx = 2.5f * (m_soilWtrSto[i][ly] - m_solFC[i][ly]) / m_solFC[i][ly];
206  sev *= Expo(xx);
207  }
208  sev = Min(sev, m_soilWtrSto[i][ly] * etco);
209  if (sev < 0.f || sev != sev) sev = 0.f;
210  if (sev > esleft) sev = esleft;
212  if (m_soilWtrSto[i][ly] > sev) {
213  esleft -= sev;
214  m_soilWtrSto[i][ly] = Max(UTIL_ZERO, m_soilWtrSto[i][ly] - sev);
215  } else {
216  esleft -= m_soilWtrSto[i][ly];
217  m_soilWtrSto[i][ly] = 0.f;
218  }
219  }
221  if (ly == 1) {
223  no3up = 0.f;
224  no3up = effnup * sev * m_solNo3[i][ly] / (m_soilWtrSto[i][ly] + UTIL_ZERO);
225  no3up = Min(no3up, m_solNo3[i][ly]);
226  m_no3Up += no3up / m_nCells;
227  m_solNo3[i][ly] -= no3up;
228  m_solNo3[i][ly - 1] += no3up;
229  }
230  }
232  m_soilWtrStoPrfl[i] = 0.f;
233  for (int ly = 0; ly < CVT_INT(m_nSoilLyrs[i]); ly++) {
234  m_soilWtrStoPrfl[i] += m_soilWtrSto[i][ly];
235  }
237  //if (esleft != esleft || es_max != es_max)
238  // cout<<"esleft: "<<esleft<<", es_max: "<<es_max<<endl;
239  if (es_max > esleft) {
240  m_soilET[i] = es_max - esleft;
241  } else {
242  m_soilET[i] = 0.f;
243  }
244  }
245  }
246  return true;
247 }
