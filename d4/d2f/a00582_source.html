45 int DepressionFSDaily::Execute() {
46  CheckInputData();
47  InitialOutputs();
48 #pragma omp parallel for
49  for (int i = 0; i < m_nCells; i++) {
51  // runoff
52  if (m_depCap[i] < 0.001f) {
53  m_sr[i] = m_pe[i];
54  m_sd[i] = 0.f;
55  } else if (m_pe[i] > 0.f) {
56  float pc = m_pe[i] - m_depCap[i] * log(1.f - m_sd[i] / m_depCap[i]);
57  float deltaSd = m_pe[i] * exp(-pc / m_depCap[i]);
58  if (deltaSd > m_depCap[i] - m_sd[i]) {
59  deltaSd = m_depCap[i] - m_sd[i];
60  }
61  m_sd[i] += deltaSd;
62  m_sr[i] = m_pe[i] - deltaSd;
63  } else {
64  m_sd[i] += m_pe[i];
65  m_sr[i] = 0.f;
66  }
67 
69  // evaporation
70  if (m_sd[i] > 0) {
74  // evaporation from depression storage
75  if (m_pet[i] - m_ei[i] < m_sd[i]) {
76  m_ed[i] = m_pet[i] - m_ei[i];
77  } else {
78  m_ed[i] = m_sd[i];
79  }
80  m_sd[i] -= m_ed[i];
81  } else {
82  m_ed[i] = 0.f;
83  m_sd[i] = 0.f;
84  }
85  if (m_impoundTriger != nullptr && FloatEqual(m_impoundTriger[i], 0.f)) {
86  if (m_potVol != nullptr) {
87  m_potVol[i] += m_sr[i];
88  m_potVol[i] += m_sd[i];
89  m_sr[i] = 0.f;
90  m_sd[i] = 0.f;
91  }
92  }
93  }
94  return true;
95 }
