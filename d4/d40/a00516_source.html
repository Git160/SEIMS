236 template<typename T>
237 bool GetNumericFromBsonIterator(bson_iter_t *iter, T &numericvalue) {
238  const bson_value_t *vv = bson_iter_value(iter);
239  if (vv->value_type == BSON_TYPE_INT32) {
240  numericvalue = (T) vv->value.v_int32;
241  } else if (vv->value_type == BSON_TYPE_INT64) {
242  numericvalue = (T) vv->value.v_int64;
243  } else if (vv->value_type == BSON_TYPE_DOUBLE) {
244  numericvalue = (T) vv->value.v_double;
245  } else if (vv->value_type == BSON_TYPE_UTF8) {
246  string tmp = vv->value.v_utf8.str;
247  numericvalue = (T) atof(tmp.c_str());
248  } else {
249  cout << "bson iterator isn't or not contains a numeric value." << endl;
250  return false;
251  }
252  return true;
253 }
