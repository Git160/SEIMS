93 int clsPI_SVSC::Execute() {
94  //check input data
95  CheckInputData();
97  InitialOutputs();
98 
99  int julian = JulianDay(m_date);
100 #pragma omp parallel for
101  for (int i = 0; i < this->m_nCells; i++) {
102  if (m_P[i] > 0.f) {
103 #ifdef STORM_MODE
104  m_P[i] = m_P[i] * m_hilldt / 3600.f * cos(atan(m_slope[i]));
106 #endif // STORM_MODE
107  //interception storage capacity
108  float degree = 2.f * PI * (julian - 87.f) / 365.f;
110  float min = m_minSt[i];
111  float max = m_maxSt[i];
112  float capacity = min + (max - min) * pow(0.5f + 0.5f * sin(degree), m_Pi_b);
113 
114  //interception, currently, m_st[i] is storage of (t-1) time step
115  float availableSpace = capacity - m_st[i];
116  if (availableSpace < 0) {
117  availableSpace = 0.f;
118  }
119 
120  if (availableSpace < m_P[i]) {
121  m_interceptionLoss[i] = availableSpace;
122  } else {
123  m_interceptionLoss[i] = m_P[i];
124  }
125 
126  //net precipitation
127  m_netPrecipitation[i] = m_P[i] - m_interceptionLoss[i];
128  m_st[i] += m_interceptionLoss[i];
129  } else {
130  m_interceptionLoss[i] = 0.f;
131  m_netPrecipitation[i] = 0.f;
132  }
133 #ifndef STORM_MODE
134  //evaporation
135  if (m_st[i] > m_PET[i]) {
136  m_evaporationLoss[i] = m_PET[i];
137  } else {
138  m_evaporationLoss[i] = m_st[i];
139  }
140  m_st[i] -= m_evaporationLoss[i];
141 #endif
142  }
143  return 0;
144 }
