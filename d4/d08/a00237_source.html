200 template<typename T>
201 void GetNumericFromBsonIterator(bson_iter_t *iter, T &numericvalue) {
202  try {
203  const bson_value_t *vv = bson_iter_value(iter);
204  if (vv->value_type == BSON_TYPE_INT32) {
205  numericvalue = (T) vv->value.v_int32;
206  } else if (vv->value_type == BSON_TYPE_INT64) {
207  numericvalue = (T) vv->value.v_int64;
208  } else if (vv->value_type == BSON_TYPE_DOUBLE) {
209  numericvalue = (T) vv->value.v_double;
210  } else if (vv->value_type == BSON_TYPE_UTF8) {
211  string tmp = vv->value.v_utf8.str;
212  numericvalue = (T) atof(tmp.c_str());
213  } else {
214  throw ModelException("MongoUtil", "GetNumericFromBsonIterator",
215  "bson iterator isn't or not contains a numeric value.\n");
216  }
217  }
218  catch (ModelException e) {
219  cout << e.toString() << endl;
220  exit(EXIT_FAILURE);
221  }
222 }
