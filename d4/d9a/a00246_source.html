755 template<typename T>
756 void utilsMath::basicStatistics(T *values, int num, double **derivedvalues, T exclude /* = (T) NODATA_VALUE */) {
757  double *tmpstats = new double[6];
758  double maxv = MISSINGFLOAT;
759  double minv = MAXIMUMFLOAT;
760  int validnum = 0;
761  double sumv = 0.;
762  double std = 0.;
763  for (int i = 0; i < num; i++) {
764  if (utilsMath::FloatEqual(values[i], exclude)) continue;
765  if (maxv < values[i]) maxv = values[i];
766  if (minv > values[i]) minv = values[i];
767  validnum += 1;
768  sumv += values[i];
769  }
770  double mean = sumv / (double) validnum;
771 #pragma omp parallel for reduction(+:std)
772  for (int i = 0; i < num; i++) {
773  if (!utilsMath::FloatEqual(values[i], exclude)) {
774  std += (values[i] - mean) * (values[i] - mean);
775  }
776  }
777  std = sqrt(std / (double) validnum);
778  tmpstats[0] = (double) validnum;
779  tmpstats[1] = mean;
780  tmpstats[2] = maxv;
781  tmpstats[3] = minv;
782  tmpstats[4] = std;
783  tmpstats[5] = maxv - minv;
784  *derivedvalues = tmpstats;
785 }

789 utilsMath::basicStatistics(T **values, int num, int lyrs, double ***derivedvalues, T exclude /* = (T) NODATA_VALUE */) {

803  utilsArray::Initialize1DArray(lyrs, sumv, 0.);
804  for (int i = 0; i < num; i++) {
805  for (int j = 0; j < lyrs; j++) {

807  if (tmpstats[2][j] < values[i][j]) tmpstats[2][j] = values[i][j];
808  if (tmpstats[3][j] > values[i][j]) tmpstats[3][j] = values[i][j];
809  tmpstats[0][j] += 1;
810  sumv[j] += values[i][j];
811  }

815  tmpstats[5][j] = tmpstats[2][j] - tmpstats[3][j];

819  double tmpstd = 0;

823  tmpstd += (values[i][j] - tmpstats[1][j]) * (values[i][j] - tmpstats[1][j]);
