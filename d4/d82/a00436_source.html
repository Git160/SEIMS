92  vector<string> vecSites = SplitString(sitesList, ',');
93  int nSites = int(vecSites.size());
94  //convert from string to int, the IDList is in order in MongoDB

96  char* end = nullptr;
97  errno = 0;
98  for (int iSite = 0; iSite < nSites; iSite++) {
99  siteIDList.push_back(strtol(vecSites[iSite].c_str(), &end, 10)); // deprecated atoi);
100  }
101  //sort(siteIDList.begin(), siteIDList.end());
102 
103  bson_t* query = bson_new();
104  build_query_bson(nSites, siteIDList, siteType, query);
105  //printf("%s\n",bson_as_json(query,NULL));
106 
107  std::unique_ptr<MongoCollection> collection(new MongoCollection(m_conn->GetCollection(hydroDBName, DB_TAB_SITES)));
108  mongoc_cursor_t* cursor = collection->ExecuteQuery(query);
109 
110  const bson_t* record = NULL;
111  float* pEle = new float[nSites];
112  float* pLat = new float[nSites];
113  float ele, lat;
114  bool hasData = false;
115  while (mongoc_cursor_more(cursor) && mongoc_cursor_next(cursor, &record)) {
116  hasData = true;
117  bson_iter_t iter;
118  long long int siteIndex = -1;
119  int curSiteID = -1;
120  if (bson_iter_init(&iter, record) && bson_iter_find(&iter, MONG_HYDRO_DATA_SITEID)) {
121  GetNumericFromBsonIterator(&iter, curSiteID);
122  } else {
123  throw ModelException("Measurement", "Measurement",
124  "The Site ID field does not exist in Sites table, Please check and retry.");
125  }
126  auto siteIDIter = find(siteIDList.begin(), siteIDList.end(), curSiteID);
127  siteIndex = distance(siteIDList.begin(), siteIDIter);
128  if (bson_iter_init(&iter, record) && bson_iter_find(&iter, MONG_HYDRO_SITE_LAT)) {
129  GetNumericFromBsonIterator(&iter, lat);
130  } else {
131  throw ModelException("InputStation", "ReadSitesInfo", "The Lat field does not exist in Sites table.");
132  }
133 
134  if (bson_iter_init(&iter, record) && bson_iter_find(&iter, MONG_HYDRO_SITE_ELEV)) {
135  GetNumericFromBsonIterator(&iter, ele);
136  } else {
137  throw ModelException("InputStation", "ReadSitesInfo", "The Lat field does not exist in Sites table.");
138  }
139  pLat[siteIndex] = lat;
140  pEle[siteIndex] = ele;
141  }
142  if (!hasData) {
143  throw ModelException("InputStation", "ReadSitesInfo", "Query failed.");
144  }
145 
146  mongoc_cursor_destroy(cursor);
147  bson_destroy(query);
148  m_elevation[siteType] = pEle;
149  m_latitude[siteType] = pLat;
150  m_numSites[siteType] = nSites;
151 }
