586 bool MGTOpt_SWAT::GetOperationCode(int i, int &factoryID, vector<int> &nOps) {
587  if (m_mgtFactory.empty()) return false;
588  int curLanduseID = int(m_landUse[i]);
589  int curMgtField = int(m_mgtFields[i]);
590  factoryID = -1;
592  for (map<int, BMPPlantMgtFactory *>::iterator it = m_mgtFactory.begin(); it != m_mgtFactory.end(); it++) {
594  if (curLanduseID == (it->first) / 100) {
597  if (it->second->GetLocations().empty() || ValueInVector(curMgtField, it->second->GetLocations())) {
598  factoryID = it->first;
599  break;
600  }
601  }
602  }
603  if (factoryID < 0) return false;
605  vector<int> tmpOpSeqences = m_mgtFactory[factoryID]->GetOperationSequence();
606  map < int, PlantManagementOperation * > tmpOperations = m_mgtFactory[factoryID]->GetOperations();
607  // get the next should be done sequence number
608  int curSeq = m_doneOpSequence[i];
609  int nextSeq = -1;
610  if (curSeq == -1 || (unsigned) curSeq == tmpOpSeqences.size() - 1) {
611  nextSeq = 0;
612  } else {
613  nextSeq = curSeq + 1;
614  }
615  int opCode = tmpOpSeqences[nextSeq];
616  // figure out the nextSeq is satisfied or not.
617  if (tmpOperations.find(opCode) != tmpOperations.end()) {
618  PlantManagementOperation *tmpOperation = tmpOperations.at(opCode);
620  bool dateDepent = false, huscDepent = false;
622  if (tmpOperation->GetMonth() != 0 && tmpOperation->GetDay() != 0) {
623  struct tm dateInfo;
624  LocalTime(m_date, &dateInfo);
625  if (dateInfo.tm_mon == tmpOperation->GetMonth() &&
626  dateInfo.tm_mday == tmpOperation->GetDay()) {
627  dateDepent = true;
628  }
629  }
631  if (tmpOperation->GetHUFraction() >= 0.f) {
632  float aphu;
633  if (FloatEqual(m_dormFlag[i], 1.f)) {
634  aphu = NODATA_VALUE;
635  } else {
636  if (tmpOperation->UseBaseHUSC() && FloatEqual(m_igro[i], 0.f)) // use base hu
637  {
638  aphu = m_phuBase[i];
639  if (aphu >= tmpOperation->GetHUFraction()) {
640  huscDepent = true;
641  }
642  } else { // use accumulated plant hu
643  aphu = m_phuAcc[i];
644  if (aphu >= tmpOperation->GetHUFraction()) {
645  huscDepent = true;
646  }
647  }
648  }
649  }
652  if (dateDepent || huscDepent) {
653  nOps.push_back(opCode);
654  m_doneOpSequence[i] = nextSeq;
655  }
656  }
657  if (nOps.empty()) return false;
658  return true;
659 }

783 void MGTOpt_SWAT::ExecuteIrrigationOperation(int i, int &factoryID, int nOp) {
784  IrrigationOperation *curOperation = (IrrigationOperation *) m_mgtFactory[factoryID]->GetOperations()[nOp];
787  int m_irrSource;
789  int m_irrNo;
791  float m_irrApplyDepth;
794  float m_irrEfficiency;
795  m_irrSource = curOperation->IRRSource();
796  m_irrNo = (curOperation->IRRNo() <= 0) ? (int) m_subBsnID[i] : curOperation->IRRNo();
797  m_irrApplyDepth = curOperation->IRRApplyDepth();
798  m_irrEfficiency = curOperation->IRREfficiency();
799  m_irrFlag[i] = 1;
800  int tmpSubbsnID = (int) m_subBsnID[i];
801  if (m_irrSource > IRR_SRC_RES)
802  {
805  float vmma = 0.f;
806  float vmm = 0.f;
807  float cnv = 0.f;
808  float vmxi = 0.f;
809  float vol = 0.f;
810  float vmms = 0.f;
811  float vmmd = 0.f;
812  if (m_nCellsSubbsn.find(m_irrNo) == m_nCellsSubbsn.end()) {
814  m_irrNo = int(m_subBsnID[i]);
815  }
816 
817  cnv = m_nAreaSubbsn[m_irrNo] * 10.f;
818  switch (m_irrSource) {
820  case IRR_SRC_SHALLOW:
821  if (m_shallowWaterDepth[tmpSubbsnID] < UTIL_ZERO) {
822  m_shallowWaterDepth[tmpSubbsnID] = 0.f;
823  }
824  vmma += m_shallowWaterDepth[tmpSubbsnID] * cnv * m_irrEfficiency;
825  vmms = vmma;
826  vmma /= m_nCellsSubbsn[m_irrNo];
827  vmm = min(m_soilSumFC[i], vmma);
828  break;
829  case IRR_SRC_DEEP:vmma += m_deepWaterDepth[tmpSubbsnID] * cnv * m_irrEfficiency;
830  vmmd = vmma;
831  vmma /= m_nCellsSubbsn[m_irrNo];
832  vmm = min(m_soilSumFC[i], vmma);
833  break;
834  case IRR_SRC_OUTWTSD:
835  vmm = m_soilSumFC[i];
836  break;
837  }
839  if (vmm > 0.f) {
840  cnv = m_cellArea * 10.f;
841  vmxi = (m_irrApplyDepth < UTIL_ZERO) ? m_soilSumFC[i] : m_irrApplyDepth;
842  if (vmm > vmxi) vmm = vmxi;
843  vol = vmm * cnv;
844  float pot_fr = 0.f;
845  if (FloatEqual(m_impoundTriger[i], 0.f) && m_potVol != NULL) {
848  pot_fr = 1.f;
849  if (m_impoundArea != NULL) {
850  m_potVol[i] += vol / (10.f * m_impoundArea[i]);
851  } else {
852  m_potVol[i] += vol / (10.f * m_cellArea);
853  }
854  m_appliedWater[i] = vmm;
855  } else {
856  pot_fr = 0.f;
858  m_appliedWater[i] = vmm * (1.f - curOperation->IRRSQfrac());
859  m_irrSurfQWater[i] = vmm * curOperation->IRRSQfrac();
860  }
862  if (pot_fr > UTIL_ZERO) {
863  vol = m_appliedWater[i] * cnv * m_irrEfficiency;
864  }
865  switch (m_irrSource) {
866  case IRR_SRC_SHALLOW:cnv = m_nAreaSubbsn[m_irrNo] * 10.f;
867  vmma = 0.f;
868  if (vmms > -0.01f) {
869  vmma = vol * m_shallowWaterDepth[tmpSubbsnID] * cnv / vmms;
870  }
871  vmma /= cnv;
872  m_shallowWaterDepth[tmpSubbsnID] -= vmma;
873  if (m_shallowWaterDepth[tmpSubbsnID] < 0.f) {
874  vmma += m_shallowWaterDepth[tmpSubbsnID];
875  m_shallowWaterDepth[tmpSubbsnID] = 0.f;
876  }
877  m_shallowIrrWater[i] += vmma;
878  break;
879  case IRR_SRC_DEEP:cnv = m_nAreaSubbsn[m_irrNo] * 10.f;
880  vmma = 0.f;
881  if (vmmd > 0.01f) {
882  vmma = vol * (m_deepWaterDepth[tmpSubbsnID] * cnv / vmmd);
883  }
884  vmma /= cnv;
885  m_deepWaterDepth[tmpSubbsnID] -= vmma;
886  if (m_deepWaterDepth[tmpSubbsnID] < 0.f) {
887  vmma += m_deepWaterDepth[tmpSubbsnID];
888  m_deepWaterDepth[tmpSubbsnID] = 0.f;
889  }
890  m_deepIrrWater[i] += vmma;
891  break;
892  }
893  }
894  }
