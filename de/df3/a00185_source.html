19 def matchSubbasin(siteDict):
20  """
21  Match the ID of subbasin
22  1. Read the coordinates of each subbasin's outlet, and
23  the outlet ID of the whole basin (not finished yet)
24  2. If the isOutlet field equals to
25  2.1 - 0, then return the subbasinID of the site's location
26  2.2 - 1, then return the outlet ID of the whole basiin
27  2.3 - 2, then return the outlet ID of nearest subbasin
28  2.4 - 3, then return the outlet IDs of the conjunct subbasins
29  :param siteDict:
30  :return:
31  """
32  subbasinRasterFile = WORKING_DIR + os.sep + DIR_NAME_GEODATA2DB + os.sep + subbasinOut
33  subbasinRaster = RasterUtilClass.ReadRaster(subbasinRasterFile)
34  localx = siteDict.get(Tag_ST_LocalX.upper())
35  localy = siteDict.get(Tag_ST_LocalY.upper())
36  siteType = siteDict.get(Tag_ST_IsOutlet.upper())
37  subbasinID = subbasinRaster.GetValueByXY(localx, localy)
38  if subbasinID is None and siteType != 1: # the site is not inside the basin and not the outlet either.
39  return False, None
40  if siteType == 0:
41  return True, [subbasinID]
42  elif siteType == 1:
43  return True, [subbasinID] # TODO, get outlet ID of the whole basin, refers to gen_subbasin.py
44  elif siteType == 2:
45  return True, [subbasinID] # TODO
