20 clsReach::clsReach(const bson_t *&bsonTable) {
21  bson_iter_t iterator;
22  // read numerical parameters
23  for (int i = 0; i < REACH_PARAM_NUM; i++) {
24  float tmp_param;
25  if (bson_iter_init_find(&iterator, bsonTable, REACH_PARAM_NAME[i])) {
26  if (GetNumericFromBsonIterator(&iterator, tmp_param)) {
27  if (REACH_PARAM_NAME[i] == REACH_BOD && tmp_param < 1.e-6f) tmp_param = 1.e-6f;
28  m_paramMap.insert(make_pair(REACH_PARAM_NAME[i], tmp_param));
29  }
30  }
31  }
32  // read group informations, DO NOT THROW EXCEPTION!
33  if (!bson_iter_init_find(&iterator, bsonTable, REACH_GROUP)) {
34  cout << "No GROUP field found!" << endl;
35  return;
36  }
37  m_groupNumber = SplitStringForInt(GetStringFromBsonIterator(&iterator), ',');
38  for (int i = 0; i < REACH_GROUP_METHOD_NUM; i++) {
39  if (!bson_iter_init_find(&iterator, bsonTable, REACH_GROUP_NAME[i])) continue;
40  vector<int> gIdx = SplitStringForInt(GetStringFromBsonIterator(&iterator), ',');
41  if (m_groupNumber.size() != gIdx.size()) {
42  cout << "The size of group method " << REACH_GROUP_NAME[i] << " is not equal to GROUP! " << endl;
43  continue;
44  }
45  map<int, int> group_index;
46  for (int j = 0; j < m_groupNumber.size(); j++) {
47  group_index.insert(make_pair(m_groupNumber[j], gIdx[j]));
48  }
49  if (group_index.empty()) continue;
50  m_groupIndex.insert(make_pair(REACH_GROUP_NAME[i], group_index));
51  }
52 }

76 clsReaches::clsReaches(MongoClient *conn, string &dbName, string collectionName,

90  this->m_reachNum = collection->QueryRecordsCount();
91  if (this->m_reachNum < 0) {
92  bson_destroy(b);
93  throw ModelException("clsReaches", "ReadAllReachInfo",
94  "Failed to get document number of collection: " + collectionName + ".\n");
95  }
96  this->m_reachUpStream.resize(this->m_reachNum + 1);
97 
98  mongoc_cursor_t *cursor = collection->ExecuteQuery(b);
99  const bson_t *bsonTable;
100  while (mongoc_cursor_more(cursor) && mongoc_cursor_next(cursor, &bsonTable)) {
101  clsReach *curReach = new clsReach(bsonTable);
102  int subID = (int) curReach->Get(REACH_SUBBASIN);
103  this->m_reachesMap.insert(make_pair(subID, curReach));
104  }
105  // In SEIMS, reach ID is the same as Index of array and vector.
106  for (int i = 1; i <= m_reachNum; i++) {
107  int downStreamId = (int) (m_reachesMap.at(i)->Get(REACH_DOWNSTREAM));
108  m_reachDownStream.insert(make_pair(i, downStreamId));
109  if (downStreamId <= 0 || downStreamId > m_reachNum) continue;
110  m_reachUpStream[downStreamId].push_back(i);
111  }
112  // Build layers of reaches according to layering method
113  m_reachLayers.clear();
114  for (int i = 1; i <= m_reachNum; i++) {
115  int order = (int) (m_reachesMap.at(i)->Get(REACH_UPDOWN_ORDER));
116  if (mtd == DOWN_UP) order = (int) (m_reachesMap.at(i)->Get(REACH_DOWNUP_ORDER));
117  if (m_reachLayers.find(order) == m_reachLayers.end()) {
118  m_reachLayers.insert(make_pair(order, vector<int>()));
119  }
120  m_reachLayers.at(order).push_back(i);
121  }
122 
123  bson_destroy(b);
124  mongoc_cursor_destroy(cursor);
