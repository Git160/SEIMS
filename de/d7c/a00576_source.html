120 float GridLayeringDinf::GetPercentage(float angle, int di, int dj) {
121  float a = 4.f * angle / PI;
122  int n = int(a) % 7;
123  float r = a - n;
124  switch (n) {
125  case 0: { return (di != 0 ? r : 1 - r); }
126  case 1: { return (dj == 0 ? r : 1 - r); }
127  case 2: { return (dj != 0 ? r : 1 - r); }
128  case 3: { return (di == 0 ? r : 1 - r); }
129  case 4: { return (di != 0 ? r : 1 - r); }
130  case 5: { return (dj == 0 ? r : 1 - r); }
131  case 6: { return (dj != 0 ? r : 1 - r); }
132  case 7: { return (di == 0 ? r : 1 - r); }
133  default: {
134  cout << "Invalid angle value: " << angle << endl;
135  exit(-1);
136  }
