48  def read_param_ranges(self):
49  """Read param_rng.def file
50 
51  name,lower_bound,upper_bound,group,dist
52  (group and dist are optional)
53 
54  e.g.,
55  Param1,0,1[,Group1][,dist1]
56  Param2,0,1[,Group2][,dist2]
57  Param3,0,1[,Group3][,dist3]
58 
59  Returns:
60  a dictionary containing:
61  - names - the names of the parameters
62  - bounds - a list of lists of lower and upper bounds
63  - num_vars - a scalar indicating the number of variables
64  (the length of names)
65  - groups - a list of group names (strings) for each variable
66  - dists - a list of distributions for the problem,
67  None if not specified or all uniform
68  """
69  client = ConnectMongoDB(self.cfg.hostname, self.cfg.port)
70  conn = client.get_conn()
71  db = conn[self.cfg.spatial_db]
72  collection = db['PARAMETERS']
73 
74  names = list()
75  bounds = list()
76  groups = list()
77  dists = list()
78  num_vars = 0
79  items = read_data_items_from_txt(self.cfg.param_range_def)
80  for item in items:
81  if len(item) < 3:
82  continue
83  # find parameter name, print warning message if not existed
84  cursor = collection.find({'NAME': item[0]}, no_cursor_timeout=True)
85  if not cursor.count():
86  print ('WARNING: parameter %s is not existed!' % item[0])
87  continue
88  num_vars += 1
89  names.append(item[0])
90  bounds.append([float(item[1]), float(item[2])])
91  # If the fourth column does not contain a group name, use
92  # the parameter name
93  if len(item) >= 4:
94  groups.append(item[3])
95  else:
96  groups.append(item[0])
97  if len(item) >= 5:
98  dists.append(item[4])
99  else:
100  dists.append('unif')
101  if groups == names:
102  groups = None
103  elif len(set(groups)) == 1:
104  raise ValueError('''Only one group defined, results will not be
105  meaningful''')
106 
107  # setting dists to none if all are uniform
108  # because non-uniform scaling is not needed
109  if all([d == 'unif' for d in dists]):
110  dists = None
111 
112  self.param_defs = {'names': names, 'bounds': bounds,
113  'num_vars': num_vars, 'groups': groups, 'dists': dists}
114  # Save as pickle
115  pickle_param_defs = open(self.cfg.psa_outpath + os.sep + 'param_defs.pickle', 'wb')
116  pickle.dump(self.param_defs, pickle_param_defs)
