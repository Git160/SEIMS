51 void BMPArealSrcFactory::ReadArealSourceManagements(MongoClient *conn, const string &bmpDBName) {
52  bson_t *b = bson_new();
53  bson_t *child1 = bson_new(), *child2 = bson_new();
54  BSON_APPEND_DOCUMENT_BEGIN(b, "$query", child1);
55  BSON_APPEND_INT32(child1, BMP_FLD_SUB, m_subScenarioId);
56  bson_append_document_end(b, child1);
57  BSON_APPEND_DOCUMENT_BEGIN(b, "$orderby", child2);
58  BSON_APPEND_INT32(child2, BMP_FLD_SEQUENCE, 1);
59  bson_append_document_end(b, child2);
60  bson_destroy(child1);
61  bson_destroy(child2);
62 
63  unique_ptr<MongoCollection> collection(new MongoCollection(conn->getCollection(bmpDBName, m_arealSrcMgtTab)));
64  mongoc_cursor_t *cursor = collection->ExecuteQuery(b);
65 
66  bson_iter_t iter;
67  const bson_t *bsonTable;
68 
70  int count = 1;
71  while (mongoc_cursor_next(cursor, &bsonTable)) {
72  m_arealSrcMgtSeqs.push_back(count);
73  m_arealSrcMgtMap[count] = new ArealSourceMgtParams(bsonTable, iter);
74  count++;
75  }
76  bson_destroy(b);
77  mongoc_cursor_destroy(cursor);
78 }

80 void BMPArealSrcFactory::ReadArealSourceLocations(MongoClient *conn, const string &bmpDBName) {
81  bson_t *b = bson_new();
82  bson_t *child1 = bson_new();
83  BSON_APPEND_DOCUMENT_BEGIN(b, "$query", child1);
84  BSON_APPEND_INT32(child1, BMP_ARSRC_FLD_CODE, m_arealSrc);
85  bson_append_document_end(b, child1);
86  bson_destroy(child1);
87 
88  unique_ptr<MongoCollection> collection(new MongoCollection(conn->getCollection(bmpDBName, m_arealSrcDistTab)));
89  mongoc_cursor_t *cursor = collection->ExecuteQuery(b);
90 
91  bson_iter_t iter;
92  const bson_t *bsonTable;
93 
94  while (mongoc_cursor_next(cursor, &bsonTable)) {
95  ArealSourceLocations *curArSrcLoc = new ArealSourceLocations(bsonTable, iter);
96  int curArSrcID = curArSrcLoc->GetArealSourceID();
97  if (ValueInVector(curArSrcID, m_arealSrcIDs)) {
98  m_arealSrcLocsMap[curArSrcID] = curArSrcLoc;
99  } else {
100  RemoveValueInVector(curArSrcID, m_arealSrcIDs);
101  }
102  }
103  bson_destroy(b);
104  mongoc_cursor_destroy(cursor);
105 }
