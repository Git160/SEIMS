53 int ReservoirMethod::Execute() {
54  CheckInputData();
55  InitialOutputs();
56  float QGConvert = 1.f * m_cellWth * m_cellWth / m_dt * 0.001f; // mm ==> m3/s
57  for (auto it = m_subbasinIDs.begin(); it != m_subbasinIDs.end(); ++it) {
58  int subID = *it;
59  Subbasin* curSub = m_subbasinsInfo->GetSubbasinByID(subID);
60  // get percolation from the bottom soil layer at the subbasin scale
61  int curCellsNum = curSub->GetCellCount();
62  int* curCells = curSub->GetCells();
63  float perco = 0.f;
64 #pragma omp parallel for reduction(+:perco)
65  for (int i = 0; i < curCellsNum; i++) {
66  int index = curCells[i];
67  float tmp_perc = m_soilPerco[index][CVT_INT(m_nSoilLyrs[index]) - 1];
68  if (tmp_perc > 0) perco += tmp_perc;
69  else m_soilPerco[index][CVT_INT(m_nSoilLyrs[index]) - 1] = 0.f;
70  }
71  perco /= curCellsNum; // mean mm
75  float ratio2gw = 1.f;
76  perco *= ratio2gw;
77  curSub->SetPerco(perco);
78  //calculate EG, i.e. Revap
79  float revap = 0.f;
80  float fPET = 0.f;
81  float fEI = 0.f;
82  float fED = 0.f;
83  float fES = 0.f;
84  float plantEP = 0.f;
85 #pragma omp parallel for reduction(+:fPET, fEI, fED, fES, plantEP)
86  for (int i = 0; i < curCellsNum; i++) {
87  if (m_pet[curCells[i]] > 0.f) fPET += m_pet[curCells[i]];
88  if (m_IntcpET[curCells[i]] > 0.f) fEI += m_IntcpET[curCells[i]];
89  if (m_deprStoET[curCells[i]] > 0.f) fED += m_deprStoET[curCells[i]];
90  if (m_soilET[curCells[i]] > 0.f) fES += m_soilET[curCells[i]];
91  if (m_actPltET[curCells[i]] > 0.f) plantEP += m_actPltET[curCells[i]];
92  }
93  fPET /= curCellsNum;
94  fEI /= curCellsNum;
95  fED /= curCellsNum;
96  fES /= curCellsNum;
97  plantEP /= curCellsNum;
98 
99  curSub->SetPet(fPET);
100 
101  //if percolation < 0.01, EG will be 0. if percolation >= 0.01, EG will be calculated by equation (why? this is not used currently. Junzhi Liu 2016-08-14).
102  //if (perco >= 0.01f)
103  //{
104  revap = (fPET - fEI - fED - fES - plantEP) * m_gwStore[subID] / m_GWMAX;
105  if (revap != revap) {
106  cout << "fPET: " << fPET << ", fEI: " << fEI << ", fED: " << fED << ", fES: " << fES << ", plantEP: "
107  << plantEP << ", " << " subbasin ID: " << subID << ", gwStore: " << m_gwStore[subID] << endl;
108  throw ModelException(MID_GWA_RE, "Execute", "revap calculation wrong!");
109  }
110  revap = Max(revap, 0.f);
111  revap = Min(revap, perco);
112  //}
113  //float prevRevap = curSub->getEG();
114  //if (prevRevap != revap)
115  //{
116  // curSub->setEG(revap);
117  // curSub->setIsRevapChanged(true);
118  //}
119  //else
120  // curSub->setIsRevapChanged(false);
121  curSub->SetEg(revap);
122 
123  //deep percolation
124  float percoDeep = perco * m_dp_co;
125  curSub->SetPerde(percoDeep);
126 
127  // groundwater runoff (mm)
128  float slopeCoef = curSub->GetSlopeCoef();
129  float kg = m_Kg * slopeCoef;
130  float groundRunoff = kg * pow(m_gwStore[subID], m_Base_ex); //mm
131  if (groundRunoff != groundRunoff) {
132  cout << groundRunoff;
133  }
134 
135  float groundQ = groundRunoff * curCellsNum * QGConvert; // groundwater discharge (m3/s)
136 
137  float groundStorage = m_gwStore[subID];
138  groundStorage += perco - revap - percoDeep - groundRunoff;
139 
140  //add the ground water from bank storage, 2011-3-14
141  float gwBank = 0.f;
142  // at the first time step m_VgroundwaterFromBankStorage is nullptr
143  if (m_VgroundwaterFromBankStorage != nullptr) {
144  gwBank = m_VgroundwaterFromBankStorage[subID];
145  }
146  groundStorage += gwBank / curSub->GetArea() * 1000.f;
147 
148  groundStorage = Max(groundStorage, 0.f);
149  if (groundStorage > m_GWMAX) {
150  groundRunoff += groundStorage - m_GWMAX;
151  groundQ = groundRunoff * curCellsNum * QGConvert; // groundwater discharge (m3/s)
152  groundStorage = m_GWMAX;
153  }
154  curSub->SetRg(groundRunoff);
155  curSub->SetGw(groundStorage);
156  curSub->SetQg(groundQ);
157  if (groundStorage != groundStorage) {
158  std::ostringstream oss;
159  oss << perco << "\t" << revap << "\t" << percoDeep << "\t" << groundRunoff << "\t" << m_gwStore[subID]
160  << "\t" << m_Kg << "\t" <<
161  m_Base_ex << "\t" << slopeCoef << endl;
162  throw ModelException("Subbasin", "setInputs", oss.str());
163  }
164  m_T_Perco[subID] = curSub->GetPerco();
165  m_T_Revap[subID] = curSub->GetEg();
166  m_T_PerDep[subID] = curSub->GetPerde();
167  m_T_RG[subID] = curSub->GetRg(); //get rg of specific subbasin
168  m_T_QG[subID] = curSub->GetQg(); //get qg of specific subbasin
169  m_petSubbasin[subID] = curSub->GetPet();
170  m_gwStore[subID] = curSub->GetGw();
171  }
172 
173  m_T_Perco[0] = m_subbasinsInfo->Subbasin2Basin(VAR_PERCO);
174  m_T_Revap[0] = m_subbasinsInfo->Subbasin2Basin(VAR_REVAP);
175  m_T_PerDep[0] = m_subbasinsInfo->Subbasin2Basin(VAR_PERDE);
176  m_T_RG[0] = m_subbasinsInfo->Subbasin2Basin(VAR_RG); // get rg of entire watershed
177  m_gwStore[0] = m_subbasinsInfo->Subbasin2Basin(VAR_GW_Q);
178  m_T_QG[0] = m_subbasinsInfo->Subbasin2Basin(VAR_QG); // get qg of entire watershed
179 
180  // output to GWWB
181  for (int i = 0; i <= m_nSubbsns; i++) {
182  m_T_GWWB[i][0] = m_T_Perco[i];
183  m_T_GWWB[i][1] = m_T_Revap[i];
184  m_T_GWWB[i][2] = m_T_PerDep[i];
185  m_T_GWWB[i][3] = m_T_RG[i];
186  m_T_GWWB[i][4] = m_gwStore[i];
187  m_T_GWWB[i][5] = m_T_QG[i];
188  }
189 
190  // update soil moisture
191  for (auto it = m_subbasinIDs.begin(); it != m_subbasinIDs.end(); ++it) {
192  Subbasin* sub = m_subbasinsInfo->GetSubbasinByID(*it);
193  int* cells = sub->GetCells();
194  int nCells = sub->GetCellCount();
195 #pragma omp parallel for
196  for (int i = 0; i < nCells; i++) {
197  m_soilWtrSto[cells[i]][CVT_INT(m_nSoilLyrs[cells[i]]) - 1] += sub->GetEg();
198  // TODO: Is it need to allocate revap to each soil layers??? By LJ
199  }
200  }
201  // DEBUG
202  //cout << "GWA_RE, cell id 17842, m_soilStorage: ";
203  //for (int i = 0; i < (int)m_soilLayers[17842]; i++)
204  // cout << m_soilStorage[17842][i] << ", ";
205  //cout << endl;
206  // END OF DEBUG
207  return 0;
208 }
