42  bson_t* b = bson_new();

44  BSON_APPEND_DOCUMENT_BEGIN(b, "$query", child1);
45  BSON_APPEND_INT32(child1, FLD_SCENARIO_SUB, m_subScenarioId);
46  bson_append_document_end(b, child1);

48  BSON_APPEND_INT32(child2, BMP_FLD_SEQUENCE, 1);
49  bson_append_document_end(b, child2);
50  bson_destroy(child1);

52 
53  std::unique_ptr<MongoCollection> collection(new MongoCollection(conn->GetCollection(bmpDBName, m_bmpCollection)));
54  mongoc_cursor_t* cursor = collection->ExecuteQuery(b);
55 
56  const bson_t* bsonTable;
57  bson_iter_t itertor;
58  int paramNum = 10;
59  m_parameters = new float[paramNum];
60  int count = 1;
61  while (mongoc_cursor_next(cursor, &bsonTable)) {
62  //int seqNo;
63  int mgtCode = -1, year = -9999, month = -9999, day = -9999;
64  float husc = 0.f;
65  bool usebaseHU = true;
66  if (bson_iter_init_find(&itertor, bsonTable, BMP_FLD_NAME)) {
67  m_name = GetStringFromBsonIterator(&itertor);
68  }
69  if (bson_iter_init_find(&itertor, bsonTable, BMP_PLTOP_FLD_LUCC)) {
70  GetNumericFromBsonIterator(&itertor, m_luccID);
71  }
72  if (m_luccID < 0) m_luccID = 1; // AGRL Agricultural Land-Generic
73  if (bson_iter_init_find(&itertor, bsonTable, BMP_PLTOP_FLD_MGTOP)) {
74  GetNumericFromBsonIterator(&itertor, mgtCode);
75  }
76  //if (bson_iter_init_find(&itertor,bsonTable,BMP_FLD_SEQUENCE))
77  // GetNumericFromBsonIterator(&itertor, seqNo);
78  if (bson_iter_init_find(&itertor, bsonTable, BMP_PLTOP_FLD_YEAR)) {
79  GetNumericFromBsonIterator(&itertor, year);
80  }
81  if (bson_iter_init_find(&itertor, bsonTable, BMP_PLTOP_FLD_MONTH)) {
82  GetNumericFromBsonIterator(&itertor, month);
83  }
84  if (bson_iter_init_find(&itertor, bsonTable, BMP_PLTOP_FLD_DAY)) {
85  GetNumericFromBsonIterator(&itertor, day);
86  }
87  if (bson_iter_init_find(&itertor, bsonTable, BMP_PLTOP_FLD_BASEHU)) {
88  usebaseHU = GetBoolFromBsonIterator(&itertor);
89  }
90  if (bson_iter_init_find(&itertor, bsonTable, BMP_PLTOP_FLD_HUSC)) {
91  GetNumericFromBsonIterator(&itertor, husc);
92  }
93  for (int i = 0; i < paramNum; i++) {
94  std::ostringstream oss;
95  oss << BMP_PLTOP_FLD_MGT_PRE << (i + 1);
96  if (bson_iter_init_find(&itertor, bsonTable, oss.str().c_str())) {
97  GetNumericFromBsonIterator(&itertor, m_parameters[i]);
98  }
99  }
100  int uniqueMgtCode = count * 1000 + mgtCode;
101  m_bmpSequence.push_back(uniqueMgtCode);
102  switch (mgtCode) {
103  case BMP_PLTOP_Plant:
104  m_bmpPlantOps[uniqueMgtCode] = new PltOp(mgtCode, usebaseHU, husc, year, month, day,
105  m_parameters);
106  break;
107  case BMP_PLTOP_Irrigation:
108  m_bmpPlantOps[uniqueMgtCode] = new IrrOp(mgtCode, usebaseHU, husc, year, month, day,
109  m_parameters);
110  break;
111  case BMP_PLTOP_Fertilizer:
112  m_bmpPlantOps[uniqueMgtCode] = new FertOp(mgtCode, usebaseHU, husc, year, month, day,
113  m_parameters);
114  break;
115  case BMP_PLTOP_Pesticide:
116  m_bmpPlantOps[uniqueMgtCode] = new PestOp(mgtCode, usebaseHU, husc, year, month, day,
117  m_parameters);
118  break;
119  case BMP_PLTOP_HarvestKill:
120  m_bmpPlantOps[uniqueMgtCode] = new HvstKillOp(mgtCode, usebaseHU, husc, year, month, day,
121  m_parameters);
122  break;
123  case BMP_PLTOP_Tillage:
124  m_bmpPlantOps[uniqueMgtCode] = new TillOp(mgtCode, usebaseHU, husc, year, month, day,
125  m_parameters);
126  break;
127  case BMP_PLTOP_Harvest:
128  m_bmpPlantOps[uniqueMgtCode] = new HvstOnlyOp(mgtCode, usebaseHU, husc, year, month, day,
129  m_parameters);
130  break;
131  case BMP_PLTOP_Kill:
132  m_bmpPlantOps[uniqueMgtCode] = new KillOp(mgtCode, usebaseHU, husc, year, month, day,
133  m_parameters);
134  break;
135  case BMP_PLTOP_Grazing:
136  m_bmpPlantOps[uniqueMgtCode] = new GrazOp(mgtCode, usebaseHU, husc, year, month, day,
137  m_parameters);
138  break;
139  case BMP_PLTOP_AutoIrrigation:
140  m_bmpPlantOps[uniqueMgtCode] = new AutoIrrOp(mgtCode, usebaseHU, husc, year, month, day,
141  m_parameters);
142  break;
143  case BMP_PLTOP_AutoFertilizer:
144  m_bmpPlantOps[uniqueMgtCode] = new AutoFertOp(mgtCode, usebaseHU, husc, year, month, day,

146  break;
