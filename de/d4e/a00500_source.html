50 bool TaskInfo::Build() {
51  if (!CheckInputData()) return false;
53  if (nullptr != subbsn_count_rank_ && !rank_subbsn_id_.empty() &&
54  !subbsn_rank_.empty() && !subbsn_layer_.empty() && !downstream_.empty() &&
55  !upstreams_.empty() && !upstreams_inrank_.empty() && !lyr_subbsns_.empty() &&
56  !srclyr_subbsns_.empty() && !nonsrclyr_subbsns_.empty())
57  return true;
59  Initialize1DArray(size_, subbsn_count_rank_, 0);
60  for (int irank = 0; irank < size_; irank++) {
61  for (int i = 0; i < max_len; i++) {
62  if (subbsn_id[irank * max_len + i] < 0) continue;
63  subbsn_count_rank_[irank]++;
64  subbsn_rank_[subbsn_id[irank * max_len + i]] = irank;
65  }
66  }
67  for (int i = rank_ * max_len; i < (rank_ + 1) * max_len; i++) {
68  if (subbsn_id[i] < 0) continue;
69  rank_subbsn_id_.emplace_back(subbsn_id[i]);
70  downstream_[subbsn_id[i]] = down_id[i] > 0 ? down_id[i] : -1;
71  }
72 #ifdef _DEBUG
73  cout << "Subbasin ID -> Rank ID" << endl;
74  for (auto it = subbsn_rank_.begin(); it != subbsn_rank_.end(); ++it) {
75  cout << it->first << " -> " << it->second << endl;
76  }
77 #endif
78  max_lyr_all_ = 0;
80  for (int i = 0; i < size_ * max_len; i++) {
81  if (subbsn_id[i] < 0) continue;
82  if (lyr_id[i] > max_lyr_all_) max_lyr_all_ = lyr_id[i];
83  subbsn_layer_[subbsn_id[i]] = lyr_id[i];
84  }
85  max_lyr_ = 0;
86  for (int i = 0; i < subbsn_count_rank_[rank_]; i++) {
87  int sub_idx = rank_ * max_len + i;
88  int sub_id = subbsn_id[sub_idx];
89  // classification according to the Layering method, i.e., up-down and down-up orders.
90  int stream_order = lyr_id[sub_idx];
91  if (stream_order > max_lyr_) { max_lyr_ = stream_order; }
92  if (lyr_subbsns_.find(stream_order) == lyr_subbsns_.end()) {
93 #ifdef HAS_VARIADIC_TEMPLATES
94  lyr_subbsns_.emplace(stream_order, vector<int>());
95 #else
96  lyr_subbsns_.insert(make_pair(stream_order, vector<int>()));
97 #endif
98  }
99  lyr_subbsns_[stream_order].emplace_back(sub_id);
100  if (up_count[sub_idx] == 0) {
101  // source subbasins of each layer
102  if (srclyr_subbsns_.find(stream_order) == srclyr_subbsns_.end()) {
103 #ifdef HAS_VARIADIC_TEMPLATES
104  srclyr_subbsns_.emplace(stream_order, vector<int>());
105 #else
106  srclyr_subbsns_.insert(make_pair(stream_order, vector<int>()));
107 #endif
108  }
109  srclyr_subbsns_[stream_order].emplace_back(sub_id);
110  } else {
111  if (nonsrclyr_subbsns_.find(stream_order) == nonsrclyr_subbsns_.end()) {
112 #ifdef HAS_VARIADIC_TEMPLATES
113  nonsrclyr_subbsns_.emplace(stream_order, vector<int>());
114 #else
115  nonsrclyr_subbsns_.insert(make_pair(stream_order, vector<int>()));
116 #endif
117  }
118  nonsrclyr_subbsns_[stream_order].emplace_back(sub_id);
119  }
120  }
121 #ifdef _DEBUG
122  cout << "Rank: " << rank_ << ", Source subbasins: " << endl;
123  for (auto it = srclyr_subbsns_.begin(); it != srclyr_subbsns_.end(); ++it) {
124  cout << " Layer ID: " << it->first << ": ";
125  for (auto it_id = it->second.begin(); it_id != it->second.end(); ++it_id) {
126  cout << *it_id << ", ";
127  }
128  cout << endl;
129  }
130  cout << endl;
131  cout << "Rank: " << rank_ << ", Non-Source subbasins: " << endl;
132  for (auto it = nonsrclyr_subbsns_.begin(); it != nonsrclyr_subbsns_.end(); ++it) {
133  cout << " Layer ID: " << it->first << ": ";
134  for (auto it_id = it->second.begin(); it_id != it->second.end(); ++it_id) {
135  cout << *it_id << ", ";
136  }
137  cout << endl;
138  }
139  cout << endl;
140 #endif /* _DEBUG */

143  for (int irank = 0; irank < size_; irank++) {
144  for (int i = 0; i < max_len; i++) {
145  int subid = subbsn_id[irank * max_len + i];
146  if (subid < 0) continue;
147  upstreams_inrank_[subid] = true; // No matter a subbasin has upstreams or not.

161  }
162  }

165  cout << "Subbasin ID -> Upstreams" << endl;

167  cout << it->first << " -> ";

169  cout << *it2 << ", ";
170  }
171  cout << endl;
172  }
173  cout << "Subbasin ID -> Is all upstream subbasins in the same rank?" << endl;
174  for (auto it = upstreams_inrank_.begin(); it != upstreams_inrank_.end(); ++it) {
175  cout << it->first << " -> " << it->second << endl;
176  }
177 #endif /* _DEBUG */
178  return true;
