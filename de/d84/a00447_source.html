28  """Import BMPs Scenario data to MongoDB
29  Args:
30  cfg: SEIMS configuration object
31  main_db: climate database
32  scenario_db: scenario database
33  Returns:
34  False if failed, otherwise True.
35  """
36  if not cfg.use_scernario:
37  return False
38  print ('Import BMP Scenario Data... ')
39  bmp_files = FileClass.get_filename_by_suffixes(cfg.scenario_dir, ['.txt'])
40  bmp_tabs = []
41  bmp_tabs_path = []
42  for f in bmp_files:
43  bmp_tabs.append(f.split('.')[0])
44  bmp_tabs_path.append(cfg.scenario_dir + os.sep + f)
45 
46  # initialize if collection not existed
47  c_list = scenario_db.collection_names()
48  for item in bmp_tabs:
49  if not StringClass.string_in_list(item.upper(), c_list):
50  scenario_db.create_collection(item.upper())
51  else:
52  scenario_db.drop_collection(item.upper())
53  # Read subbasin.tif and dist2Stream.tif
54  subbasin_r = RasterUtilClass.read_raster(cfg.spatials.subbsn)
55  dist2stream_r = RasterUtilClass.read_raster(cfg.spatials.dist2stream_d8)
56  # End reading
57  for j, bmp_txt in enumerate(bmp_tabs_path):
58  bmp_tab_name = bmp_tabs[j]
59  data_array = read_data_items_from_txt(bmp_txt)
60  field_array = data_array[0]
61  data_array = data_array[1:]
62  for item in data_array:
63  dic = {}
64  for i, field_name in enumerate(field_array):
65  if MathClass.isnumerical(item[i]):
66  v = float(item[i])
67  if v % 1. == 0.:
68  v = int(v)
69  dic[field_name.upper()] = v
70  else:
71  dic[field_name.upper()] = str(item[i]).upper()
72  if StringClass.string_in_list(ImportScenario2Mongo._LocalX, dic.keys()) and \
73  StringClass.string_in_list(ImportScenario2Mongo._LocalY, dic.keys()):
74  subbsn_id = subbasin_r.get_value_by_xy(
75  dic[ImportScenario2Mongo._LocalX.upper()],
76  dic[ImportScenario2Mongo._LocalY.upper()])
77  distance = dist2stream_r.get_value_by_xy(
78  dic[ImportScenario2Mongo._LocalX.upper()],
79  dic[ImportScenario2Mongo._LocalY.upper()])
80  if subbsn_id is not None and distance is not None:
81  dic[ImportScenario2Mongo._SUBBASINID] = int(subbsn_id)
82  dic[ImportScenario2Mongo._DISTDOWN] = float(distance)
83  scenario_db[bmp_tab_name.upper()].find_one_and_replace(dic, dic,
84  upsert=True)
85  else:
86  scenario_db[bmp_tab_name.upper()].find_one_and_replace(dic, dic,
87  upsert=True)
88  # print 'BMP tables are imported.'
89  # Write BMP database name into Model workflow database
90  c_list = main_db.collection_names()
91  if not StringClass.string_in_list(DBTableNames.main_scenario, c_list):
92  main_db.create_collection(DBTableNames.main_scenario)
93 
94  bmp_info_dic = dict()
95  bmp_info_dic[ImportScenario2Mongo._FLD_DB] = cfg.bmp_scenario_db
96  main_db[DBTableNames.main_scenario].find_one_and_replace(bmp_info_dic, bmp_info_dic,
97  upsert=True)
98  return True
