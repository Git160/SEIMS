411 void DataCenter::SetRaster(string &paraName, string &remoteFileName, SimulationModule *pModule) {
412  int n, lyrs;

427  (StringMatch(tmpParam->Change, PARAM_CHANGE_AC) && !FloatEqual(tmpParam->Impact, 0.f)) ||

429  (StringMatch(tmpParam->Change, PARAM_CHANGE_NC))) {
430  adjust_data = true;

432  }
434  if (raster->is2DRaster()) {

438  if (nullptr != data2D && adjust_data) {
439  m_initParameters[upperName]->Adjust2DRaster(n, raster->getLayers(), data2D);

442  else {
443  if (!raster->getRasterData(&n, &data)) {
444  throw ModelException("DataCenter", "SetRaster", "Load " + remoteFileName + " failed!");
445  }

447  m_initParameters[upperName]->Adjust1DRaster(n, data);
448  }
449  }
450  }
451  else {
452  raster = m_rsMap.at(remoteFileName);

458  }
459  string upperName = GetUpper(paraName);
460  pModule->Set2DData(paraName.c_str(), n, lyrs, data2D);
461  }
462  else {
463  if (!raster->getRasterData(&n, &data)) {
464  throw ModelException("DataCenter", "SetRaster", "Load " + remoteFileName + " failed!");
