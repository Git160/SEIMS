75 int PETPriestleyTaylor::Execute() {
76  CheckInputData();
77  InitialOutputs();
78  m_jday = JulianDay(m_date);
79 #pragma omp parallel for
80  for (int i = 0; i < m_nCells; i++) {
82  /* update base zero total heat units, src code from SWAT, subbasin.f
83  if (tmpav(j) > 0. .and. phutot(hru_sub(j)) > 0.01) then
84  phubase(j) = phubase(j) + tmpav(j) / phutot(hru_sub(j))
85  end if*/
86  if (m_jday == 1) m_phuBase[i] = 0.f;
87  if (m_meanTemp[i] > 0.f && m_phuAnn[i] > 0.01f) {
88  m_phuBase[i] += m_meanTemp[i] / m_phuAnn[i];
89  }
92  float raShortWave = m_sr[i] * (1.0f - 0.23f);
93  //if the mean T < T_snow, consider the snow depth is larger than 0.5mm.
94  if (m_meanTemp[i] < m_snowTemp) {
95  raShortWave = m_sr[i] * (1.0f - 0.8f);
96  }
97 
99  MaxSolarRadiation(m_jday, m_cellLat[i], m_dayLen[i], m_srMax);
100 
103  float satVaporPressure = SaturationVaporPressure(m_meanTemp[i]);
104  float actualVaporPressure = 0.f;
105  if (m_rhd[i] > 1) {
107  m_rhd[i] *= 0.01f;
108  } else if (m_rhd[i] < UTIL_ZERO) {
109  m_rhd[i] = UTIL_ZERO;
110  }
111  actualVaporPressure = m_rhd[i] * satVaporPressure;
112  if (actualVaporPressure < UTIL_ZERO) {
113  actualVaporPressure = UTIL_ZERO;
114  }
115  m_vpd[i] = satVaporPressure - actualVaporPressure;
116  if (m_vpd[i] < 0.f) {
117  m_vpd[i] = 0.f;
118  }
119  float rbo = -(0.34f - 0.139f * sqrt(actualVaporPressure));
120  //cloud cover factor
121  float rto = 0.0f;
122  if (m_srMax >= 1.0e-4f) {
123  rto = 0.9f * (m_sr[i] / m_srMax) + 0.1f;
124  }
125  //net long-wave radiation
126  float tk = m_meanTemp[i] + 273.15f;
127  float raLongWave = rbo * rto * 4.9e-9f * pow(tk, 4.f);
128 
129  float raNet = raShortWave + raLongWave;
130 
132  //calculate the slope of the saturation vapor pressure curve
133  float dlt = 4098.f * satVaporPressure / pow(m_meanTemp[i] + 237.3f, 2.f);
134 
135  //calculate latent heat of vaporization(MJ/kg, from swat)
136  float latentHeat = LatentHeatVapor(m_meanTemp[i]);
137 
138  //calculate mean barometric pressure(kPa)
139  float pb = MeanBarometricPressure(m_dem[i]);
140  //psychrometric constant(kPa/deg C)
141  float gma = 1.013e-3f * pb / (0.622f * latentHeat);
142  float pet_alpha = 1.28f;
143 
144  float petValue = pet_alpha * (dlt / (dlt + gma)) * raNet / latentHeat;
145  m_pet[i] = m_petFactor * Max(0.f, petValue);
146  if (m_pet[i] != m_pet[i]) {
147  cout << "cell id: " << i << ", pet: " << m_pet[i] << ", meanT: " << m_meanTemp[i] <<
148  ", rhd: " << m_rhd[i] << ", rbo: " << rbo << ", sr: " << m_sr[i] << ", m_srMax: "
149  << m_srMax << ", rto: " << rto << ", satVaporPressure: " << satVaporPressure << endl;
150  throw ModelException(MID_PET_PT, "Execute", "Calculation error occurred!\n");
151  }
152  }
153  return 0;
154 }
