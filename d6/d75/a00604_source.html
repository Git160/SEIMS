52 int PETHargreaves::Execute() {
53  CheckInputData();
54  InitialOutputs();
55  m_jday = utils_time::JulianDay(m_date);
56  //cout<<m_jday<<","m_tMean[0]<<","<<m_tMax[0]<<","<<m_tMin[0]<<endl;
57 #pragma omp parallel for
58  for (int i = 0; i < m_nCells; ++i) {
60  /* update base zero total heat units, src code from SWAT, subbasin.f
61  if (tmpav(j) > 0. .and. phutot(hru_sub(j)) > 0.01) then
62  phubase(j) = phubase(j) + tmpav(j) / phutot(hru_sub(j))
63  end if*/
64  if (m_jday == 1) {
65  m_phuBase[i] = 0.f;
66  }
67  if (m_tMean[i] > 0.f && m_phutot[i] > 0.01f) {
68  m_phuBase[i] += m_tMean[i] / m_phutot[i];
69  }
70  MaxSolarRadiation(m_jday, m_cellLat[i], m_dayLen[i], m_srMax);
72  float latentHeat = 2.501f - 0.002361f * m_tMean[i];
75  float h0 = m_srMax * 37.59f / 30.0f;
78  float petValue = m_HCoef_pet * h0 * pow(Abs(m_tMax[i] - m_tMin[i]), 0.5f)
79  * (m_tMean[i] + 17.8f) / latentHeat;
80  m_pet[i] = m_petFactor * Max(0.0f, petValue);
82  float satVaporPressure = SaturationVaporPressure(m_tMean[i]);
83  float actualVaporPressure = 0.f;
84  if (m_rhd[i] > 1) {
85  actualVaporPressure = m_rhd[i] * satVaporPressure * 0.01f;
86  } else {
87  actualVaporPressure = m_rhd[i] * satVaporPressure;
88  }
89  m_vpd[i] = satVaporPressure - actualVaporPressure;
90  }
91  return 0;
92 }
