101  try:
102  # parallel on multiprocesor or clusters using SCOOP
103  from scoop import futures
104  fitnesses = futures.map(toolbox.evaluate, [cfg] * len(invalid_ind), invalid_ind)

132  offspring = [toolbox.clone(ind) for ind in offspring]

139  else:
