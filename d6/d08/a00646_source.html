236 int NutrientTransportSediment::Execute() {
237  CheckInputData();
238  if (m_cbnModel == 1) {
239  if (!CheckInputDataCFarmModel()) return false;
240  }
241  if (m_cbnModel == 2) {
242  if (!CheckInputDataCenturyModel()) return false;
243  }
244  InitialOutputs();
245  // initial nutrient to channel for each day
246  for (int i = 0; i < m_nSubbsns + 1; i++) {
247  m_surfRfSedOrgNToCh[i] = 0.f;
248  m_surfRfSedOrgPToCh[i] = 0.f;
249  m_surfRfSedAbsorbMinPToCh[i] = 0.f;
250  m_surfRfSedSorbMinPToCh[i] = 0.f;
251  }
252 
253 #pragma omp parallel for
254  for (int i = 0; i < m_nCells; i++) {
255  if (m_olWtrEroSed[i] < 1.e-4f) m_olWtrEroSed[i] = 0.f;
256  // CREAMS method for calculating enrichment ratio
257  m_enratio[i] = CalEnrichmentRatio(m_olWtrEroSed[i], m_surfRf[i], m_cellArea);
258 
259  //Calculates the amount of organic nitrogen removed in surface runoff
260  if (m_cbnModel == 0) {
261  OrgNRemovedInRunoffStaticMethod(i);
262  } else if (m_cbnModel == 1) {
263  OrgNRemovedInRunoffCFarmOneCarbonModel(i);
264  } else if (m_cbnModel == 2) {
265  OrgNRemovedInRunoffCenturyModel(i);
266  }
267  //Calculates the amount of organic and mineral phosphorus attached to sediment in surface runoff. psed.f of SWAT
268  OrgPAttachedtoSed(i);
269  }
270  // sum by subbasin
271  // See https://github.com/lreis2415/SEIMS/issues/36 for more descriptions. By lj
272 #pragma omp parallel
273  {
274  float* tmp_orgn2ch = new(nothrow) float[m_nSubbsns + 1];
275  float* tmp_orgp2ch = new(nothrow) float[m_nSubbsns + 1];
276  float* tmp_minpa2ch = new(nothrow) float[m_nSubbsns + 1];
277  float* tmp_minps2ch = new(nothrow) float[m_nSubbsns + 1];
278  for (int i = 0; i <= m_nSubbsns; i++) {
279  tmp_orgn2ch[i] = 0.f;
280  tmp_orgp2ch[i] = 0.f;
281  tmp_minpa2ch[i] = 0.f;
282  tmp_minps2ch[i] = 0.f;
283  }
284 #pragma omp for
285  for (int i = 0; i < m_nCells; i++) {
286  tmp_orgn2ch[CVT_INT(m_subbsnID[i])] += m_surfRfSedOrgN[i];
287  tmp_orgp2ch[CVT_INT(m_subbsnID[i])] += m_surfRfSedOrgP[i];
288  tmp_minpa2ch[CVT_INT(m_subbsnID[i])] += m_surfRfSedAbsorbMinP[i];
289  tmp_minps2ch[CVT_INT(m_subbsnID[i])] += m_surfRfSedSorbMinP[i];
290  }
291 #pragma omp critical
292  {
293  for (int i = 1; i <= m_nSubbsns; i++) {
294  m_surfRfSedOrgNToCh[i] += tmp_orgn2ch[i];
295  m_surfRfSedOrgPToCh[i] += tmp_orgp2ch[i];
296  m_surfRfSedAbsorbMinPToCh[i] += tmp_minpa2ch[i];
297  m_surfRfSedSorbMinPToCh[i] += tmp_minps2ch[i];
298  }
299  }
300  delete[] tmp_orgn2ch;
301  delete[] tmp_orgp2ch;
302  delete[] tmp_minpa2ch;
303  delete[] tmp_minps2ch;
304  tmp_orgn2ch = nullptr;
305  tmp_orgp2ch = nullptr;
306  tmp_minpa2ch = nullptr;
307  tmp_minps2ch = nullptr;
308  } /* END of #pragma omp parallel */
309  // sum all the subbasins and put the sum value in the zero-index of the array
310  for (int i = 1; i < m_nSubbsns + 1; i++) {
311  m_surfRfSedOrgNToCh[i] *= m_cellArea;
312  m_surfRfSedOrgPToCh[i] *= m_cellArea;
313  m_surfRfSedAbsorbMinPToCh[i] *= m_cellArea;
314  m_surfRfSedSorbMinPToCh[i] *= m_cellArea;
315  m_surfRfSedOrgNToCh[0] += m_surfRfSedOrgNToCh[i];
316  m_surfRfSedOrgPToCh[0] += m_surfRfSedOrgPToCh[i];
317  m_surfRfSedAbsorbMinPToCh[0] += m_surfRfSedAbsorbMinPToCh[i];
318  m_surfRfSedSorbMinPToCh[0] += m_surfRfSedSorbMinPToCh[i];
319  }
320  return 0;
321 }

323 void NutrientTransportSediment::OrgNRemovedInRunoffStaticMethod(const int i) {
324  //amount of organic N in first soil layer (orgninfl)
325  float orgninfl = 0.f;
326  //conversion factor (wt)
327  float wt = 0.f;
328  orgninfl = m_soilStabOrgN[i][0] + m_soilActvOrgN[i][0] + m_soilFrshOrgN[i][0];
329  wt = m_soilBD[i][0] * m_soilThk[i][0] * 0.01f;
330  //concentration of organic N in soil (concn)
331  float concn = orgninfl * m_enratio[i] / wt;
332  //Calculate the amount of organic nitrogen transported with sediment to the stream, equation 4:2.2.1 in SWAT Theory 2009, p271
333  m_surfRfSedOrgN[i] = 0.001f * concn * m_olWtrEroSed[i] * 0.001f / m_cellArea;
334  //update soil nitrogen pools
335  if (orgninfl > 1.e-6f) {
336  m_soilActvOrgN[i][0] = m_soilActvOrgN[i][0] - m_surfRfSedOrgN[i] * (m_soilActvOrgN[i][0] / orgninfl);
337  m_soilStabOrgN[i][0] = m_soilStabOrgN[i][0] - m_surfRfSedOrgN[i] * (m_soilStabOrgN[i][0] / orgninfl);
338  m_soilFrshOrgN[i][0] = m_soilFrshOrgN[i][0] - m_surfRfSedOrgN[i] * (m_soilFrshOrgN[i][0] / orgninfl);
339  if (m_soilActvOrgN[i][0] < 0.f) {
340  m_surfRfSedOrgN[i] = m_surfRfSedOrgN[i] + m_soilActvOrgN[i][0];
341  m_soilActvOrgN[i][0] = 0.f;
342  }
343  if (m_soilStabOrgN[i][0] < 0.f) {
344  m_surfRfSedOrgN[i] = m_surfRfSedOrgN[i] + m_soilStabOrgN[i][0];
345  m_soilStabOrgN[i][0] = 0.f;
346  }
347  if (m_soilFrshOrgN[i][0] < 0.f) {
348  m_surfRfSedOrgN[i] = m_surfRfSedOrgN[i] + m_soilFrshOrgN[i][0];
349  m_soilFrshOrgN[i][0] = 0.f;
350  }
351  }

358 void NutrientTransportSediment::OrgNRemovedInRunoffCenturyModel(const int i) {
359  float totOrgN_lyr0 = 0.f;
360  float wt1 = 0.f;
361  float er = 0.f;
362  float conc = 0.f;
363  float QBC = 0.f;
364  float VBC = 0.f;
365  float YBC = 0.f;
366  float YOC = 0.f;
367  float YW = 0.f;
368  float TOT = 0.f;
369  float YEW = 0.f;
370  float X1 = 0.f, PRMT_21 = 0.f;
371  float PRMT_44 = 0.f;
372  float XX = 0.f, DK = 0.f, V = 0.f, X3 = 0.f;
373  float CO = 0.f;
374  float CS = 0.f;
375  float perc_clyr = 0.f, latc_clyr = 0.f;
376 
377  totOrgN_lyr0 = m_sol_LSN[i][0] + m_sol_LMN[i][0] + m_sol_HPN[i][0] + m_sol_HSN[i][0];
378  wt1 = m_soilBD[i][0] * m_soilThk[i][0] * 0.01f;
379  er = m_enratio[i];
380  conc = totOrgN_lyr0 * er / wt1;
381  m_surfRfSedOrgN[i] = 0.001f * conc * m_olWtrEroSed[i] * 0.001f / m_cellArea;
383  if (totOrgN_lyr0 > UTIL_ZERO) {
384  float xx1 = 1.f - m_surfRfSedOrgN[i] / totOrgN_lyr0;
385  m_sol_LSN[i][0] *= xx1;
386  m_sol_LMN[i][0] *= xx1;
387  m_sol_HPN[i][0] *= xx1;
388  m_sol_HSN[i][0] *= xx1;
389  }
392  TOT = m_sol_HPC[i][0] + m_sol_HSC[i][0] + m_sol_LMC[i][0] + m_sol_LSC[i][0];
394  YEW = Min((m_olWtrEroSed[i] / m_cellArea + YW / m_cellArea) / m_soilMass[i][0], 0.9f);
395 
396  X1 = 1.f - YEW;
397  YOC = YEW * TOT;
398  m_sol_HSC[i][0] *= X1;
399  m_sol_HPC[i][0] *= X1;
400  m_sol_LS[i][0] *= X1;
401  m_sol_LM[i][0] *= X1;
402  m_sol_LSL[i][0] *= X1;
403  m_sol_LSC[i][0] *= X1;
404  m_sol_LMC[i][0] *= X1;
405  m_sol_LSLC[i][0] *= X1;
406  m_sol_LSLNC[i][0] = m_sol_LSC[i][0] - m_sol_LSLC[i][0];
407 
408  if (m_sol_BMC[i][0] > 0.01f) {
410  PRMT_21 = 1000.f;
411  m_sol_WOC[i][0] = m_sol_LSC[i][0] + m_sol_LMC[i][0] + m_sol_HPC[i][0] + m_sol_HSC[i][0] + m_sol_BMC[i][0];
412  DK = 0.0001f * PRMT_21 * m_sol_WOC[i][0];
413  X1 = m_soilSat[i][0];
414  if (X1 <= 0.f) X1 = 0.01f;
415  XX = X1 + DK;
416  V = m_surfRf[i] + m_soilPerco[i][0] + m_subSurfRf[i][0];
417  if (V > 1.e-10f) {
418  X3 = m_sol_BMC[i][0] * (1.f - exp(-V / XX));
419  PRMT_44 = 0.5;
420  CO = X3 / (m_soilPerco[i][0] + PRMT_44 * (m_surfRf[i] + m_subSurfRf[i][0]));
421  CS = PRMT_44 * CO;
422  VBC = CO * m_soilPerco[i][0];
423  m_sol_BMC[i][0] -= X3;
424  QBC = CS * (m_surfRf[i] + m_subSurfRf[i][0]);
426  if (YEW > 0.f) {
427  CS = DK * m_sol_BMC[i][0] / XX;
428  YBC = YEW * CS;
429  }
430  }
431  }
432  m_sol_BMC[i][0] -= YBC;
434  m_soilIfluCbn[i][0] = QBC * (m_subSurfRf[i][0] / (m_surfRf[i] + m_subSurfRf[i][0] + UTIL_ZERO));
435  m_soilPercoCbn[i][0] = VBC;
436  m_sedLossCbn[i] = YOC + YBC;
437 
438  latc_clyr += m_soilIfluCbn[i][0];
439  for (int k = 1; k < CVT_INT(m_nSoilLyrs[i]); k++) {
440  m_sol_WOC[i][k] = m_sol_LSC[i][k] + m_sol_LMC[i][k] + m_sol_HPC[i][k] + m_sol_HSC[i][k];
441  float Y1 = m_sol_BMC[i][k] + VBC;
442  VBC = 0.f;
443  if (Y1 > 0.01f) {
444  V = m_soilPerco[i][k] + m_subSurfRf[i][k];
445  if (V > 0.f) {
446  VBC = Y1 * (1.f - exp(-V / (m_soilSat[i][k] + 0.0001f * PRMT_21 * m_sol_WOC[i][k])));
447  }
448  }
449  m_soilIfluCbn[i][k] = VBC * (m_subSurfRf[i][k] / (m_subSurfRf[i][k] + m_soilPerco[i][k] + UTIL_ZERO));
450  m_soilPercoCbn[i][k] = VBC - m_soilIfluCbn[i][k];
451  m_sol_BMC[i][k] = Y1 - VBC;
452 
454  perc_clyr += m_soilPercoCbn[i][k];
455  latc_clyr += m_soilIfluCbn[i][k];
456  }
457  m_soilIfluCbnPrfl[i] = latc_clyr;
458  m_soilPercoCbnPrfl[i] = perc_clyr;

461 void NutrientTransportSediment::OrgPAttachedtoSed(const int i) {
462  //amount of phosphorus attached to sediment in soil (sol_attp)
463  float sol_attp = 0.f;
464  //fraction of active mineral/organic/stable mineral phosphorus in soil (sol_attp_o, sol_attp_a, sol_attp_s)
465  float sol_attp_o = 0.f;
466  float sol_attp_a = 0.f;
467  float sol_attp_s = 0.f;
468  //Calculate sediment
469  sol_attp = m_soilHumOrgP[i][0] + m_soilFrshOrgP[i][0] + m_soilActvMinP[i][0] + m_soilStabMinP[i][0];
470  if (m_soilManP != nullptr) {
471  sol_attp += m_soilManP[i][0];
472  }
473  if (sol_attp > 1.e-3f) {
474  sol_attp_o = (m_soilHumOrgP[i][0] + m_soilFrshOrgP[i][0]) / sol_attp;
475  if (m_soilManP != nullptr) {
476  sol_attp_o += m_soilManP[i][0] / sol_attp;
477  }
478  sol_attp_a = m_soilActvMinP[i][0] / sol_attp;
479  sol_attp_s = m_soilStabMinP[i][0] / sol_attp;
480  }
481  //conversion factor (mg/kg => kg/ha) (wt)
482  float wt = m_soilBD[i][0] * m_soilThk[i][0] * 0.01f;
483  //concentration of organic P in soil (concp)
484  float concp = 0.f;
485  concp = sol_attp * m_enratio[i] / wt;
486  //total amount of P removed in sediment erosion (sedp)
487  float sedp = 1.e-6f * concp * m_olWtrEroSed[i] / m_cellArea;
488  m_surfRfSedOrgP[i] = sedp * sol_attp_o;
489  m_surfRfSedAbsorbMinP[i] = sedp * sol_attp_a;
490  m_surfRfSedSorbMinP[i] = sedp * sol_attp_s;
491 
492  //if(i==100)cout << "sedp: " << sedp<< ",sol_attp_o: " << sol_attp_o << endl;
493  //modify phosphorus pools
494 
495  //total amount of P in mineral sediment pools prior to sediment removal (psedd) // Not used
496  //float psedd = 0.f;
497  //psedd = m_sol_actp[i][0] + m_sol_stap[i][0];
498 
499  //total amount of P in organic pools prior to sediment removal (porgg)
500  float porgg = 0.f;
501  porgg = m_soilHumOrgP[i][0] + m_soilFrshOrgP[i][0];
502  if (porgg > 1.e-3f) {
503  m_soilHumOrgP[i][0] = m_soilHumOrgP[i][0] - m_surfRfSedOrgP[i] * (m_soilHumOrgP[i][0] / porgg);
504  m_soilFrshOrgP[i][0] = m_soilFrshOrgP[i][0] - m_surfRfSedOrgP[i] * (m_soilFrshOrgP[i][0] / porgg);
505  }
506  m_soilActvMinP[i][0] = m_soilActvMinP[i][0] - m_surfRfSedAbsorbMinP[i];
507  m_soilStabMinP[i][0] = m_soilStabMinP[i][0] - m_surfRfSedSorbMinP[i];
508  if (m_soilHumOrgP[i][0] < 0.f) {
509  m_surfRfSedOrgP[i] = m_surfRfSedOrgP[i] + m_soilHumOrgP[i][0];
510  m_soilHumOrgP[i][0] = 0.f;
511  }
512  if (m_soilFrshOrgP[i][0] < 0.f) {
513  m_surfRfSedOrgP[i] = m_surfRfSedOrgP[i] + m_soilFrshOrgP[i][0];
514  m_soilFrshOrgP[i][0] = 0.f;
515  }
516  if (m_soilActvMinP[i][0] < 0.f) {
517  m_surfRfSedAbsorbMinP[i] = m_surfRfSedAbsorbMinP[i] + m_soilActvMinP[i][0];
518  m_soilActvMinP[i][0] = 0.f;
519  }
520  if (m_soilStabMinP[i][0] < 0.f) {
521  m_surfRfSedSorbMinP[i] = m_surfRfSedSorbMinP[i] + m_soilStabMinP[i][0];
522  m_soilStabMinP[i][0] = 0.f;
523  }
