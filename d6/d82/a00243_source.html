128  def evaluate_parallel(invalid_pops):
129  """Evaluate model by SCOOP or map, and set fitness of individuals
130  according to calibration step."""
131  popnum = len(invalid_pops)
132  try: # parallel on multi-processors or clusters using SCOOP
133  from scoop import futures
134  invalid_pops = list(futures.map(toolbox.evaluate, [cali_obj] * popnum, invalid_pops))
135  except ImportError or ImportWarning: # Python build-in map (serial)
136  invalid_pops = list(toolbox.map(toolbox.evaluate, [cali_obj] * popnum, invalid_pops))
137  for tmpind in invalid_pops:
138  if step == 'Q': # Step 1 Calibrating discharge
139  tmpind.fitness.values = tmpind.cali.efficiency_values('Q')
140  elif step == 'SED': # Step 2 Calibrating sediment

142  [tmpind.cali.efficiency_values('Q')[0]] # NSE-Q

144  # NSE > 0 is the preliminary condition to be a valid solution!
