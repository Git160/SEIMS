57  """Import HydroClimate sites table"""
58  sites_loc = dict()
59  site_data_items = read_data_items_from_txt(site_file)
60  site_flds = site_data_items[0]
61  for i in range(1, len(site_data_items)):
62  dic = dict()
63  for j in range(len(site_data_items[i])):
64  if StringClass.string_match(site_flds[j], StationFields.id):
65  dic[StationFields.id] = int(site_data_items[i][j])
66  elif StringClass.string_match(site_flds[j], StationFields.name):
67  dic[StationFields.name] = site_data_items[i][j]
68  elif StringClass.string_match(site_flds[j], StationFields.x):
69  dic[StationFields.x] = float(site_data_items[i][j])
70  elif StringClass.string_match(site_flds[j], StationFields.y):
71  dic[StationFields.y] = float(site_data_items[i][j])
72  elif StringClass.string_match(site_flds[j], StationFields.lon):
73  dic[StationFields.lon] = float(site_data_items[i][j])
74  elif StringClass.string_match(site_flds[j], StationFields.lat):
75  dic[StationFields.lat] = float(site_data_items[i][j])
76  elif StringClass.string_match(site_flds[j], StationFields.elev):
77  dic[StationFields.elev] = float(site_data_items[i][j])
78  elif StringClass.string_match(site_flds[j], StationFields.outlet):
79  dic[StationFields.outlet] = float(site_data_items[i][j])
80  dic[StationFields.type] = site_type
81  curfilter = {StationFields.id: dic[StationFields.id],
82  StationFields.type: dic[StationFields.type]}
83  hydro_clim_db[DBTableNames.sites].find_one_and_replace(curfilter, dic, upsert=True)
84 
85  if dic[StationFields.id] not in list(sites_loc.keys()):
86  sites_loc[dic[StationFields.id]] = SiteInfo(dic[StationFields.id],
87  dic[StationFields.name],
88  dic[StationFields.lat],
89  dic[StationFields.lon],
90  dic[StationFields.x],
91  dic[StationFields.y],
92  dic[StationFields.elev])
93  hydro_clim_db[DBTableNames.sites].create_index([(StationFields.id, ASCENDING),
94  (StationFields.type, ASCENDING)])
95  return sites_loc

113  @staticmethod
114  def ogrwkt2shapely(input_shape, id_field):
115  """Return shape objects list and ids list"""
116  # CAUTION, IMPORTANT
117  # Because shapely is dependent on sqlite, and the version is not consistent
118  # with GDAL executable (e.g., located in C:\GDAL_x64\bin), thus the shapely
119  # must be locally imported here.
120  from shapely.wkt import loads as shapely_loads
121  shapely_objects = list()
122  id_list = list()
123  # print(input_shape)
124  shp = ogr_Open(input_shape)
125  if shp is None:
126  raise RuntimeError('The input ESRI Shapefile: %s is not existed or has '
127  'no read permission!' % input_shape)
128  lyr = shp.GetLayer()
129  for n in range(0, lyr.GetFeatureCount()):
130  feat = lyr.GetFeature(n)
131  # This function may print Failed `CDLL(/opt/local/lib/libgeos_c.dylib)` in macOS
132  # Don't worry about that!
133  wkt_feat = shapely_loads(feat.geometry().ExportToWkt())
134  shapely_objects.append(wkt_feat)
135  if isinstance(id_field, text_type):
136  id_field = str(id_field)
137  id_index = feat.GetFieldIndex(id_field)
138  id_list.append(feat.GetField(id_index))
139  return shapely_objects, id_list

201  cfg.thiessen_field, type_list, cfg.storm_mode)
202 
203  # 2. Import geographic information of each sites to Hydro-Climate database
204  c_list = clim_db.collection_names()
205  tables = [DBTableNames.sites, DBTableNames.var_desc]
206  for tb in tables:
207  if not StringClass.string_in_list(tb, c_list):
208  clim_db.create_collection(tb)
209  ImportHydroClimateSites.variable_table(clim_db, cfg.hydro_climate_vars)
210  site_m_loc = ImportHydroClimateSites.sites_table(clim_db, cfg.Meteo_sites, DataType.m)
211  site_p_loc = ImportHydroClimateSites.sites_table(clim_db, cfg.prec_sites, DataType.p)
212  # print(site_m_loc, site_p_loc)
