230 bool GetNumericFromBsonIterator(bson_iter_t *iter, T &numericvalue) {
231  const bson_value_t *vv = bson_iter_value(iter);
232  if (vv->value_type == BSON_TYPE_INT32) {
233  numericvalue = (T) vv->value.v_int32;
234  } else if (vv->value_type == BSON_TYPE_INT64) {
235  numericvalue = (T) vv->value.v_int64;
236  } else if (vv->value_type == BSON_TYPE_DOUBLE) {
237  numericvalue = (T) vv->value.v_double;
238  } else if (vv->value_type == BSON_TYPE_UTF8) {
239  string tmp = vv->value.v_utf8.str;
240  numericvalue = (T) atof(tmp.c_str());
241  } else {
242  cout << "bson iterator isn't or not contains a numeric value." << endl;
243  return false;
244  }
245  return true;
246 }
