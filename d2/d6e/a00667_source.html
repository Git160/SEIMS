88 int NPS_Management::Execute() {
89  CheckInputData();
90  if (m_cellArea < 0.f) m_cellArea = m_cellWth * m_cellWth;
91  for (auto it = m_arealSrcFactory.begin(); it != m_arealSrcFactory.end(); ++it) {
93  if (!it->second->GetLocationLoadStatus()) {
94  it->second->SetArealSrcLocsMap(m_nCells, m_mgtFields);
95  }
97  vector<int>& tmpArealFieldIDs = it->second->GetArealSrcIDs();
98  vector<int>& tmpMgtSeqs = it->second->GetArealSrcMgtSeqs();
99  for (auto mgtSeqIter = tmpMgtSeqs.begin(); mgtSeqIter != tmpMgtSeqs.end(); ++mgtSeqIter) {
100  ArealSourceMgtParams* tmpMgtParams = it->second->GetArealSrcMgtMap().at(*mgtSeqIter);
101  if (tmpMgtParams->GetStartDate() != 0 && tmpMgtParams->GetEndDate() != 0) {
102  if (m_date < tmpMgtParams->GetStartDate() || m_date > tmpMgtParams->GetEndDate()) {
103  continue;
104  }
105  }
106  for (auto fIDIter = tmpArealFieldIDs.begin(); fIDIter != tmpArealFieldIDs.end(); ++fIDIter) {
107  float deltaWtrMM = 0.f, deltaNH4 = 0.f, deltaNO3 = 0.f, deltaOrgN = 0.f;
108  float deltaMinP = 0.f, deltaOrgP = 0.f;
109  map<int, ArealSourceLocations *>& tmpLocMap = it->second->GetArealSrcLocsMap();
110  if (tmpLocMap.find(*fIDIter) == tmpLocMap.end()) {
111  continue;
112  }
113  ArealSourceLocations* tmpLoc = tmpLocMap.at(*fIDIter);
114  float tmpSize = tmpLoc->GetSize();
115  int tmpNCells = tmpLoc->GetValidCells();
116  if (tmpMgtParams->GetWaterVolume() > 0.f) {
118  deltaWtrMM = tmpMgtParams->GetWaterVolume() * tmpSize * m_timestep *
119  1.1574074074074073e-05f / tmpNCells / m_cellArea * 1000.f;
120  }
122  float cvt = tmpSize * m_timestep * 1.1574074074074073e-05f / tmpNCells / m_cellArea * 10000.f;
123  if (tmpMgtParams->GetNH4() > 0.f) {
124  deltaNH4 = cvt * tmpMgtParams->GetNH4();
125  }
126  if (tmpMgtParams->GetNO3() > 0.f) {
127  deltaNO3 = cvt * tmpMgtParams->GetNO3();
128  }
129  if (tmpMgtParams->GetOrgN() > 0.f) {
130  deltaOrgN = cvt * tmpMgtParams->GetOrgN();
131  }
132  if (tmpMgtParams->GetMinP() > 0.f) {
133  deltaMinP = cvt * tmpMgtParams->GetMinP();
134  }
135  if (tmpMgtParams->GetOrgP() > 0.f) {
136  deltaOrgP = cvt * tmpMgtParams->GetOrgP();
137  }
138  vector<int>& tmpCellIdx = tmpLoc->GetCellsIndex();
139  for (auto idxIter = tmpCellIdx.begin(); idxIter != tmpCellIdx.end(); ++idxIter) {
141  if (deltaWtrMM > 0.f && nullptr != m_soilWtrSto) {
142  m_soilWtrSto[*idxIter][0] += deltaWtrMM;
143  }
144  if (deltaNO3 > 0.f && nullptr != m_soilNO3) {
145  m_soilNO3[*idxIter][0] += deltaNO3;
146  }
147  if (deltaNH4 > 0.f && nullptr != m_soilNH4) {
148  m_soilNH4[*idxIter][0] += deltaNH4;
149  }
150  if (deltaMinP > 0.f && nullptr != m_soilSolP) {
151  m_soilSolP[*idxIter][0] += deltaMinP;
152  }
153  if (deltaOrgN > 0.f && nullptr != m_soilStabOrgN) {
154  m_soilStabOrgN[*idxIter][0] += deltaOrgN;
155  }
156  if (deltaOrgP > 0.f && nullptr != m_soilHumOrgP) {
157  m_soilHumOrgP[*idxIter][0] += deltaOrgP;
158  }
159  }
160  }
161  }
162  }
163  return true;
