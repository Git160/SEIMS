57  @staticmethod
58  def nashcoef(obsvalues, simvalues):
59  """Calculate Nash coefficient.
60  Args:
61  obsvalues: observe values array
62  simvalues: simulate values array
63 
64  Returns:
65  NSE, or raise exception
66  """
67  if len(obsvalues) != len(simvalues):
68  raise ValueError("The size of observed and simulated values must be"
69  " the same for NSE calculation!")
70  n = len(obsvalues)
71  ave = sum(obsvalues) / n
72  a1 = 0.
73  a2 = 0.
74  for i in range(n):
75  a1 += pow(float(obsvalues[i]) - float(simvalues[i]), 2.)
76  a2 += pow(float(obsvalues[i]) - ave, 2.)
77  if a2 == 0.:
78  return 1.
79  return 1. - a1 / a2

81  @staticmethod
82  def rsquare(obsvalues, simvalues):
83  """Calculate R-square.
84  Args:
85  obsvalues: observe values array
86  simvalues: simulate values array
87 
88  Returns:
89  R-square value, or raise exception
90  """
91  if len(obsvalues) != len(simvalues):
92  raise ValueError("The size of observed and simulated values must be "
93  "the same for R-square calculation!")
94  n = len(obsvalues)
95  obs_avg = sum(obsvalues) / n
96  pred_avg = sum(simvalues) / n
97  obs_minus_avg_sq = 0.
98  pred_minus_avg_sq = 0.
99  obs_pred_minus_avgs = 0.
100  for i in range(n):
101  obs_minus_avg_sq += pow((obsvalues[i] - obs_avg), 2.)
102  pred_minus_avg_sq += pow((simvalues[i] - pred_avg), 2.)
103  obs_pred_minus_avgs += (obsvalues[i] - obs_avg) * (simvalues[i] - pred_avg)
104  # Calculate R-square
105  yy = (pow(obs_minus_avg_sq, 0.5) * pow(pred_minus_avg_sq, 0.5))
106  if yy == 0.:
107  return 1.
108  return pow((obs_pred_minus_avgs / yy), 2.)

349  17-07-04 lj - Handling subprocess crash in Windows, refers to

351  Args:
352  commands: string or list
353 
354  Returns:
355  output lines
356  """
357  print (commands)
358  use_shell = False
359  if isinstance(commands, list) or isinstance(commands, tuple):
360  use_shell = False
361  commands = ' '.join(c for c in commands)
362  if sysstr == 'Windows':
363  import ctypes
364  SEM_NOGPFAULTERRORBOX = 0x0002 # From MSDN
365  ctypes.windll.kernel32.SetErrorMode(SEM_NOGPFAULTERRORBOX)
366  subprocess_flags = 0x8000000 # win32con.CREATE_NO_WINDOW?
367  else:
368  subprocess_flags = 0
369  process = subprocess.Popen(commands, shell=use_shell, stdout=subprocess.PIPE,
370  stdin=open(os.devnull),
371  stderr=subprocess.STDOUT, universal_newlines=True,
372  creationflags=subprocess_flags)
373 
374  if process.stdout is None:
375  return ['']
376  out_lines = process.stdout.readlines()
377  err = False
378  for line in out_lines:
379  if 'ERROR' in line.upper():
380  err = True
381  break
382  if err or (process.returncode is not None and process.returncode != 0):
383  raise subprocess.CalledProcessError(-1, commands,
384  "ERROR occurred when running subprocess!")
