22  m_landuse = SplitStringForInt(landuse_str, '-');
23  }
24  if (bson_iter_init_find(&iter, bsonTable, BMP_ARSTRUCT_FLD_PARAMS)) {
25  string params_str = GetStringFromBsonIterator(&iter);
26  vector<string> params_strs = SplitString(params_str, '-');
27  for (vector<string>::iterator it = params_strs.begin(); it != params_strs.end(); it++) {
28  vector<string> tmp_param_items = SplitString(*it, ':');
29  assert(tmp_param_items.size() == 4);
30  ParamEffect tmp_pe;
31  tmp_pe.paramID = tmp_param_items[0];
32  tmp_pe.paramDesc = tmp_param_items[1];
33  tmp_pe.change = tmp_param_items[2];
34  tmp_pe.impact = (float)atof(tmp_param_items[3].c_str());
35  if (!m_parameters.insert(make_pair(tmp_pe.paramID, tmp_pe)).second) {
36  cout << "WARNING: Read BMPArealStruct failed: BMPID: " << m_id << ", param_name: "
37  << tmp_pe.paramID << endl;
38  }
39  }
