220 int NutrientTransportSediment::Execute() {
221  CheckInputData();
222  if (m_CbnModel == 1) {
223  if (!CheckInputData_CFARM()) { return false; }
224  }
225  if (m_CbnModel == 2) {
226  if (!CheckInputData_CENTURY()) { return false; }
227  }
228  InitialOutputs();
229  // initial nutrient to channel for each day
230  for (int i = 0; i < m_nSubbasins + 1; i++) {
231  m_sedorgnToCh[i] = 0.f;
232  m_sedorgpToCh[i] = 0.f;
233  m_sedminpaToCh[i] = 0.f;
234  m_sedminpsToCh[i] = 0.f;
235  }
236 
237 #pragma omp parallel for
238  for (int i = 0; i < m_nCells; i++) {
239  if (m_sedEroded[i] < 1.e-4f) m_sedEroded[i] = 0.f;
240  // CREAMS method for calculating enrichment ratio
241  m_enratio[i] = CalEnrichmentRatio(m_sedEroded[i], m_surfaceRunoff[i], m_cellArea);
242  //if(i == 1000) cout << ""<< m_sedEroded[i]<<","<< m_surfaceRunoff[i] << "," << m_enratio[i]<<endl;
243 
244  //Calculates the amount of organic nitrogen removed in surface runoff
245  if (m_CbnModel == 0) {
246  OrgNRemovedInRunoff_StaticMethod(i);
247  } else if (m_CbnModel == 1) {
248  OrgNRemovedInRunoff_CFARMOneCarbonModel(i);
249  } else if (m_CbnModel == 2) {
250  OrgNRemovedInRunoff_CENTURY(i);
251  }
252  //Calculates the amount of organic and mineral phosphorus attached to sediment in surface runoff. psed.f of SWAT
253  OrgPAttachedtoSed(i);
254  }
255  // sum by subbasin
256  // See https://github.com/lreis2415/SEIMS/issues/36 for more descriptions. By lj
257 #pragma omp parallel
258  {
259  float *tmp_orgn2ch = new float[m_nSubbasins + 1];
260  float *tmp_orgp2ch = new float[m_nSubbasins + 1];
261  float *tmp_minpa2ch = new float[m_nSubbasins + 1];
262  float *tmp_minps2ch = new float[m_nSubbasins + 1];
263  for (int i = 0; i <= m_nSubbasins; i++) {
264  tmp_orgn2ch[i] = 0.f;
265  tmp_orgp2ch[i] = 0.f;
266  tmp_minpa2ch[i] = 0.f;
267  tmp_minps2ch[i] = 0.f;
268  }
269 #pragma omp for
270  for (int i = 0; i < m_nCells; i++) {
271  tmp_orgn2ch[(int) m_subbasin[i]] += m_sedorgn[i];
272  tmp_orgp2ch[(int) m_subbasin[i]] += m_sedorgp[i];
273  tmp_minpa2ch[(int) m_subbasin[i]] += m_sedminpa[i];
274  tmp_minps2ch[(int) m_subbasin[i]] += m_sedminps[i];
275  }
276 #pragma omp critical
277  {
278  for (int i = 1; i <= m_nSubbasins; i++) {
279  m_sedorgnToCh[i] += tmp_orgn2ch[i];
280  m_sedorgpToCh[i] += tmp_orgp2ch[i];
281  m_sedminpaToCh[i] += tmp_minpa2ch[i];
282  m_sedminpsToCh[i] += tmp_minps2ch[i];
283  }
284  }
285  delete[] tmp_orgn2ch;
286  delete[] tmp_orgp2ch;
287  delete[] tmp_minpa2ch;
288  delete[] tmp_minps2ch;
289  } /* END of #pragma omp parallel */
290  // sum all the subbasins and put the sum value in the zero-index of the array
291  for (int i = 1; i < m_nSubbasins + 1; i++) {
292  m_sedorgnToCh[i] *= m_cellArea;
293  m_sedorgpToCh[i] *= m_cellArea;
294  m_sedminpaToCh[i] *= m_cellArea;
295  m_sedminpsToCh[i] *= m_cellArea;
296  m_sedorgnToCh[0] += m_sedorgnToCh[i];
297  m_sedorgpToCh[0] += m_sedorgpToCh[i];
298  m_sedminpaToCh[0] += m_sedminpaToCh[i];
299  m_sedminpsToCh[0] += m_sedminpsToCh[i];
300  }
301  return 0;
302 }

304 void NutrientTransportSediment::OrgNRemovedInRunoff_StaticMethod(int i) {
305  //amount of organic N in first soil layer (orgninfl)
306  float orgninfl = 0.f;
307  //conversion factor (wt)
308  float wt = 0.f;
309  orgninfl = m_sol_orgn[i][0] + m_sol_aorgn[i][0] + m_sol_fon[i][0];
310  wt = m_sol_bd[i][0] * m_soilThick[i][0] / 100.f;
311  //concentration of organic N in soil (concn)
312  float concn = 0.f;
313  concn = orgninfl * m_enratio[i] / wt;
314  //Calculate the amount of organic nitrogen transported with sediment to the stream, equation 4:2.2.1 in SWAT Theory 2009, p271
315  m_sedorgn[i] = 0.001f * concn * m_sedEroded[i] / 1000.f / m_cellArea;
316  //update soil nitrogen pools
317  if (orgninfl > 1.e-6f) {
318  m_sol_aorgn[i][0] = m_sol_aorgn[i][0] - m_sedorgn[i] * (m_sol_aorgn[i][0] / orgninfl);
319  m_sol_orgn[i][0] = m_sol_orgn[i][0] - m_sedorgn[i] * (m_sol_orgn[i][0] / orgninfl);
320  m_sol_fon[i][0] = m_sol_fon[i][0] - m_sedorgn[i] * (m_sol_fon[i][0] / orgninfl);
321  if (m_sol_aorgn[i][0] < 0.f) {
322  m_sedorgn[i] = m_sedorgn[i] + m_sol_aorgn[i][0];
323  m_sol_aorgn[i][0] = 0.f;
324  }
325  if (m_sol_orgn[i][0] < 0.f) {
326  m_sedorgn[i] = m_sedorgn[i] + m_sol_orgn[i][0];
327  m_sol_orgn[i][0] = 0.f;
328  }
329  if (m_sol_fon[i][0] < 0.f) {
330  m_sedorgn[i] = m_sedorgn[i] + m_sol_fon[i][0];
331  m_sol_fon[i][0] = 0.f;
332  }
333  }

340 void NutrientTransportSediment::OrgNRemovedInRunoff_CENTURY(int i) {
341  float totOrgN_lyr0 = 0.f;
342  float wt1 = 0.f;
343  float er = 0.f;
344  float conc = 0.f;
345  float sol_mass = 0.f;
346  float QBC = 0.f;
347  float VBC = 0.f;
348  float YBC = 0.f;
349  float YOC = 0.f;
350  float YW = 0.f;
351  float TOT = 0.f;
352  float YEW = 0.f;
353  float X1 = 0.f, PRMT_21 = 0.f;
354  float PRMT_44 = 0.f;
355  float XX = 0.f, DK = 0.f, V = 0.f, X3 = 0.f;
356  float CO = 0.f;
357  float CS = 0.f;
358  float perc_clyr = 0.f, latc_clyr = 0.f;
359 
360  totOrgN_lyr0 = m_sol_LSN[i][0] + m_sol_LMN[i][0] + m_sol_HPN[i][0] + m_sol_HSN[i][0];
361  wt1 = m_sol_bd[i][0] * m_soilThick[i][0] / 100.f;
362  er = m_enratio[i];
363  conc = totOrgN_lyr0 * er / wt1;
364  m_sedorgn[i] = 0.001f * conc * m_sedEroded[i] / 1000.f / m_cellArea;
366  if (totOrgN_lyr0 > UTIL_ZERO) {
367  float xx1 = (1.f - m_sedorgn[i] / totOrgN_lyr0);
368  m_sol_LSN[i][0] *= xx1;
369  m_sol_LMN[i][0] *= xx1;
370  m_sol_HPN[i][0] *= xx1;
371  m_sol_HSN[i][0] *= xx1;
372  }
374  sol_mass = m_soilThick[i][0] / 1000.f * 10000.f * m_sol_bd[i][0] * 1000.f *
375  (1.f - m_sol_rock[i][0] / 100.f);
376  TOT = m_sol_HPC[i][0] + m_sol_HSC[i][0] + m_sol_LMC[i][0] + m_sol_LSC[i][0];
379  YEW = Min((m_sedEroded[i] / m_cellArea + YW / m_cellArea) / sol_mass, 0.9f);
380 
381  X1 = 1.f - YEW;
382  YOC = YEW * TOT;
383  m_sol_HSC[i][0] *= X1;
384  m_sol_HPC[i][0] *= X1;
385  m_sol_LS[i][0] *= X1;
386  m_sol_LM[i][0] *= X1;
387  m_sol_LSL[i][0] *= X1;
388  m_sol_LSC[i][0] *= X1;
389  m_sol_LMC[i][0] *= X1;
390  m_sol_LSLC[i][0] *= X1;
391  m_sol_LSLNC[i][0] = m_sol_LSC[i][0] - m_sol_LSLC[i][0];
392 
393  if (m_sol_BMC[i][0] > 0.01f) {
395  PRMT_21 = 1000.f;
396  m_sol_WOC[i][0] = m_sol_LSC[i][0] + m_sol_LMC[i][0] + m_sol_HPC[i][0] + m_sol_HSC[i][0] + m_sol_BMC[i][0];
397  DK = 0.0001f * PRMT_21 * m_sol_WOC[i][0];
398  X1 = m_sol_wsatur[i][0];
399  if (X1 <= 0.f) X1 = 0.01f;
400  XX = X1 + DK;
401  V = m_surfaceRunoff[i] + m_sol_perco[i][0] + m_sol_laterq[i][0];
402  if (V > 1.e-10f) {
403  X3 = m_sol_BMC[i][0] * (1.f - exp(-V / XX));
404  PRMT_44 = 0.5;
405  CO = X3 / (m_sol_perco[i][0] + PRMT_44 * (m_surfaceRunoff[i] + m_sol_laterq[i][0]));
406  CS = PRMT_44 * CO;
407  VBC = CO * m_sol_perco[i][0];
408  m_sol_BMC[i][0] -= X3;
409  QBC = CS * (m_surfaceRunoff[i] + m_sol_laterq[i][0]);
411  if (YEW > 0.f) {
412  CS = DK * m_sol_BMC[i][0] / XX;
413  YBC = YEW * CS;
414  }
415  }
416  }
417  m_sol_BMC[i][0] -= YBC;
419  m_sol_latC[i][0] = QBC * (m_sol_laterq[i][0] / (m_surfaceRunoff[i] + m_sol_laterq[i][0] + UTIL_ZERO));
420  m_sol_percoC[i][0] = VBC;
421  m_sedCLoss[i] = YOC + YBC;
422 
423  latc_clyr += m_sol_latC[i][0];
424  for (int k = 1; k < (int) m_nSoilLayers[i]; k++) {
425  m_sol_WOC[i][k] = m_sol_LSC[i][k] + m_sol_LMC[i][k] + m_sol_HPC[i][k] + m_sol_HSC[i][k];
426  float Y1 = m_sol_BMC[i][k] + VBC;
427  VBC = 0.f;
428  if (Y1 > 0.01f) {
429  V = m_sol_perco[i][k] + m_sol_laterq[i][k];
430  if (V > 0.f) {
431  VBC = Y1 * (1.f - exp(-V / (m_sol_wsatur[i][k] + 0.0001f * PRMT_21 * m_sol_WOC[i][k])));
432  }
433  }
434  m_sol_latC[i][k] = VBC * (m_sol_laterq[i][k] / (m_sol_laterq[i][k] + m_sol_perco[i][k] + UTIL_ZERO));
435  m_sol_percoC[i][k] = VBC - m_sol_latC[i][k];
436  m_sol_BMC[i][k] = Y1 - VBC;
437 
439  perc_clyr += m_sol_percoC[i][k];
440  latc_clyr += m_sol_latC[i][k];
441  }
442  m_laterC[i] = latc_clyr;
443  m_percoC[i] = perc_clyr;
444 }

446 void NutrientTransportSediment::OrgPAttachedtoSed(int i) {
447  //amount of phosphorus attached to sediment in soil (sol_attp)
448  float sol_attp = 0.f;
449  //fraction of active mineral/organic/stable mineral phosphorus in soil (sol_attp_o, sol_attp_a, sol_attp_s)
450  float sol_attp_o = 0.f;
451  float sol_attp_a = 0.f;
452  float sol_attp_s = 0.f;
453  //Calculate sediment
454  sol_attp = m_sol_orgp[i][0] + m_sol_fop[i][0] + m_sol_actp[i][0] + m_sol_stap[i][0];
455  if (m_sol_mp != nullptr) {
456  sol_attp += m_sol_mp[i][0];
457  }
458  if (sol_attp > 1.e-3f) {
459  sol_attp_o = (m_sol_orgp[i][0] + m_sol_fop[i][0]) / sol_attp;
460  if (m_sol_mp != nullptr) {
461  sol_attp_o += m_sol_mp[i][0] / sol_attp;
462  }
463  sol_attp_a = m_sol_actp[i][0] / sol_attp;
464  sol_attp_s = m_sol_stap[i][0] / sol_attp;
465  }
466  //conversion factor (mg/kg => kg/ha) (wt)
467  float wt = m_sol_bd[i][0] * m_soilThick[i][0] / 100.f;
468  //concentration of organic P in soil (concp)
469  float concp = 0.f;
470  concp = sol_attp * m_enratio[i] / wt;
471  //total amount of P removed in sediment erosion (sedp)
472  float sedp = 1.e-6f * concp * m_sedEroded[i] / m_cellArea;
473  m_sedorgp[i] = sedp * sol_attp_o;
474  m_sedminpa[i] = sedp * sol_attp_a;
475  m_sedminps[i] = sedp * sol_attp_s;
476 
477  //if(i==100)cout << "sedp: " << sedp<< ",sol_attp_o: " << sol_attp_o << endl;
478  //modify phosphorus pools
479 
480  //total amount of P in mineral sediment pools prior to sediment removal (psedd) // Not used
481  //float psedd = 0.f;
482  //psedd = m_sol_actp[i][0] + m_sol_stap[i][0];
483 
484  //total amount of P in organic pools prior to sediment removal (porgg)
485  float porgg = 0.f;
486  porgg = m_sol_orgp[i][0] + m_sol_fop[i][0];
487  if (porgg > 1.e-3f) {
488  m_sol_orgp[i][0] = m_sol_orgp[i][0] - m_sedorgp[i] * (m_sol_orgp[i][0] / porgg);
489  m_sol_fop[i][0] = m_sol_fop[i][0] - m_sedorgp[i] * (m_sol_fop[i][0] / porgg);
490  }
491  m_sol_actp[i][0] = m_sol_actp[i][0] - m_sedminpa[i];
492  m_sol_stap[i][0] = m_sol_stap[i][0] - m_sedminps[i];
493  if (m_sol_orgp[i][0] < 0.f) {
494  m_sedorgp[i] = m_sedorgp[i] + m_sol_orgp[i][0];
495  m_sol_orgp[i][0] = 0.f;
496  }
497  if (m_sol_fop[i][0] < 0.f) {
498  m_sedorgp[i] = m_sedorgp[i] + m_sol_fop[i][0];
499  m_sol_fop[i][0] = 0.f;
500  }
501  if (m_sol_actp[i][0] < 0.f) {
502  m_sedminpa[i] = m_sedminpa[i] + m_sol_actp[i][0];
503  m_sol_actp[i][0] = 0.f;
504  }
505  if (m_sol_stap[i][0] < 0.f) {
506  m_sedminps[i] = m_sedminps[i] + m_sol_stap[i][0];
507  m_sol_stap[i][0] = 0.f;
508  }
