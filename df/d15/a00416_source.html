31 void Scenario::loadScenarioName() {
32  auto it = find(m_bmpCollections.begin(), m_bmpCollections.end(), string(TAB_BMP_SCENARIO));
33  if (it == m_bmpCollections.end()) {
34  throw ModelException("BMP Scenario", "loadScenarioName", "The BMP database '" + m_bmpDBName +
35  "' does not exist or there is not a table named '" +
36  TAB_BMP_SCENARIO + "' in BMP database.");
37  }
38  mongoc_collection_t* sceCollection = m_conn->GetCollection(m_bmpDBName, TAB_BMP_SCENARIO);
40  bson_t *query = bson_new(), *reply = bson_new();
41  query = BCON_NEW("distinct", BCON_UTF8(TAB_BMP_SCENARIO), "key", FLD_SCENARIO_NAME,
42  "query", "{", FLD_SCENARIO_ID, BCON_INT32(m_sceneID), "}");
43  bson_iter_t iter, sub_iter;
44  bson_error_t* err = NULL;
45  if (mongoc_collection_command_simple(sceCollection, query, NULL, reply, err)) {
46  //cout<<bson_as_json(reply,NULL)<<endl;
47  if (bson_iter_init_find(&iter, reply, "values") &&
48  (BSON_ITER_HOLDS_DOCUMENT(&iter) || BSON_ITER_HOLDS_ARRAY(&iter)) &&
49  bson_iter_recurse(&iter, &sub_iter)) {
50  while (bson_iter_next(&sub_iter)) {
51  m_name = GetStringFromBsonIterator(&sub_iter);
52  break;
53  }
54  } else {
55  throw ModelException("BMP Scenario", "loadScenarioName",
56  "There is not scenario existed with the ID: " +
57  ValueToString(m_sceneID) + " in " + TAB_BMP_SCENARIO +
58  " table in BMP database.");
59  }
60  }
61  bson_destroy(query);
62  bson_destroy(reply);
63  mongoc_collection_destroy(sceCollection);
64 }

66 void Scenario::loadBMPs() {
67  auto it = find(m_bmpCollections.begin(), m_bmpCollections.end(), string(TAB_BMP_INDEX));
68  if (it == m_bmpCollections.end()) {
69  throw ModelException("BMP Scenario", "loadScenarioName", "The BMP database '" + m_bmpDBName +
70  "' does not exist or there is not a table named '" +
71  TAB_BMP_INDEX + "' in BMP database.");
72  }
73  bson_t* query = bson_new();
74  BSON_APPEND_INT32(query, FLD_SCENARIO_ID, m_sceneID);
75  //cout<<bson_as_json(query, NULL)<<endl;
76  std::unique_ptr<MongoCollection> collection(new MongoCollection(m_conn->GetCollection(m_bmpDBName, TAB_BMP_SCENARIO)));
77  std::unique_ptr<MongoCollection> collbmpidx(new MongoCollection(m_conn->GetCollection(m_bmpDBName, TAB_BMP_INDEX)));
78  mongoc_cursor_t* cursor = collection->ExecuteQuery(query);
79 
80  bson_error_t* err = NULL;
81  if (mongoc_cursor_error(cursor, err)) {
82  throw ModelException("BMP Scenario", "loadBMPs",
83  "There are no record with scenario ID: " + ValueToString(m_sceneID));
84  }
85  const bson_t* info;
86  while (mongoc_cursor_more(cursor) && mongoc_cursor_next(cursor, &info)) {
87  //cout<<bson_as_json(info,0)<<endl;
88  bson_iter_t iter;
89  int BMPID = -1;
90  int subScenario = -1;
91  string distribution = "";
92  string collectionName = "";
93  string location = "";
94  if (bson_iter_init_find(&iter, info, FLD_SCENARIO_BMPID)) GetNumericFromBsonIterator(&iter, BMPID);
95  if (bson_iter_init_find(&iter, info, FLD_SCENARIO_SUB)) GetNumericFromBsonIterator(&iter, subScenario);
96  if (bson_iter_init_find(&iter, info, FLD_SCENARIO_DIST)) distribution = GetStringFromBsonIterator(&iter);
97  if (bson_iter_init_find(&iter, info, FLD_SCENARIO_TABLE)) collectionName = GetStringFromBsonIterator(&iter);
98  if (bson_iter_init_find(&iter, info, FLD_SCENARIO_LOCATION)) location = GetStringFromBsonIterator(&iter);
99 
101  vector<string> dist = SplitString(distribution, '|');
102  if (dist.size() >= 2 && StringMatch(dist[0], FLD_SCENARIO_DIST_RASTER)) {
103  string gridfs_name = ValueToString(m_subbsnID) + "_" + GetUpper(dist[1]);
104  if (m_sceneRsMap.find(gridfs_name) == m_sceneRsMap.end()) {
105  m_sceneRsMap.insert(make_pair(gridfs_name, nullptr));
106  }
107  dist[1] = gridfs_name;
108  }
109 
110  int BMPType = -1;
111  int BMPPriority = -1;
112  bson_t* queryBMP = bson_new();
113  BSON_APPEND_INT32(queryBMP, FLD_BMP_ID, BMPID);
114 
115  mongoc_cursor_t* cursor2 = collbmpidx->ExecuteQuery(queryBMP);
116 
117  if (mongoc_cursor_error(cursor2, err)) {
118  throw ModelException("BMP Scenario", "loadBMPs",
119  "There are no record with BMP ID: " + ValueToString(BMPID));
120  }
121  const bson_t* info2;
122  while (mongoc_cursor_more(cursor2) && mongoc_cursor_next(cursor2, &info2)) {
123  bson_iter_t sub_iter;
124  if (bson_iter_init_find(&sub_iter, info2, FLD_BMP_TYPE)) {
125  GetNumericFromBsonIterator(&sub_iter, BMPType);
126  }
127  if (bson_iter_init_find(&sub_iter, info2, FLD_BMP_PRIORITY)) {
128  GetNumericFromBsonIterator(&sub_iter, BMPPriority);
129  }
130  }
131  //cout<<BMPID<<","<<BMPType<<","<<distribution<<","<<parameter<<endl;
132  bson_destroy(queryBMP);
133  mongoc_cursor_destroy(cursor2);
135  int uniqueBMPID = BMPID * 100000 + subScenario;
136  if (this->m_bmpFactories.find(uniqueBMPID) == this->m_bmpFactories.end()) {
137  if (BMPID == BMP_TYPE_POINTSOURCE) {
138  this->m_bmpFactories[uniqueBMPID] = new BMPPointSrcFactory(m_sceneID, BMPID, subScenario, BMPType,
139  BMPPriority, dist,
140  collectionName, location);
141  }
142  if (BMPID == BMP_TYPE_PLANT_MGT) {
143  this->m_bmpFactories[uniqueBMPID] = new BMPPlantMgtFactory(m_sceneID, BMPID, subScenario, BMPType,
144  BMPPriority, dist,
145  collectionName, location);
146  }
147  if (BMPID == BMP_TYPE_AREALSOURCE) {
148  this->m_bmpFactories[uniqueBMPID] = new BMPArealSrcFactory(m_sceneID, BMPID, subScenario, BMPType,
149  BMPPriority, dist,
150  collectionName, location);
151  }
152  if (BMPID == BMP_TYPE_AREALSTRUCT) {
153  this->m_bmpFactories[uniqueBMPID] = new BMPArealStructFactory(m_sceneID, BMPID, subScenario, BMPType,
154  BMPPriority, dist,
155  collectionName, location);
156  }
157  }
158  }
159  bson_destroy(query);
160  mongoc_cursor_destroy(cursor);
161 }
