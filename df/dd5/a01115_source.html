106 void MUSK_CH::initialOutputs() {
107  if (m_nreach <= 0) {
108  throw ModelException(MID_MUSK_CH, "initialOutputs", "The reach number can not be less than zero.");
109  }
110 
111  if (m_reachLayers.empty()) {
112  for (int i = 1; i <= m_nreach; i++) {
113  if (m_chOrder == NULL) {
114  throw ModelException(MID_MUSK_CH, "initialOutputs",
115  "Stream order is not loaded successful from Reach table.");
116  }
117  int order = (int) m_chOrder[i];
118  m_reachLayers[order].push_back(i);
119  }
120  }
121  if (m_outletID < 0) {
122  m_outletID = m_reachLayers.rbegin()->second[0];
123  }
124  //initial channel storage
125  if (m_chStorage == NULL) {
126  m_chStorage = new float[m_nreach + 1];
127  m_preChStorage = new float[m_nreach + 1];
128  m_qIn = new float[m_nreach + 1];
129  m_qOut = new float[m_nreach + 1];
130  m_bankStorage = new float[m_nreach + 1];
131  m_seepage = new float[m_nreach + 1];
132  m_qsCh = new float[m_nreach + 1];
133  m_qiCh = new float[m_nreach + 1];
134  m_qgCh = new float[m_nreach + 1];
135  m_chWTdepth = new float[m_nreach + 1];
136  m_preChWTDepth = new float[m_nreach + 1];
137  m_chWTWidth = new float[m_nreach + 1];
138  m_chBtmWidth = new float[m_nreach + 1];
139 
140  for (int i = 1; i <= m_nreach; i++) {
141  float qiSub = 0.f;
142  float qgSub = 0.f;
143  if (m_qiSub != NULL) { // interflow (subsurface flow)
144  qiSub = m_qiSub[i];
145  }
146  if (m_qgSub != NULL) { // groundwater
147  qgSub = m_qgSub[i];
148  }
149  m_seepage[i] = 0.f;
150  m_bankStorage[i] = m_Bnk0 * m_chLen[i];
151  m_chBtmWidth[i] = m_chWidth[i] - 2.f * m_chSideSlope[i] * m_chDepth[i];
152  if (m_chBtmWidth[i] <= UTIL_ZERO) {
153  m_chBtmWidth[i] = 0.5f * m_chWidth[i];
154  m_chSideSlope[i] = (m_chWidth[i] - m_chBtmWidth[i]) / 2.f / m_chDepth[i];
155  }
156  m_chWTdepth[i] = m_chDepth[i] * m_Chs0_perc;
157  m_chWTWidth[i] = m_chBtmWidth[i] + 2.f * m_chSideSlope[i] * m_chWTdepth[i];
158  m_preChWTDepth[i] = m_chWTWidth[i];
159  // m_chWTWidth[i] = m_chWidth[i];
160  // m_chStorage[i] = m_chWTdepth[i] * m_chWTWidth[i] * m_chLen[i];
161  m_chStorage[i] = m_chLen[i] * m_chWTdepth[i] * (m_chBtmWidth[i] + m_chSideSlope[i] * m_chWTdepth[i]);
162  m_preChStorage[i] = m_chStorage[i];
163  m_qIn[i] = 0.f;
164  m_qOut[i] = m_qsSub[i] + qiSub + qgSub;
165  m_qsCh[i] = m_qsSub[i];
166  m_qiCh[i] = qiSub;
167  m_qgCh[i] = qgSub;
168  }
169  }
171  if (m_ptSub == NULL) {
172  Initialize1DArray(m_nreach + 1, m_ptSub, 0.f);
173  }

176 void MUSK_CH::PointSourceLoading() {
178  for (map<int, BMPPointSrcFactory *>::iterator it = m_ptSrcFactory.begin(); it != m_ptSrcFactory.end(); it++) {
180  for (int i = 0; i <= m_nreach; i++) {
181  m_ptSub[i] = 0.f;
182  }
183  //cout<<"unique Point Source Factory ID: "<<it->first<<endl;
184  vector<int> m_ptSrcMgtSeqs = it->second->GetPointSrcMgtSeqs();
185  map < int, PointSourceMgtParams * > m_pointSrcMgtMap = it->second->GetPointSrcMgtMap();
186  vector<int> m_ptSrcIDs = it->second->GetPointSrcIDs();
187  map < int, PointSourceLocations * > m_pointSrcLocsMap = it->second->GetPointSrcLocsMap();
188  // 1. looking for management operations from m_pointSrcMgtMap
189  for (vector<int>::iterator seqIter = m_ptSrcMgtSeqs.begin(); seqIter != m_ptSrcMgtSeqs.end(); seqIter++) {
190  PointSourceMgtParams *curPtMgt = m_pointSrcMgtMap.at(*seqIter);
191  // 1.1 If current day is beyond the date range, then continue to next management
192  if (curPtMgt->GetStartDate() != 0 && curPtMgt->GetEndDate() != 0) {
193  if (m_date < curPtMgt->GetStartDate() || m_date > curPtMgt->GetEndDate()) {
194  continue;
195  }
196  }
197  // 1.2 Otherwise, get the water volume
198  float per_wtrVol = curPtMgt->GetWaterVolume();
199  // 1.3 Sum up all point sources
200  for (vector<int>::iterator locIter = m_ptSrcIDs.begin(); locIter != m_ptSrcIDs.end(); locIter++) {
201  if (m_pointSrcLocsMap.find(*locIter) != m_pointSrcLocsMap.end()) {
202  PointSourceLocations *curPtLoc = m_pointSrcLocsMap.at(*locIter);
203  int curSubID = curPtLoc->GetSubbasinID();
204  m_ptSub[curSubID] += per_wtrVol * curPtLoc->GetSize() / 86400.f;
205  }
206  }
207  }
208  }

427 void MUSK_CH::SetReaches(clsReaches *reaches) {

449  m_chVel[i] = (float) tmpReach->GetV0();
450  m_area[i] = (float) tmpReach->GetArea();
451  m_chSideSlope[i] = (float) tmpReach->GetSideSlope();
452  }

454  m_reachUpStream.resize(m_nreach + 1);
455  if (m_nreach > 1) {
456  for (int i = 1; i <= m_nreach; i++)// index of the reach is the equal to streamlink ID(1 to nReaches)

459  if (downStreamId <= 0) {

461  }
462  m_reachUpStream[downStreamId].push_back(i);

464  }

467  }

488 void MUSK_CH::GetCoefficients(float reachLength, float v0, MuskWeights &weights) {
489  v0 = m_vScalingFactor * v0;
490  float K = (4.64f - 3.64f * m_co1) * reachLength / (5.f * v0 / 3.f);
491 
492  float min = 2.f * K * m_x;
493  float max = 2.f * K * (1.f - m_x);

502  weights.c2 = (dt + min) / temp;

504  weights.c4 = 2 * dt / temp;
505  weights.n = n;

509  weights.c2 += weights.c1;
510  weights.c1 = 0.f;

532 void MUSK_CH::ChannelFlow(int i) {
533  float st0 = m_chStorage[i];
534  float qiSub = 0.f;
535  if (m_qiSub != NULL && m_qiSub[i] >= 0.f) {
536  qiSub = m_qiSub[i];
537  }
538  float qgSub = 0.f;
539  if (m_qgSub != NULL && m_qgSub[i] >= 0.f) {
540  qgSub = m_qgSub[i];
541  }
542  float ptSub = 0.f;
543  if (m_ptSub != NULL && m_ptSub[i] >= 0.f) {
544  ptSub = m_ptSub[i];
545  }
547  // first add all the inflow water
548  // 1. water from this subbasin
549  float qIn = m_qsSub[i] + qiSub + qgSub + ptSub + m_deepGroundwater;
550  //if (i == m_outletID) /// this should be added to each channel. By lj
551  // qIn += m_deepGroundwater;
552  // 2. water from upstream reaches
553  float qsUp = 0.f;
554  float qiUp = 0.f;
555  float qgUp = 0.f;
556  for (size_t j = 0; j < m_reachUpStream[i].size(); ++j) {
557  int upReachId = m_reachUpStream[i][j];
558  qsUp += m_qsCh[upReachId];
559  qiUp += m_qiCh[upReachId];
560  qgUp += m_qgCh[upReachId];
561  }
562  qIn += qsUp + qiUp + qgUp;
563  //qIn is equivalent to the wtrin variable in rtmusk.f of SWAT
564  qIn += m_qUpReach;
565  // m_qUpReach is zero for not-parallel program and qsUp, qiUp and qgUp are zero for parallel computing
566  //if(i == 12)
567  // cout <<"surfaceQ: "<< m_qsSub[i] << ", subsurfaceQ: " << qiSub << ", groundQ: " << qgSub << ", pointQ: " << ptSub <<
568  // ", UPsurfaceQ: "<<qsUp<<", UPsubsurface: "<<qiUp<<", UPground: "<<qgUp<<", \n";
569  // 3. water from bank storage
570  float bankOut = m_bankStorage[i] * (1.f - exp(-m_aBank));
571 
572  m_bankStorage[i] -= bankOut;
573  qIn += bankOut / m_dt;
574 
575  // add inflow water to storage
576  //if (i==12) cout<<"initial chStorage: "<<m_chStorage[i]<<", ";
577  m_chStorage[i] += qIn * m_dt;
579  updateWaterWidthDepth(i);
580  //if (i==12) cout<<"added chStorage: "<<m_chStorage[i]<<endl;
581  //if(i == 2) cout <<"qIn:"<< qIn<<", chStorage: "<<m_chStorage[i]<<endl;
583  // then subtract all the outflow water
584  // 1. transmission losses to deep aquifer, which is lost from the system
585  // the unit of kchb is mm/hr
586  float seepage = m_Kchb / 1000.f / 3600.f * m_chBtmWidth[i] * m_chLen[i] * m_dt;
587  //if(i == 2) cout << "seepage: " << seepage << endl;
588  if (qgSub < UTIL_ZERO) {
589  if (m_chStorage[i] > seepage) {
590  m_seepage[i] = seepage;
591  m_chStorage[i] -= seepage;
592  } else {
593  m_seepage[i] = m_chStorage[i];
594  m_chStorage[i] = 0.f;
595  }
596  } else {
597  m_seepage[i] = 0.f;
598  }
599 
600  // 2. calculate transmission losses to bank storage
601  //float dch = m_chStorage[i] / (m_chWTWidth[i] * m_chLen[i]);
602  float dch = m_chWTdepth[i];
603  float bankLen = dch * sqrt(1.f + m_chSideSlope[i] * m_chSideSlope[i]);
604  float bankInLoss = 2.f * m_Kbank / 1000.f / 3600.f * bankLen * m_chLen[i] * m_dt; // m^3
605  bankInLoss = 0.f; //TODO
606  if (m_chStorage[i] > bankInLoss) {
607  m_chStorage[i] -= bankInLoss;
608  } else {
609  bankInLoss = m_chStorage[i];
610  m_chStorage[i] = 0.f;
611  }
612  // water balance of the bank storage
613  // loss the water from bank storage to the adjacent unsaturated zone and groundwater storage
614  float bankOutGw = m_bankStorage[i] * (1.f - exp(-m_bBank));
615  bankOutGw = 0.f; //TODO
616  m_bankStorage[i] = m_bankStorage[i] + bankInLoss - bankOutGw;
617  if (m_gwStorage != NULL) {
618  m_gwStorage[i] += bankOutGw / m_area[i] * 1000.f;
619  } // updated groundwater storage
620 
621  // 3. evaporation losses
622  float et = 0.f;
623  if (m_petCh != NULL) {
624  et = m_Epch * m_petCh[i] / 1000.0f * m_chWTWidth[i] * m_chLen[i]; //m3
625  if (m_chStorage[i] > et) {
626  m_chStorage[i] -= et;
627  } else {
628  et = m_chStorage[i];
629  m_chStorage[i] = 0.f;
630  }
631  }
632  if (FloatEqual(m_chStorage[i], 0.f)) {
633  m_qOut[i] = 0.f;
634  m_qsCh[i] = 0.f;
635  m_qiCh[i] = 0.f;
636  m_qgCh[i] = 0.f;
637  m_chWTdepth[i] = 0.f;
638  m_chWTWidth[i] = 0.f;
639  return;
640  }
641 
642  //if(i == 2) cout << "chStorage before routing " << m_chStorage[i] << endl;
644  // routing, there are water in the channel after inflow and transmission loss
645  float totalLoss = m_seepage[i] + bankInLoss + et;
646  //if(i == 12) cout << ", m_petCh: " << m_petCh[i] << ", et: " << et << ", m_chStorage: " << m_chStorage[i] << ", \n";
647  //if (m_chStorage[i] >= 0.f)
648  //{
649 // qIn -= totalLoss / m_dt;// average loss rate during m_dt
650 // if(qIn < 0.f) qIn = 0.f;
651 
652  m_preChStorage[i] = m_chStorage[i];
653  m_preChWTDepth[i] = m_chWTdepth[i];
654  m_chStorage[i] = st0;
655 
656  // calculate coefficients
657  MuskWeights wt;
658  GetCoefficients(m_chLen[i], m_chVel[i], wt);
659  int n = wt.n;
660  float q = 0.f;
661  for (int j = 0; j < n; j++) {
662  m_qOut[i] = wt.c1 * qIn + wt.c2 * m_qIn[i] + wt.c3 * m_qOut[i];
663  m_qIn[i] = qIn;
664  float tmp = m_chStorage[i] + (qIn - totalLoss / m_dt - m_qOut[i]) * wt.dt;
665  if (tmp < 0.f) {
666  m_qOut[i] = m_chStorage[i] / wt.dt + qIn;
667  m_chStorage[i] = 0.f;
668  } else {
669  m_chStorage[i] = tmp;
670  }
671  q += m_qOut[i];
672  }
673  m_qOut[i] = q / n;
674  //}
675  //else
676  //{
677  // m_qOut[i] = 0.f;
678  // m_chStorage[i] = 0.f;
679  // qIn = 0.f;
680  //}
681  float qInSum = m_qsSub[i] + qiSub + qgSub + qsUp + qiUp + qgUp;
682  m_qsCh[i] = m_qOut[i] * (m_qsSub[i] + qsUp) / qInSum;
683  m_qiCh[i] = m_qOut[i] * (qiSub + qiUp) / qInSum;
684  m_qgCh[i] = m_qOut[i] * (qgSub + qgUp) / qInSum;
685 
686  // set variables for next time step
687  m_qIn[i] = qIn;
688  //m_chWTdepth[i] = m_chStorage[i] / (m_chWTWidth[i] * m_chLen[i]);
689  updateWaterWidthDepth(i);
