25 int SoilTemperatureFINPL::Execute() {
27  CheckInputData();
29  initialOutputs();
30  m_julianDay = JulianDay(m_date);
31  size_t errCount = 0;
32 #pragma omp parallel for reduction(+: errCount)
33  for (int i = 0; i < m_nCells; ++i) {
34  float t = m_tMean[i];
35  float t1 = m_t1[i];
36  float t2 = m_t2[i];
37  if ((t > 60.f || t < -90.f) || (t1 > 60.f || t1 < -90.f) || (t2 > 60.f || t2 < -90.f)) {
38  cout << "cell index: " << i << ", t1: " << t1 << ", t2: " << t2 << endl;
39  errCount++;
40  }
41  else{
42  if (FloatEqual((int) m_landuse[i], LANDUSE_ID_WATR)) {
44  m_soilTemp[i] = t;
45  } else {
46  float t10 = m_a0 + m_a1 * t2 + m_a2 * t1 + m_a3 * t
47  + m_b1 * sin(w * m_julianDay) + m_d1 * cos(w * m_julianDay)
48  + m_b2 * sin(2.f * w * m_julianDay) + m_d2 * cos(2.f * w * m_julianDay);
49  m_soilTemp[i] = t10 * m_kSoil10 + m_relativeFactor[i];
50  if (m_soilTemp[i] > 60.f || m_soilTemp[i] < -90.f) {
51  cout << "The calculated soil temperature at cell (" << i
52  << ") is out of reasonable range: " << m_soilTemp[i]
53  << ". JulianDay: " << m_julianDay << ",t: "<< t << ", t1: "
54  << t1 << ", t2: " << t2 << ", relativeFactor: " << m_relativeFactor[i] << endl;
55  errCount++;
56  }
57  }
58  //save the temperature
59  m_t2[i] = m_t1[i];
60  m_t1[i] = t;
61  }
62  }
63  if (errCount > 0) {
64  throw ModelException(MID_STP_FP, "Execute", "The calculation of soil temperature failed!");
65  }
66  return 0;
67 }
