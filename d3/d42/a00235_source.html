213 template<typename T>
214 bool GetNumericFromBsonIterator(bson_iter_t *iter, T &numericvalue) {
215  const bson_value_t *vv = bson_iter_value(iter);
216  if (vv->value_type == BSON_TYPE_INT32) {
217  numericvalue = (T) vv->value.v_int32;
218  } else if (vv->value_type == BSON_TYPE_INT64) {
219  numericvalue = (T) vv->value.v_int64;
220  } else if (vv->value_type == BSON_TYPE_DOUBLE) {
221  numericvalue = (T) vv->value.v_double;
222  } else if (vv->value_type == BSON_TYPE_UTF8) {
223  string tmp = vv->value.v_utf8.str;
224  numericvalue = (T) atof(tmp.c_str());
225  } else {
226  cout << "bson iterator isn't or not contains a numeric value." << endl;
227  return false;
228  }
229  return true;
230 }
