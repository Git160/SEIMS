114 int AET_PT_H::Execute() {
115  CheckInputData();
116  InitialOutputs();
117 #pragma omp parallel for
118  for (int i = 0; i < m_nCells; i++) {
120  float esd = 0.f, etco = 0.f, effnup = 0.f;
121  float no3up = 0.f, es_max = 0.f, eos1 = 0.f, xx = 0.f;
122  float cej = 0.f, eaj = 0.f, pet = 0.f, esleft = 0.f;
123  float evzp = 0.f, eosl = 0.f, dep = 0.f, evz = 0.f, sev = 0.f;
124  pet = m_pet[i] - m_IntcpET[i];
125  esd = 500.f;
126  etco = 0.8f;
127  effnup = 0.1f;
128 
129  if (pet < UTIL_ZERO) {
130  pet = 0.f;
131  m_ppt[i] = 0.f; // i.e., ep_max
132  es_max = 0.f;
133  } else {
135  if (m_lai[i] <= 3.f) {
136  m_ppt[i] = m_lai[i] * pet / 3.f;
137  } else {
138  m_ppt[i] = pet;
139  }
140  if (m_ppt[i] < 0.f) m_ppt[i] = 0.f;
142  cej = -5.e-5f;
143  eaj = 0.f;
144  es_max = 0.f;
145  eos1 = 0.f;
146  if (m_snowAccum[i] >= 0.5f) {
147  eaj = 0.5f;
148  } else {
149  eaj = exp(cej * (m_solCov[i] + 0.1f));
150  }
151  es_max = pet * eaj;
152  eos1 = pet / (es_max + m_ppt[i] + 1.e-10f);
153  eos1 = es_max * eos1;
154  es_max = Min(es_max, eos1);
155  es_max = Max(es_max, 0.f);
157  if (pet < es_max + m_ppt[i] && !FloatEqual(es_max + m_ppt[i], 0.f)) {
158  es_max = pet * es_max / (es_max + m_ppt[i]);
159  m_ppt[i] = pet * m_ppt[i] / (es_max + m_ppt[i]);
160  }
161  if (pet < es_max + m_ppt[i]) {
162  es_max = pet - m_ppt[i] - UTIL_ZERO;
163  }
164 
166  esleft = es_max;
168  if (m_tMean[i] > 0.f) {
169  if (m_snowAccum[i] >= esleft) {
171  m_snowAccum[i] -= esleft;
172  m_snowSB[i] += esleft;
173  esleft = 0.f;
174  } else {
176  esleft -= m_snowAccum[i];
177  m_snowSB[i] += m_snowAccum[i];
178  m_snowAccum[i] = 0.f;
179  }
180  }
181  // take soil evap from each soil layer
182  evzp = 0.f;
183  eosl = esleft;
184  for (int ly = 0; ly < CVT_INT(m_nSoilLayers[i]); ly++) {
185  dep = 0.f;
187  if (ly == 0) {
188  dep = m_soilDepth[i][ly];
189  } else {
190  dep = m_soilDepth[i][ly - 1];
191  }
192  if (dep < esd) {
194  evz = 0.f;
195  sev = 0.f;
196  xx = 0.f;
197  evz = eosl * m_soilDepth[i][ly] /
198  (m_soilDepth[i][ly] + exp(2.374f - 0.00713f * m_soilDepth[i][ly]));
199  sev = evz - evzp * m_esco[i];
200  evzp = evz;
201  if (m_soilStorage[i][ly] < m_solFC[i][ly]) {
202  xx = 2.5f * (m_soilStorage[i][ly] - m_solFC[i][ly]) / m_solFC[i][ly];
203  sev *= Expo(xx);
204  }
205  sev = Min(sev, m_soilStorage[i][ly] * etco);
206  if (sev < 0.f || sev != sev) sev = 0.f;
207  if (sev > esleft) sev = esleft;
209  if (m_soilStorage[i][ly] > sev) {
210  esleft -= sev;
211  m_soilStorage[i][ly] = Max(UTIL_ZERO, m_soilStorage[i][ly] - sev);
212  } else {
213  esleft -= m_soilStorage[i][ly];
214  m_soilStorage[i][ly] = 0.f;
215  }
216  }
218  if (ly == 1) {
220  no3up = 0.f;
221  no3up = effnup * sev * m_solNo3[i][ly] / (m_soilStorage[i][ly] + UTIL_ZERO);
222  no3up = Min(no3up, m_solNo3[i][ly]);
223  m_no3Up += no3up / m_nCells;
224  m_solNo3[i][ly] -= no3up;
225  m_solNo3[i][ly - 1] += no3up;
226  }
227  }
229  m_soilStorageProfile[i] = 0.f;
230  for (int ly = 0; ly < CVT_INT(m_nSoilLayers[i]); ly++) {
231  m_soilStorageProfile[i] += m_soilStorage[i][ly];
232  }
234  //if (esleft != esleft || es_max != es_max)
235  // cout<<"esleft: "<<esleft<<", es_max: "<<es_max<<endl;
236  if (es_max > esleft) {
237  m_soilESDay[i] = es_max - esleft;
238  } else {
239  m_soilESDay[i] = 0.f;
240  }
241  }
242  }
243  return true;
244 }
