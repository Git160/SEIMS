48 int IUH_OL::Execute() {
49  CheckInputData();
50  InitialOutputs();
51  // delete value of last time step
52  for (int n = 0; n <= m_nSubbsns; n++) {
53  m_Q_SBOF[n] = 0.f;
54  }
55 #pragma omp parallel for
56  for (int i = 0; i < m_nCells; i++) {
57  //forward one time step
58  for (int j = 0; j < m_cellFlowCols - 1; j++) {
59  m_cellFlow[i][j] = m_cellFlow[i][j + 1];
60  }
61  m_cellFlow[i][m_cellFlowCols - 1] = 0.f;
62 
63  if (m_surfRf[i] <= 0.f) continue;
64 
65  int min = CVT_INT(m_iuhCell[i][0]);
66  int max = CVT_INT(m_iuhCell[i][1]);
67  int col = 2;
68  for (int k = min; k <= max; k++) {
69  m_cellFlow[i][k] += m_surfRf[i] * 0.001f * m_iuhCell[i][col] * m_cellArea / m_TimeStep;
70  col++;
71  }
72  }
73  // See https://github.com/lreis2415/SEIMS/issues/36 for more descriptions. By lj
74 #pragma omp parallel
75  {
76  float* tmp_qsSub = new float[m_nSubbsns + 1];
77  for (int i = 0; i <= m_nSubbsns; i++) {
78  tmp_qsSub[i] = 0.f;
79  }
80 #pragma omp for
81  for (int i = 0; i < m_nCells; i++) {
82  tmp_qsSub[CVT_INT(m_subbsnID[i])] += m_cellFlow[i][0]; //get new value
83  m_OL_Flow[i] = m_cellFlow[i][0];
84  m_OL_Flow[i] = m_OL_Flow[i] * m_TimeStep * 1000.f / m_cellArea; // m3/s -> mm
85  }
86 #pragma omp critical
87  {
88  for (int i = 1; i <= m_nSubbsns; i++) {
89  m_Q_SBOF[i] += tmp_qsSub[i];
90  }
91  }
92  delete[] tmp_qsSub;
93  tmp_qsSub = nullptr;
94  } /* END of #pragma omp parallel */
95 
96  for (int n = 1; n <= m_nSubbsns; n++) {
97  //get overland flow routing for entire watershed.
98  m_Q_SBOF[0] += m_Q_SBOF[n];
99  }
100  return 0;
101 }
