183  def generate_lat_raster(cfg):
184  """Generate latitude raster"""
185  dem_file = cfg.spatials.filldem
186  ds = RasterUtilClass.read_raster(dem_file)
187  src_srs = ds.srs
188  if not src_srs.ExportToProj4():
189  raise ValueError("The source raster %s has not coordinate, "
190  "which is required!" % dem_file)
191  dst_srs = osr_SpatialReference()
192  dst_srs.ImportFromEPSG(4326) # WGS84
193  # dst_wkt = dst_srs.ExportToWkt()
194  transform = osr_CoordinateTransformation(src_srs, dst_srs)
195 
196  point_ll = ogr_CreateGeometryFromWkt("POINT (%f %f)" % (ds.xMin, ds.yMin))
197  point_ur = ogr_CreateGeometryFromWkt("POINT (%f %f)" % (ds.xMax, ds.yMax))
198 
199  point_ll.Transform(transform)
200  point_ur.Transform(transform)
201 
202  lower_lat = point_ll.GetY()
203  up_lat = point_ur.GetY()
204 
205  rows = ds.nRows
206  cols = ds.nCols
207  delta_lat = (up_lat - lower_lat) / float(rows)
208 
209  def cal_cell_lat(row, col):
210  """calculate latitude of cell by row number"""
211  return up_lat - (row + 0.5) * delta_lat
212 
213  data_lat = fromfunction(cal_cell_lat, (rows, cols))
214  data_lat = where(ds.validZone, data_lat, ds.data)
215  RasterUtilClass.write_gtiff_file(cfg.spatials.cell_lat, rows, cols, data_lat,
216  ds.geotrans, ds.srs,
217  ds.noDataValue, GDT_Float32)
