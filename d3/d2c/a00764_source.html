46  @staticmethod
47  def run(function_name, in_files, in_params=None, out_files=None, mpi_params=None,
48  log_params=None):
49  """
50  Run TauDEM function.
51  The command will not execute if:
52  1) any input file does not exist;
53  2) all output files exist and were last modified no earlier than any input file.
54  An error will be detected after running the TauDEM command if:
55  1) any output file does not exist ;
56  2) any output file was last modified earlier than any input file.
57 
58  Args:
59  function_name: TauDEM function name (with or without path) as string
60  in_files: Required. Dict of pairs of parameter id (string) and file path (string)
61  for input files.
62  in_params: Optional. Dict of pairs of parameter id (string) and value (or None for a
63  flag parameter without a value) for input parameters.
64  out_files: Optional. Dict of pairs of parameter id (string) and file path (string) for
65  output files.
66  mpi_params: Optional. Dict of pairs of parameter id (string) and value or path for
67  MPI setting.
68  e.g. {'mpipath':'/soft/bin','hostfile':'/soft/bin/cluster.node','n':4} or
69  {'mpipath':'/soft/bin', 'n':4} or {'n':4}
70  log_params: Optional. Dict of pairs of parameter id (string) and value or path for
71  runtime and log output parameters. e.g. {'logfile': '/home/user/log.txt'}
72 
73  Returns:
74  True if TauDEM run successfully, otherwise False.
75  """
76  # check the log parameter
77  log_file = None
78  if log_params is not None:
79  if not isinstance(log_params, dict):
80  TauDEM.error("The log parameter must be a dict!\n")
81  if 'logfile' in log_params and log_params['logfile'] is not None:
82  log_file = log_params['logfile']
83 
84  # check input files
85  if in_files is None:
86  TauDEM.error("Input files parameter is required!\n", log_file)
87  if not isinstance(in_files, dict):
88  TauDEM.error("The input files parameter must be a dict!\n")
89  for (pid, path) in in_files.items():
90  if path is not None and not os.path.exists(path):
91  TauDEM.error("Input files parameter %s: %s is not existed!\n" % (pid, path),
92  log_file)

109  if out_files is not None:

113  if out_file is not None:

116  # concatenate command line
117  commands = []
118  # MPI header
119  if mpi_params is not None:
120  if not isinstance(mpi_params, dict):
121  TauDEM.error("The MPI settings parameter must be a dict!\n")
122  if 'mpipath' in mpi_params and mpi_params['mpipath'] is not None:
123  commands.append(mpi_params['mpipath'] + os.sep + 'mpiexec')

129  if 'n' in mpi_params and mpi_params['n'] > 1:

132  else: # If number of processor is less equal than 1, then do not call mpiexec.
133  commands = []

135  commands.append(function_name)

138  if infile is None:

149  if pid[0] != '-':

152  # allow for parameter which is an flag without value

154  if MathClass.isnumerical(v):

156  else:
157  commands.append(v)

164  pid = '-' + pid

166  commands.append(outfile)

170  TauDEM.log(runmsg, log_file)
