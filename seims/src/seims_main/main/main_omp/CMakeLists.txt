    SET(LIBRARY_OUTPUT_PATH ${PROJECT_BINARY_DIR})

    INSTALL(TARGETS main_mongodb DESTINATION bin)
    IF (MSVC)
        SET_PROPERTY(TARGET main_mongodb PROPERTY FOLDER "main")
    ENDIF ()
ELSE ()
    ADD_EXECUTABLE(seims_omp ${SRC_LIST})
    SET(EXECUTABLE_OUTPUT_PATH ${PROJECT_BINARY_DIR})
    TARGET_LINK_LIBRARIES(seims_omp util data bmps module_setting)
    INSTALL(TARGETS seims_omp DESTINATION bin)
    IF (MSVC)
        SET_PROPERTY(TARGET seims_omp PROPERTY FOLDER "main")
    ENDIF ()
ENDIF ()
