30 void Scenario::loadScenarioName() {
31  vector<string>::iterator it = find(m_bmpCollections.begin(), m_bmpCollections.end(), string(TAB_BMP_SCENARIO));
32  if (it == m_bmpCollections.end()) {
33  throw ModelException("BMP Scenario", "loadScenarioName", "The BMP database '" + m_bmpDBName +
34  "' does not exist or there is not a table named '" +
35  TAB_BMP_SCENARIO + "' in BMP database.");
36  }
37  mongoc_collection_t *sceCollection = m_conn->getCollection(m_bmpDBName, TAB_BMP_SCENARIO);
39  bson_t *query = bson_new(), *reply = bson_new();
40  query = BCON_NEW("distinct", BCON_UTF8(TAB_BMP_SCENARIO), "key", FLD_SCENARIO_NAME,
41  "query", "{", FLD_SCENARIO_ID, BCON_INT32(m_id), "}");
42  bson_iter_t iter, sub_iter;
43  bson_error_t *err = NULL;
44  if (mongoc_collection_command_simple(sceCollection, query, NULL, reply, err)) {
45  //cout<<bson_as_json(reply,NULL)<<endl;
46  if (bson_iter_init_find(&iter, reply, "values") &&
47  (BSON_ITER_HOLDS_DOCUMENT(&iter) || BSON_ITER_HOLDS_ARRAY(&iter)) &&
48  bson_iter_recurse(&iter, &sub_iter)) {
49  while (bson_iter_next(&sub_iter)) {
50  m_name = GetStringFromBsonIterator(&sub_iter);
51  break;
52  }
53  } else {
54  throw ModelException("BMP Scenario", "loadScenarioName",
55  "There is not scenario existed with the ID: " + ValueToString(m_id)
56  + " in " + TAB_BMP_SCENARIO + " table in BMP database.");
57  }
58  }
59  bson_destroy(query);
60  bson_destroy(reply);
61  mongoc_collection_destroy(sceCollection);
62 }

64 void Scenario::loadBMPs() {
65  vector<string>::iterator it = find(m_bmpCollections.begin(), m_bmpCollections.end(), string(TAB_BMP_INDEX));
66  if (it == m_bmpCollections.end()) {
67  throw ModelException("BMP Scenario", "loadScenarioName", "The BMP database '" + m_bmpDBName +
68  "' does not exist or there is not a table named '" +
69  TAB_BMP_INDEX + "' in BMP database.");
70  }
71  bson_t *query = bson_new();
72  BSON_APPEND_INT32(query, FLD_SCENARIO_ID, m_id);
73  //cout<<bson_as_json(query, NULL)<<endl;
74  unique_ptr<MongoCollection> collection(new MongoCollection(m_conn->getCollection(m_bmpDBName, TAB_BMP_SCENARIO)));
75  unique_ptr<MongoCollection> collbmpidx(new MongoCollection(m_conn->getCollection(m_bmpDBName, TAB_BMP_INDEX)));
76  mongoc_cursor_t* cursor = collection->ExecuteQuery(query);
77 
78  bson_error_t *err = NULL;
79  if (mongoc_cursor_error(cursor, err)) {
80  throw ModelException("BMP Scenario", "loadBMPs",
81  "There are no record with scenario ID: " + ValueToString(m_id));
82  }
83  const bson_t *info;
84  while (mongoc_cursor_more(cursor) && mongoc_cursor_next(cursor, &info)) {
85  //cout<<bson_as_json(info,0)<<endl;
86  bson_iter_t iter;
87  int BMPID = -1;
88  int subScenario = -1;
89  string distribution = "";
90  string collectionName = "";
91  string location = "";
92  if (bson_iter_init_find(&iter, info, FLD_SCENARIO_BMPID)) GetNumericFromBsonIterator(&iter, BMPID);
93  if (bson_iter_init_find(&iter, info, FLD_SCENARIO_SUB)) GetNumericFromBsonIterator(&iter, subScenario);
94  if (bson_iter_init_find(&iter, info, FLD_SCENARIO_DIST)) distribution = GetStringFromBsonIterator(&iter);
95  if (bson_iter_init_find(&iter, info, FLD_SCENARIO_TABLE)) collectionName = GetStringFromBsonIterator(&iter);
96  if (bson_iter_init_find(&iter, info, FLD_SCENARIO_LOCATION)) location = GetStringFromBsonIterator(&iter);
97 
98  int BMPType = -1;
99  int BMPPriority = -1;
100  bson_t *queryBMP = bson_new();
101  BSON_APPEND_INT32(queryBMP, FLD_BMP_ID, BMPID);
102 
103  mongoc_cursor_t* cursor2 = collbmpidx->ExecuteQuery(queryBMP);
104 
105  if (mongoc_cursor_error(cursor2, err)) {
106  throw ModelException("BMP Scenario", "loadBMPs",
107  "There are no record with BMP ID: " + ValueToString(BMPID));
108  }
109  const bson_t *info2;
110  while (mongoc_cursor_more(cursor2) && mongoc_cursor_next(cursor2, &info2)) {
111  bson_iter_t sub_iter;
112  if (bson_iter_init_find(&sub_iter, info2, FLD_BMP_TYPE)) GetNumericFromBsonIterator(&sub_iter, BMPType);
113  if (bson_iter_init_find(&sub_iter, info2, FLD_BMP_PRIORITY)) {
114  GetNumericFromBsonIterator(&sub_iter, BMPPriority);
115  }
116  }
117  //cout<<BMPID<<","<<BMPType<<","<<distribution<<","<<parameter<<endl;
118  bson_destroy(queryBMP);
119  mongoc_cursor_destroy(cursor2);
121  int uniqueBMPID = BMPID * 100000 + subScenario;
122  if (this->m_bmpFactories.find(uniqueBMPID) == this->m_bmpFactories.end()) {
123  if (BMPID == BMP_TYPE_POINTSOURCE) {
124  this->m_bmpFactories[uniqueBMPID] = new BMPPointSrcFactory(m_id, BMPID, subScenario, BMPType,
125  BMPPriority, distribution,
126  collectionName, location);
127  }
128  if (BMPID == BMP_TYPE_PLANT_MGT) {
129  this->m_bmpFactories[uniqueBMPID] = new BMPPlantMgtFactory(m_id, BMPID, subScenario, BMPType,
130  BMPPriority, distribution,
131  collectionName, location);
132  }
133  if (BMPID == BMP_TYPE_AREALSOURCE) {
134  this->m_bmpFactories[uniqueBMPID] = new BMPArealSrcFactory(m_id, BMPID, subScenario, BMPType,
135  BMPPriority, distribution,
136  collectionName, location);
137  }
138  }
139  }
140  bson_destroy(query);
141  mongoc_cursor_destroy(cursor);
142 }
