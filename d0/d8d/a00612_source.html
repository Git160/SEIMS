91 int SNO_WB::Execute() {
92  this->CheckInputData();
93 
94  this-> InitialOutputs();
95 
96  /*if(m_subbasinList == NULL && this->m_subbasinSelected != NULL && this->m_subbasinSelectedCount > 0)
97  {
98  getSubbasinList(this->m_nCells,this->m_subbasin,this->m_subbasinSelectedCount ,this->m_subbasinSelected);
99  }*/
100 
101  if (m_SR == NULL)
102  {
103  m_SR = new float[m_nCells];
104  for (int i = 0; i < this->m_nCells; i++) {
105  m_SR[i] = 0.f;
106  }
107  }
108 
110  for (int rw = 0; rw < this->m_nCells; rw++) {
111  float dT = this->m_tMean[rw];
112  float dPnet = this->m_Pnet[rw];
113  float dSE = this->m_SE[rw];
114  float dSR = this->m_SR[rw];
115  float dSA = this->m_SA[rw];
116  float dSM = this->m_SM[rw];
117 
118  float dtmp2 = dSA + dSR - dSE - dSM;
119  if (dT <= this->m_tsnow) //snowfall
120  {
121  dtmp2 += (1 + this->m_kblow) * dPnet;
122  } else if (dT > this->m_tsnow && dT < this->m_t0 && dSA >
123  dPnet) //rain on snow, if the snow accumulation is larger than net precipitation, assume all the precipitation is accumulated in snow pack.
124  {
125  dtmp2 += dPnet;
126  }
127 
128  this->m_SA[rw] = max(dtmp2, 0.0f);
129 
130  this->m_SWE += this->m_SA[rw];
131  }
132 
133  this->m_SWE /= this->m_nCells;
134  return 0;
135 }
