91 int GWaterReservoir::Execute(void) {
92  InitOutputs();
93  CheckInputData();
94 #pragma omp parallel for
95  for (int i = 0; i <= m_nReaches; i++) {
96  m_percSubbasin[i] = 0.f;
97  }
98 
99  // get percolation for each subbasin
100  for (int i = 0; i < m_nCells; i++) {
101 #ifdef MULTIPLY_REACHES
102  int subbasinId = (int) m_subbasin[i];
103 #else
104  int subbasinId = 1;
105 #endif /* MULTIPLY_REACHES */
106  m_percSubbasin[subbasinId] += m_recharge[i];
107  }
108 
109  //float sum = 0.f;
110 #pragma omp parallel for //reduction(+:sum)
111  for (int i = 1; i <= m_nReaches; i++) {
112  float percolation = m_percSubbasin[i] * (1.f - m_deepCoefficient) / m_nCellsSubbasin[i];
113  // depth of groundwater runoff(mm)
114  float outFlowDepth = m_recessionCoefficient * pow(m_storage[i], m_recessionExponent);
115  // groundwater flow out of the subbasin at time t (m3/s)
116  m_qg[i] = outFlowDepth / 1000.f * m_nCellsSubbasin[i] * m_CellWidth * m_CellWidth / m_dt;
117  //sum = sum + m_qg[i];
118 
119  // water balance (mm)
120  m_storage[i] += percolation - outFlowDepth;
121  }
122  //m_qg[0] = sum;
123 
124  return 0;
125 }
