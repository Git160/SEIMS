83 void InputStation::ReadSitesInfo(string siteType, string hydroDBName, string sitesList) {
84  vector <string> vecSites = SplitString(sitesList, ',');
85  int nSites = (int) vecSites.size();
86  //convert from string to int, the IDList is in order in MongoDB

88  for (int iSite = 0; iSite < nSites; iSite++) {
89  siteIDList.push_back(atoi(vecSites[iSite].c_str()));
90  }
91  //sort(siteIDList.begin(), siteIDList.end());
92 
93  bson_t *query = bson_new();
94  build_query_bson(nSites, siteIDList, siteType, query);
95  //printf("%s\n",bson_as_json(query,NULL));
96 
97  unique_ptr<MongoCollection> collection(new MongoCollection(m_conn->getCollection(hydroDBName, DB_TAB_SITES)));
98  mongoc_cursor_t* cursor = collection->ExecuteQuery(query);
99 
100  const bson_t *record = NULL;
101  float *pEle = new float[nSites];
102  float *pLat = new float[nSites];
103  float ele, lat;
104  bool hasData = false;
105  vector<int>::iterator siteIDIter;
106  while (mongoc_cursor_more(cursor) && mongoc_cursor_next(cursor, &record)) {
107  hasData = true;
108  bson_iter_t iter;
109  int siteIndex = -1;
110  int curSiteID = -1;
111  if (bson_iter_init(&iter, record) && bson_iter_find(&iter, MONG_HYDRO_DATA_SITEID)) {
112  GetNumericFromBsonIterator(&iter, curSiteID);
113  } else {
114  throw ModelException("Measurement", "Measurement",
115  "The Site ID field does not exist in Sites table, Please check and retry.");
116  }
117  siteIDIter = find(siteIDList.begin(), siteIDList.end(), curSiteID);
118  siteIndex = distance(siteIDList.begin(), siteIDIter);
119  if (bson_iter_init(&iter, record) && bson_iter_find(&iter, MONG_HYDRO_SITE_LAT)) {
120  GetNumericFromBsonIterator(&iter, lat);
121  } else {
122  throw ModelException("InputStation", "ReadSitesInfo", "The Lat field does not exist in Sites table.");
123  }
124 
125  if (bson_iter_init(&iter, record) && bson_iter_find(&iter, MONG_HYDRO_SITE_ELEV)) {
126  GetNumericFromBsonIterator(&iter, ele);
127  } else {
128  throw ModelException("InputStation", "ReadSitesInfo", "The Lat field does not exist in Sites table.");
129  }
130  pLat[siteIndex] = lat;
131  pEle[siteIndex] = ele;
132  }
133  if (!hasData) {
134  throw ModelException("InputStation", "ReadSitesInfo", "Query failed.");
135  }
136 
137  mongoc_cursor_destroy(cursor);
138  bson_destroy(query);
139  m_elevation[siteType] = pEle;
140  m_latitude[siteType] = pLat;
141  m_numSites[siteType] = nSites;
142 }
