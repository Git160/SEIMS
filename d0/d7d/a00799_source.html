805 template<typename T>
806 void utilsMath::basicStatistics(const T *values, int num, double **derivedvalues, T exclude /* = (T) NODATA_VALUE */) {
807  double *tmpstats = new double[6];
808  double maxv = MISSINGFLOAT;
809  double minv = MAXIMUMFLOAT;
810  int validnum = 0;
811  double sumv = 0.;
812  double std = 0.;
813  for (int i = 0; i < num; i++) {
814  if (utilsMath::FloatEqual(values[i], exclude)) continue;
815  if (maxv < values[i]) maxv = values[i];
816  if (minv > values[i]) minv = values[i];
817  validnum += 1;
818  sumv += values[i];
819  }
820  double mean = sumv / (double) validnum;
821 #pragma omp parallel for reduction(+:std)
822  for (int i = 0; i < num; i++) {
823  if (!utilsMath::FloatEqual(values[i], exclude)) {
824  std += (values[i] - mean) * (values[i] - mean);
825  }
826  }
827  std = sqrt(std / (double) validnum);
828  tmpstats[0] = (double) validnum;
829  tmpstats[1] = mean;
830  tmpstats[2] = maxv;
831  tmpstats[3] = minv;
832  tmpstats[4] = std;
833  tmpstats[5] = maxv - minv;
834  *derivedvalues = tmpstats;
835 }

859  tmpstats[0][j] += 1;
