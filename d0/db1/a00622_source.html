238 int IMP_SWAT::Execute() {
239  CheckInputData();
240  InitialOutputs();
241 
242  for (int ilyr = 0; ilyr < m_nRteLyrs; ilyr++) {
243  // There are not any flow relationship within each routing layer.
244  // So parallelization can be done here.
245  int ncells = CVT_INT(m_rteLyrs[ilyr][0]);
246 #pragma omp parallel for
247  for (int icell = 1; icell <= ncells; icell++) {
248  int id = CVT_INT(m_rteLyrs[ilyr][icell]); // cell index
249  if (FloatEqual(m_impoundTrig[id], 0.f)) {
251  PotholeSimulate(id);
252  } else {
253  ReleaseWater(id);
254  }
255  }
256  }
258 #pragma omp parallel for
259  for (int i = 0; i <= m_nSubbasins; i++) {
260  m_surfqToCh[i] = 0.f;
261  m_sedToCh[i] = 0.f;
262  m_surNO3ToCh[i] = 0.f;
263  m_surNH4ToCh[i] = 0.f;
264  m_surSolPToCh[i] = 0.f;
265  m_surCodToCh[i] = 0.f;
266  m_sedOrgNToCh[i] = 0.f;
267  m_sedOrgPToCh[i] = 0.f;
268  m_sedMinPAToCh[i] = 0.f;
269  m_sedMinPSToCh[i] = 0.f;
270  }
271  // See https://github.com/lreis2415/SEIMS/issues/36 for more descriptions. By lj
272 #pragma omp parallel
273  {
274  float* tmp_surfq2ch = new(nothrow) float[m_nSubbasins + 1];
275  float* tmp_sed2ch = new(nothrow) float[m_nSubbasins + 1];
276  float* tmp_sno32ch = new(nothrow) float[m_nSubbasins + 1];
277  float* tmp_snh42ch = new(nothrow) float[m_nSubbasins + 1];
278  float* tmp_solp2ch = new(nothrow) float[m_nSubbasins + 1];
279  float* tmp_cod2ch = new(nothrow) float[m_nSubbasins + 1];
280  float* tmp_orgn2ch = new(nothrow) float[m_nSubbasins + 1];

282  float* tmp_minpa2ch = new(nothrow) float[m_nSubbasins + 1];
283  float* tmp_minps2ch = new(nothrow) float[m_nSubbasins + 1];
284  for (int i = 0; i <= m_nSubbasins; i++) {
285  tmp_surfq2ch[i] = 0.f;
286  tmp_sed2ch[i] = 0.f;
287  tmp_sno32ch[i] = 0.f;
288  tmp_snh42ch[i] = 0.f;
289  tmp_solp2ch[i] = 0.f;
290  tmp_cod2ch[i] = 0.f;
291  tmp_orgn2ch[i] = 0.f;
292  tmp_orgp2ch[i] = 0.f;
293  tmp_minpa2ch[i] = 0.f;
294  tmp_minps2ch[i] = 0.f;
295  }
296 #pragma omp for
297  for (int i = 0; i < m_nCells; i++) {
298  int subi = CVT_INT(m_subbasin[i]);
299  tmp_surfq2ch[subi] += m_surfaceRunoff[i] * 10.f / m_timestep;
300  tmp_sed2ch[subi] += m_sedYield[i];

302  tmp_snh42ch[subi] += m_surqNH4[i];
303  tmp_solp2ch[subi] += m_surqSolP[i];
304  tmp_cod2ch[subi] += m_surqCOD[i];

306  tmp_orgp2ch[subi] += m_sedOrgP[i];

309  }
310 #pragma omp critical
311  {
312  for (int i = 1; i <= m_nSubbasins; i++) {
313  m_surfqToCh[i] += tmp_surfq2ch[i] * m_cellArea;
314  m_sedToCh[i] += tmp_sed2ch[i];
315  m_surNO3ToCh[i] += tmp_sno32ch[i] * m_cellArea;
316  m_surNH4ToCh[i] += tmp_snh42ch[i] * m_cellArea;
317  m_surSolPToCh[i] += tmp_solp2ch[i] * m_cellArea;

320  m_sedOrgPToCh[i] += tmp_orgp2ch[i] * m_cellArea;
321  m_sedMinPAToCh[i] += tmp_minpa2ch[i] * m_cellArea;

324  }
325  delete[] tmp_surfq2ch;

329  delete[] tmp_solp2ch;

332  delete[] tmp_orgp2ch;
333  delete[] tmp_minpa2ch;
334  delete[] tmp_minps2ch;
335  } /* END of #pragma omp parallel */
336 
337  for (int i = 1; i <= m_nSubbasins; i++) {
338  m_surfqToCh[0] += m_surfqToCh[i];
339  m_sedToCh[0] += m_sedToCh[i];
340  m_surNO3ToCh[0] += m_surNO3ToCh[i];
