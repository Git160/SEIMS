64 int IUH_SED_OL::Execute() {
65  CheckInputData();
66  initialOutputs();
67  // delete value of last time step
68 #pragma omp parallel for
69  for (int i = 0; i < m_nSubbasins + 1; i++) {
70  m_sedtoCh[i] = 0.f;
71  }
72 
73 #pragma omp parallel for
74  for (int i = 0; i < m_nCells; i++) {
75  //forward one time step
76  for (int j = 0; j < m_cellFlowCols; j++) {
77  if (j != m_cellFlowCols - 1) {
78  m_cellSed[i][j] = m_cellSed[i][j + 1];
79  } else {
80  m_cellSed[i][j] = 0.f;
81  }
82  }
83 
84  if (m_sedYield[i] > 0.f) {
85  int min = int(this->m_iuhCell[i][0]);
86  int max = int(this->m_iuhCell[i][1]);
87  int col = 2;
88  for (int k = min; k <= max; k++) {
89  m_cellSed[i][k] += m_sedYield[i] * m_iuhCell[i][col];
90  col++;
91  }
92  }
93  }
94 
95  for (int i = 0; i < m_nCells; i++) {
96  //add today's flow
97  int subi = (int) m_subbasin[i];
98  if (m_nSubbasins == 1) {
99  subi = 1;
100  } else if (subi >= m_nSubbasins + 1) {
101  throw ModelException(MID_IUH_SED_OL, "Execute", "The subbasin " + ValueToString(subi) + " is invalid.");
102  }
103  m_sedtoCh[subi] += m_cellSed[i][0];
104  m_sedOL[i] = m_cellSed[i][0];
105  //if(i == 1852) cout << m_cellSed[i][0] << endl;
106  }
107  //cout << m_sedtoCh[1] << endl;
108  float tmp = 0.f;
109 #pragma omp parallel for reduction(+:tmp)
110  for (int i = 1; i < m_nSubbasins + 1; i++) {
111  tmp += m_sedtoCh[i]; //get overland flow routing for entire watershed.
112  }
113  m_sedtoCh[0] = tmp;
114 
115  return 0;
116 }
