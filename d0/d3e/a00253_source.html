156  def calculate_95ppu_efficiency(obs_data_list, obs_dates_list, sim_dates_list):
157  count = 0
158  ylows_obs = list()
159  yups_obs = list()
160  for oi, ov in enumerate(obs_data_list):
161  try:
162  si = sim_dates_list.index(obs_dates_list[oi])
163  ylows_obs.append(ylows[si])
164  yups_obs.append(yups[si])
165  if ylows[si] <= ov <= yups[si]:
166  count += 1
167  except Exception:
168  continue
169  p = float(count) / len(obs_data_list)
170  ylows_obs = numpy.array(ylows_obs)
171  yups_obs = numpy.array(yups_obs)
172  r = numpy.mean(yups_obs - ylows_obs) / numpy.std(numpy.array(obs_data_list))
173  return p, r
