66 int DepressionFSDaily::Execute() {
67  //check the data
68  CheckInputData();
69  initialOutputs();
70 
71 #pragma omp parallel for
72  for (int i = 0; i < m_nCells; ++i) {
74  // runoff
75  if (m_depCap[i] < 0.001f) {
76  m_sr[i] = m_pe[i];
77  m_sd[i] = 0.f;
78  } else if (m_pe[i] > 0.f) {
79  float pc = m_pe[i] - m_depCap[i] * log(1.f - m_sd[i] / m_depCap[i]);
80  float deltaSd = m_pe[i] * exp(-pc / m_depCap[i]);
81  if (deltaSd > m_depCap[i] - m_sd[i]) {
82  deltaSd = m_depCap[i] - m_sd[i];
83  }
84  m_sd[i] += deltaSd;
85  m_sr[i] = m_pe[i] - deltaSd;
86  } else {
87  m_sd[i] += m_pe[i];
88  m_sr[i] = 0.f;
89  }
90 
92  // evaporation
93  if (m_sd[i] > 0) {
97  // evaporation from depression storage
98  if (m_pet[i] - m_ei[i] < m_sd[i]) {
99  m_ed[i] = m_pet[i] - m_ei[i];
100  } else {
101  m_ed[i] = m_sd[i];
102  }
103  m_sd[i] -= m_ed[i];
104  } else {
105  m_ed[i] = 0.f;
106  m_sd[i] = 0.f;
107  }
108  if (m_impoundTriger != NULL && FloatEqual(m_impoundTriger[i], 0.f)) {
109  if (m_potVol != NULL) {
110  m_potVol[i] += m_sr[i];
111  m_potVol[i] += m_sd[i];
112  m_sr[i] = 0.f;
113  m_sd[i] = 0.f;
114  }
115  }
116  }
117  return true;
118 }
