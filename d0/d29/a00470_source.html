18 def lhs(n, samples=None, criterion=None, iterations=None):
19  """
20  Generate a latin-hypercube design
21 
22  Parameters
23  ----------
24  n : int
25  The number of factors to generate samples for
26 
27  Optional
28  --------
29  samples : int
30  The number of samples to generate for each factor (Default: n)
31  criterion : str
32  Allowable values are "center" or "c", "maximin" or "m",
33  "centermaximin" or "cm", and "correlation" or "corr". If no value
34  given, the design is simply randomized.
35  iterations : int
36  The number of iterations in the maximin and correlations algorithms
37  (Default: 5).
38 
39  Returns
40  -------
41  H : 2d-array
42  An n-by-samples design matrix that has been normalized so factor values
43  are uniformly spaced between zero and one.
44 
45  Example
46  -------
47  A 3-factor design (defaults to 3 samples)::
48 
49  >>> lhs(3)
50  array([[ 0.40069325, 0.08118402, 0.69763298],
51  [ 0.19524568, 0.41383587, 0.29947106],
52  [ 0.85341601, 0.75460699, 0.360024 ]])
53 
54  A 4-factor design with 6 samples::
55 
56  >>> lhs(4, samples=6)
57  array([[ 0.27226812, 0.02811327, 0.62792445, 0.91988196],
58  [ 0.76945538, 0.43501682, 0.01107457, 0.09583358],
59  [ 0.45702981, 0.76073773, 0.90245401, 0.18773015],
60  [ 0.99342115, 0.85814198, 0.16996665, 0.65069309],
61  [ 0.63092013, 0.22148567, 0.33616859, 0.36332478],
62  [ 0.05276917, 0.5819198 , 0.67194243, 0.78703262]])
63 
64  A 2-factor design with 5 centered samples::
65 
66  >>> lhs(2, samples=5, criterion='center')
67  array([[ 0.3, 0.5],
68  [ 0.7, 0.9],
69  [ 0.1, 0.3],
70  [ 0.9, 0.1],
71  [ 0.5, 0.7]])
72 
73  A 3-factor design with 4 samples where the minimum distance between
74  all samples has been maximized::
75 
76  >>> lhs(3, samples=4, criterion='maximin')
77  array([[ 0.02642564, 0.55576963, 0.50261649],
78  [ 0.51606589, 0.88933259, 0.34040838],
79  [ 0.98431735, 0.0380364 , 0.01621717],
80  [ 0.40414671, 0.33339132, 0.84845707]])
81 
82  A 4-factor design with 5 samples where the samples are as uncorrelated
83  as possible (within 10 iterations)::
84 
85  >>> lhs(4, samples=5, criterion='correlate', iterations=10)
86 
87  """
88  H = None
89 
90  if samples is None:
91  samples = n
92 
93  if criterion is not None:
94  assert criterion.lower() in ('center', 'c', 'maximin', 'm',
95  'centermaximin', 'cm', 'correlation',
96  'corr'), 'Invalid value for "criterion": {}'.format(criterion)
97  else:
98  H = _lhsclassic(n, samples)
99 
100  if criterion is None:
101  criterion = 'center'
102 
103  if iterations is None:
104  iterations = 5
105 
106  if H is None:
107  if criterion.lower() in ('center', 'c'):
108  H = _lhscentered(n, samples)
109  elif criterion.lower() in ('maximin', 'm'):
110  H = _lhsmaximin(n, samples, iterations, 'maximin')
111  elif criterion.lower() in ('centermaximin', 'cm'):
112  H = _lhsmaximin(n, samples, iterations, 'centermaximin')
113  elif criterion.lower() in ('correlate', 'corr'):
114  H = _lhscorrelate(n, samples, iterations)
115 
116  return H

121 def _lhsclassic(n, samples):
122  # Generate the intervals
123  cut = np.linspace(0, 1, samples + 1)
124 
125  # Fill points uniformly in each interval
126  u = np.random.rand(samples, n)
127  a = cut[:samples]
128  b = cut[1:samples + 1]
129  rdpoints = np.zeros_like(u)
130  for j in range(n):
131  rdpoints[:, j] = u[:, j] * (b - a) + a
132 
133  # Make the random pairings
134  H = np.zeros_like(rdpoints)
135  for j in range(n):
136  order = np.random.permutation(range(samples))
137  H[:, j] = rdpoints[order, j]
138 
139  return H

144 def _lhscentered(n, samples):
145  # Generate the intervals
146  cut = np.linspace(0, 1, samples + 1)
147 
148  # Fill points uniformly in each interval
149  u = np.random.rand(samples, n)
150  a = cut[:samples]
151  b = cut[1:samples + 1]
152  _center = (a + b) / 2
153 
154  # Make the random pairings
155  H = np.zeros_like(u)
156  for j in range(n):
157  H[:, j] = np.random.permutation(_center)
158 
159  return H

164 def _lhsmaximin(n, samples, iterations, lhstype):
165  maxdist = 0
166 
167  # Maximize the minimum distance between points
168  for i in range(iterations):
169  if lhstype == 'maximin':
170  Hcandidate = _lhsclassic(n, samples)
171  else:
172  Hcandidate = _lhscentered(n, samples)
173 
174  d = _pdist(Hcandidate)
175  if maxdist < np.min(d):
176  maxdist = np.min(d)
177  H = Hcandidate.copy()
178 
179  return H

184 def _lhscorrelate(n, samples, iterations):
185  mincorr = np.inf
186 
187  # Minimize the components correlation coefficients
188  for i in range(iterations):
189  # Generate a random LHS
190  Hcandidate = _lhsclassic(n, samples)
191  R = np.corrcoef(Hcandidate)
192  if np.max(np.abs(R[R != 1])) < mincorr:
193  mincorr = np.max(np.abs(R - np.eye(R.shape[0])))
194  print('new candidate solution found with max,abs corrcoef = {}'.format(mincorr))
195  H = Hcandidate.copy()
196 
197  return H

202 def _pdist(x):
203  """
204  Calculate the pair-wise point distances of a matrix
205 
206  Parameters
207  ----------
208  x : 2d-array
209  An m-by-n array of scalars, where there are m points in n dimensions.
210 
211  Returns
212  -------
213  d : array
214  A 1-by-b array of scalars, where b = m*(m - 1)/2. This array contains
215  all the pair-wise point distances, arranged in the order (1, 0),
216  (2, 0), ..., (m-1, 0), (2, 1), ..., (m-1, 1), ..., (m-1, m-2).
217 
218  Examples
219  --------
220  ::
221 
222  >>> x = np.array([[0.1629447, 0.8616334],
223  ... [0.5811584, 0.3826752],
224  ... [0.2270954, 0.4442068],
225  ... [0.7670017, 0.7264718],
226  ... [0.8253975, 0.1937736]])
227  >>> _pdist(x)
228  array([ 0.6358488, 0.4223272, 0.6189940, 0.9406808, 0.3593699,
229  0.3908118, 0.3087661, 0.6092392, 0.6486001, 0.5358894])
230 
231  """
232 
233  x = np.atleast_2d(x)
234  assert len(x.shape) == 2, 'Input array must be 2d-dimensional'
235 
236  m, n = x.shape
237  if m < 2:
238  return []
239 
240  d = []
241  for i in range(m - 1):
242  for j in range(i + 1, m):
243  d.append((sum((x[j, :] - x[i, :]) ** 2)) ** 0.5)
244 
245  return np.array(d)
