42  """Algorithm modified from Whitebox GAT v3.4.0.
43  source code: https://github.com/jblindsay/whitebox-geospatial-analysis-tools/
44  blob/master/HydroTools/src/plugins/Hillslopes.java
45  Args:
46  stream_raster: Stream cell value greater than 0 is identified by stream
47  The input stream are recommended sequenced as 1, 2, 3...
48  flow_dir_raster: D8 flow direction in TauDEM code
49  hillslope_out: With the sequenced stream IDs, the output hillslope will be numbered:
50  - Header hillslope: MaxStreamID + (current_id - 1) * 3 + 1
51  - Right hillslope: MaxStreamID + (current_id - 1) * 3 + 2
52  - Left hillslope: MaxStreamID + (current_id - 1) * 3 + 3
53  d8alg: Currently, "TauDEM", "ArcGIS", and "Whitebox" are supported.
54  stream_value_method: stream value assigned method, depend on this parameter,
55  the output hillslope will be appended as follows:
56  -1 - all the four files will be output.
57  0 - keep stream link code, which has the default file name
58  1 - Set to the value of right hillslope and head hillslope, <name>_r.tif
59  2 - Set to the value of left hillslope and head hillslope, <name>_l.tif
60  3 - Set stream cell to NoData, <name>_n.tif
61  """

81  def inflow_stream_number(vrow, vcol, flowmodel="taudem"):
82  """
83  Count the inflow stream cell number and coordinates of all inflow cells
84  Args:
85  vrow: row number
86  vcol: col number
87  flowmodel: D8 flow direction algorithm.
88  Returns:
89  neighb_stream_cell_num: inflow cells number that is stream
90  cell_coors: inflow cell coordinates, the size() is equal or greater
91  than neighb_stream_cell_num
92  """
93  neighb_stream_cell_num = 0
94  cell_coors = []
95  for c in range(8):
96  newrow = vrow + FlowModelConst.ccw_drow[c]
97  newcol = vcol + FlowModelConst.ccw_dcol[c]
98  if newrow < 0 or newrow >= nrows or newcol < 0 or newcol >= ncols:
99  continue
100  if flowd8_data[newrow][newcol] == FlowModelConst.d8_inflows.get(flowmodel)[c]:
101  cell_coors.append((newrow, newcol))
102  if stream_data[newrow][newcol] > 0 \
103  and stream_data[newrow][newcol] != stream_nodata:
104  neighb_stream_cell_num += 1
105  return neighb_stream_cell_num, cell_coors

107  def assign_sequenced_stream_ids(c_id, vrow, vcol, flowmodel="taudem"):
108  """set sequenced stream IDs"""
109  in_strm_num, in_coors = inflow_stream_number(vrow, vcol, flowmodel)
110  if in_strm_num == 0:
111  # it's a headwater location so start a downstream flowpath
112  c_id += 1
113  tmp_row = vrow
114  tmp_col = vcol
115  sequenced_stream_d[tmp_row][tmp_col] = c_id
116  searched_flag = True
117  while searched_flag:
118  # find the downslope neighbour
119  tmpflowd8 = flowd8_data[tmp_row][tmp_col]
120  if tmpflowd8 < 0 or tmpflowd8 == flowd8_nodata:
121  if stream_data[tmp_row][tmp_col] > 0 \
122  and stream_data[tmp_row][tmp_col] != stream_nodata:
123  # it is a valid stream cell and probably just has no downslope
124  # neighbour (e.g. at the edge of the grid)
125  sequenced_stream_d[tmp_row][tmp_col] = c_id
126  break
127  tmp_row, tmp_col = D8Util.downstream_index(tmpflowd8, tmp_row,
128  tmp_col, flowmodel)
129  if tmp_row < 0 or tmp_row >= nrows or tmp_col < 0 or tmp_col >= ncols:
130  break
131  if stream_data[tmp_row][tmp_col] <= 0:
132  searched_flag = False # it is not a stream cell
133  else:
134  if sequenced_stream_d[tmp_row][tmp_col] > 0:
135  # run into a larger stream, end the downstream search
136  break
137  # is it a confluence (conjunction node)
138  in_strm_num, in_coors = inflow_stream_number(tmp_row, tmp_col, flowmodel)
139  if in_strm_num >= 2:
140  c_id += 1
141  sequenced_stream_d[tmp_row][tmp_col] = c_id
142  return c_id

144  def assign_hillslope_code_of_neighbors(vrow, vcol, flowmodel="taudem"):
145  """set hillslope code for neighbors of current stream cell."""
146  stream_coors.append((vrow, vcol))
147  in_strm_num, in_coors = inflow_stream_number(vrow, vcol, flowmodel)
148  strm_id = stream_data[vrow][vcol]
149  # print ("Assign hillslope code for stream cell, r: %d, c: %d, ID: %d" % (vrow, vcol,
150  # int(strm_id)))
151  # set hillslope IDs
152  hillslp_ids = DelineateHillslope.cal_hs_codes(max_id, strm_id)
153  cur_d8_value = flowd8_data[vrow][vcol]
154  if in_strm_num == 0: # it is a one-order stream head
155  headstream_coors.append((vrow, vcol))
156  for (in_nostrm_row, in_nostrm_col) in in_coors:
157  hillslope_mtx[in_nostrm_row][in_nostrm_col] = hillslp_ids[0]
158  else: # search the 3*3 neighbors by clockwise and counterclockwise separately
159  if cur_d8_value <= 0 or cur_d8_value == flowd8_nodata:
160  return
161  dirv = int(cur_d8_value) # direction code
162  d_idx = FlowModelConst.d8_dirs.get(flowmodel).index(dirv) # direction index
163  # look to the right side, i.e. clockwise
164  d_idx_r = d_idx
165  while True and len(in_coors) > 0:
166  d_idx_r -= 1
167  if d_idx_r > 7:
168  d_idx_r = 0
169  if d_idx_r < 0:
170  d_idx_r = 7
171  tmp_row = vrow + FlowModelConst.ccw_drow[d_idx_r]
172  tmp_col = vcol + FlowModelConst.ccw_dcol[d_idx_r]
173  if (tmp_row, tmp_col) not in in_coors: # not inflow to this cell
174  continue
175  tmpstream = stream_data[tmp_row][tmp_col]
176  in_coors.remove((tmp_row, tmp_col))
177  if tmpstream <= 0 or tmpstream == stream_nodata:
178  hillslope_mtx[tmp_row][tmp_col] = hillslp_ids[1] # right hillslope
179  else: # encounter another in flow stream
180  break
181 
182  # look to the left side, i.e. counterclockwise
183  d_idx_l = d_idx
184  while True and len(in_coors) > 0:
185  d_idx_l += 1
186  if d_idx_l > 7:
187  d_idx_l = 0
188  if d_idx_l < 0:
189  d_idx_l = 7
190  tmp_row = vrow + FlowModelConst.ccw_drow[d_idx_l]
191  tmp_col = vcol + FlowModelConst.ccw_dcol[d_idx_l]
192  if (tmp_row, tmp_col) not in in_coors: # not inflow to this cell
193  continue
194  tmpstream = stream_data[tmp_row][tmp_col]
195  in_coors.remove((tmp_row, tmp_col))
196  if tmpstream <= 0 or tmpstream == stream_nodata:
197  hillslope_mtx[tmp_row][tmp_col] = hillslp_ids[2] # left hillslope
198  else: # encounter another in flow stream
199  break
200  # if any inflow cells existed?
201  if len(in_coors) > 0:
202  for (tmp_row, tmp_col) in in_coors:
203  tmpstream = stream_data[tmp_row][tmp_col]
204  if tmpstream <= 0 or tmpstream == stream_nodata:
205  hillslope_mtx[tmp_row][tmp_col] = hillslp_ids[0]
206  # add the current cell as head stream
207  headstream_coors.append((vrow, vcol))

209  def output_hillslope(method_id):
210  """Output hillslope according different stream cell value method."""
211  for (tmp_row, tmp_col) in stream_coors:
212  tmp_hillslp_ids = DelineateHillslope.cal_hs_codes(max_id,
213  stream_data[tmp_row][tmp_col])
214  if 0 < method_id < 3:
215  hillslope_mtx[tmp_row][tmp_col] = tmp_hillslp_ids[method_id]
216  # is head stream cell?

279  for row in range(nrows):

281  if hillslope_mtx[row][col] > 0 or flowd8_data[row][col] == flowd8_nodata:
282  continue
283  flag = False
284  tmprow = row
285  tmpcol = col

291  if curflowdir <= 0 or curflowdir == flowd8_nodata:
292  break

295  if tmprow < 0 or tmprow >= nrows or tmpcol < 0 or tmpcol >= ncols:
296  break
297  # if the new cell already has a hillslope value, use that
298  if hillslope_mtx[tmprow][tmpcol] > 0:
299  hillslp_id = hillslope_mtx[tmprow][tmpcol]
300  flag = True

302  tmpcoors.append((tmprow, tmpcol))
