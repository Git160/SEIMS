47 int FindBoundingBox(IntRaster* rs_subbasin, map<int, SubBasin>& bbox_map) {
48  int n_xsize = rs_subbasin->GetCols();
49  int n_ysize = rs_subbasin->GetRows();
50 
51  int* p_data = rs_subbasin->GetRasterDataPointer();
52  int nodata_value = rs_subbasin->GetNoDataValue();
53  // find bounding box for each subbasin
54  for (int i = 0; i < n_ysize; i++) {
55  for (int j = 0; j < n_xsize; j++) {
56  int id = p_data[i * n_xsize + j];
57  if (nodata_value != id) {
58  if (bbox_map.find(id) == bbox_map.end()) {
59  bbox_map[id] = SubBasin(j, i, j, i);
60  } else {
61  if (j < bbox_map[id].x_min) {
62  bbox_map[id].x_min = j;
63  } else if (j > bbox_map[id].x_max) {
64  bbox_map[id].x_max = j;
65  }
66  if (i > bbox_map[id].y_max) {
67  bbox_map[id].y_max = i;
68  }
69  }
70  bbox_map[id].cell_count += 1;
71  }
72  }
73  }
74  return 0;
75 }

261  const char* dst_file, const char* tmp_folder) {

264

267 
268  float* rs_data = rs->GetRasterDataPointer();

272  GDALDriver* po_driver = GetGDALDriverManager()->GetDriverByName(psz_format);
273 
274  string core_name = GetCoreFileName(dst_file);
275  for (auto it = bbox_map.begin(); it != bbox_map.end(); ++it) {
276  int id = it->first;
277  int subbasin_id = id;
278  if (bbox_map.size() == 1) {
279  // means not for Cluster
280  id = 0;
281  }
282  SubBasin& subbasin = it->second;
283  int subXSize = subbasin.x_max - subbasin.x_min + 1;
284  int subYSize = subbasin.y_max - subbasin.y_min + 1;
285 
286  std::ostringstream oss;
287  oss << tmp_folder << "/" << id << "/" << GetUpper(core_name) << ".tif";
288  string subbasin_file = oss.str();
289  //cout << subbasinFile << endl;
290  // create a new raster for the subbasin
291  char** papsz_options = nullptr;
292  GDALDataset* po_dst_ds = po_driver->Create(subbasin_file.c_str(), subXSize, subYSize,
293  1, GDT_Float32, papsz_options);
294 
295  float* sub_data = reinterpret_cast<float *>(CPLMalloc(sizeof(float) * subXSize * subYSize));
296 #pragma omp parallel for
297  for (int i = subbasin.y_min; i <= subbasin.y_max; i++) {
298  for (int j = subbasin.x_min; j <= subbasin.x_max; j++) {
299  int index = i * n_xsize + j;
300  int sub_index = (i - subbasin.y_min) * subXSize + (j - subbasin.x_min);
301  if (subbasin_data[index] == subbasin_id) {
302  sub_data[sub_index] = rs_data[index];
303  } else {
304  sub_data[sub_index] = nodata_value;
305  }
306  }
307  }
308 
309  //write the data to new file
310  GDALRasterBand* po_dst_band = po_dst_ds->GetRasterBand(1);
311  po_dst_band->RasterIO(GF_Write, 0, 0, subXSize, subYSize,
312  sub_data, subXSize, subYSize, GDT_Float32, 0, 0);
313  po_dst_band->SetNoDataValue(nodata_value);
314 
315  double geo_trans[6];
316  float cell_size = CVT_FLT(rs->GetCellWidth());
317  geo_trans[0] = rs->GetXllCenter() + (subbasin.x_min - 0.5f) * cell_size;
318  geo_trans[1] = cell_size;
319  geo_trans[2] = 0;
320  geo_trans[3] = rs->GetYllCenter() + (rs->GetRows() - subbasin.y_min - 0.5f) * cell_size;
321  geo_trans[4] = 0;
322  geo_trans[5] = -cell_size;
323  po_dst_ds->SetGeoTransform(geo_trans);
324 
325  CPLFree(sub_data);
326  GDALClose(po_dst_ds);
327  }
328  delete rs;
329  return 0;
330 }
