100 /************ Implementation of template functions ******************/
101 template <typename T>
102 bool SplitStringForValues(const string& items, const char delimiter, vector<T>& values) {
103  vector<string> value_strs = SplitString(items, delimiter);
104  char* end = nullptr;
105  for (vector<string>::iterator it = value_strs.begin(); it != value_strs.end(); ++it) {
106  if ((*it).find_first_of("0123456789") == string::npos) {
107  continue;
108  }
109  values.emplace_back(static_cast<T>(strtod((*it).c_str(), &end)));
110  }
111  vector<T>(values).swap(values);
112  return true;
113 }
